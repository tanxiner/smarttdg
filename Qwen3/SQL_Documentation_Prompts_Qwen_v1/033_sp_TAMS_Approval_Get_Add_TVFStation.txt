CREATE PROCEDURE [dbo].[sp_TAMS_Approval_Get_Add_TVFStation]
	@TARID		BIGINT
AS

BEGIN

-- >> TVF Station

-- >> Table[0]
SELECT a.StationName AS StrStation, b.TVFDirection AS StrDirection, b.ID AS TVFID
	FROM TAMS_Station a, TAMS_TAR_TVF b
		WHERE a.ID = b.TVFStationId
			AND b.TARID = @TARID
	ORDER BY a.ID

-- >> 1
-- >> TVF Run Mode
SELECT TVFMode 
	FROM TAMS_TAR
		WHERE ID = @TARID


END
