create PROCEDURE [dbo].[sp_TAMS_RGS_Update_QTS_bak20221229]
	@TARID			AS BIGINT,
	@InchargeNRIC	AS NVARCHAR(50),
	@UserID			AS NVARCHAR(500),
	@Message		AS NVARCHAR(500) = NULL OUTPUT,
	@QTSQCode		AS NVARCHAR(50) = NULL OUTPUT,
	@QTSLine		AS NVARCHAR(10) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT

SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

CREATE TABLE #tmpnric 
(id BIGINT IDENTITY(1,1) NOT NULL, 
nric NVARCHAR(50),
namestr NVARCHAR(500),
line NVARCHAR(50),
qualdate NVARCHAR(50),
qualcode NVARCHAR(100),
qualstatus NVARCHAR(100))

TRUNCATE TABLE #tmpnric

DECLARE @ExecString	NVARCHAR(MAX)
SET @ExecString = ''
DECLARE @QualDate	NVARCHAR(20)
SET @QualDate = ''

DECLARE @InChargeName	NVARCHAR(500)
DECLARE @InChargeStatus	NVARCHAR(20)

SET @InChargeName = ''
SET @InChargeStatus = ''

DECLARE @Line NVARCHAR(20)
DECLARE @AccessDate DATE
DECLARE @AccessType NVARCHAR(50)
DECLARE @TOAID BIGINT

SELECT @Line = a.Line,
	@AccessDate = a.AccessDate,
	@TOAID = a.Id,
	@AccessType = b.AccessType
FROM TAMS_TOA a, TAMS_TAR b 
	WHERE a.TARId = @TARID
		AND a.TARId = b.Id

DECLARE @QTSQualCode		NVARCHAR(20)
DECLARE @QTSQualCodeProt	NVARCHAR(20)

SELECT @QTSQualCode = ParaValue2
	FROM TAMS_Parameters
		WHERE ParaCode = 'QTSQualCode'
			AND ParaValue1 = @Line
			AND ParaValue3 = 'Possession'
			AND GETDATE() BETWEEN EffectiveDate AND ExpiryDate

SELECT @QTSQualCodeProt = ParaValue2
	FROM TAMS_Parameters
		WHERE ParaCode = 'QTSQualCode'
			AND ParaValue1 = @Line
			AND ParaValue3 = ''
			AND GETDATE() BETWEEN EffectiveDate AND ExpiryDate

SET @QualDate = CONVERT(NVARCHAR(20), @AccessDate, 103)
SET @ExecString = 'sp_TAMS_TOA_QTS_Chk '+ '''' + @InchargeNRIC + ''', ''' + @QualDate + ''', ''' + @Line + ''', ''' + @QTSQualCode + ''''

INSERT INTO #tmpnric
EXEC(@ExecString)

SELECT @InChargeName = namestr,
	@InChargeStatus = qualstatus
FROM #tmpnric

DECLARE @QTSFinStatus NVARCHAR(20)
SET @QTSFinStatus = ''

DECLARE @QTSFinQualCode NVARCHAR(50)
SET @QTSFinQualCode = ''

IF @InChargeStatus	= 'InValid'
	BEGIN
		IF @AccessType = 'Protection'
			BEGIN
				TRUNCATE TABLE #tmpnric
				SET @InChargeName = ''
				SET @InChargeStatus = ''

				SET @ExecString = 'sp_TAMS_TOA_QTS_Chk '+ '''' + @InchargeNRIC + ''', ''' + @QualDate + ''', ''' + @Line + ''', ''' + @QTSQualCodeProt + ''''

				INSERT INTO #tmpnric
				EXEC(@ExecString)

				SELECT @InChargeName = namestr,
					@InChargeStatus = qualstatus
				FROM #tmpnric

				IF @InChargeStatus	= 'InValid'
					BEGIN
						SET @QTSFinStatus = 'InValid'
						SET @QTSFinQualCode = ''
					END
				ELSE
					BEGIN
						SET @QTSFinStatus = 'Valid'
						SET @QTSFinQualCode = @QTSQualCodeProt
					END
			END
		ELSE
			BEGIN
				SET @QTSFinStatus = 'InValid'
				SET @QTSFinQualCode = ''
			END
	END
ELSE
	BEGIN
		SET @QTSFinStatus = 'Valid'
		SET @QTSFinQualCode = @QTSQualCode
	END

DROP TABLE #tmpnric

--SELECT @QTSFinQualCode

IF @QTSFinStatus = 'InValid'
	BEGIN
		SET @Message = '1' -- >> Invalid Qualification
		SET @QTSQCode = ''
		SET @QTSLine = ''
	END
ELSE
	BEGIN

		UPDATE TAMS_TOA
			SET UpdateQTSTime = GETDATE(),
				UpdatedOn = GETDATE(),
				UpdatedBy = @UserID
			WHERE TARId = @TARID

		INSERT INTO TAMS_TOA_Audit
		SELECT @UserID, GETDATE(), 'U', * FROM TAMS_TOA
			WHERE TARId = @TARID

		SET @Message = '0' -- >> Updated
		SET @QTSQCode = @QTSFinQualCode
		SET @QTSLine = @Line

		----DECLARE @RetMsg NVARCHAR(500)
		----SET @RetMsg = ''
		------ >> EXEC [mssqldevpsvr1].[QTSDB].[dbo].[sp_api_tams_qts_upd_accessdate] @InchargeNRIC, @QTSFinQualCode, @Line, @UserID, @RetMsg OUTPUT
		----EXEC [mssqldevpsvr1].[QTSDB].[dbo].[sp_api_tams_qts_upd_accessdate_m] @InchargeNRIC, @QTSFinQualCode, @Line, @UserID
		------SELECT @RetMsg

		--IF @RetMsg = ''
			--BEGIN
			--	UPDATE TAMS_TOA
			--		SET UpdateQTSTime = GETDATE(),
			--			UpdatedOn = GETDATE(),
			--			UpdatedBy = @UserID
			--		WHERE TARId = @TARID

			--	INSERT INTO TAMS_TOA_Audit
			--	SELECT @UserID, GETDATE(), 'U', * FROM TAMS_TOA
			--		WHERE TARId = @TARID

			--	SET @Message = '0' -- >> Updated
			--END
		--ELSE
		--	BEGIN
		--		SET @Message = '2' -- >> Error Updating
		--	END

	END

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR INSERTING INTO TAMS_TOA'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END
