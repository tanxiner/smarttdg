CREATE PROCEDURE [dbo].[sp_TAMS_Update_Company_Details_By_ID] 	
	@CompID				INT,
	@Company			NVARCHAR(100),
	@BizOwner			NVARCHAR(200),
	@CompanyOfficeNo	NVARCHAR(20),
	@CompanyMobileNo	NVARCHAR(20),
	@CompanyEmail		NVARCHAR(200),
	@IsActive			BIT,
	@UpdatedBy			INT
AS
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION;
			IF EXISTS(SELECT * FROM TAMS_Company WHERE ID = @CompID) 
			BEGIN
				UPDATE TAMS_Company 
				SET Company = @Company, 
					BizOwner = @BizOwner, 
					CompanyOfficeNo = @CompanyOfficeNo, 
					CompanyMobileNo = @CompanyMobileNo, 
					CompanyEmail = @CompanyEmail,
					UpdatedBy = @UpdatedBy, 
					UpdatedOn = GETDATE()
				WHERE ID = @CompID;

			END;

		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		ROLLBACK TRANSACTION;
	END CATCH

END;
