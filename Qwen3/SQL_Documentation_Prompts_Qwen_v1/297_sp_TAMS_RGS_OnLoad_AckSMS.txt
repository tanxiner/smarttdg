### ⬇️ RAW SQL INPUT ⬇️
CREATE PROCEDURE [dbo].[sp_TAMS_RGS_OnLoad_AckSMS]
	@TARID		AS BIGINT = 0
AS

BEGIN

	SELECT b.AccessType, 
			ISNULL(CONVERT(NVARCHAR(10), a.AckGrantTOATime, 103) + ' ' + CONVERT(NVARCHAR(8), a.AckGrantTOATime, 108), '00:00:00') AS AckGrantTOATime, 
			ISNULL(CONVERT(NVARCHAR(10), a.AckProtectionLimitTime, 103) + ' ' + CONVERT(NVARCHAR(8), a.AckProtectionLimitTime, 108), '00:00:00') AS AckProtectionLimitTime,
			b.TARNo,
			a.TOANo
		FROM TAMS_TOA a, TAMS_TAR b
			WHERE b.Id = a.TARId
				AND a.TARId = @TARID
END
