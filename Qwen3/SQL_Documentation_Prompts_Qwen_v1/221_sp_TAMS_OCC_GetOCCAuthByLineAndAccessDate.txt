CREATE PROCEDURE [dbo].[sp_TAMS_OCC_GetOCCAuthByLineAndAccessDate]
	@Line AS nvarchar(10) = NULL,
	@AccessDate as nvarchar(50) = NULL


AS
BEGIN
	SELECT [ID]
	,[Line]
	,[OperationDate]
	,[AccessDate]
	,[TractionPowerId]
	,[Remark]
	,[PFRRemark]
	,[OCCAuthStatusId]
	,[IsBuffer]
	,[PowerOn]
	,[PowerOffTime]
	,[RackedOutTime]
	FROM [dbo].[TAMS_OCC_Auth]
	WHERE [Line] = @Line
	AND [AccessDate] = convert(datetime,@AccessDate,103)
	ORDER BY [ID] ASC
END
