CREATE procedure  [dbo].[sp_TAMS_RGS_OnLoad_20250128]
	@Line		NVARCHAR(20) = NULL,
	@TrackType	NVARCHAR(50) = NULL
AS

BEGIN

DECLARE @NewLine AS NVARCHAR(6)
SET @NewLine = '<br />'

DECLARE @RGSPossBG			NVARCHAR(20)
DECLARE @RGSProtBG			NVARCHAR(20)
DECLARE @RGSCancBG			NVARCHAR(20)

--DECLARE @trainLine as NVARCHAR(10)
--SET @trainLine='DTL'
--DECLARE @trackType as nvarchar(10)
--SET @trackType='mainline'

DECLARE @accessDate as Date
DECLARE @OperationDate as Date

DECLARE @CurDate		DATE
DECLARE @CurTime		TIME

SET @CurDate =  GETDATE()
SET @CurTime = CONVERT(TIME, GETDATE(), 108)


DECLARE @CutOff	TIME
SET @CutOff = '06:00:00'

	IF @CurTime > @CutOff
		BEGIN
			SET @OperationDate	= @CurDate
			SET @AccessDate		= DATEADD(DAY, 1, @CurDate)
		END
	ELSE
		BEGIN
			SET @OperationDate	= DATEADD(DAY, -1, @CurDate)
			SET @AccessDate		= @CurDate
		END
	
--SET @OperationDate = '2023-11-17'
--SET @AccessDate = '2023-11-18'

SET @OperationDate = '2024-08-01'
SET @AccessDate = '2024-08-30'

SELECT @RGSPossBG = ParaValue2 
	FROM TAMS_Parameters
		WHERE ParaCode = 'RGSPossessionBG'
			AND ParaValue1 = @Line

SELECT @RGSProtBG = ParaValue2 
	FROM TAMS_Parameters
		WHERE ParaCode = 'RGSProtectionBG'
			AND ParaValue1 = @Line

SELECT @RGSCancBG = ParaValue2 
	FROM TAMS_Parameters
		WHERE ParaCode = 'RGSCancelledBG'
			AND ParaValue1 = @Line


DECLARE @TOACallBackTime	NVARCHAR(10)
SET @TOACallBackTime = '04:00'


			
			DECLARE @possession_ctrl as bit=0

			SET @possession_ctrl=IIF(EXISTS(
			(
			
			SELECT counttotal from (
			SELECT Count(1) as counttotal 
			FROM TAMS_TAR a, TAMS_TOA b
			WHERE a.Id = b.TARId
			AND a.AccessDate = @accessDate
			AND b.TOAStatus NOT IN (0, 6)
			AND a.Line = 'DTL'
			and a.TrackType = @TrackType
			AND a.AccessType = 'Possession'
			) B where B.counttotal=(
			SELECT COUNT(*) 
			FROM TAMS_TAR a, TAMS_TOA b
			WHERE a.Id = b.TARId
			AND a.AccessDate = @accessDate
			AND b.TOAStatus NOT IN (0, 6)
			AND a.Line = 'DTL'
			and a.TrackType = @TrackType
			AND a.AccessType = 'Possession' 
			AND NOT (ISNULL(CONVERT(NVARCHAR(5), b.AckProtectionLimitTime, 108), '') = ''
			OR ISNULL(CONVERT(NVARCHAR(5), b.AckProtectionLimitTime, 108), '')= '00:00:00')) )),1,0)
			

			PRINT @possession_ctrl

SELECT ROW_NUMBER() OVER (ORDER BY a.AccessType, b.TOAStatus DESC, a.Id) as Sno,a.TARNo as TARNo, dbo.TAMS_Get_ES_NoBufferZone(a.Id) as ElectricalSections,
		CASE WHEN (a.Line='NEL' AND a.PowerOn = 1 AND a.AccessType='Possession')THEN (
			''
		)  WHEN (a.Line='NEL' ) THEN (
			(SELECT  ISNULL(CONVERT(NVARCHAR(5), max(c.PowerOffTime), 108), '')
					FROM TAMS_OCC_Auth c, TAMS_Traction_Power_Detail d
						WHERE c.TractionPowerId = d.TractionPowerId
							AND d.PowerSectionId in (
							SELECT  PowerSectorId from TAMS_TAR_Power_Sector where TARId = a.Id AND  ((a.PowerOn  = 0) or (a.PowerOn  = 1 and TAMS_TAR_Power_Sector.IsBuffer<>1))
								 )
							AND d.TractionPowerType = 'Sector'
							AND c.OperationDate = @OperationDate
							AND c.AccessDate = @accessDate
							 AND c.PowerOffTime is not null  and c.PowerOffType = 'Completed'
			)
		)  
		ELSE (
					SELECT  ISNULL(CONVERT(NVARCHAR(5), max(c.PowerOffTime), 108), '')
					FROM TAMS_OCC_Auth c, TAMS_Traction_Power_Detail d
						WHERE c.TractionPowerId = d.TractionPowerId
							 AND d.SectorId in (
												SELECT e.SectorId
											FROM TAMS_TAR_Sector e, TAMS_Sector f
												WHERE e.SectorId = f.ID
													AND e.TARId = a.Id
													AND ISNULL(f.IsGap, 0) = 0
													AND e.IsBuffer = 0
											)
							AND d.TractionPowerType = 'Sector'
							AND c.OperationDate = @OperationDate
							AND c.AccessDate = @accessDate
							 AND c.PowerOffTime IS NOT NULL and c.PowerOffType='Completed'
		)END as PowerOffTime, 
		CASE WHEN a.Line='DTL' THEN (
					SELECT  ISNULL(CONVERT(NVARCHAR(5), max(g.RackedOutTime), 108), '')
						FROM TAMS_OCC_Auth g, TAMS_Traction_Power_Detail h 
							WHERE g.TractionPowerId = h.TractionPowerId
								 AND h.SectorId in (
													SELECT g1.SectorId
												FROM TAMS_TAR_Sector g1, TAMS_Sector h1
													WHERE g1.SectorId = h1.ID
														AND g1.TARId = a.Id
														AND ISNULL(h1.IsGap, 0) = 0
														AND g1.IsBuffer = 0
												)
								AND h.TractionPowerType = 'Sector'
								AND g.OperationDate = @OperationDate
								AND g.AccessDate = @accessDate
								 AND g.RackedOutTime IS NOT NULL and g.RackedOutType='Completed'
				
		
		) ELSE (
					SELECT  ISNULL(CONVERT(NVARCHAR(5), max(i.RackedOutTime), 108), '')
					FROM TAMS_OCC_Auth i, TAMS_Traction_Power_Detail j, (
					SELECT i11.TARId 
					FROM TAMS_TAR_AccessReq i11, TAMS_Access_Requirement j11
						WHERE i11.IsPower = 1
							AND i11.OperationRequirement = j11.ID
							AND j11.ID = 27	-- >> 27 FOR NEL
							AND i11.IsSelected = 1
						) j2
						WHERE i.TractionPowerId = j.TractionPowerId
							AND j.PowerSectionId in (
								select PowerSectorId from TAMS_TAR_Power_Sector where TARId = a.Id
							)
							AND j2.TARId=a.Id
							AND j.TractionPowerType = 'Sector'
							AND i.OperationDate = @OperationDate
							AND i.AccessDate = @accessDate
							 AND i.RackedOutTime is not null and i.RackedOutType = 'Completed' 
						) END as CircuitBreakOutTime,
						dbo.TAMS_Get_TOA_Parties(b.Id) as PartiesName,
						b.NoOfParties  as NoOfPersons,a.DescOfWork as WorkDescription, (b.MobileNo + @NewLine + b.TetraRadioNo) as ContactNo
						,ISNULL(b.TOANo, '') as TOANo,
		--b.Id, a.TARNo, a.ARRemark, a.TVFMode, a.AccessType, b.TOAStatus, 
		-- >> ISNULL(CONVERT(NVARCHAR(20), b.AckProtectionLimitTime, 108), '00:00:00'), 
		--ISNULL(CONVERT(NVARCHAR(5), b.AckProtectionLimitTime, 108), ''), 
		--b.NoOfParties as NoOfPersons,
		--a.DescOfWork as WorkDescription, b.MobileNo as ContactNo, b.TetraRadioNo, ISNULL(b.TOANo, '') as TOANo,
		CASE WHEN ISNULL(b.TOANo, '') = '' THEN '' ELSE '04:00' END as CallBackTime,
		ISNULL(CONVERT(NVARCHAR(5), b.GrantTOATime, 108), '') as RadioMsgTime,
		ISNULL(CONVERT(NVARCHAR(5), b.AckSurrenderTime, 108), '') as LineClearMsgTime,
		[dbo].[SP_TAMS_RGS_Comments](a.Id,b.Id,'NEL') as Remarks,
		b.TOAStatus,
		CONVERT(bit,
		CASE WHEN a.Line='DTL' THEN 
		(
			SELECT IIF(EXISTS(SELECT * FROM TAMS_OCC_Auth a1, TAMS_Traction_Power_Detail b1
				WHERE a1.TractionPowerId = b1.TractionPowerId AND b1.SectorId in (
												SELECT a2.SectorId
											FROM TAMS_TAR_Sector a2, TAMS_Sector b2
												WHERE a2.SectorId = b2.ID
													AND a2.TARId = a.Id
													AND ISNULL(b2.IsGap, 0) = 0
													AND a2.IsBuffer = 0
											)
											AND b1.TractionPowerType = 'Sector'
											AND a1.OperationDate = @OperationDate
											AND a1.AccessDate = @accessDate
											AND ((a1.OCCAuthStatusId < 10 and a1.IsBuffer = 0) OR ((a1.OCCAuthStatusId < 7 and a1.IsBuffer = 1)))	 
										) or not EXISTS(SELECT aocc1.ID
									FROM TAMS_OCC_Auth aocc1 
									WHERE  aocc1.Line = @Line
									and aocc1.TrackType = @TrackType
									AND aocc1.OperationDate = @OperationDate
									AND aocc1.AccessDate = @AccessDate),0,1) 
		) ELSE (
			SELECT IIF(EXISTS(SELECT *
					FROM TAMS_OCC_Auth aa1, TAMS_Traction_Power_Detail bb1 
						WHERE aa1.TractionPowerId = bb1.TractionPowerId
							AND bb1.PowerSectionId in (
								select PowerSectorId from TAMS_TAR_Power_Sector where TARId = a.Id
							)
							AND bb1.TractionPowerType = 'Sector'
							AND aa1.OperationDate = @OperationDate
							AND aa1.AccessDate = @accessDate
							 AND aa1.OCCAuthStatusId < 8) or not exists(
								select aocc.ID FROM TAMS_OCC_Auth aocc, TAMS_Traction_Power_Detail bpd 
								WHERE aocc.TractionPowerId = bpd.TractionPowerId
									AND bpd.PowerSectionId in (
										select PowerSectorId from TAMS_TAR_Power_Sector where TARId = a.Id
									)
									AND bpd.TractionPowerType = 'Sector'
									AND aocc.OperationDate = @OperationDate
									AND aocc.AccessDate = @AccessDate
							 ),0,1)
		
		) END) as IsTOAAuth,
		CASE WHEN TOAStatus=6 THEN @RGSCancBG ELSE CASE WHEN a.AccessType='Possession' THEN @RGSPossBG ELSE @RGSProtBG END END as ColourCode,
		--Grant TOA
		CONVERT(bit,IIF(b.TOAStatus=6,1,IIF(a.Line='DTL',IIF(a.AccessType='Possession',1, IIF(@possession_ctrl=1,1,0)),1))) as IsGrantTOAEnable,
		ISNULL(CONVERT(NVARCHAR(5), b.UpdateQTSTime, 108), '') as UpdQTSTime,
		a.AccessType as AccessType,
		ISNULL(CONVERT(NVARCHAR(8), b.AckGrantTOATime, 108), '') as AckGrantTOATime,
		ISNULL(CONVERT(NVARCHAR(5), b.AckProtectionLimitTime, 108), '') as AckProtLimitTIme, 
		a.Id as TARID,
		b.Id as TOAID,
		dbo.DecryptString(b.InChargeNRIC) as InChargeNRIC
	FROM TAMS_TAR a, TAMS_TOA b
		WHERE a.Id = b.TARId
			AND a.AccessDate = @accessDate
			and a.TrackType = @TrackType
			AND b.TOAStatus <> 0
			AND a.Line = @Line
			AND UPPER(a.TrackType)=UPPER(@TrackType)
	-- >> ORDER BY a.AccessType, b.TOAType, a.Id
	--ORDER BY a.AccessType,a.TARNo
	ORDER BY a.AccessType, b.TOAStatus DESC, a.Id

-- >> Cancel List
SELECT a.Id AS Val, a.TARNo AS Txt
	FROM TAMS_TAR a, TAMS_TOA b
		WHERE a.Id = b.TARId
			AND b.TOAStatus NOT IN (0, 5, 6)
			AND a.AccessDate = @accessDate
			AND a.Line = @Line
			AND a.TrackType = @TrackType
	ORDER BY 1

SELECT CONVERT(NVARCHAR(20), @OperationDate, 103) AS OperationDate,  
	CONVERT(NVARCHAR(20), @accessDate, 103) AS AccessDate

	
END
