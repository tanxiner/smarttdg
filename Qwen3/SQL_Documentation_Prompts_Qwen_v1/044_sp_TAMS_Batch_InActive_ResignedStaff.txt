CREATE PROCEDURE [dbo].[sp_TAMS_Batch_InActive_ResignedStaff]
AS	
	
BEGIN	

UPDATE TAMS_User 
SET IsActive = 0 
WHERE IsActive = 1 
AND LoginId IN (
	SELECT s_lanid 
		FROM VMSDBSVR.ACRS.dbo.ResignedStaff 
			WHERE s_eff_day BETWEEN CONVERT(DATETIME, CONVERT(NVARCHAR(10), GETDATE() - 7, 103), 103) AND CONVERT(DATETIME, CONVERT(NVARCHAR(10), GETDATE(), 103), 103)
			-- WHERE s_eff_day < CONVERT(DATETIME, CONVERT(NVARCHAR(10), GETDATE(), 103), 103)
				AND ISNULL(s_lanid, '') <> ''
)


INSERT INTO [dbo].[TAMS_User_InActive]
([LoginId]
	,[Name]
	,[Email]
	,[OfficeNo]
	,[MobileNo]
	,[SAPNo]
	,[Department]
	,[ValidFrom]
	,[ValidTo]
	,[IsExternal]
	,[Password]
	,[CompanyID]
	,[SBSTContactPersonName]
	,[SBSTContactPersonDept]
	,[SBSTContactPersonOfficeNo]
	,[LastLogin]
	,[PasswordChangedDate]
	,[IsActive]
	,[CreatedOn]
	,[CreatedBy]
	,[UpdatedOn]
	,[UpdatedBy])
SELECT [LoginId]
	,[Name]
	,[Email]
	,[OfficeNo]
	,[MobileNo]
	,[SAPNo]
	,[Department]
	,[ValidFrom]
	,[ValidTo]
	,[IsExternal]
	,[Password]
	,[CompanyID]
	,[SBSTContactPersonName]
	,[SBSTContactPersonDept]
	,[SBSTContactPersonOfficeNo]
	,[LastLogin]
	,[PasswordChangedDate]
	,[IsActive]
	,[CreatedOn]
	,[CreatedBy]
	,[UpdatedOn]
	,[UpdatedBy]
	FROM [dbo].[TAMS_User]
WHERE LoginId IN (
	SELECT s_lanid 
		FROM VMSDBSVR.ACRS.dbo.ResignedStaff 
			WHERE s_eff_day BETWEEN CONVERT(DATETIME, CONVERT(NVARCHAR(10), GETDATE() - 7, 103), 103) AND CONVERT(DATETIME, CONVERT(NVARCHAR(10), GETDATE(), 103), 103)
			-- WHERE s_eff_day < CONVERT(DATETIME, CONVERT(NVARCHAR(10), GETDATE(), 103), 103)
				AND ISNULL(s_lanid, '') <> ''
)

DELETE FROM [dbo].[TAMS_User]
WHERE LoginId IN (
	SELECT s_lanid 
		FROM VMSDBSVR.ACRS.dbo.ResignedStaff 
			WHERE s_eff_day BETWEEN CONVERT(DATETIME, CONVERT(NVARCHAR(10), GETDATE() - 7, 103), 103) AND CONVERT(DATETIME, CONVERT(NVARCHAR(10), GETDATE(), 103), 103)
			-- WHERE s_eff_day < CONVERT(DATETIME, CONVERT(NVARCHAR(10), GETDATE(), 103), 103)
				AND ISNULL(s_lanid, '') <> ''
)

END;
