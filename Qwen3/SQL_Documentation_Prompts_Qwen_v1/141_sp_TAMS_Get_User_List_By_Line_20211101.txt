CREATE PROCEDURE [dbo].[sp_TAMS_Get_User_List_By_Line_20211101] 
	@CurrentUser	NVARCHAR(100),
	@SearchRail		NVARCHAR(10),
	@SearchUserType NVARCHAR(10),
	@SearchActive	NVARCHAR(10), 
	@SearchModule	NVARCHAR(10),
	@SearchUserID	NVARCHAR(100),
	@SearchUserName	NVARCHAR(200)
AS
BEGIN
	--To Keep Role ID List	
	CREATE TABLE #UserTable
	(
		ROWID [int] IDENTITY(1,1),
		ID INT,
		UserLine NVARCHAR(20),
		UserType NVARCHAR(20),
		UserID NVARCHAR(200),
		UserName NVARCHAR(100),
		Module	NVARCHAR(20),
		IsActive	BIT
	);	

	DECLARE @ID AS INT
	DECLARE @UserID AS NVARCHAR(100);
	DECLARE @UserType AS NVARCHAR(20);
	DECLARE @UserName AS NVARCHAR(100);
	DECLARE @Module AS NVARCHAR(20);
	DECLARE @IsActive AS BIT;

	DECLARE @ret AS NVARCHAR(100); 
	SET @ret = ''; 
	DECLARE @retmsg AS NVARCHAR(500); 
	SET @retmsg = ''; 

	DECLARE cur CURSOR FOR 
		SELECT u.UserId, u.LoginID, IIF(u.IsExternal = 0, 'Internal', 'External'), u.Name, u.IsActive
		FROM TAMS_User u, TAMS_User_Role ur, TAMS_Role r
		WHERE u.Userid = ur.Userid
		AND ur.RoleID = r.ID
		AND r.Line IN 
		(
			SELECT DISTINCT(r.Line) 
			FROM TAMS_User u, TAMS_User_Role ur, TAMS_Role r
			WHERE u.Userid = ur.Userid
			AND ur.RoleID = r.ID
			AND u.LoginID = @CurrentUser
		)
		AND u.NAME LIKE @SearchUserName
		AND u.LoginID LIKE @SearchUserID
		AND (r.Line LIKE @SearchRail OR r.Line LIKE '%ALL%')
		AND u.IsExternal LIKE @SearchUserType
		AND u.IsActive LIKE @SearchActive
		AND r.Module LIKE @SearchModule;
	

	OPEN cur;

	FETCH NEXT FROM cur INTO @ID, @UserID, @UserType, @UserName, @IsActive;

	WHILE @@FETCH_STATUS = 0
	BEGIN

		-- GET Modules TAR/OCC
		DECLARE @UModule AS NVARCHAR(10);
		DECLARE @HasTAR AS BIT;
		DECLARE @HasOCC AS BIT;

		SET @HasTAR = 0;
		SET @HasOCC = 0;

		DECLARE u_cur CURSOR FOR
			SELECT DISTINCT(r.Module) FROM TAMS_User u, TAMS_User_Role ur, TAMS_Role r
			WHERE u.Userid = ur.Userid
			AND ur.RoleID = r.ID
			AND u.LoginID = @UserID;

		OPEN u_cur;

		FETCH NEXT FROM u_cur INTO @UModule;

		WHILE @@FETCH_STATUS = 0
		BEGIN
			IF @UModule = 'TAR'
			BEGIN
				SET @HasTAR = 1;
			END
			ELSE IF @UModule = 'OCC'
			BEGIN
				SET @HasOCC = 1;
			END
			ELSE
			BEGIN
				SET @HasTAR = 1;
				SET @HasOCC = 1;
			END;

			FETCH NEXT FROM u_cur INTO @UModule;
		END;

		CLOSE u_cur;
		DEALLOCATE u_cur;

		IF @HasTAR = 1 AND @HasOCC = 1
		BEGIN
			SET @UModule = 'TAR, OCC';
		END
		ELSE IF @HasTAR = 1 
		BEGIN 
			SET @UModule = 'TAR';
		END
		ELSE IF @HasOCC = 1
		BEGIN
			SET @UModule = 'OCC';
		END
			
		--IF NOT EXISTS(SELECT * FROM #UserTable WHERE UserID = @UserID)
		--BEGIN
		--	INSERT INTO #UserTable VALUES
		--		(@ID, @UserType, @UserID, @UserName, @UModule, @IsActive);
		--END;
		
		--FETCH NEXT FROM cur INTO @ID, @UserID, @UserType, @UserName, @IsActive;

		-- GET Rail
		-- GET Modules TAR/OCC
		DECLARE @URail AS NVARCHAR(10);
		DECLARE @UserRail AS NVARCHAR(20);
		SET @UserRail = '';

		DECLARE u_curRail CURSOR FOR
			SELECT DISTINCT(r.Line) FROM TAMS_User u, TAMS_User_Role ur, TAMS_Role r
			WHERE u.Userid = ur.Userid
			AND ur.RoleID = r.ID
			AND u.LoginID = @UserID;

		OPEN u_curRail;

		FETCH NEXT FROM u_curRail INTO @URail;

		WHILE @@FETCH_STATUS = 0
		BEGIN
			IF @URail = 'All'
			BEGIN
				SET @UserRail = 'DTL, NEL, SPLRT';
				BREAK;
			END;
			
			IF LEN(@UserRail) = 0
			BEGIN
				SET @UserRail = @URail;
				
			END
			ELSE
			BEGIN
				SET @UserRail = @UserRail + ', ' + @URail;
			END;

			FETCH NEXT FROM u_curRail INTO @URail;
		END;

		CLOSE u_curRail;
		DEALLOCATE u_curRail;
			
		IF NOT EXISTS(SELECT * FROM #UserTable WHERE UserID = @UserID)
		BEGIN
			INSERT INTO #UserTable VALUES
				(@ID, @UserRail, @UserType, @UserID, @UserName, @UModule, @IsActive);
		END;
		
		FETCH NEXT FROM cur INTO @ID, @UserID, @UserType, @UserName, @IsActive;
	END

	CLOSE cur;
	DEALLOCATE cur;

	SELECT * FROM #UserTable;

	DROP TABLE #UserTable;
	
END
