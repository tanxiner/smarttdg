CREATE PROCEDURE [dbo].[sp_TAMS_OCC_AddTVFAckRemarks]
	@UserId			as int,
	@TVFAckId		as int,
	@TVFRemarks		as nvarchar(1000)

AS
	DECLARE @NewID AS int;
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION;
			INSERT INTO TAMS_TVF_Ack_Remark(
			TVFAckId,
			Remark,
			CreatedOn,
			CreatedBy,
			UpdatedOn,
			UpdatedBy)
			VALUES (
			@TVFAckId,
			@TVFRemarks,
			GETDATE(),
			@UserId,
			GETDATE(),
			@UserId)

			SELECT @NewID = SCOPE_IDENTITY();

			INSERT INTO TAMS_TVF_Ack_Remark_Audit(
			   [ActionBy]
			  ,[ActionOn]
			  ,[AuditAction]
			  ,[TVFAckRemarkId]
			  ,[TVFAckId]
			  ,[Remark]
			  ,[CreatedOn]
			  ,[CreatedBy]
			  ,[UpdatedOn]
			  ,[UpdatedBy])
			  SELECT 
			   @UserID
			  ,GETDATE()
			  ,'I'
			  ,@NewID
			  ,[TVFAckId]
			  ,[Remark]
			  ,[CreatedOn]
			  ,[CreatedBy]
			  ,[UpdatedOn]
			  ,[UpdatedBy]
			  FROM TAMS_TVF_Ack_Remark
		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		ROLLBACK TRANSACTION;
	END CATCH

END
