CREATE PROCEDURE [dbo].[sp_TAMS_TOA_Add_Parties]
	@PartiesFIN		AS NVARCHAR(50) = NULL,
	@PartiesName	AS NVARCHAR(200) = NULL,
	@IsTMC			AS NVARCHAR(5) = NULL,
	@NoOfParties	AS BIGINT = 0,
	@TOAID			AS BIGINT = 0,
	@Message		AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

DECLARE @PartCtr	BIGINT
DECLARE @DerIsTMC	BIGINT
SET @PartCtr = 0
SET @Message = ''

IF @IsTMC = 'Y'
	BEGIN
		SET @DerIsTMC = 1
	END
ELSE
	BEGIN
		SET @DerIsTMC = 0
	END
SELECT @PartCtr = COUNT(*) 
	FROM TAMS_TOA_Parties
		WHERE TOAId = @TOAID
			AND dbo.DecryptString(NRIC) = @PartiesFIN

	IF @PartCtr > 0
		BEGIN
			SET @Message = '1'
		END
	ELSE
		BEGIN
			UPDATE TAMS_TOA
				SET NoOfParties = @NoOfParties
			WHERE Id = @TOAID

			INSERT INTO [dbo].[TAMS_TOA_Parties]
				([TOAId], [Name], [NRIC], [IsInCharge], [IsWitness], [IsTMC], 
					[BookInTime], [BookOutTime], [BookInStatus])
			VALUES
				(@TOAID, @PartiesName, dbo.EncryptString(@PartiesFIN), 0, 0, @DerIsTMC,
					GETDATE(), NULL, 'In')

			SET @Message = ''
		END

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR INSERTING TAMS_TOA_Parties'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END
