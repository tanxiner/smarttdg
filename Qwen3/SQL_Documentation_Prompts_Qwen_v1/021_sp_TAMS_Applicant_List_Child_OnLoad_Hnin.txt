	@Line			AS NVARCHAR(10) = NULL,

INSERT INTO #TmpAppList
SELECT t.Id, t.TARNo, t.TARType, t.AccessType, t.Company, ws.WFStatus, t.AccessDate, ts.SectorId, s.Sector, CASE WHEN s.Direction IN ('BB', 'NB') THEN 1 ELSE 2 END, ISNULL(ts.ColourCode, '')
	FROM TAMS_TAR t, TAMS_TAR_Sector ts, TAMS_WFStatus ws, TAMS_Sector s
		WHERE t.Id = ts.TARId
			AND t.TARStatusId = ws.WFStatusId
			AND t.Line = ws.Line
			AND t.TrackType = ws.TrackType
			AND ts.SectorId = s.ID
			AND ws.WFType = 'TARWFStatus'
			AND t.Line = @Line
			AND t.TARStatusId <> 0
			AND s.IsActive = 1
			AND @CurrDate BETWEEN s.EffectiveDate AND s.ExpiryDate
			AND (t.AccessDate >= CONVERT(DATE, @ToAccessDate, 103) AND t.AccessDate <= CONVERT(DATE, @FromAccessDate, 103))
			AND (t.TARType = @TARType OR ISNULL(@TARType, '') = '')

SELECT b.TARID, b.TARNo, b.TARType, b.AccessDate, b.AccessType, b.Company, b.WFStatus, b.ColorCode 
	FROM #TmpAppList b
		WHERE b.SectorID = @SectorID
	GROUP BY b.TARID, b.TARNo, b.TARType, b.AccessDate, b.AccessType, b.Company, b.WFStatus, b.ColorCode 
	ORDER BY b.TARID
