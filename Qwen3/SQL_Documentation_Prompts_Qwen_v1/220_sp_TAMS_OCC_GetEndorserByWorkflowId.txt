CREATE PROCEDURE [dbo].[sp_TAMS_OCC_GetEndorserByWorkflowId]
	@ID AS INT = 0
AS
BEGIN

	SELECT ID, Line, WorkflowId, [Level], RoleId, Title, RequireValidation, RequireBuffer, RequireTVF, WFStatusId FROM TAMS_Endorser
	WHERE WorkflowId = @ID
	AND [Level] = 1
	AND EffectiveDate <= GETDATE()
	AND ExpiryDate >= GETDATE()
	AND IsActive = 1
	ORDER BY ID
END
