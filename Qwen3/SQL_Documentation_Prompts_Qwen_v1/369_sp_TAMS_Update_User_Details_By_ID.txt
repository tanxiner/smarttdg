CREATE PROCEDURE [dbo].[sp_TAMS_Update_User_Details_By_ID] 	
	@UserID			INT,
	@Name			NVARCHAR(100),
	@Email			NVARCHAR(200),
	@Mobile			NVARCHAR(20),
	@OfficeTel		NVARCHAR(20),
	@Dept			NVARCHAR(100),
	@ValidTo		NVARCHAR(20),
	@IsActive		BIT,
	@UpdatedBy		INT
AS
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION;
			IF EXISTS(SELECT * FROM TAMS_User WHERE UserID = @UserID) 
			BEGIN
				UPDATE TAMS_User 
				SET Name = @Name, 
					Email = @Email, 
					Department = @Dept, 
					OfficeNo = @OfficeTel,
					MobileNo = @Mobile,
					ValidTo = @ValidTo, 
					IsActive = @IsActive,
					UpdatedBy = @UpdatedBy, 
					UpdatedOn = GETDATE()
				WHERE UserID = @UserID;

			END;

		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		ROLLBACK TRANSACTION;
	END CATCH

END;
