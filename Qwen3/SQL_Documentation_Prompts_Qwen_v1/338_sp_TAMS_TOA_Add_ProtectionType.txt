CREATE PROCEDURE [dbo].[sp_TAMS_TOA_Add_ProtectionType]
	@pointno as dbo.Point readonly,
	@protectiontype as char(5),
	@toaid as int,
	@Message		AS NVARCHAR(500) = NULL OUTPUT,
	@CreatedBy as nvarchar(50)
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

SET @Message = ''

		BEGIN
			UPDATE TAMS_TOA
			SET ProtectionType=@protectiontype
			where Id=@toaid

			DELETE FROM TAMS_TOA_PointNo
			where TOAID=@toaid
		
			INSERT INTO [dbo].[TAMS_TOA_PointNo]
				(TOAID,PointNo,CreatedDate,CreatedBy)
			SELECT @toaid, pointno, getDate(), @CreatedBy from @pointno

			SET @Message = ''
		END

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR INSERTING TAMS_TOA_ProtectionType'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END
