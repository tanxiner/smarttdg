CREATE procedure [dbo].[EAS_Form_Check_View_Access_Lists]

	@P_Guid				varchar(225),
	@P_UserID		    varchar(15), 
	@P_ErrorMsg			varchar(500)  output	
 
AS
BEGIN

	set @P_ErrorMsg=''
	
	IF (Select COUNT(*)
		from EAS_User
		--where formguid=@P_Guid
		where userid=@P_UserID
		and active=1
		)= 0 
	BEGIN
	 set @p_Errormsg='You Are Not Authorised To View This Page.'
	 return
	END
		

END
