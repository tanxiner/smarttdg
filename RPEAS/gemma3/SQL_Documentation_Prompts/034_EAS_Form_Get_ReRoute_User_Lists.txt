CREATE procedure [dbo].[EAS_Form_Get_ReRoute_User_Lists]

	@P_Guid					varchar(225),
	@P_ReRoute_lvl		    varchar(1)  
	 
 
AS
BEGIN

	select userid,Name
	from EAS_USER
	where sysid='RPEAS'
	and active=1
	and userid not in (Select userid
			from EAS_Form_Approve_Lvl
			where formguid=@P_Guid)
	order by Name
		

END
