CREATE procedure [dbo].[EAS_Form_Update_Binary_File]

 
	 
	@p_formguid		nvarchar(225),
	@p_FileName	    nvarchar(255),
	@p_FileContentType nvarchar(50),
	@p_FileContent  varbinary(MAX),
	@p_errormsg		varchar(500)  output	
 
AS
BEGIN

  
 set @p_errormsg=''
 
 BEGIN TRY
 --BEGIN TRANSACTION;	
 ------------------------
  Update [dbo].[EAS_Form_Attach_Document]
         set [FileContent]=@p_FileContent,[FileContentType]=@p_FileContentType

  where [FormGuid]=@p_formguid
  and [FileName]=@p_FileName

--COMMIT TRANSACTION;
	END TRY
	 --Catch Error Message
	BEGIN CATCH 
 		set @P_ErrorMsg= ERROR_MESSAGE(); 		 
		--ROLLBACK TRANSACTION;
	END CATCH; 
------------------------------
END
