### ⬇️ RAW SQL INPUT ⬇️
create procedure [dbo].[EAS_Form_Get_Detail_Info_ExportPDF]

	@P_Guid				varchar(225) 
 
	 
AS
BEGIN

--Get Form Master (Table 0)
Select  [FormGuid],[DocRefNo],[DocType],[Company],[Title],[FormStatus],
        [VendorName],[ApprovedAmt],[Active]
from  dbo.EAS_Form_Master
where FormGuid=@P_Guid
and active=1

--Get Attached Document (Table 1,2)
SELECT '1' as SortOrder, ID as SNO,FormGuid,FileName,FileType,FileURL,FileSize,CreatedOn,Createdby,[FileContent],[FileContentType]
FROM EAS_Form_Attach_Document
where [FormGuid]=@P_Guid
and FileType='ApprovalDoc'
union all
SELECT  '2' as SortOrder, ID as SNO,FormGuid,FileName,FileType,FileURL,FileSize,CreatedOn,Createdby,[FileContent],[FileContentType]
FROM EAS_Form_Attach_Document
where [FormGuid]=@P_Guid
and FileType='SupportDoc'
order by SortOrder,ID
---------------------
-------------------
--Get Approver List - table 3

select dbo.EAS_User_Name(Userid) as UserName, 
[dbo].[EAS_User_Dept_Design](Userid,'1') as UserDept, 
[dbo].[EAS_User_Dept_Design](Userid,'2') as UserDesign, 
a.Action,a.ActionOn,
       --case when  isnull(a.Action,'x')='x' then 'N' else 'Y' end CompleteStatus,
	   --case when  isnull(a.ChkDeclareConflit,'x')='Y' then 'True' else 'False' end ChkDeclareTrueFlase, 
	   isnull(b.Remarks,'')Remarks,
	   case a.GroupLevel when 1 then 'Prepared by'
					     when 2 then 'Submitted By (Optional)'
						 when 3 then 'Submitted Thru (Optional)'
						 when 4 then 'Supported By (Optional)'
						 when 5 then 'Endorsed By (Optional)'
						 when 6 then 'Approved By' end ApproverRole
from EAS_Form_Approve_Lvl a  left outer join 
	dbo.EAS_Form_Log_History b on a.FormGuid=b.FormGuid
							   and a.ActionBy=b.ActionBy
 
where a.FormGuid=@P_Guid
order by  2,3
--------------------

END
