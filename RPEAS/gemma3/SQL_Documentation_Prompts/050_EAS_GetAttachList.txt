CREATE   PROCEDURE [dbo].[EAS_GetAttachList]
 
 
AS
BEGIN
	 
	 
	 
	 -------------------------

	 select FileRefNo,[FILE_ID],[FILE_NAME],FILE_PATH,FILE_CONTENT,CONT_TYPE,
	 'D:\CanBusOutput\REAPS_UploadFile\' + SUBSTRING(FileURL,1, CHARINDEX('/',FileURL) -1) + '\' as FilePath
	 from ( 
	select  b.FormGuid as FileRefNo,a.id as [FILE_ID], 
	     a.FileName as  [FILE_NAME],   a.FileURL as FILE_PATH,
		a.[FileContent] as FILE_CONTENT, a.FileContentType as CONT_TYPE,
		--('D:\CanBusOutput\ODTS_UploadFile\'  )FilePath
		REPLACE(a.FileURL, 'https://apps.sbstransit.com.sg/RPEAS_Document/','')  as FileURL 

	From [dbo].[EAS_Form_Attach_Document] a ,EAS_Form_Master b
	where a.FormGuid=b.FormGuid
	and a.[FileContent] is null 
	--and a.[FILE_ID]	 <=9

	) z 
	order by 	[FILE_ID]
