CREATE procedure [dbo].[EAS_Form_Get_Outstanding_Lists]

	@P_UserID varchar(15)
	 
AS
BEGIN


 select a.FormGuid,a.DocRefNo,a.DocType,a.Company,a.Title,a.updatedon as SubmitDate,
	     (
	     select (case NoDays when 0 then 
				 'Pending Approval by ' + [dbo].[EAS_User_Name](Userid) 
				   else 'Pending Approval by ' + [dbo].[EAS_User_Name](Userid)  +' For ' + cast(NoDays as varchar) +' Day(s)'
				   End )FormStatus
			FROM (
				Select top 1 * ,[dbo].[EAS_Form_Pending_NODays](FormGuid)as NoDays				 
				from dbo.EAS_Form_Approve_Lvl  
				where active=1
				and FormGuid=a.FormGuid
				and action is null
				order by applevel asc 
			)z
	     ) as FormStatus,
	     [dbo].[EAS_Form_Get_User_Authorized_Functios](a.FormGuid,@P_UserID,'View') as Show_View,
	     [dbo].[EAS_Form_Get_User_Authorized_Functios](a.FormGuid,@P_UserID,'ReRoute') as Show_ReRoute,
	     [dbo].[EAS_Form_Get_User_Authorized_Functios](a.FormGuid,@P_UserID,'Withdraw') as Show_Withdraw,
	     [dbo].[EAS_Form_Get_User_Authorized_Functios](a.FormGuid,@P_UserID,'ForApproval') as Show_ForApproval
	     
	     
 from EAS_Form_Master a 
-- where a.FormGuid='1F3E3A8F-1839-4B72-9A79-68D426B43CEC'
 where   a.active=1
 and  a.FormGuid in 
	   ( select FormGuid
	     from EAS_Form_Approve_Lvl	 
	     where userid=@P_UserID
	     and active=1)
and a.FormStatus not in ('Closed','Rejected','Withdrawn')
 order by a.updatedon asc
 ------------------------------------
 
END
