CREATE PROCEDURE [dbo].[EAS_PARAM_GET_REF]
(
 @P_ParamRecType varchar(50)=null
)
AS
BEGIN
	SELECT PARAM_REC_TYPE,PARAM_DESC
	FROM dbo.EAS_PARAM_REF
	where PARAM_REC_TYPE=isnull(@P_ParamRecType ,PARAM_REC_TYPE)
	ORDER BY PARAM_REC_TYPE ASC
END
