CREATE procedure [dbo].[EAS_Form_Get_UserLogid_GUID] 
(
 @P_SysID    varchar(50) ,
 @p_UserID   varchar(10) ,
 @p_GUID	 varchar(225) output
 )	 
	 
AS
BEGIN

declare @pNewGuiD varchar(225)

set @pNewGuiD=NEWID() 
set @p_GUID=@pNewGuiD

Insert into  [dbo].[EAS_User_login]
(
    [SysID] ,
    [UserID],
	[StartDate],
	[EndTime] ,
	[SessionGUID],
	[Active],
	[CreatedOn] ,
	[CreatedBy] 
)

Values('RPEAS',@p_UserID,GETDATE(),dateadd (hour , 1 , getdate()),
	   @pNewGuiD,'Y',GETDATE(),@p_UserID)


END
