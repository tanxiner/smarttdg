CREATE procedure [dbo].[EAS_Get_MenuChild] 
	@p_UserID as nvarchar(100) = null,
	@p_MenuID as nvarchar(100) = null,
	@p_Sysid  as nvarchar(30)
AS
BEGIN

	--declare @UserID as nvarchar(100);
	--set @UserID = 'csapex72';

	declare @urole as nvarchar(255);
	set @urole = '';

	Select  @urole = @urole + CONVERT(varchar,  Role)  + ','
	from  dbo.EAS_User_Role a , dbo.EAS_User b
	WHERE  a.UserID=b.UserID
	and a.Sysid=b.sysid
	and a.active=b.active
	and a.active=1
	and a.userid=@p_UserID
	AND a.Sysid = @p_sysid;

    SET @urole = LEFT(@urole, NULLIF(LEN(@urole)-1,-1));
    
    
    if @urole is not NULL and @urole != '' 
    begin 
		declare @rsql as nvarchar(1000);
		set @rsql = '
		Select  MenuID as m_id ,MenuLevel,DispText as m_displaytext,
				TargetFrame,ToolTip as m_tooltip,ParentMenuID as m_parentid,ProgURL as m_url
	
		From dbo.EAS_Menu
		WHERE  active =1 		 
		AND ParentMenuID = ''' + @p_MenuID +'''
		AND Sysid = ''' + @p_Sysid +'''
		AND MenuID IN(		
		Select Distinct(MenuID)
		From EAS_Menu_Role
		WHERE active =1 
		AND  Sysid = ''' + @p_Sysid +'''
		AND Role in (''' + @urole + ''')) order by MenuLevel';
		Exec (@rsql);
    end;

END
