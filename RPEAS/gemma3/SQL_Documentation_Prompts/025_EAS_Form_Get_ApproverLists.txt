CREATE procedure [dbo].[EAS_Form_Get_ApproverLists] 
(@P_ExclUser  varchar(15))	 
	 
AS
BEGIN

	 
	select userid,name
	from EAS_User
	where active=1
	and sysid='RPEAS'
	and userid <>@P_ExclUser 
	and userid not in  (select a.userid
						From EAS_User a,EAS_USER_ROLE b
						where a.userid=b.userid
						and b.Role='PA'
						and a.active=1)
	order by name 


END
