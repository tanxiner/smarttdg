CREATE procedure [dbo].[EAS_Form_Get_Company] 
	 
	 
AS
BEGIN

	 
	SELECT  [PARAM_VAL_C1] as Company
	FROM [dbo].[EAS_PARAM_REF_DET]
	WHERE [PARAM_REC_TYPE]='EAS_COMP_NAME'
	and [PARAM_TYPE]='COMP_NAME' 
	and GETDATE() between EFF_FROM_DATE and EFF_TO_DATE
	 


END
