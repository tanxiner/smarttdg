CREATE PROCEDURE [dbo].[EAS_PARAM_GET_PARAM_TYPE]
(@P_Param_RecType varchar(50))
	 
AS
BEGIN
		
		select distinct PARAM_TYPE 
		from EAS_PARAM_REF_DET
		where PARAM_REC_TYPE=@P_Param_RecType
		and CONVERT(date,getdate()) between EFF_FROM_DATE and EFF_TO_DATE
		--order by PARAM_VAL_C1		 

		 
END
