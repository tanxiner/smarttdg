### ⬇️ RAW SQL INPUT ⬇️
CREATE procedure [dbo].[EAS_Get_UserInfo] 
	@p_uid nvarchar(100) = null,
	@p_sysid varchar(30) 
AS
BEGIN
	SELECT   [UserID] ,[Name] ,[Email] ,Designation,
			[BusinessArea], Department,Section
	FROM EAS_User 
	WHERE UserID  = @p_uid
	and active = 1
	and sysid=@p_sysid

 



END
