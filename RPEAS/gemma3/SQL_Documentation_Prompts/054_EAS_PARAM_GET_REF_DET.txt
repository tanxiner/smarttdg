@P_ParamRecType NVARCHAR(50)=NULL,
@P_ParamType NVARCHAR(50)=NULL

AS
BEGIN
	SELECT ROW_NUMBER() OVER (ORDER BY PARAM_REC_TYPE ASC, PARAM_CODE ASC,EFF_FROM_DATE DESC)  AS Sno,
	 *
	FROM dbo.EAS_PARAM_REF_DET 
	--WHERE EFF_TO_DATE >= GETDATE()

	and PARAM_TYPE =isnull(@P_ParamType,PARAM_TYPE)

END
