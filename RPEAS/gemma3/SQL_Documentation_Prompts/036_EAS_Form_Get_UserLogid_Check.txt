CREATE procedure [dbo].[EAS_Form_Get_UserLogid_Check] 
(
 @P_SysID    varchar(50) ,
 @p_GUID     varchar(225) ,
 @p_Cnt		 varchar(225) output
 )	 
	 
AS
BEGIN

 
select @p_Cnt=COUNT(*)
from [EAS_User_login]
where [SessionGUID]=@p_GUID
and sysid=@P_SysID
and active='Y'
and GETDATE() between [StartDate] and [EndTime]

 

END
