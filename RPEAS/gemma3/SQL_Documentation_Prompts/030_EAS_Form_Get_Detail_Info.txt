CREATE procedure [dbo].[EAS_Form_Get_Detail_Info]

	@P_Guid				varchar(225),
	@P_UserID		    varchar(15)
	 
AS
BEGIN

--Get Form Master (Table 0)
 Select DocRefNo,DocType,Company,Title,createdOn,FormStatus,VendorName,[ApprovedAmt],
--[dbo].[EAS_Form_Get_Approver_Name](FormGuid,4,'A')ApproverName,
--[dbo].[EAS_Form_Get_Approver_Name](FormGuid,3,'ST')SubmitThruName,
--[dbo].[EAS_Form_Get_Approver_Name](FormGuid,2,'SB')SubmitByName,
[dbo].[EAS_Form_Get_Approver_Name](FormGuid,1,'P1')PrepByName1,
[dbo].[EAS_Form_Get_Approver_Name](FormGuid,2,'P2')PrepByName2,
[dbo].[EAS_Form_Get_Form_Level](FormGuid,@P_UserID,'F')FormLevel,
[dbo].[EAS_Form_Get_Form_Level](FormGuid,@P_UserID,'U')UserLevel

from  dbo.EAS_Form_Master
where FormGuid=@P_Guid
and active=1

--Get Attached Document (Table 1,2)
exec [dbo].[EAS_Form_Get_Attach_Files]  @P_Guid
-------------------
--Get Approver List - table 3

select *,dbo.EAS_User_Name(Userid) as UserName,
       case when  isnull(a.Action,'x')='x' then 'N' else 'Y' end CompleteStatus,
	   case when  isnull(a.ChkDeclareConflit,'x')='Y' then 'True' else 'False' end ChkDeclareTrueFlase, 
	   b.Remarks,
	   case a.GroupLevel when 1 then 'Prepared by'
					     when 2 then 'Submitted By (Optional)'
						 when 3 then 'Submitted Thru (Optional)'
						 when 4 then 'Supported By (Optional)'
						 when 5 then 'Endorsed By (Optional)'
						 when 6 then 'Approved By' end Approverdescription
from EAS_Form_Approve_Lvl a  left outer join 
	dbo.EAS_Form_Log_History b on a.FormGuid=b.FormGuid
							   and a.ActionBy=b.ActionBy
 
where a.FormGuid=@P_Guid  
order by  2,3
--------------------
--Get Current User Action - table 4
select  top 1 *,dbo.EAS_User_Name(Userid) as UserName,
       case when  isnull(Action,'x')='x' then 'N' else 'Y' end CompleteStatus
from EAS_Form_Approve_Lvl
where FormGuid=@P_Guid 
and  isnull(Action,'x')='x'
order by  2,3 asc
