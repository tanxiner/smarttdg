### ⬇️ RAW SQL INPUT ⬇️
CREATE PROCEDURE [dbo].[EAS_PARAM_REF_DET_Delete]
@ParaRecType VARCHAR(50),
@ParaType	 VARCHAR(50),
@ParaCode	 VARCHAR(50),
@ParaDesc	 VARCHAR(250),
@EffFromDate DATETIME,
@EffToDate	 DATETIME,
@P_ErrorMsg	 NVARCHAR(MAX) OUTPUT
AS
BEGIN

	BEGIN TRY
	BEGIN TRANSACTION

	
		DELETE
		FROM  [EAS_PARAM_REF_DET]
		WHERE  [PARAM_TYPE] = @ParaType
		AND	   [PARAM_CODE] = @ParaCode
		AND	   [PARAM_REC_TYPE] = @ParaRecType
		AND	   [PARAM_DESC] = @ParaDesc 		 
		AND	   [EFF_FROM_DATE] = @EffFromDate
		AND	   [EFF_TO_DATE] = @EffToDate
			  
			  	
	EXECUTE  EAS_GET_ErrorMessage '102',@P_ErrorMsg OUTPUT
		  
	COMMIT TRANSACTION
	END TRY
	BEGIN CATCH 
		set @P_ErrorMsg= ERROR_MESSAGE(); 
		 
		ROLLBACK TRANSACTION;
	END CATCH; 
END
