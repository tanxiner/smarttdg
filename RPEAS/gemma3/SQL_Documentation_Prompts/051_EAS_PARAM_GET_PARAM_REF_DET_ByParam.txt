### ⬇️ RAW SQL INPUT ⬇️
CREATE PROCEDURE [dbo].[EAS_PARAM_GET_PARAM_REF_DET_ByParam]
@Param_Rec_Type VARCHAR(50)=NULL ,
@ParaType VARCHAR(50)=NULL,
@ParaCode VARCHAR(50)=NULL

AS
BEGIN
	SELECT *
	FROM dbo.EAS_PARAM_REF_DET 
	WHERE PARAM_REC_TYPE =@Param_Rec_Type 
	AND PARAM_TYPE = @ParaType
	AND PARAM_CODE =@ParaCode
	ORDER BY PARAM_REC_TYPE ASC,CreatedOn DESC
END
