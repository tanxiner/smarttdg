CREATE procedure [dbo].[EAS_Form_Save_ReRoute_ProceedNxtLevel]

	@P_Guid varchar(225),
	@P_Userid varchar(15),
	@P_FormLevel int,
	@P_ReRouteType  varchar(15),
	@P_ReRouteTo varchar(15),
	@P_Remarks varchar(1000),	 
	@P_ErrorMsge varchar(500) output
	 
AS
BEGIN

set @P_ErrorMsge=''
declare @p_NextReRoute as varchar(15)
declare @p_NextLevel as  int

  BEGIN TRY  
  BEGIN TRANSACTION;     
     
 --update Form Master
 Update dbo.EAS_Form_Master  
 set  UpdatedBy =@P_Userid ,UpdatedOn =GETDATE ()  
 where FormGuid=@P_Guid
 ------------------------   
 
 IF @P_ReRouteType='R' 
 BEGIN
 --Update Existing Record
 Update  dbo.EAS_Form_Approve_Lvl
 set Action ='Re-Routed', Remarks=@P_Remarks, 
	 ReRouteType=@P_ReRouteType,	ReRouteTo =@P_ReRouteTo,
     active=0,
	 ActionBy  =@P_Userid ,
	 ActionOn  =GETDATE ()  
 where FormGuid=@P_Guid
 and AppLevel=@P_FormLevel 
 and active=1
 --------------------------
 --Create new record
	Insert into EAS_Form_Approve_Lvl
  ([FormGuid],[AppLevel],[UserID]
   ,[ReRouteType]  
   ,[Active],[CreatedBy],[CreatedOn])  

	Values(@p_guid,@P_FormLevel,@P_ReRouteTo,0,
	1,@P_Userid,GETDATE())
	--------------------
	--Log history
	 Insert into dbo.EAS_Form_Log_History
	 (FormGuid,Remarks,Action,ActionBy,ActionOn)
	 values(@P_Guid,@P_Remarks,
	 case @P_ReRouteType when 'R' then 'Re-Routed by' when 'N' then 'Proceed to Next Level by'
	 else '' end ,
	 @P_Userid,GETDATE () )
	-----------------------
 
	--Send Re-Route Email
	EXEC	 [dbo].[EAS_Send_ReRoute_Email] 
			 @P_Guid ,
			 @P_Userid,
			 @P_FormLevel,
			 @P_ErrorMsge OUTPUT
			 
 END
 ------------------------------
 IF @P_ReRouteType='N' --Proceed to Next Level
 BEGIN
 --Get Next User Level ID
 select top 1 @p_NextReRoute=userid,@p_NextLevel=AppLevel
 FROM EAS_Form_Approve_Lvl
 where FormGuid=@P_Guid
 and AppLevel > @P_FormLevel	 
 and active=1
 and action is null
 order by AppLevel asc
 -----------------------
	 --Update Existing Record
	 Update  dbo.EAS_Form_Approve_Lvl
	 set Action ='Proceed to Next Level', Remarks=@P_Remarks, 
		 ReRouteType=@P_ReRouteType,ReRouteTo =@p_NextReRoute,      
		 active=0,
		 ActionBy  =@P_Userid ,
		 ActionOn  =GETDATE ()  
	 where FormGuid=@P_Guid
	 and AppLevel=@P_FormLevel	 
	 and active=1
	 -----------------------
	 --Log history
	  Insert into dbo.EAS_Form_Log_History
	 (FormGuid,Remarks,Action,ActionBy,ActionOn)
	 values(@P_Guid,@P_Remarks,
	 case @P_ReRouteType when 'R' then 'Re-Routed' when 'N' then 'Proceed to Next Level'
	 else '' end ,
	 @P_Userid,GETDATE () )
	 -----------------------
 
	 --Send  Proceed to Next Level Email
	 EXEC [dbo].[EAS_Send_ProcdNxtLvl_Email]	 
			 @P_Guid ,
			 @P_Userid,
			 @P_FormLevel,
			 @p_NextLevel,
			 @P_ErrorMsge OUTPUT	 
 
 END
 ------------------------------  

   
  COMMIT TRANSACTION;  
 END TRY  
  --Catch Error Message  
 BEGIN CATCH   
   set @P_ErrorMsge= ERROR_MESSAGE();      
  ROLLBACK TRANSACTION;  
 END CATCH;   
 Set @P_ErrorMsge=isnull(@P_ErrorMsge,'')  
    
 
END
