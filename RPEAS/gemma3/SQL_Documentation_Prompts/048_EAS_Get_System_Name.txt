### ⬇️ RAW SQL INPUT ⬇️
CREATE procedure [dbo].[EAS_Get_System_Name] 
	@P_SysID	  varchar(30) ,
	@P_SysName   varchar(200) output
	 
AS
BEGIN

	 
	set @P_SysName = '';
	
	SELECT @P_SysName= [PARAM_VAL_C1]
	FROM [dbo].[EAS_PARAM_REF_DET]
	WHERE [PARAM_REC_TYPE]='EAS_APPL'
	and [PARAM_TYPE]='APPL_NAME'
	and [PARAM_CODE]=@P_SysID
	and GETDATE() between EFF_FROM_DATE and EFF_TO_DATE


END
