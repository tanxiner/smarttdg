CREATE procedure [dbo].[EAS_Form_Save_Approved_Action]

	@P_Guid varchar(225),
	@P_Userid varchar(15),
	@P_GroupLvl int,
	@P_ActionID int,
	@P_Status  varchar(50),
	@P_Remarks varchar(1000),
	@P_ChkConflict varchar(1),
	@P_ErrorMsge varchar(500) output
	 
AS
BEGIN

set @P_ErrorMsge=''
declare @p_ActionLog varchar(50)

 
BEGIN
	set @p_ActionLog =case @P_GroupLvl  when 1 then 'Prepared by'
										when 2 then 'Submitted by '
										when 3 then 'Submitted Thru '
										when 4 then 'Supported by '
										when 5 then 'Endorsed by ' 
										when 6 then 'Approved by '  end
END
-----------------
  BEGIN TRY  
  BEGIN TRANSACTION;     
     
 --update Form Master
 Update dbo.EAS_Form_Master  
 set FormStatus =@P_Status,
	 UpdatedBy =@P_Userid ,UpdatedOn =GETDATE ()  
 where FormGuid=@P_Guid
 ------------------------   
 --update Form App Level
 Update  dbo.EAS_Form_Approve_Lvl
 set Action ='Approved',ChkDeclareConflit=@P_ChkConflict,Remarks=@P_Remarks,
	 ActionBy  =@P_Userid ,
	 ActionOn  =GETDATE ()  
 where FormGuid=@P_Guid
 and id=@P_ActionID
 --and AppLevel=@P_AppLevel
 --and UserID=@P_Userid
 --and active=1
------------------------    
 Insert into dbo.EAS_Form_Log_History
 (FormGuid,Remarks,Action,ActionBy,ActionOn)
 values(@P_Guid,@P_Remarks,@p_ActionLog,@P_Userid,GETDATE () )
 -----------------------
 ----Send email for next approval
 if @P_Status <>'Closed'
 BEGIN
	
	EXEC	 [dbo].[EAS_Send_Approval_Email]
			 @P_Guid ,
			 @P_Userid,
			 --@P_AppLevel,
			 @P_ErrorMsge OUTPUT

	 
 END
 --Notified all users once approved
 if @P_Status ='Closed'
 BEGIN
	 
	exec [dbo].[EAS_Send_Final_Approved_Email] 
	@P_Guid,@P_Userid,@P_ErrorMsge output
	
 END
 ------------------------
   
  COMMIT TRANSACTION;  
 END TRY  
  --Catch Error Message  
 BEGIN CATCH   
   set @P_ErrorMsge= ERROR_MESSAGE();      
  ROLLBACK TRANSACTION;  
 END CATCH;   
 Set @P_ErrorMsge=isnull(@P_ErrorMsge,'')  
    
 
END
