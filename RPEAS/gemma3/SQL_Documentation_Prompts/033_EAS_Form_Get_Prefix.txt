CREATE procedure [dbo].[EAS_Form_Get_Prefix] 
	@P_DocPrefix   varchar(50) output
	 
AS
BEGIN

	 
	set @P_DocPrefix = '';
	
	SELECT @P_DocPrefix= [PARAM_VAL_C1]
	FROM [dbo].[EAS_PARAM_REF_DET]
	WHERE [PARAM_REC_TYPE]='EAS_DOC_PREFIX'
	and [PARAM_TYPE]='DOC_PREFIX'
	and [PARAM_CODE]='PREFIX_NAME'
	and GETDATE() between EFF_FROM_DATE and EFF_TO_DATE


END
