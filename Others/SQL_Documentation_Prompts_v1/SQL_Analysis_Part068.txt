CREATE PROCEDURE [dbo].[sp_TAMS_Get_UserAccessStatusInfo_by_LoginID] 
	@LoginID NVARCHAR(100) = NULL
AS
BEGIN
	
	CREATE TABLE #AccessStatus
	(
		Line NVARCHAR(10),
		TrackType NVARCHAR(50),
		Module NVARCHAR(10),
		Status NVARCHAR(100)
	)
	
	DECLARE @Line NVARCHAR(10);
	DECLARE @TrackType NVARCHAR(50);
	DECLARE @Module NVARCHAR(10);
	DECLARE @UserID NVARCHAR(200);

	IF EXISTS (SELECT * FROM TAMS_User WHERE LoginID = @LoginID)
	BEGIN
		SELECT @UserID = UserID FROM TAMS_User WHERE LoginID = @LoginID

		DECLARE cur CURSOR FOR 
			SELECT DISTINCT(Line),TrackType, Module FROM TAMS_Role WHERE Line NOT IN ('All') AND IsActive = 1
		
		OPEN cur;

		FETCH NEXT FROM cur INTO @Line,@TrackType, @Module;

		WHILE @@FETCH_STATUS = 0
		BEGIN
			IF EXISTS(SELECT * FROM TAMS_User_Role WHERE UserID = @UserID AND RoleID IN (
				SELECT ID FROM TAMS_Role WHERE Line = @Line AND Module = @Module))
			BEGIN
				INSERT INTO #AccessStatus VALUES (@Line, @TrackType,@Module, 'Approved')
			END
			ELSE
			BEGIN
				IF EXISTS(SELECT * FROM TAMS_Registration r, TAMS_Reg_Module rm 
					WHERE r.ID = rm.RegID
					AND r.LoginID = @LoginID
					AND rm.Line = @Line
					AND rm.TrackType = @TrackType
					AND rm.Module = @Module
					AND rm.RegStatus IN (SELECT ID FROM TAMS_WFStatus
										WHERE Line = @Line
										and TrackType = @TrackType
										AND WFType = 'UserRegStatus'
										AND IsActive = 1
										AND WFStatus NOT IN ('Approved', 'Rejected')))
				BEGIN
					INSERT INTO #AccessStatus VALUES (@Line,@TrackType, @Module, 'Pending Approval')
				END
			END;

			FETCH NEXT FROM cur INTO @Line, @TrackType,@Module;
		END;

		CLOSE cur;
		DEALLOCATE cur;

	END
	ELSE
	BEGIN
		DECLARE cur CURSOR FOR 
			SELECT DISTINCT(Line),TrackType, Module FROM TAMS_Role WHERE Line NOT IN ('All') AND IsActive = 1
		
		OPEN cur;

		FETCH NEXT FROM cur INTO @Line,@TrackType, @Module;

		WHILE @@FETCH_STATUS = 0
		BEGIN
			IF EXISTS(SELECT * FROM TAMS_Registration r, TAMS_Reg_Module rm 
				WHERE r.ID = rm.RegID
				AND r.LoginID = @LoginID
				AND rm.Line = @Line
				and rm.TrackType = @TrackType
				AND rm.Module = @Module
				AND rm.RegStatus IN (SELECT ID FROM TAMS_WFStatus
									WHERE Line = @Line
									and TrackType = @TrackType
									AND WFType = 'UserRegStatus'
									AND IsActive = 1
									AND WFStatus NOT IN ('Approved', 'Rejected')))
			BEGIN
				INSERT INTO #AccessStatus VALUES (@Line,@TrackType, @Module, 'Pending Approval')
			END
			ELSE

			FETCH NEXT FROM cur INTO @Line,@TrackType, @Module;
		END;

		CLOSE cur;
		DEALLOCATE cur;
	END;

	SELECT * FROM #AccessStatus;
	DROP TABLE #AccessStatus;
END

CREATE PROCEDURE [dbo].[sp_TAMS_Get_UserInfo] 
	@uid NVARCHAR(100) = NULL
AS
BEGIN

	DECLARE @ret AS NVARCHAR(100); 
	SET @ret = ''; 
	DECLARE @retmsg AS NVARCHAR(500); 
	SET @retmsg = ''; 

	IF EXISTS(SELECT UserID FROM [TAMS_User] WHERE loginID = @uid AND IsActive = 1 AND ValidTo > GETDATE()) 
	BEGIN 
		UPDATE [TAMS_User] 
		SET lastlogin = GETDATE()
		WHERE LoginID = @uid
		AND IsActive = 1
	END;	 
	ELSE IF EXISTS(SELECT UserID FROM [TAMS_User] WHERE loginID = @uid AND ValidTo < GETDATE())
	BEGIN
		SET @ret = 'Expired';
		SET @retmsg = 'Your account has expired. Please approach system administrator.';
	END
	ELSE IF EXISTS(SELECT UserID FROM [TAMS_User] WHERE loginID = @uid AND IsActive = 0) 
	BEGIN 
		SET @ret = 'DeActivate';
		SET @retmsg = 'Your account has been deactivated due to no login for the past 6 month. Please approach system administrator.'
	END;
	ELSE 
	BEGIN 
		SET @ret = 'NoAccess';
	END; 

	SELECT * FROM [TAMS_User] WHERE loginID = @uid AND IsActive = 1  AND ValidTo > GETDATE();

	SELECT @ret AS 'AccountStatus', @retmsg AS 'MessageToDisplay';

	SELECT r.Role FROM TAMS_User_Role ur, TAMS_User u, TAMS_Role r
	WHERE ur.userid = u.Userid
	AND ur.roleID = r.ID
	AND LoginID = @uid
END
