CREATE PROCEDURE [dbo].[sp_TAMS_RGS_AckReg_20230807_M]
	@TARID		AS BIGINT = 0,
	@UserID		AS NVARCHAR(500),
	@Message	AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

SET @Message = ''

DECLARE @NewLine AS CHAR(2) = CHAR(30) + CHAR(10)

DECLARE @TOAStatus INT

SELECT @TOAStatus = b.TOAStatus
	FROM TAMS_TAR a, TAMS_TOA b	
		WHERE a.Id = b.TARId
			AND a.Id = @TARID

IF @TOAStatus = 1
	BEGIN
		UPDATE TAMS_TOA
			SET	TOAStatus = 2,
				AckRegisterTime = GETDATE(),
				UpdatedOn = GETDATE(),
				UpdatedBy = @UserID
		WHERE TARId = @TARID

		DECLARE @SMSMsg	NVARCHAR(4000)
		SET @SMSMsg = ''

		DECLARE @AckRegTime NVARCHAR(20)
		DECLARE @TARNo	NVARCHAR(50)
		DECLARE @Line NVARCHAR(5)
		DECLARE @CurrDate NVARCHAR(20)
		DECLARE @HPNo NVARCHAR(20)

		SET @AckRegTime = ''
		SET @TARNo = ''
		SET @Line = ''
		SET @CurrDate = ''
		SET @HPNo = ''

		SELECT @TARNo = a.TARNo, 
				@Line = a.Line, 
				@AckRegTime = CONVERT(NVARCHAR(20), b.AckRegisterTime, 108),
				@HPNo = b.MobileNo
			FROM TAMS_TAR a, TAMS_TOA b	
				WHERE a.Id = b.TARId
					AND a.Id = @TARID

		SET @CurrDate = CONVERT(NVARCHAR(20), GETDATE(), 103)

		IF @Line = 'DTL'
			BEGIN
				SET @SMSMsg = CAST(@TARNo AS NVARCHAR(50)) + ' registration has been acknowledged by DTL OCC @ ' + CAST(@AckRegTime AS NVARCHAR(20))
					+ ' hrs on ' + CAST(@CurrDate AS NVARCHAR(20)) + '. DO NOT GO DOWN TO TRACK YET.'
			END
		ELSE
			BEGIN
				SET @SMSMsg = CAST(@TARNo AS NVARCHAR(50)) + ' registration has been acknowledged by NEL OCC @ ' + CAST(@AckRegTime AS NVARCHAR(20))
					+ ' hrs on ' + CAST(@CurrDate AS NVARCHAR(20)) + '. DO NOT GO DOWN TO TRACK YET.'
			END

		-- >> SEND SMS
		DECLARE @RetVal NVARCHAR(5)
		SET @RetVal = ''

			IF LTRIM(RTRIM(ISNULL(@HPNo, ''))) <> '' AND LTRIM(RTRIM(ISNULL(@SMSMsg, ''))) <> ''
				BEGIN
					EXEC sp_api_send_sms @HPNo, 'TAMS RGS', @SMSMsg, @RetVal OUTPUT
				END

		-- >> 20220818 Quickly trigger SMS Sending
			DECLARE @SMSOutputMsg NVARCHAR(500) 

			SET @SMSOutputMsg = ''

			EXEC SP_Call_SMTP_Send_SMSAlert @SMSOutputMsg OUTPUT

			IF @SMSOutputMsg <> ''
				BEGIN
					SET @Message = 'Error SMS Sending'
					GOTO TRAP_ERROR
				END
		-- >> 20220818 Quickly trigger SMS Sending
	END
ELSE
	BEGIN
		IF @TOAStatus = 2
			BEGIN
				SET @Message = '1'
				-- TOA already granted. Please refresh RGS.
				GOTO TRAP_ERROR
			END
		ELSE
			BEGIN
				SET @Message = 'Invalid TAR status. Please refresh RGS.'
				GOTO TRAP_ERROR
			END
	END

IF @@ERROR <> 0 BEGIN
	SET @Message = 'Error RGS Ack Reg'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END
