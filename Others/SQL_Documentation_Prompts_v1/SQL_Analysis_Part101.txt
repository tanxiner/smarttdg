CREATE PROCEDURE [dbo].[sp_TAMS_OCC_InsertTVFAckByParameters]
	@OperationDate datetime = null,
	@AccessDate datetime = null,
	@UserID as int = 0,
	@StationId as int = 0,	
	@TVFMode as varchar(10) = null,	
	@TVFDirection1 as bit = null,	
	@TVFDirection2 as bit = null



AS
	DECLARE @NewID AS int;
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION;
			INSERT INTO TAMS_TVF_Acknowledge(
			AccessDate,
			OperationDate,
			StationId,
			TVFMode,
			TVFDirection1,
			TVFDirection2,
			AcknowledgedBy,
			AcknowledgedOn,
			VerifiedBy,
			VerifiedOn,
			OperatedBy,
			TVFOnTime,
			CreatedOn,
			CreatedBy,
			UpdatedOn,
			UpdatedBy)
			VALUES (
			@AccessDate,
			@OperationDate,
			@StationId,
			@TVFMode,
			@TVFDirection1,
			@TVFDirection2,
			@UserID,
			GETDATE(),
			NULL,
			NULL,
			NULL,
			NULL,
			GETDATE(),
			@UserID,
			NULL,
			NULL)

			SELECT @NewID = SCOPE_IDENTITY();

			UPDATE TAMS_TVF_Acknowledge
			SET AcknowledgedBy = NULL, AcknowledgedOn = NULL, CreatedOn = NULL, CreatedBy = NULL
			WHERE TVFMode = 'Select'

			INSERT INTO TAMS_TVF_Acknowledge_Audit(
			   [ActionBy]
			  ,[ActionOn]
			  ,[AuditAction]
			  ,[TVFAckId]
			  ,[AccessDate]
			  ,[OperationDate]
			  ,[StationId]
			  ,[TVFMode]
			  ,[TVFDirection1]
			  ,[TVFDirection2]
			  ,[AcknowledgedBy]
			  ,[AcknowledgedOn]
			  ,[VerifiedBy]
			  ,[VerifiedOn]
			  ,[OperatedBy]
			  ,[TVFOnTime]
			  ,[CreatedOn]
			  ,[CreatedBy]
			  ,[UpdatedOn]
			  ,[UpdatedBy])
			  SELECT 
			   @UserID
			  ,GETDATE()
			  ,'A'
			  ,@NewID
			  ,[AccessDate]
			  ,[OperationDate]
			  ,[StationId]
			  ,[TVFMode]
			  ,[TVFDirection1]
			  ,[TVFDirection2]
			  ,[AcknowledgedBy]
			  ,[AcknowledgedOn]
			  ,[VerifiedBy]
			  ,[VerifiedOn]
			  ,[OperatedBy]
			  ,[TVFOnTime]
			  ,[CreatedOn]
			  ,[CreatedBy]
			  ,[UpdatedOn]
			  ,[UpdatedBy]
			  FROM TAMS_TVF_Acknowledge
			  WHERE ID = @NewID

			  --ActionOn = GETDATE(),

			  --TVFAckId = @NewID
			  --WHERE TVFAckId = @NewID;

		COMMIT TRANSACTION;

CREATE PROCEDURE  [dbo].[sp_TAMS_OCC_InsertToDutyOCCRosterTable]
	@TAMS_OCC_DutyRoster [dbo].[TAMS_OCC_DutyRoster] READONLY



AS
BEGIN

	declare @count as int = 0;
	declare @loop as int = 0;
	declare @operationdate as date;
	declare @shift as nvarchar(1);
	declare @line as nvarchar(10);
	declare @tracktype as nvarchar(50);

	select @operationdate = operationdate, @shift = [shift], @line = [line],@tracktype=TrackType from @TAMS_OCC_DutyRoster where 1=1;
		

	select @count = count(*) from TAMS_OCC_Duty_Roster where UPPER(TrackType)=UPPER(@tracktype) and operationdate = @operationdate and [shift] = @shift and [line] = @line;


	if @count = 0
	begin
		INSERT INTO TAMS_OCC_Duty_Roster(Line,
		TrackType,
		OperationDate,
		[Shift],
		RosterCode,
		DutyStaffId,
		IsActive,
		CreatedOn,
		CreatedBy,
		UpdatedOn,
		UpdatedBy
		)
		SELECT Line,
		TrackType,
		OperationDate,
		[Shift],
		RosterCode,
		DutyStaffId,
		IsActive,
		CreatedOn,
		CreatedBy,
		UpdatedOn,
		UpdatedBy
		FROM @TAMS_OCC_DutyRoster

		INSERT INTO TAMS_OCC_Duty_Roster_Audit(
		ActionBy,
		ActionOn,
		AuditAction,
		DutyRosterId,
		Line,
		TrackType,
		OperationDate,
		[Shift],
		RosterCode,
		DutyStaffId,
		IsActive,
		CreatedOn,
		CreatedBy,
		UpdatedOn,
		UpdatedBy
		)
		SELECT 
		CreatedBy,
		UpdatedOn,
		'I',
		DutyStaffId,
		Line,
		TrackType,
		OperationDate,
		[Shift],
		RosterCode,
		DutyStaffId,
		IsActive,
		CreatedOn,
		CreatedBy,
		UpdatedOn,
		UpdatedBy
		FROM @TAMS_OCC_DutyRoster
	end;
	else
	begin
		update TAMS_OCC_Duty_Roster
		set DutyStaffId = dr.DutyStaffId,
		UpdatedOn = dr.UpdatedOn,
		UpdatedBy = dr.UpdatedBy
		from  @TAMS_OCC_DutyRoster dr
		where TAMS_OCC_Duty_Roster.ID = dr.ID
		and TAMS_OCC_Duty_Roster.Line = dr.Line
		and TAMS_OCC_Duty_Roster.TrackType = dr.TrackType
		and TAMS_OCC_Duty_Roster.OperationDate = dr.operationdate
		and TAMS_OCC_Duty_Roster.[shift] = dr.[shift]
		and TAMS_OCC_Duty_Roster.RosterCode = dr.RosterCode

		INSERT INTO TAMS_OCC_Duty_Roster_Audit(
		ActionBy,
		ActionOn,
		AuditAction,
		DutyRosterId,
		Line,
		TrackType,
		OperationDate,
		[Shift],
		RosterCode,
		DutyStaffId,
		IsActive,
		CreatedOn,
		CreatedBy,
		UpdatedOn,
		UpdatedBy
		)
		SELECT 
		CreatedBy,
		UpdatedOn,
		'U',
		DutyStaffId,
		Line,
		TrackType,
		OperationDate,
		[Shift],
		RosterCode,
		DutyStaffId,
		IsActive,
		CreatedOn,
		CreatedBy,
		UpdatedOn,
		UpdatedBy
		FROM @TAMS_OCC_DutyRoster
	end;
END

CREATE PROCEDURE [dbo].[sp_TAMS_OCC_RejectTVFAckByParameters_PFR]
	@OperationDate datetime = null,
	@AccessDate datetime = null,
	@UserID as int = 0,
	@StationId as int = 0,	
	@TVFMode as varchar(10) = null,	
	@TVFDirection1 as bit = null,	
	@TVFDirection2 as bit = null



AS
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION;

			UPDATE TAMS_TVF_Acknowledge
			SET TVFMode = @TVFMode, TVFDirection1 = @TVFDirection1, TVFDirection2 = @TVFDirection2,
			VerifiedBy = NULL, VerifiedOn = NULL, UpdatedOn = GETDATE(), UpdatedBy = @UserID
			WHERE StationId = @StationId
			AND OperationDate = @OperationDate
			AND AccessDate = @AccessDate

			INSERT INTO TAMS_TVF_Acknowledge_Audit(
			   [ActionBy]
			  ,[ActionOn]
			  ,[AuditAction]
			  ,[TVFAckId]
			  ,[AccessDate]
			  ,[OperationDate]
			  ,[StationId]
			  ,[TVFMode]
			  ,[TVFDirection1]
			  ,[TVFDirection2]
			  ,[AcknowledgedBy]
			  ,[AcknowledgedOn]
			  ,[VerifiedBy]
			  ,[VerifiedOn]
			  ,[OperatedBy]
			  ,[TVFOnTime]
			  ,[CreatedOn]
			  ,[CreatedBy]
			  ,[UpdatedOn]
			  ,[UpdatedBy])
			  SELECT 
			   @UserID
			  ,GETDATE()
			  ,'R'
			  ,[ID]
			  ,[AccessDate]
			  ,[OperationDate]
			  ,[StationId]
			  ,[TVFMode]
			  ,[TVFDirection1]
			  ,[TVFDirection2]
			  ,[AcknowledgedBy]
			  ,[AcknowledgedOn]
			  ,[VerifiedBy]
			  ,[VerifiedOn]
			  ,[OperatedBy]
			  ,[TVFOnTime]
			  ,[CreatedOn]
			  ,[CreatedBy]
			  ,[UpdatedOn]
			  ,[UpdatedBy]
			  FROM TAMS_TVF_Acknowledge
			  WHERE StationId = @StationId
			  AND OperationDate = @OperationDate
			  AND AccessDate = @AccessDate


		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		PRINT ERROR_MESSAGE();
		ROLLBACK TRANSACTION;
	END CATCH
END
