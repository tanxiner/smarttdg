CREATE PROCEDURE  [dbo].[sp_TAMS_OCC_GetOCCAuthorisationPFRByParameters_bak20230727]
	@UserID		as int,
	@Line AS nvarchar(10) = NULL,
	@TrackType			AS nvarchar(50) = NULL,
	@OperationDate as date,
	@AccessDate as date

AS

DECLARE @WorkflowId as int,
		@OCCAuthID as int,
		@EndorserID as int,
		@EndorserLevel as int,
		@EndorserTitle as nvarchar(250),
		@WFStatus as nvarchar(50),
		@ActionOn as datetime,
		@StationName as nvarchar(50),
		@FISTestResult as nvarchar(50)

CREATE TABLE #TMP
(
	TractionPowerId int not null,	
	StationName nvarchar(50)
);

CREATE TABLE #TMP_Endorser
(
	ID int not null,	
	Line nvarchar(10),
	[Level] int,
	RoleId int,
	Title nvarchar(250)
);

CREATE TABLE #TMP_OCCAuthPFR
(
	SNO int not null,
	ID int not null,	
	OCCAuthID int not null,
	Remark nvarchar(1000),
	PFRRemark nvarchar(1000),
	IsBuffer bit,
	PowerOn bit,
	Line nvarchar(10),
	TractionPowerSection nvarchar(100),
	HSCBName nvarchar(100),
	StationName nvarchar(50),
	TC nvarchar(10),
	Train nvarchar(50),
	MainlineTractionCurrentSwitchOff_TractionCurrentOff nvarchar(8),
	MainlineTractionCurrentSwitchOff_RackOut nvarchar(8),
	MainlineTractionCurrentSwitchOff_STA nvarchar(8),
	PermanentClosingVLD_Station nvarchar(8),
	PermanentClosingVLD_Time nvarchar(8),
	LineClearCertification_STA nvarchar(8),
	LineClearCertification_RackIn nvarchar(8),
	NormalisationVLD_Station nvarchar(8),
	NormalisationVLD_Time nvarchar(8),
	MainlineTractionCurrentSwitchOn_TractionCurrentOn nvarchar(8),
	MainlineTractionCurrentSwitchOn_FISTestResult nvarchar(8),
	RemarksCC nvarchar(1000),
	RemarksPFR nvarchar(1000)
);

IF @Line = 'DTL'
BEGIN
	SELECT @WorkflowId = ID FROM [TAMS_Workflow]
						WHERE Line = @Line
						and TrackType = @TrackType
						AND  WorkflowType = 'OCCAuth'
						AND IsActive = 1
						AND EffectiveDate <= GETDATE()
						AND ExpiryDate >= GETDATE()

	INSERT INTO #TMP_Endorser(ID, Line, [Level], RoleId, Title)
	SELECT ID, Line, [Level], RoleId, Title FROM [TAMS_Endorser]
	WHERE Line = 'DTL'
	AND WorkflowId = @WorkflowId
	AND RoleId = 14
	--AND Level in (1,9,10,11,20)


	INSERT INTO #TMP(TractionPowerId, StationName)
	SELECT 
	   Id,
	   STUFF((SELECT ';' + S.StationName
			  FROM [TAMS_Traction_Power_Detail] TPD, [TAMS_Station] S
			  WHERE TPD.TractionPowerId = TP.ID
			  AND TPD.StationId = S.ID
			  AND  TPD.TractionPowerType = 'Station'
			  FOR XML PATH('')), 1, 1, '') [StationName]
	FROM [TAMS_Traction_Power] TP
	WHERE TP.Line = 'DTL'
	and TP.TrackType = @TrackType 
	GROUP BY TP.ID

	INSERT INTO #TMP_OCCAuthPFR(
	SNO, 
	ID, 
	OCCAuthID,
	Remark,
	PFRRemark,
	IsBuffer,
	PowerOn,
	Line, 
	TractionPowerSection,  
	HSCBName,
	StationName,
	TC,
	Train,
	MainlineTractionCurrentSwitchOff_TractionCurrentOff,
	MainlineTractionCurrentSwitchOff_RackOut,
	MainlineTractionCurrentSwitchOff_STA,
	PermanentClosingVLD_Station,
	PermanentClosingVLD_Time,
	LineClearCertification_STA,
	LineClearCertification_RackIn,
	NormalisationVLD_Station,
	NormalisationVLD_Time,
	MainlineTractionCurrentSwitchOn_TractionCurrentOn,
	MainlineTractionCurrentSwitchOn_FISTestResult,
	RemarksCC,
	RemarksPFR)
	SELECT ROW_NUMBER() OVER(ORDER BY tp.ID ASC) AS SNO
		  ,tp.[ID]
		  ,oa.[ID]
		  ,oa.[Remark]
		  ,oa.[PFRRemark]
		  ,oa.[IsBuffer]
		  ,oa.[PowerOn]
		  ,tp.[Line]
		  ,[TractionPowerSection]
		  ,[HSCBName]
		  ,[StationName]
		  ,[TC]
		  ,[Train]
		  ,'' AS MainlineTractionCurrentSwitchOff_TractionCurrentOff
		  ,'' AS MainlineTractionCurrentSwitchOff_RackOut
		  ,'' AS MainlineTractionCurrentSwitchOff_STA
		  ,'' AS PermanentClosingVLD_Station
		  ,'' AS PermanentClosingVLD_Time	
		  ,'' AS LineClearCertification_STA
		  ,'' AS LineClearCertification_RackIn
		  ,'' AS NormalisationVLD_Station
		  ,'' AS NormalisationVLD_Time
		  ,'' AS MainlineTractionCurrentSwitchOn_TractionCurrentOn
		  ,'' AS MainlineTractionCurrentSwitchOn_FISTestResult
		  ,'' AS RemarksCC
		  ,'' AS RemarksPFR
	  FROM [TAMS_Traction_Power] tp, #TMP tmp, [TAMS_OCC_Auth] oa
	  WHERE tp.ID = tmp.TractionPowerId
	  AND tmp.TractionPowerId = oa.TractionPowerId
	  AND EffectiveDate <= GETDATE() and ExpiryDate >= GETDATE()
	  AND IsActive = 1
	  AND oa.OperationDate = @OperationDate
	  AND oa.AccessDate = @AccessDate
	  AND TC in 
	  (SELECT RosterCode FROM TAMS_OCC_Duty_Roster 
	   WHERE OperationDate = CAST(@OperationDate as DATE)
	   and [shift] = 3
	   and IsActive = 1
	   )


	DECLARE cur CURSOR FOR
	SELECT OCCAuthID FROM #TMP_OCCAuthPFR


	OPEN cur;

	FETCH NEXT FROM cur INTO @OCCAuthID
	WHILE @@FETCH_STATUS = 0
	BEGIN

		DECLARE cur1 CURSOR FOR
		SELECT ID, [Level], Title FROM #TMP_Endorser

		OPEN Cur1;

		FETCH NEXT FROM Cur1 INTO @EndorserID, @EndorserLevel, @EndorserTitle;
		WHILE @@FETCH_STATUS = 0
		BEGIN
			IF @EndorserID = 100
			BEGIN
				SELECT @WFStatus = WFStatus, @ActionOn = ActionOn 
				FROM [TAMS_OCC_Auth_Workflow] 
				WHERE OCCAuthId = @OCCAuthID AND OCCAuthEndorserId = 100;
				IF @WFStatus = 'Pending'
				BEGIN
					UPDATE #TMP_OCCAuthPFR
					SET MainlineTractionCurrentSwitchOff_TractionCurrentOff = @WFStatus
					WHERE OCCAuthID = @OCCAuthID
				END;
				IF @WFStatus = 'Completed'
				BEGIN
					UPDATE #TMP_OCCAuthPFR
					SET MainlineTractionCurrentSwitchOff_TractionCurrentOff = CONVERT(TIME, @ActionOn)
					WHERE OCCAuthID = @OCCAuthID
				END;

				SET @WFStatus = '';
				SET @ActionOn = '';
			END;

			IF @EndorserID = 101
			BEGIN
				SELECT @WFStatus = WFStatus, @ActionOn = ActionOn 
				FROM [TAMS_OCC_Auth_Workflow] 
				WHERE OCCAuthId = @OCCAuthID AND OCCAuthEndorserId = 101;
				IF @WFStatus = 'Pending'
				BEGIN
					UPDATE #TMP_OCCAuthPFR
					SET MainlineTractionCurrentSwitchOff_RackOut = @WFStatus
					WHERE OCCAuthID = @OCCAuthID
				END;
				IF @WFStatus = 'Completed' OR @WFStatus = 'N.A.'
				BEGIN
					UPDATE #TMP_OCCAuthPFR
					SET MainlineTractionCurrentSwitchOff_RackOut = CONVERT(TIME, @ActionOn)
					WHERE OCCAuthID = @OCCAuthID
				END;
				SET @WFStatus = '';
				SET @ActionOn = '';
			END;
			IF @EndorserID = 102
			BEGIN
				SELECT @WFStatus = WFStatus, @ActionOn = ActionOn 
				FROM [TAMS_OCC_Auth_Workflow] 
				WHERE OCCAuthId = @OCCAuthID AND OCCAuthEndorserId = 102;
				IF @WFStatus = 'Pending'
				BEGIN
					UPDATE #TMP_OCCAuthPFR
					SET MainlineTractionCurrentSwitchOff_STA = @WFStatus
					WHERE OCCAuthID = @OCCAuthID
				END;
				IF @WFStatus = 'Completed' OR @WFStatus = 'N.A.'
				BEGIN
					UPDATE #TMP_OCCAuthPFR
					SET MainlineTractionCurrentSwitchOff_STA = CONVERT(TIME, @ActionOn)
					WHERE OCCAuthID = @OCCAuthID
				END;
				SET @WFStatus = '';
				SET @ActionOn = '';
			END;

			IF @EndorserID = 103
			BEGIN
				SELECT @WFStatus = WFStatus, @ActionOn = ActionOn 
				FROM [TAMS_OCC_Auth_Workflow] 
				WHERE OCCAuthId = @OCCAuthID AND OCCAuthEndorserId = 103;
				IF @WFStatus = 'Pending'
				BEGIN
					UPDATE #TMP_OCCAuthPFR
					SET PermanentClosingVLD_Time = @WFStatus
					WHERE OCCAuthID = @OCCAuthID
				END;
				IF @WFStatus = 'Completed'
				BEGIN

					SELECT @StationName = StationName FROM [TAMS_Station]
					WHERE ID IN (SELECT StationId FROM [TAMS_OCC_Auth_Workflow]  WHERE OCCAuthId = @OCCAuthID AND  OCCAuthEndorserId = 103)

					IF @StationName = '' OR @StationName = NULL
					BEGIN
						SET @StationName = 'N.A.';
					END;
					
					UPDATE #TMP_OCCAuthPFR
					SET PermanentClosingVLD_Time = CONVERT(TIME, @ActionOn), 
					PermanentClosingVLD_Station = @StationName
					WHERE OCCAuthID = @OCCAuthID
				END;
				SET @WFStatus = '';
				SET @ActionOn = '';
				SET @StationName = '';

			END;

			IF @EndorserID = 108
			BEGIN
				SELECT @WFStatus = WFStatus, @ActionOn = ActionOn 
				FROM [TAMS_OCC_Auth_Workflow] 
				WHERE OCCAuthId = @OCCAuthID AND OCCAuthEndorserId = 108;
				IF @WFStatus = 'Pending'
				BEGIN
					UPDATE #TMP_OCCAuthPFR
					SET LineClearCertification_STA = @WFStatus
					WHERE OCCAuthID = @OCCAuthID
				END;
				IF @WFStatus = 'Completed' OR @WFStatus = 'N.A.'
				BEGIN
					UPDATE #TMP_OCCAuthPFR
					SET LineClearCertification_STA = CONVERT(TIME, @ActionOn)
					WHERE OCCAuthID = @OCCAuthID
				END;
				SET @WFStatus = '';
				SET @ActionOn = '';
			END;

			IF @EndorserID = 109
			BEGIN
				SELECT @WFStatus = WFStatus, @ActionOn = ActionOn 
				FROM [TAMS_OCC_Auth_Workflow] 
				WHERE OCCAuthId = @OCCAuthID AND OCCAuthEndorserId = 109;
				IF @WFStatus = 'Pending'
				BEGIN
					UPDATE #TMP_OCCAuthPFR
					SET LineClearCertification_RackIn = @WFStatus
					WHERE OCCAuthID = @OCCAuthID
				END;
				IF @WFStatus = 'Completed' OR @WFStatus = 'N.A.'
				BEGIN
					UPDATE #TMP_OCCAuthPFR
					SET LineClearCertification_RackIn = CONVERT(TIME, @ActionOn)
					WHERE OCCAuthID = @OCCAuthID
				END;
				SET @WFStatus = '';
				SET @ActionOn = '';
			END;

			IF @EndorserID = 111
			BEGIN
				SELECT @WFStatus = WFStatus, @ActionOn = ActionOn 
				FROM [TAMS_OCC_Auth_Workflow] 
				WHERE OCCAuthId = @OCCAuthID AND OCCAuthEndorserId = 111;
				IF @WFStatus = 'Pending'
				BEGIN
					UPDATE #TMP_OCCAuthPFR
					SET NormalisationVLD_Time = @WFStatus
					WHERE OCCAuthID = @OCCAuthID
				END;
				IF @WFStatus = 'Completed'
				BEGIN
					UPDATE #TMP_OCCAuthPFR
					SET NormalisationVLD_Time = CONVERT(TIME, @ActionOn)
					WHERE OCCAuthID = @OCCAuthID
				END;
				SET @WFStatus = '';
				SET @ActionOn = '';
			END;
			IF @EndorserID = 113
			BEGIN
				SELECT @WFStatus = WFStatus, @ActionOn = ActionOn 
				FROM [TAMS_OCC_Auth_Workflow] 
				WHERE OCCAuthId = @OCCAuthID AND OCCAuthEndorserId = 113;
				IF @WFStatus = 'Pending'
				BEGIN
					UPDATE #TMP_OCCAuthPFR
					SET MainlineTractionCurrentSwitchOn_TractionCurrentOn = @WFStatus
					WHERE OCCAuthID = @OCCAuthID
				END;
				IF @WFStatus = 'Completed'
				BEGIN
					UPDATE #TMP_OCCAuthPFR
					SET MainlineTractionCurrentSwitchOn_TractionCurrentOn = CONVERT(TIME, @ActionOn)
					WHERE OCCAuthID = @OCCAuthID
				END;
				SET @WFStatus = '';
				SET @ActionOn = '';
			END;

			IF @EndorserID = 114
			BEGIN
				SELECT @WFStatus = WFStatus, @FISTestResult = FISTestResult, @ActionOn = ActionOn 
				FROM [TAMS_OCC_Auth_Workflow] 
				WHERE OCCAuthId = @OCCAuthID AND OCCAuthEndorserId = 114;
				IF @WFStatus = 'Pending'
				BEGIN
					UPDATE #TMP_OCCAuthPFR
					SET MainlineTractionCurrentSwitchOn_FISTestResult = @WFStatus
					WHERE OCCAuthID = @OCCAuthID
				END;
				IF @WFStatus = 'Completed'
				BEGIN
					UPDATE #TMP_OCCAuthPFR
					SET MainlineTractionCurrentSwitchOn_FISTestResult = @FISTestResult
					WHERE OCCAuthID = @OCCAuthID
				END;
				--IF @WFStatus = 'Pass' OR @WFStatus = 'Fail' OR @WFStatus = 'Not Done'
				--BEGIN
				--	UPDATE #TMP_OCCAuthPFR
				--	SET MainlineTractionCurrentSwitchOn_FISTestResult = @WFStatus
				--	WHERE OCCAuthID = @OCCAuthID
				--END;
				SET @WFStatus = '';
				SET @ActionOn = '';
			END;

			FETCH NEXT FROM Cur1 INTO @EndorserID, @EndorserLevel, @EndorserTitle;
		END

		CLOSE Cur1;
		DEALLOCATE Cur1;

		FETCH NEXT FROM cur INTO @OCCAuthID;

	END;

	CLOSE cur;
	DEALLOCATE cur;

	SELECT * FROM #TMP_OCCAuthPFR;

END

DROP TABLE #TMP;

DROP TABLE #TMP_Endorser;

DROP TABLE #TMP_OCCAuthPFR;
