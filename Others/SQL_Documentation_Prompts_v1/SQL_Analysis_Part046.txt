CREATE   procedure [dbo].[sp_TAMS_Form_Save_Possession_DepotSector]
	@Sector	AS NVARCHAR(4000) = NULL,
	@PowerOff		AS INT = 0,
	@NoOFSCD		AS INT = 0,
	@BreakerOut		AS NVARCHAR(5) = NULL,
	@PossID			AS BIGINT = 0,
	@Message		AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT

SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

IF (SELECT COUNT(*) 
		FROM [dbo].[TAMS_Possession_DepotSector]
			WHERE PossessionId = @PossID
				AND Sector = LTRIM(RTRIM(@Sector))) = 0
	BEGIN
		INSERT INTO [dbo].[TAMS_Possession_DepotSector]
		([PossessionId], [Sector],PowerOff, [NoOFSCD], [BreakerOut])
		VALUES (@PossID, LTRIM(RTRIM(@Sector)),@PowerOff, @NoOFSCD, CASE WHEN @BreakerOut = 'Y' THEN 1 ELSE 0 END)
	END

-- >>

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR INSERTING INTO TAMS_TAR'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message


END

CREATE PROCEDURE  [dbo].[sp_TAMS_Form_Save_Temp_Attachment]
(@TARId			INTEGER			= NULL, 
@TARAccessReqId	INTEGER			= NULL, 
@FileName		NVARCHAR(50)	= NULL,
@FileType		NVARCHAR(50)	= NULL,
@FileUpload		VARBINARY(MAX)	= NULL)

AS
BEGIN

DECLARE @ret AS NVARCHAR(50); 
SET @ret = ''; 

BEGIN TRY
	BEGIN TRANSACTION;

	IF (SELECT COUNT(*) 
			FROM TAMS_TAR_Attachment_Temp
				WHERE [TARId] = @TARId
					AND [TARAccessReqId] = @TARAccessReqId) = 0
		BEGIN
			INSERT INTO [dbo].[TAMS_TAR_Attachment_Temp]
				([TARId], [TARAccessReqId], [FileName], [FileType], [FileUpload])
			VALUES (@TARId, @TARAccessReqId,@FileName, @FileType, @FileUpload)
		END

	COMMIT TRANSACTION;
END TRY

BEGIN CATCH
	PRINT 'Inside exception';
	SELECT ERROR_NUMBER() AS ErrorNumber, 
		ERROR_MESSAGE() AS ErrorMessage;
	ROLLBACK TRANSACTION;
	SET @ret = 'Errror' 
END CATCH

SELECT @ret AS ReturnValue 

End;
