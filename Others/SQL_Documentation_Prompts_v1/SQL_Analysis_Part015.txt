create PROCEDURE  [dbo].[sp_TAMS_Approval_Proceed_To_App_20231009]
	@TARID		AS INTEGER,						-- >> TAR ID

DECLARE @TARWFCtr	INTEGER

	END
-- >>>

IF @TVFRunModeUpdInd = 1
	BEGIN
		UPDATE TAMS_TAR
			SET TVFMode = @TVFRunMode
		WHERE ID = @TARID
	END

SELECT @Line		= Line,
		@AccessDate		= AccessDate,
		@TARType		= TARType,
		@AccessType		= AccessType,
		@IsExclusive	= IsExclusive,
		@EMToSend = Email,		-- >> ToSendEmail
		@EMCCSend = Email,
		@EMTARID = Id,
		@EMTARNo = TARNo,
		@EMRemarks = ARRemark,
		@EMCompany = Company
	FROM TAMS_TAR
		WHERE ID = @TARID

SELECT @WFID = WorkflowId 
	FROM TAMS_TAR_Workflow
		WHERE ID = @TARWFID

-- >> Current Level Endorser
SELECT @CurEndID = ID, 
		@CurEndTitle = Title,
		@CurWFID = WorkflowId,
		@CurWFStatID = WFStatusId,
		@CurRoleID = RoleId,
		@CurReqVal = RequireValidation,
		@CurReqTVF = RequireTVF

			AND [Level] = @ELevel

	CREATE TABLE #TmpExc 
	(TARID			BIGINT,
	TARNo			NVARCHAR(20),
	TARType			NVARCHAR(10),
	AccessDate		NVARCHAR(20),
	AccessType		NVARCHAR(20),
	IsExclusive		INT,
	EmailAdd		NVARCHAR(200))

	CREATE TABLE #TmpExcSector 
	(TARID			BIGINT,
	TARNo			NVARCHAR(20),
	TARType			NVARCHAR(10),
	AccessDate		NVARCHAR(20),
	AccessType		NVARCHAR(20),
	IsExclusive		INT,
	SectorID		INT,
	SectorStr		NVARCHAR(50),
	IsBuffer		INT,
	EmailAdd		NVARCHAR(200))
	
	TRUNCATE TABLE #TmpExc
	TRUNCATE TABLE #TmpExcSector

	DECLARE @Cur01 CURSOR;

	DECLARE @CID			INTEGER
	DECLARE @CSectorID		INTEGER
	DECLARE @CIsBuffer		BIT
	DECLARE @CColourCode	NVARCHAR(20)
	DECLARE @CIsAddBuffer	BIT

	DECLARE @Cur02 CURSOR;

	DECLARE @C2TARID		INTEGER 
	DECLARE @C2TARNo		NVARCHAR(20) 
	DECLARE @C2TARType		NVARCHAR(20)
	DECLARE @C2AccessDate	DATE
	DECLARE @C2AccessType	NVARCHAR(20) 
	DECLARE @C2IsExclusive	BIT
	DECLARE @C2SectorID		INTEGER 
	DECLARE @C2SectorStr	NVARCHAR(20) 
	DECLARE @C2IsBuffer		BIT 
	DECLARE @C2EmailAdd		NVARCHAR(200) 

	SET @Cur01 = CURSOR FAST_FORWARD 
	FOR
		SELECT ID, SectorId, IsBuffer, ColourCode, IsAddedBuffer FROM TAMS_TAR_Sector
			WHERE TARId = @TARID
				ORDER BY SectorId

	OPEN @Cur01
	
	FETCH NEXT FROM @Cur01 INTO @CID, @CSectorID, @CIsBuffer, @CColourCode, @CIsAddBuffer
	
	WHILE @@FETCH_STATUS = 0 
	
		BEGIN

			IF @CIsBuffer = 1
				BEGIN
					-- >> Is Buffer
					-- PRINT 1
					INSERT INTO #TmpExcSector
						(TARID, TARNo, TARType, AccessDate, AccessType, IsExclusive, SectorID, SectorStr, IsBuffer, EmailAdd)
					SELECT t.Id, t.TARNo, t.TARType, t.AccessDate, t.AccessType, t.IsExclusive,
							ts.SectorId, s.Sector, ts.IsBuffer, t.Email 
						FROM TAMS_TAR t, TAMS_TAR_Sector ts, TAMS_Sector s
							WHERE t.Id = ts.TARId
								AND s.ID = ts.SectorId
								AND t.AccessDate = @AccessDate
								AND ts.IsBuffer = 0
								AND t.TARStatusId = 8
								AND ts.SectorId = @CSectorID
								AND t.Id <> @TARID
								AND s.IsActive = 1
								AND @AccessDate BETWEEN s.EffectiveDate and S.ExpiryDate
				END
			ELSE
				BEGIN
					-- >> Not Buffer
					-- PRINT 2
					INSERT INTO #TmpExcSector
						(TARID, TARNo, TARType, AccessDate, AccessType, IsExclusive, SectorID, SectorStr, IsBuffer, EmailAdd)
					SELECT t.Id, t.TARNo, t.TARType, t.AccessDate, t.AccessType, t.IsExclusive,
							ts.SectorId, s.Sector, ts.IsBuffer, t.Email 
						FROM TAMS_TAR t, TAMS_TAR_Sector ts, TAMS_Sector s
							WHERE t.Id = ts.TARId
								AND s.ID = ts.SectorId
								AND t.AccessDate = @AccessDate
								-- >> AND ts.IsBuffer = 0
								AND t.TARStatusId = 8
								AND ts.SectorId = @CSectorID
								AND t.Id <> @TARID
								AND s.IsActive = 1
								AND @AccessDate BETWEEN s.EffectiveDate and S.ExpiryDate
				END
				
			-- >> #TmpExcSector
			SET @Cur02 = CURSOR FAST_FORWARD 
			FOR
				SELECT TARID, TARNo, TARType, AccessDate, AccessType, IsExclusive, SectorID, SectorStr, IsBuffer, EmailAdd 
					FROM #TmpExcSector
							ORDER BY TARID, SectorID

			OPEN @Cur02
	
			FETCH NEXT FROM @Cur02 INTO @C2TARID, @C2TARNo, @C2TARType, @C2AccessDate, @C2AccessType, @C2IsExclusive, @C2SectorID, @C2SectorStr, @C2IsBuffer, @C2EmailAdd
	
			WHILE @@FETCH_STATUS = 0 
	
				BEGIN
					IF @AccessType = 'Protection' AND @IsExclusive = 0
						BEGIN
							print 1
							IF @C2AccessType = 'Protection' AND @C2IsExclusive = 0
								BEGIN
									PRINT 'DO NOTHING'
								END
							ELSE
								BEGIN
									IF (SELECT COUNT(*) FROM #TmpExc WHERE TARID = @C2TARID) = 0
										BEGIN
											INSERT INTO #TmpExc
												(TARID, TARNo, TARType, AccessDate, AccessType, IsExclusive, EmailAdd)
											VALUES 
												(@C2TARID, @C2TARNo, @C2TARType, @C2AccessDate, @C2AccessType, @C2IsExclusive, @C2EmailAdd)
										END
								END
						END
					ELSE
						BEGIN
							IF (SELECT COUNT(*) FROM #TmpExc WHERE TARID = @C2TARID) = 0
								BEGIN
									INSERT INTO #TmpExc
										(TARID, TARNo, TARType, AccessDate, AccessType, IsExclusive, EmailAdd)
									VALUES 
										(@C2TARID, @C2TARNo, @C2TARType, @C2AccessDate, @C2AccessType, @C2IsExclusive, @C2EmailAdd)
								END
						END

					FETCH NEXT FROM @Cur02 INTO @C2TARID, @C2TARNo, @C2TARType, @C2AccessDate, @C2AccessType, @C2IsExclusive, @C2SectorID, @C2SectorStr, @C2IsBuffer, @C2EmailAdd
				END 	

			CLOSE @Cur02
	
			DEALLOCATE @Cur02
			-- >> #TmpExcSector

			FETCH NEXT FROM @Cur01 INTO @CID, @CSectorID, @CIsBuffer, @CColourCode, @CIsAddBuffer
		END

DECLARE @C3AccessType	NVARCHAR(20)

-- >> Action Log
DECLARE @LogMsg	NVARCHAR(100)


-- >> #TmpExcSector
SET @Cur03 = CURSOR FAST_FORWARD 
FOR
	SELECT TARID, TARNo, TARType, AccessDate, AccessType, IsExclusive, EmailAdd 
		FROM #TmpExc
			ORDER BY TARID

OPEN @Cur03
	
FETCH NEXT FROM @Cur03 INTO @C3TARID, @C3TARNo, @C3TARType, @C3AccessDate, @C3AccessType, @C3IsExclusive, @C3EmailAdd
	
WHILE @@FETCH_STATUS = 0 
	
	BEGIN
		
		-- >> NEL Cancel = 10 Else DTL or LRt = 9
		UPDATE TAMS_TAR
			SET TARStatusId = CASE WHEN @Line = 'NEL' THEN 10 ELSE 9 END,
				UpdatedOn = GETDATE(),
				UpdatedBy = @UserID
		WHERE Id = @C3TARID

		SET @LogMsg = 'TAR Form Cancelled by : ' + CAST(@UserName AS NVARCHAR(500)) + ' On ' + CAST(CONVERT(NVARCHAR(20), GETDATE(), 103) AS NVARCHAR(20)) + ' ' + CAST(CONVERT(NVARCHAR(20), GETDATE(), 108) AS NVARCHAR(20))
		INSERT INTO [dbo].[TAMS_Action_Log]
			([Line], [Module], [Function], [TransactionID], [LogMessage], [CreatedOn], [CreatedBy])
		VALUES 
			(@Line, 'TAR', 'Approval - Cancel', @C3TARID, @LogMsg, GETDATE(), @UserID)

		IF @TARType = 'Urgent' -- >> If Current TAR Type is Urgent then send Email
			BEGIN
				-- >> Cancel Urgent TAR
				SET @EMTARStatus = 'Cancelled'
				SET @EMMsg = ''

				EXEC sp_TAMS_Email_Cancel_TAR @C3TARID, @EMTARStatus, @C3TARNo, @C3EmailAdd, @EMMsg OUTPUT

			END
		
		FETCH NEXT FROM @Cur03 INTO @C3TARID, @C3TARNo, @C3TARType, @C3AccessDate, @C3AccessType, @C3IsExclusive, @C3EmailAdd
	END 	

CLOSE @Cur03

		@NextRoleID = RoleId

IF ISNULL(@NextEndTitle, '') = ''
	BEGIN
		-- >> If NEL Approved = 9 Else DTL or LRT = 8
		UPDATE TAMS_TAR
			SET TARStatusId = CASE WHEN @Line = 'NEL' THEN 9 ELSE 8 END,
				UpdatedOn = GETDATE(),
				UpdatedBy = @UserID
			WHERE ID = @TARID

		IF @TARType = 'Urgent'
			BEGIN
				SET @EMTARStatus = 'Approved'
				SET @EMMsg = ''

				EXEC sp_TAMS_Email_Urgent_TAR @EMTARID, @EMTARStatus, @EMTARNo, @EMRemarks, @EMActor, @EMToSend, @EMMsg OUTPUT
			END

	END
ELSE
	BEGIN

		IF (SELECT COUNT(*) FROM TAMS_TAR_Workflow
			WHERE TARId = @TARID
				AND WorkflowId = @NextWFID
				AND EndorserId = @NextEndID) = 0
			BEGIN	
				INSERT INTO [dbo].[TAMS_TAR_Workflow]
					([TARId], [WorkflowId], [EndorserId], [UserId], [WFStatus], [Remark], [ActionOn], [ActionBy])
				VALUES
					(@TARID, @NextWFID, @NextEndID, NULL, 'Pending', '', GETDATE(), @UserID)
			END

		UPDATE TAMS_TAR
			SET TARStatusId = @NextWFStatID,
				UpdatedOn = GETDATE(),
				UpdatedBy = @UserID
			WHERE Id = @TARID

		IF @TARType = 'Urgent'
			BEGIN
				-- >> NEXT ROLE ID EMAIL ALERT
				-- >> PRINT 'Urgent Email'
				DECLARE @RoleEmail	NVARCHAR(1000)
				SET @RoleEmail = ''
				SET @EMMsg = ''

				SELECT @RoleEmail = @RoleEmail + ISNULL(a.Email + ', ', '') 
					FROM TAMS_User a, TAMS_User_Role b
						WHERE a.Userid = b.UserID
							AND a.IsActive = 1
							AND @TARDate BETWEEN a.ValidFrom AND a.ValidTo
							AND b.RoleID = @NextEndID

				IF LEN(LTRIM(RTRIM(@RoleEmail))) > 0
					BEGIN
						SET @RoleEmail = LEFT(LTRIM(RTRIM(@RoleEmail)), LEN(LTRIM(RTRIM(@RoleEmail))) - 1)
					END

				-- >> Select all email where role id = next role id
				SET @EMToSend = @RoleEmail

				EXEC sp_TAMS_Email_Apply_Urgent_TAR 2, @EMCompany, @EMTARNo, @NextEndTitle, @EMToSend, @EMCCSend, @EMMsg OUTPUT
			END
	END

	-- >> SET @LogMsg = 'TAR Form has approved by : ' + '' + ' On ' + GETDATE()	
	SET @LogMsg = 'TAR Form Approved by : ' + CAST(@UserName AS NVARCHAR(500)) + ' On ' + CAST(CONVERT(NVARCHAR(20), GETDATE(), 103) AS NVARCHAR(20)) + ' ' + CAST(CONVERT(NVARCHAR(20), GETDATE(), 108) AS NVARCHAR(20))
		
	INSERT INTO [dbo].[TAMS_Action_Log]
		([Line], [Module], [Function], [TransactionID], [LogMessage], [CreatedOn], [CreatedBy])
	VALUES 
		(@Line, 'TAR', 'Approved TAR', @TARID, @LogMsg, GETDATE(), @UserID)

DECLARE @WFType	NVARCHAR(50)
DECLARE @InvPow	BIT

SELECT @WFType = WorkflowType,
		@InvPow = InvolvePower 
	FROM TAMS_Workflow
		WHERE ID = @WFID

IF @WFType = 'UrgentAfter' AND @InvPow = 1 AND @ELevel = 2
	BEGIN
		-- >> TAMS_DTL_PFR
		-- >> PRINT 'EMAIL'

		DECLARE @OCCEmail	NVARCHAR(1000)
		DECLARE @OCCRole	NVARCHAR(50)

		SET @OCCEmail = ''
		SET @OCCRole = @Line + '_PFR'
		SET @EMTARStatus = 'Approved'
		SET @EMMsg = ''

		SELECT @OCCEmail = @OCCEmail + ISNULL(a.Email + ', ', '') 
			FROM TAMS_User a, TAMS_User_Role b, TAMS_Role c
				WHERE a.Userid = b.UserID
					AND b.RoleID = c.ID
					AND a.IsActive = 1
					AND @TARDate BETWEEN a.ValidFrom AND a.ValidTo
					AND c.Role = @OCCRole

		IF LEN(LTRIM(RTRIM(@OCCEmail))) > 0
			BEGIN
				SET @OCCEmail = LEFT(LTRIM(RTRIM(@OCCEmail)), LEN(LTRIM(RTRIM(@OCCEmail))) - 1)
			END

		EXEC sp_TAMS_Email_Urgent_TAR_OCC @TARID, @EMTARStatus, @EMTARNo, @Remarks, @OCCEmail, @EMMsg OUTPUT
	END

-- >>

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR INSERTING INTO TAMS_TAR'
	GOTO TRAP_ERROR
END
