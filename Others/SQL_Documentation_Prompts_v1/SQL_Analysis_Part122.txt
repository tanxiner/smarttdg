CREATE PROCEDURE [dbo].[sp_TAMS_RGS_Cancel_OSReq]
	@TARID		AS BIGINT = 0,
	@CancelRemarks	AS NVARCHAR(1000) = NULL,
	@UserID		AS NVARCHAR(500),
	@Message	AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

SET @Message = ''

DECLARE @NewLine AS CHAR(2) = CHAR(30) + CHAR(10)

UPDATE TAMS_TOA
	SET	TOAStatus = 6,
		CancelRemark = @CancelRemarks,
		UpdatedOn = GETDATE(),
		UpdatedBy = @UserID
WHERE TARId = @TARID

DECLARE @SMSMsg	NVARCHAR(4000)
SET @SMSMsg = ''

DECLARE @lv_IsAllAckSurrender BIT
DECLARE @TARNo NVARCHAR(50)
DECLARE @TOANo NVARCHAR(50)
DECLARE @Line NVARCHAR(5)
DECLARE @CurrDate NVARCHAR(20)
DECLARE @CurrTime NVARCHAR(20)
DECLARE @AccessDate DATE
DECLARE @OperationDate DATE

SET @lv_IsAllAckSurrender = 1

SET @TARNo = ''
SET @Line = ''
SET @CurrDate = ''

SET @CurrDate = CONVERT(NVARCHAR(20), GETDATE(), 103)
SET @CurrTime = CONVERT(NVARCHAR(20), GETDATE(), 108)

SELECT @TARNo = a.TARNo,
		@TOANo = b.TOANo, 
		@Line = a.Line, 
		@AccessDate = b.AccessDate,
		@OperationDate = b.OperationDate
	FROM TAMS_TAR a, TAMS_TOA b	
		WHERE a.Id = b.TARId
			AND a.Id = @TARID

DECLARE @Endorser1 INT
DECLARE @Endorser2 INT
DECLARE @Endorser3 INT

IF @Line = 'DTL'
	BEGIN
		SELECT @Endorser1 = a.ID 
			FROM TAMS_Endorser a, TAMS_Workflow b
				WHERE b.WorkflowType = 'OCCAuth'
					AND a.WorkflowId = b.ID
					AND b.Line = 'DTL'
					AND a.Level = 10

		SELECT @Endorser2 = a.ID 
			FROM TAMS_Endorser a, TAMS_Workflow b
				WHERE b.WorkflowType = 'OCCAuth'
					AND a.WorkflowId = b.ID
					AND b.Line = 'DTL'
					AND a.Level = 11

		SELECT @Endorser3 = a.ID 
			FROM TAMS_Endorser a, TAMS_Workflow b
				WHERE b.WorkflowType = 'OCCAuth'
					AND a.WorkflowId = b.ID
					AND b.Line = 'DTL'
					AND a.Level = 12
	END
ELSE
	BEGIN
		SELECT @Endorser1 = a.ID 
			FROM TAMS_Endorser a, TAMS_Workflow b
				WHERE b.WorkflowType = 'OCCAuth'
					AND a.WorkflowId = b.ID
					AND b.Line = 'NEL'
					AND a.Level = 7
	END

DECLARE @Cur01 CURSOR;
DECLARE @TOAStatus	INT

SET @Cur01 = CURSOR FAST_FORWARD 
FOR
SELECT b.TOAStatus
	FROM TAMS_TAR a, TAMS_TOA b
		WHERE a.Id = b.TARId
			AND b.AccessDate = @AccessDate
			AND b.TOAStatus NOT IN (0, 6)
			AND a.Line = @Line
	ORDER BY b.Id
OPEN @Cur01
FETCH NEXT FROM @Cur01 INTO @TOAStatus
WHILE @@FETCH_STATUS = 0 
	BEGIN
		IF @TOAStatus <> 5
			BEGIN
				SET @lv_IsAllAckSurrender = 0
			END
		FETCH NEXT FROM @Cur01 INTO @TOAStatus
	END 	
CLOSE @Cur01
DEALLOCATE @Cur01

IF @lv_IsAllAckSurrender = 1
	BEGIN
		IF @Line = 'DTL'
			BEGIN
				DECLARE @Cur02 CURSOR;
				DECLARE @Cur03 CURSOR;
				DECLARE @DTLOCCAuthID	INT

				SET @Cur02 = CURSOR FAST_FORWARD 
				FOR
				SELECT a.ID
					FROM TAMS_OCC_Auth a
						WHERE a.Line = @Line
							AND a.OperationDate = @OperationDate
							AND a.AccessDate = @AccessDate
							AND a.OCCAuthStatusId = 10
							AND a.PowerOn = 0
				OPEN @Cur02
				FETCH NEXT FROM @Cur02 INTO @DTLOCCAuthID
				WHILE @@FETCH_STATUS = 0 
					BEGIN
						UPDATE TAMS_OCC_Auth
							SET OCCAuthStatusId = 11,
								UpdatedOn = GETDATE(),
								UpdatedBy = @UserID
							WHERE ID = @DTLOCCAuthID

						INSERT INTO [dbo].[TAMS_OCC_Auth_Workflow]
							([OCCAuthId], [OCCAuthEndorserId], [WFStatus], 
								[StationId], [FISTestResult], [ActionOn], [ActionBy])
						VALUES (@DTLOCCAuthID, @Endorser1, 'Pending',
							0, NULL, GETDATE(), @UserID)

						FETCH NEXT FROM @Cur02 INTO @DTLOCCAuthID
					END 	
				CLOSE @Cur02
				DEALLOCATE @Cur02

				-- >> 
				SET @Cur03 = CURSOR FAST_FORWARD 
				FOR
				SELECT a.ID
					FROM TAMS_OCC_Auth a
						WHERE a.Line = @Line
							AND a.OperationDate = @OperationDate
							AND a.AccessDate = @AccessDate
							AND a.OCCAuthStatusId = 8 -- >> Check with Reno how to know this one end already
				OPEN @Cur03
				FETCH NEXT FROM @Cur03 INTO @DTLOCCAuthID
				WHILE @@FETCH_STATUS = 0 
					BEGIN
						UPDATE TAMS_OCC_Auth
							SET OCCAuthStatusId = 12,
								UpdatedOn = GETDATE(),
								UpdatedBy = @UserID
							WHERE ID = @DTLOCCAuthID

						INSERT INTO [dbo].[TAMS_OCC_Auth_Workflow]
							([OCCAuthId], [OCCAuthEndorserId], [WFStatus], 
								[StationId], [FISTestResult], [ActionOn], [ActionBy])
						VALUES (@DTLOCCAuthID, @Endorser1, 'Terminated',
							0, NULL, GETDATE(), @UserID)

						FETCH NEXT FROM @Cur03 INTO @DTLOCCAuthID
					END 	
				CLOSE @Cur03
				DEALLOCATE @Cur03

				-- >> 
				SET @Cur03 = CURSOR FAST_FORWARD 
				FOR
				SELECT a.ID
					FROM TAMS_OCC_Auth a
						WHERE a.Line = @Line
							AND a.OperationDate = @OperationDate
							AND a.AccessDate = @AccessDate
							AND a.OCCAuthStatusId = 8 -- >> Check with Reno how to know this one end already
				OPEN @Cur03
				FETCH NEXT FROM @Cur03 INTO @DTLOCCAuthID
				WHILE @@FETCH_STATUS = 0 
					BEGIN
						INSERT INTO [dbo].[TAMS_OCC_Auth_Workflow]
							([OCCAuthId], [OCCAuthEndorserId], [WFStatus], 
								[StationId], [FISTestResult], [ActionOn], [ActionBy])
						VALUES (@DTLOCCAuthID, @Endorser2, 'Terminated',
							0, NULL, GETDATE(), @UserID)

						FETCH NEXT FROM @Cur03 INTO @DTLOCCAuthID
					END 	
				CLOSE @Cur03
				DEALLOCATE @Cur03

				-- >> 
				SET @Cur03 = CURSOR FAST_FORWARD 
				FOR
				SELECT a.ID
					FROM TAMS_OCC_Auth a
						WHERE a.Line = @Line
							AND a.OperationDate = @OperationDate
							AND a.AccessDate = @AccessDate
							AND a.OCCAuthStatusId = 8 -- >> Check with Reno how to know this one end already
				OPEN @Cur03
				FETCH NEXT FROM @Cur03 INTO @DTLOCCAuthID
				WHILE @@FETCH_STATUS = 0 
					BEGIN
						INSERT INTO [dbo].[TAMS_OCC_Auth_Workflow]
							([OCCAuthId], [OCCAuthEndorserId], [WFStatus], 
								[StationId], [FISTestResult], [ActionOn], [ActionBy])
						VALUES (@DTLOCCAuthID, @Endorser3, 'Pending',
							0, NULL, GETDATE(), @UserID)

						FETCH NEXT FROM @Cur03 INTO @DTLOCCAuthID
					END 	
				CLOSE @Cur03
				DEALLOCATE @Cur03
			END
		ELSE
			BEGIN
				DECLARE @Cur04 CURSOR;
				DECLARE @NELOCCAuthID	INT

				SET @Cur02 = CURSOR FAST_FORWARD 
				FOR
				SELECT a.ID
					FROM TAMS_OCC_Auth a
						WHERE a.Line = @Line
							AND a.OperationDate = @OperationDate
							AND a.AccessDate = @AccessDate
							AND a.OCCAuthStatusId = 7
							AND a.PowerOn = 0
				OPEN @Cur02
				FETCH NEXT FROM @Cur02 INTO @NELOCCAuthID
				WHILE @@FETCH_STATUS = 0 
					BEGIN
						UPDATE TAMS_OCC_Auth
							SET OCCAuthStatusId = 8,
								UpdatedOn = GETDATE(),
								UpdatedBy = @UserID
							WHERE ID = @NELOCCAuthID

						INSERT INTO [dbo].[TAMS_OCC_Auth_Workflow]
							([OCCAuthId], [OCCAuthEndorserId], [WFStatus], 
								[StationId], [FISTestResult], [ActionOn], [ActionBy])
						VALUES (@NELOCCAuthID, @Endorser1, 'Pending',
							0, NULL, GETDATE(), @UserID)

						FETCH NEXT FROM @Cur02 INTO @NELOCCAuthID
					END 	
				CLOSE @Cur02
				DEALLOCATE @Cur02
			END
	END

IF @Line = 'DTL'
	BEGIN
		SET @SMSMsg = CAST(@TOANo AS NVARCHAR(50)) + ' surrender has been acknowledged by DTL OCC @ ' + CAST(@CurrTime AS NVARCHAR(20))
			+ ' hrs on ' + CAST(@CurrDate AS NVARCHAR(20)) + '. Thank you!'
	END
ELSE
	BEGIN
		SET @SMSMsg = CAST(@TOANo AS NVARCHAR(50)) + ' surrender has been acknowledged by NEL OCC @ ' + CAST(@CurrTime AS NVARCHAR(20))
			+ ' hrs on ' + CAST(@CurrDate AS NVARCHAR(20)) + '. Thank you!'
	END

-- >> SEND SMS

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR INSERTING TAMS_TOA_Parties'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END

CREATE PROCEDURE [dbo].[sp_TAMS_RGS_GrantTOA]
	@TARID		AS BIGINT = 0,
	@EncTARID	AS NVARCHAR(250),
	@UserID		AS NVARCHAR(500),
	@Message	AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

SET @Message = ''

DECLARE @NewLine AS CHAR(2) = CHAR(30) + CHAR(10)

DECLARE @SMSMsg	NVARCHAR(4000)
SET @SMSMsg = ''

DECLARE @OPDate NVARCHAR(20)
DECLARE @TARNo NVARCHAR(50)
DECLARE @Line NVARCHAR(5)
DECLARE @CurrDate NVARCHAR(20)
DECLARE @AccessType NVARCHAR(20)
DECLARE @Site NVARCHAR(200)
DECLARE @HPNo NVARCHAR(20)
DECLARE @TOAStatus INT

SET @TARNo = ''
SET @Line = ''
SET @CurrDate = ''
SET @AccessType = ''
SET @Site = ''
SET @HPNo = ''
SET @TOAStatus = 0

SELECT @TARNo = a.TARNo, 
		@Line = a.Line, 
		@OPDate = CONVERT(NVARCHAR(20), b.OperationDate, 103),
		@AccessType = a.AccessType,
		@HPNo = b.MobileNo,
		@TOAStatus = b.TOAStatus
	FROM TAMS_TAR a, TAMS_TOA b	
		WHERE a.Id = b.TARId
			AND a.Id = @TARID

IF @TOAStatus = 2
	BEGIN
		DECLARE @RefNum		NVARCHAR(20)
		DECLARE @RefNumMsg	NVARCHAR(20)
		SET @RefNum = ''
		SET @RefNumMsg = ''
		EXEC sp_Generate_Ref_Num_TOA 'TOA', @Line,@TARID, @OPDate, 'Mainline' , @RefNum OUTPUT, @RefNumMsg OUTPUT

		UPDATE TAMS_TOA
			SET	TOAStatus = 3,
				TOANo = @RefNum,
				GrantTOATime = GETDATE(),
				UpdatedOn = GETDATE(),
				UpdatedBy = @UserID
		WHERE TARId = @TARID

		INSERT INTO TAMS_TOA_Audit
		SELECT @UserID, GETDATE(), 'U', * FROM TAMS_TOA
			WHERE TARId = @TARID

		SET @CurrDate = CONVERT(NVARCHAR(20), GETDATE(), 103)

		--IF @AccessType = 'Possession'

		--			+ @Site + 'https://sbstweb.sbstransit.com.sg/TAMS/RGSAckSMS.aspx?TARID=' + CAST(@EncTARID AS NVARCHAR(500)) + '&SMSType=2'

		--ELSE
		--	BEGIN
		--		SET @SMSMsg = CAST(@RefNum AS NVARCHAR(50)) + ' has been granted for TAR ID ' + CAST(@TARNo AS NVARCHAR(20))
		--			+ '. You may now proceed to work on track. Click on the link to acknowledge TOA :~'
		--			+ @Site + 'https://sbstweb.sbstransit.com.sg/TAMS/RGSAckSMS.aspx?TARID=' + CAST(@EncTARID AS NVARCHAR(500)) + '&SMSType=1'
		--	END

		IF @AccessType = 'Possession'
			BEGIN
				SET @SMSMsg = CAST(@RefNum AS NVARCHAR(50)) + ' has been granted for TAR ID ' + CAST(@TARNo AS NVARCHAR(20))
					+ '. Click on the link to acknowledge TOA :~'
					+ @Site + 'https://sbstweb.sbstransit.com.sg/TAMS/RGSAckSMS.aspx?TARID=' + CAST(@EncTARID AS NVARCHAR(500)) + '&SMSType=2'
			END
		ELSE
			BEGIN
				SET @SMSMsg = CAST(@RefNum AS NVARCHAR(50)) + ' has been granted for TAR ID ' + CAST(@TARNo AS NVARCHAR(20))
					+ '. Click on the link to acknowledge TOA :~'
					+ @Site + 'https://sbstweb.sbstransit.com.sg/TAMS/RGSAckSMS.aspx?TARID=' + CAST(@EncTARID AS NVARCHAR(500)) + '&SMSType=1'
			END


		-- >> SEND SMS
		DECLARE @RetVal NVARCHAR(5)
		SET @RetVal = ''

			IF @HPNo <> ''
				BEGIN
					EXEC sp_api_send_sms @HPNo, 'TAMS RGS', @SMSMsg, @RetVal OUTPUT
				END

		------ >> 20220818 Quickly trigger SMS Sending

ELSE
	BEGIN
		IF @TOAStatus = 3

				SET @Message = '1'

				GOTO TRAP_ERROR
			END
		ELSE
			BEGIN
				SET @Message = 'Invalid TAR status. Please refresg RGS.'
				GOTO TRAP_ERROR
			END
	END

IF @@ERROR <> 0 BEGIN
