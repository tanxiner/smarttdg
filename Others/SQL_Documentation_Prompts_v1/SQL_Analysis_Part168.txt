CREATE PROCEDURE [dbo].[sp_TAMS_TOA_Submit_Register]
	@UserID			AS NVARCHAR(20),
	@PartiesWitness	AS BIGINT = 0,
	@TOAID			AS BIGINT = 0,
	@Message		AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

-- >>
UPDATE [dbo].[TAMS_TOA]
SET TOAStatus	= 1,
	RegisteredTime	=	GETDATE(),
	UpdatedOn		=	GETDATE(),
	UpdatedBy		=	[InChargeName]
WHERE ID = @TOAID

INSERT INTO TAMS_TOA_Audit
SELECT [InChargeName], GETDATE(), 'I', * FROM TAMS_TOA
	WHERE ID = @TOAID

UPDATE [dbo].[TAMS_TOA_Parties]
SET IsWitness	=	1
WHERE TOAId	=	@TOAID
	AND Id	=	@PartiesWitness

UPDATE [dbo].[TAMS_TOA_Parties]
SET BookInTime	=	GETDATE()
WHERE TOAId	=	@TOAID

-- >>

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR UPDATING TAMS_TOA'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END

CREATE PROCEDURE [dbo].[sp_TAMS_Update_Company_Details_By_ID] 	
	@CompID				INT,
	@Company			NVARCHAR(100),
	@BizOwner			NVARCHAR(200),
	@CompanyOfficeNo	NVARCHAR(20),
	@CompanyMobileNo	NVARCHAR(20),
	@CompanyEmail		NVARCHAR(200),
	@IsActive			BIT,
	@UpdatedBy			INT
AS
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION;
			IF EXISTS(SELECT * FROM TAMS_Company WHERE ID = @CompID) 
			BEGIN
				UPDATE TAMS_Company 
				SET Company = @Company, 
					BizOwner = @BizOwner, 
					CompanyOfficeNo = @CompanyOfficeNo, 
					CompanyMobileNo = @CompanyMobileNo, 
					CompanyEmail = @CompanyEmail,
					UpdatedBy = @UpdatedBy, 
					UpdatedOn = GETDATE()
				WHERE ID = @CompID;

			END;

		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		ROLLBACK TRANSACTION;
	END CATCH

END;

CREATE PROCEDURE [dbo].[sp_TAMS_Update_External_UserPasswordByUserID] 	
	@UserID						INT,
	@Password					NVARCHAR(200)
AS
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION;
			IF EXISTS(SELECT * FROM TAMS_User WHERE UserID = @UserID) 
			BEGIN


				UPDATE TAMS_User 
				SET Password = dbo.EncryptString(@Password), PasswordChangedDate = GETDATE()
				WHERE UserID = @UserID;

			END;

		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		ROLLBACK TRANSACTION;
	END CATCH

END;
