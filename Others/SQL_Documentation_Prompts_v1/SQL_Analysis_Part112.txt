									AND a.AccessDate = @AccessDate

									-- >> AND a.OCCAuthStatusId = 7 -- >> Check with Reno how to know this one end already

						OPEN @Cur03
						FETCH NEXT FROM @Cur03 INTO @DTLOCCAuthID

								UPDATE TAMS_OCC_Auth

										UpdatedOn = GETDATE(),

									WHERE ID = @DTLOCCAuthID

								([ActionBy], [ActionOn], [AuditAction], 
									[OCCAuthID], [Line], [OperationDate], [AccessDate], [TractionPowerId], [Remark], [PFRRemark],

								([AuditActionBy], [AuditActionOn], [AuditAction],

							WHERE  WFStatus = 'Pending' and OCCAuthId in (
							Select ID
							FROM TAMS_OCC_Auth WHERE TractionPowerId = @TractionPowerID

						AND OCCAuthStatusId = 8

							UpdatedOn = GETDATE(),
							UpdatedBy = @UserIDID
						WHERE TractionPowerId = @TractionPowerID
						AND OperationDate = @OperationDate 
						and AccessDate = @AccessDate
						AND OCCAuthStatusId = 8
						

					-- >> Insert to Audit
					INSERT INTO [dbo].[TAMS_OCC_Auth_Audit]

						[CreatedOn], [CreatedBy], [UpdatedOn], [UpdatedBy]
						FROM [dbo].[TAMS_OCC_Auth]
						WHERE TractionPowerId = @TractionPowerID
						AND OperationDate = @OperationDate 
						and AccessDate = @AccessDate
						AND OCCAuthStatusId = 8
					-- >> Insert to Audit

				END;


				 
				FETCH NEXT FROM @CurNELOCC INTO @TractionPowerID

		ELSE
			BEGIN
				SET @SMSMsg = CAST(@TOANo AS NVARCHAR(50)) + ' surrender has been acknowledged by NEL OCC @ ' + CAST(@CurrTime AS NVARCHAR(20))
