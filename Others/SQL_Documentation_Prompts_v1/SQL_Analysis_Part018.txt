CREATE PROCEDURE [dbo].[sp_TAMS_Block_Date_Delete]
	@BlockID	AS INTEGER,
	@Message	AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

-- >>
SET @Message = ''

INSERT INTO [dbo].[TAMS_Block_TARDate_Audit]
SELECT '', GETDATE(), 'D', * FROM [dbo].[TAMS_Block_TARDate]
	WHERE ID = @BlockID

DELETE 
	FROM TAMS_Block_TARDate
		WHERE ID = @BlockID

-- >>

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR DELETING TAMS_BLOCK_TARDATE'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message


END

CREATE PROCEDURE  [dbo].[sp_TAMS_Block_Date_OnLoad]
	@Line		AS NVARCHAR(20) = NULL,
	@TrackType  AS NVARCHAR(50) = NULL,
	@BlockDate	AS NVARCHAR(20) = NULL
AS

BEGIN
	SELECT ID, Line,TrackType, CONVERT(NVARCHAR(20), BlockDate, 103) AS BlockDate, BlockReason, IsActive 
		FROM TAMS_Block_TARDate
			WHERE (Line = @Line OR ISNULL(@Line, '') = '')
			AND  (TrackType = @TrackType OR ISNULL(@TrackType, '') = '')
				AND (BlockDate = CONVERT(DATE, @BlockDate, 103) OR ISNULL(@BlockDate, '') = '')
		ORDER BY CONVERT(DATE, BlockDate, 103) DESC
END

CREATE PROCEDURE  [dbo].[sp_TAMS_Block_Date_Save]
	@Line			AS NVARCHAR(20) = NULL,
	@TrackType		AS NVARCHAR(50) = NULL,
	@BlockDate		AS NVARCHAR(20) = NULL,
	@BlockReason	AS NVARCHAR(100) = NULL,
	@UserLI			AS NVARCHAR(50) = NULL,
	@Message		AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

-- >>
SET @Message = ''

DECLARE @CDWeekNo	INTEGER
DECLARE @BDWeekNo	INTEGER
DECLARE @CDYear		INTEGER
DECLARE @BDYear		INTEGER
DECLARE @UserID		INTEGER
DECLARE @BlockDateID	INTEGER

SELECT @UserID = Userid
FROM TAMS_User
	WHERE LoginId = @UserLI

SET @CDWeekNo = DATEPART(WEEK, GETDATE())
SET @BDWeekNo = DATEPART(WEEK, CONVERT(DATE, @BlockDate, 103))

SET @CDYear	= YEAR(GETDATE())
SET @BDYear = YEAR(CONVERT(DATE, @BlockDate, 103))

IF @CDYear = @BDYear
	BEGIN
		IF (@BDWeekNo - @CDWeekNo) < 5
			BEGIN
				-- "Block Date must be N-5 weeks."
				-- SET @Message = 'Block Date must be N-5 weeks.'
				SET @Message = '1'
			END
	END
ELSE
	BEGIN
		IF @CDYear < @BDYear
			BEGIN
				SET @BDWeekNo = @BDWeekNo + 52
				IF (@BDWeekNo - @CDWeekNo) < 5
					BEGIN
						-- "Block Date must be N-5 weeks."
						-- SET @Message = 'Block Date must be N-5 weeks.'
						SET @Message = '2'
					END
			END
		ELSE
			BEGIN
				-- "Block Date must future date"
				-- SET @Message = 'Block Date must future date.'
				SET @Message = '3'
			END
	END

	IF (SELECT COUNT(*) FROM TAMS_Block_TARDate 
			WHERE BlockDate = CONVERT(DATE, @BlockDate, 103)
				AND Line = @Line
				AND  TrackType = @TrackType
				AND IsActive = 1) > 0
		BEGIN
			-- "Block Date Already Existing."
			-- SET @Message = 'Block Date Already Existing.'
			SET @Message = '4'
		END

	IF @Message = ''
		BEGIN
			INSERT INTO [dbo].[TAMS_Block_TARDate]
				([Line],[TrackType], [BlockDate], [BlockReason], [IsActive],
				[CreatedOn], [CreatedBy], [UpdatedOn], [UpdatedBy])
			VALUES
				(@Line,@TrackType, CONVERT(DATE, @BlockDate, 103), @BlockReason, 1,
				GETDATE(), @UserID, GETDATE(), @UserID)

			SELECT @BlockDateID = SCOPE_IDENTITY();

			INSERT INTO [dbo].[TAMS_Block_TARDate_Audit]
			SELECT @UserID, GETDATE(), 'I', * FROM [dbo].[TAMS_Block_TARDate]
				WHERE ID = @BlockDateID
		END

-- >>

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR INSERTING INTO TAMS_TAR'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message


END

CREATE PROCEDURE [dbo].[sp_TAMS_CancelTarByTarID] 	
	@TarId AS integer = 0,
	@UID As integer = 0
AS
BEGIN
	DECLARE @Line as varchar(3),
			@StatusId as int,
			@Name as nvarchar(500)
	BEGIN TRY
		BEGIN TRANSACTION;
			BEGIN
				select @Line = line from TAMS_TAR where Id = @TarId;
				select @StatusId = WFStatusId from TAMS_WFStatus where Line = @Line and WFType = 'TARWFStatus' and WFStatus = 'Cancel';
				select @Name = [Name] from TAMS_User where userid = @UID;

				UPDATE TAMS_TAR 
				SET TARStatusId = @StatusId
				WHERE Id = @TarId;



				insert into TAMS_Action_Log
				values(@Line, 'TAR', 'Enquiry-Cancel', @TarId, 'TAR Form has been cancelled by ' + @Name + ' on ' + convert(varchar, getdate(), 103), GETDATE(), @UID);
			END;

		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		ROLLBACK TRANSACTION;
	END CATCH

END;

CREATE PROCEDURE [dbo].[sp_TAMS_Delete_RegQueryDept_SysOwnerApproval] 	
	@RegModID			INT,
	@RegRoleID			INT
AS
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION;

				DECLARE @Module NVARCHAR(20);
				DECLARE @RegStatus INT;
				DECLARE @RegId INT;
				DECLARE @Line NVARCHAR(20);

			-- TAMS_User_Role
			SELECT @RegId=RegID, @Line=Line, @Module=Module, @RegStatus=RegStatus FROM TAMS_Reg_Module WHERE ID = @RegModID

			SET @RegStatus = @RegStatus - 1;

			SELECT @RegModID=ID FROM TAMS_Reg_Module WHERE RegId = @RegId AND Line=@Line AND Module=@Module AND RegStatus=@RegStatus;

			IF EXISTS
			(
				SELECT * FROM TAMS_Reg_QueryDept WHERE RegModID = @RegModID AND RegRoleID = @RegRoleID
			)
			BEGIN
				DELETE FROM TAMS_Reg_QueryDept WHERE RegModID = @RegModID AND RegRoleID = @RegRoleID;

			END;
		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		ROLLBACK TRANSACTION;
	END CATCH

END;
