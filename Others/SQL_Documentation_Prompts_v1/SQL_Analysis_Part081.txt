CREATE PROCEDURE [dbo].[sp_TAMS_Insert_RegQueryDept_SysOwnerApproval] 	
	@RegModID			INT,
	@RegRoleID			INT,
	@Dept				NVARCHAR(200),
	@UpdatedBy			INT
AS
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION;

		DECLARE @UserID INT;
		SELECT @UserID = UserID FROM TAMS_User WHERE LoginID = (SELECT LoginID FROM TAMS_Registration WHERE ID = 
		(SELECT RegID FROM TAMS_Reg_Module WHERE ID = @RegModID))

		INSERT INTO TAMS_Reg_QueryDept 
		VALUES (@RegModID, @RegRoleID, @Dept, GETDATE(), @UpdatedBy, GETDATE(), @UpdatedBy);

		IF NOT EXISTS(SELECT * FROM TAMS_User_QueryDept WHERE UserID = @UserID AND RoleID = @RegRoleID AND 
		TARQueryDept = @Dept)
		BEGIN
			INSERT INTO TAMS_User_QueryDept 
			VALUES (@UserID, @RegRoleID, @Dept, GETDATE(), @UpdatedBy, GETDATE(), @UpdatedBy);
		END;

		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		ROLLBACK TRANSACTION;
	END CATCH

END;

CREATE PROCEDURE [dbo].[sp_TAMS_Insert_UserQueryDeptByUserID] 	
	@UserID			INT,
	@Dept			NVARCHAR(100),
	@UpdatedBy		INT
AS
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION;

		IF NOT EXISTS(SELECT * FROM TAMS_User_QueryDept WHERE UserID = @UserID AND TARQueryDept = @Dept)
		BEGIN
			DECLARE @RoleID INT;
			DECLARE @Line NVARCHAR(100);

			-- Get Dept Line
			SELECT @Line = ParaValue1 
			FROM TAMS_Parameters
			WHERE ParaCode = 'Company'
			AND ParaValue2 = @Dept
			AND EffectiveDate <= GETDATE()
			AND ExpiryDate >= GETDATE()

			-- Get Role ID
			SELECT @RoleID = ID 
			FROM TAMS_Role 
			WHERE Line = @Line AND 
			Role LIKE @Line + '%' + 'ApplicantHOD'

			INSERT INTO TAMS_User_QueryDept 
			VALUES (@UserID, @RoleID, @Dept, GETDATE(), @UpdatedBy, GETDATE(), @UpdatedBy);

		END

		

		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		ROLLBACK TRANSACTION;
	END CATCH

END;

CREATE PROCEDURE [dbo].[sp_TAMS_OCC_AddTVFAckRemarks]
	@UserId			as int,
	@TVFAckId		as int,
	@TVFRemarks		as nvarchar(1000)

AS
	DECLARE @NewID AS int;
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION;
			INSERT INTO TAMS_TVF_Ack_Remark(
			TVFAckId,
			Remark,
			CreatedOn,
			CreatedBy,
			UpdatedOn,
			UpdatedBy)
			VALUES (
			@TVFAckId,
			@TVFRemarks,
			GETDATE(),
			@UserId,
			GETDATE(),
			@UserId)

			SELECT @NewID = SCOPE_IDENTITY();

			INSERT INTO TAMS_TVF_Ack_Remark_Audit(
			   [ActionBy]
			  ,[ActionOn]
			  ,[AuditAction]
			  ,[TVFAckRemarkId]
			  ,[TVFAckId]
			  ,[Remark]
			  ,[CreatedOn]
			  ,[CreatedBy]
			  ,[UpdatedOn]
			  ,[UpdatedBy])
			  SELECT 
			   @UserID
			  ,GETDATE()
			  ,'I'
			  ,@NewID
			  ,[TVFAckId]
			  ,[Remark]
			  ,[CreatedOn]
			  ,[CreatedBy]
			  ,[UpdatedOn]
			  ,[UpdatedBy]
			  FROM TAMS_TVF_Ack_Remark
		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		ROLLBACK TRANSACTION;
	END CATCH

END
