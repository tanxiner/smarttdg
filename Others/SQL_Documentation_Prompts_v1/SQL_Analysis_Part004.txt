CREATE PROCEDURE [dbo].[SP_TAMS_Depot_SaveDTCAuthComments]
	-- Add the parameters for the stored procedure here
	--@authid as int,
	@str TAMS_DTC_AUTH_COMMENTS readonly,
	@success as bit=NULL OUTPUT,
	@Message AS NVARCHAR(500) = NULL OUTPUT
AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	SET NOCOUNT ON;

	
	DECLARE	@IntrnlTrans	TINYINT
	SELECT 	@IntrnlTrans 	= 0	

	IF (@@TRANCOUNT = 0) 
	BEGIN
		SET @IntrnlTrans = 1
		BEGIN TRAN
	END

			-- insert comments

		DECLARE @authid as int;
		DECLARE @comments as nvarchar(max);

		DECLARE C CURSOR FOR
		SELECT authid,comments from @str 
		OPEN C;

		FETCH NEXT FROM C INTO @authid, @comments
		WHILE @@FETCH_STATUS=0
		BEGIN 
		
			DECLARE @remarkid as int=-1;
			Select @remarkid=RemarkID from TAMS_Depot_Auth where id=@authid and RemarkID is not null

			if @remarkid<>-1
			BEGIN
				UPDATE TAMS_Depot_Auth_Remark
				SET Remark=@comments
				where RemarkID=@remarkid
			END
			ELSE
			BEGIN
				INSERT INTO TAMS_Depot_Auth_Remark
				VALUES(@comments)

				SET @remarkid=SCOPE_IDENTITY();

				UPDATE TAMS_Depot_Auth
				SET RemarkID=@remarkid
				where id=@authid;
			END
		
	
		FETCH NEXT FROM C INTO @authid, @comments
	END
	CLOSE C;
	DEALLOCATE C;

		
	IF @@ERROR <> 0 BEGIN
		SET @Message = 'ERROR INSERTING INTO Depot Authorization module.'
		
		PRINT 'trap error'
		GOTO TRAP_ERROR
	END

	FORCE_EXIT_PROC:	
	
		PRINT 'trap error 1'
		IF @IntrnlTrans = 1 COMMIT TRAN
		set @success=1;
		RETURN;
		--RETURN @Message 

	TRAP_ERROR:
	
		PRINT 'trap error 2'
		IF @IntrnlTrans = 1 ROLLBACK TRAN
		set @success=0;
		RETURN;
		--RETURN @Message
END

CREATE PROCEDURE [dbo].[getUserInformationByID] 
	@UserID NVARCHAR(100) = NULL
AS
BEGIN

	IF EXISTS (SELECT * FROM TAMS_User WHERE UserID = @UserID)
	BEGIN
		SELECT * 
		FROM TAMS_User u, TAMS_User_Role ur, TAMS_Role r
		WHERE u.UserID = ur.UserID
		AND ur.RoleID = r.ID
		AND u.UserID = @UserID

	END
	
END

CREATE   procedure [dbo].[sp_Generate_Ref_Num]
	@FormType	AS NVARCHAR(20) = NULL,
	@Line		AS NVARCHAR(20) = NULL,
	@TrackType	AS NVARCHAR(50) = NULL,
	@RefNum		AS NVARCHAR(20) = NULL OUTPUT,
	@Message	AS NVARCHAR(500) = NULL OUTPUT
AS
BEGIN

DECLARE	@IntrnlTrans	TINYINT,
	@ErrorMsg		NVARCHAR(500)			

SET @ErrorMsg = ''

SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

-- >>



DECLARE @MaxNum	BIGINT

SET @MaxNum = 1

IF @FormType = 'TAR'
	BEGIN
		IF (SELECT COUNT(*) 
			FROM [dbo].[TAMS_RefSerialNumber]
				WHERE FormType = @FormType
					AND Line = @Line
					and TrackType = @TrackType
					AND [Year] = YEAR(GETDATE())) = 0
			BEGIN
				INSERT INTO [dbo].[TAMS_RefSerialNumber]
					([Line],[TrackType], [Year], [OperationDate], [FormType], [MaxNum])
				VALUES 
					(@Line,@TrackType, YEAR(GETDATE()), GETDATE(), @FormType, @MaxNum)
			END
		ELSE
			BEGIN
				SELECT @MaxNum = [MaxNum] 
					FROM [dbo].[TAMS_RefSerialNumber]
						WHERE FormType = @FormType
							AND Line = @Line
							and TrackType = @TrackType
							AND [Year] = YEAR(GETDATE())

				SET @MaxNum = @MaxNum + 1

				UPDATE [dbo].[TAMS_RefSerialNumber]
					SET MaxNum = @MaxNum
				WHERE FormType = @FormType
					AND Line = @Line
					AND TrackType = @TrackType
					AND [Year] = YEAR(GETDATE())
			END

			if @TrackType = 'Depot' 
			begin
				set @Line = @Line + 'D';
			end;
			SET @RefNum = @Line + '-' + @FormType + '-' + CAST(YEAR(GETDATE()) AS NVARCHAR(20)) + '-' + RIGHT('00000' + CAST(@MaxNum AS VARCHAR(5)), 5)
	END

-- >>

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR GENERATING REF NUM'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END

CREATE PROCEDURE [dbo].[sp_Generate_Ref_Num_TOA]
	@FormType	AS NVARCHAR(20) = NULL,
	@Line		AS NVARCHAR(20) = NULL,
	@TARID		AS Int = 0, 
	@OperationDate	AS NVARCHAR(20) = NULL,
	@TrackType	AS NVARCHAR(50) = 'MainLine',
	@RefNum		AS NVARCHAR(20) = NULL OUTPUT,
	@Message	AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT,
	@ErrorMsg		NVARCHAR(500)			

SET @ErrorMsg = ''

SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

-- >>

DECLARE @MaxNum	BIGINT
SET @MaxNum = 1

DECLARE @OPDateStr			NVARCHAR(20)
DECLARE @CurrTimeStr		NVARCHAR(20)
DECLARE @RefNumOPDateStr	NVARCHAR(20)
DECLARE @RefNumCurrStr		NVARCHAR(20)

Declare @AccessDate	Date;
Select @AccessDate = AccessDate 
From TAMS_TAR 
Where ID = @TARID;


SET @OPDateStr = CONVERT(NVARCHAR(20), @AccessDate, 103)
SET @CurrTimeStr = CONVERT(NVARCHAR(20), CONVERT(TIME, GETDATE(), 108), 108)

SET @RefNumOPDateStr = CAST(LEFT(@OPDateStr, 2) AS NVARCHAR(2)) + CAST(SUBSTRING(@OPDateStr, 4, 2) AS NVARCHAR(2))
SET @RefNumCurrStr = CAST(LEFT(@CurrTimeStr, 2) AS NVARCHAR(2)) + CAST(SUBSTRING(@CurrTimeStr, 4, 2) AS NVARCHAR(2))

IF @FormType = 'TOA'
	BEGIN
		IF (SELECT COUNT(*) 
			FROM [dbo].[TAMS_RefSerialNumber]
				WHERE FormType = @FormType
					AND Line = @Line
					and UPPER(TrackType) = UPPER(@TrackType)
					AND [Year] = YEAR(GETDATE())
					AND OperationDate = CONVERT(DATE, CONVERT(DATETIME, @OperationDate, 103))) = 0
			BEGIN
				INSERT INTO [dbo].[TAMS_RefSerialNumber]
					([Line], [TrackType], [Year], [OperationDate], [FormType], [MaxNum])
				VALUES 
					(@Line, @TrackType, YEAR(GETDATE()), CONVERT(DATE, CONVERT(DATETIME, @OperationDate, 103)), @FormType, @MaxNum)
			END
		ELSE
			BEGIN
				--SELECT @MaxNum = [MaxNum] 
				--	FROM [dbo].[TAMS_RefSerialNumber]
				--		WHERE FormType = @FormType
				--			AND Line = @Line
				--			AND [Year] = YEAR(GETDATE())
				--			AND OperationDate = CONVERT(DATE, @OperationDate, 103)

				--SET @MaxNum = @MaxNum + 1

				UPDATE [dbo].[TAMS_RefSerialNumber]
					SET MaxNum = ISNULL(MaxNum, 0) + 1
				WHERE FormType = @FormType
					AND Line = @Line
					AND [Year] = YEAR(GETDATE())
					and UPPER(TrackType) = UPPER(@TrackType)
					AND OperationDate = CONVERT(DATE, CONVERT(DATETIME, @OperationDate, 103))

				SELECT @MaxNum = ISNULL(MaxNum, 0)
					FROM [dbo].[TAMS_RefSerialNumber]
					WHERE FormType = @FormType
					AND Line = @Line
					and UPPER(TrackType) = UPPER(@TrackType)
					AND [Year] = YEAR(GETDATE())
					AND OperationDate = CONVERT(DATE, CONVERT(DATETIME, @OperationDate, 103))

			END


			--XXX-TOA-DDMM-hhmm-# 

			--where XXX – Line (DTL, NEL)
			--		DDMM – access date in day & month
			--		hhmm – TOA generation time.
			--			   hh in 24 hrs format
			--			   mm minutes
			--		# - Running no. Reset to 1 daily.

			--Eg: DTL-TOA-1504-0123-1
			--	DTL-TOA-1504-0126-2
			--	DTL-TOA-1504-0130-3
			--	DTL-TOA-1504-0130-4


		if @TrackType = 'Depot' 
		begin
				set @Line = @Line + 'D';
		end;
		
		SET @RefNum = @Line + '-' + 
			@FormType + '-' + 
			CAST(@RefNumOPDateStr AS NVARCHAR(5)) + '-' + 
			CAST(@RefNumCurrStr AS NVARCHAR(5)) + '-' + 
			CAST(@MaxNum AS VARCHAR(2))
	END

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR GENERATING REF NUM'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END

CREATE PROCEDURE  [dbo].[sp_Get_TypeOfWorkByLine]
	@Line AS nvarchar(10) = NULL,
	@TrackType AS nvarchar(50) = NULL
AS
BEGIN
	SELECT ID, Line, TypeOfWork, ColourCode, ForSelection, [Order]
	FROM TAMS_Type_Of_Work
	WHERE Line = @Line
	and TrackType = @TrackType
	AND IsActive = 1 
	ORDER BY [Order] ASC
END
