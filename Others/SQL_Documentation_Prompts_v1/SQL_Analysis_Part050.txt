CREATE PROCEDURE [dbo].[sp_TAMS_Form_Update_Access_Details]
	@Company		AS NVARCHAR(50) = NULL,
	@Designation	AS NVARCHAR(50) = NULL,
	@Name			AS NVARCHAR(50) = NULL,
	@OfficeNo		AS NVARCHAR(50) = NULL,
	@MobileNo		AS NVARCHAR(50) = NULL,
	@Email			AS NVARCHAR(50) = NULL,
	@AccessTimeFrom AS NVARCHAR(50) = NULL,
	@AccessTimeTo	AS NVARCHAR(50) = NULL,
	@IsExclusive	AS INT= 0,
	@DescOfWork		AS NVARCHAR(100) = NULL,
	@ARRemark		AS NVARCHAR(1000) = NULL,
	@InvolvePower	AS INT = 0,
	@PowerOn		AS INT = 0,
	@Is13ASocket	AS INT = 0,
	@CrossOver		AS INT = 0, 
	@UserID			AS NVARCHAR(100) = NULL,
	@TARID			AS BIGINT = 0,
	@Message		AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

DECLARE @UserIDID	INT = 1;

SELECT @UserIDID = Userid
	FROM TAMS_User
		WHERE LoginId = @UserID
-- >>

UPDATE [dbo].[TAMS_TAR]
SET [Company] = @Company, 
	[Designation] = @Designation, 
	[Name] = @Name, 
	[OfficeNo] = @OfficeNo, 
	[MobileNo] = @MobileNo, 
	[Email] = @Email,
	[AccessTimeFrom] = @AccessTimeFrom, 
	[AccessTimeTo] = @AccessTimeTo,
	[IsExclusive] = @IsExclusive,
	[DescOfWork] = @DescOfWork, 
	[ARRemark] = @ARRemark,
	[InvolvePower] = @InvolvePower, 
	[PowerOn] = @PowerOn, 
	[Is13ASocket] = @Is13ASocket, 
	[CrossOver] = @CrossOver,
	[UpdatedOn] = GETDATE(), 
	[UpdatedBy] = @UserIDID
WHERE ID = @TARID

-- >>

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR INSERTING INTO TAMS_TAR'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END

CREATE PROCEDURE  [dbo].[sp_TAMS_GetBlockedTarDates]
	@Line AS nvarchar(10) = NULL,
	@TrackType as nvarchar(50) = NULL,
	@AccessDate AS date = NULL
AS
BEGIN
	SELECT ID, Line,TrackType, BlockDate, BlockReason
	FROM TAMS_Block_TARDate
	WHERE Line = @Line
	AND TrackType = @TrackType
	AND BlockDate = @AccessDate
	AND IsActive = 1 
	ORDER BY BlockDate ASC
END

CREATE PROCEDURE  [dbo].[sp_TAMS_GetDutyOCCRosterByParameters]
	@Line AS nvarchar(10) = NULL,
	@TrackType AS nvarchar(50) = NULL,
	@OperationDate as date,
	@Shift as nvarchar(1) = NULL,
	@RosterCode as nvarchar(50) = NULL,
	@ID as int = 0

AS
BEGIN

  select ID, Line, OperationDate, [Shift], RosterCode, DutyStaffId, [Name] as DutyStaffName
  from TAMS_OCC_Duty_Roster r , TAMS_User u
  where r.Line = @Line 
  and r.TrackType = @TrackType
  and r.DutyStaffId = u.Userid
  and r.OperationDate = @OperationDate
  and r.[shift] = @Shift
  and r.RosterCode = @RosterCode
  and r.id = @ID
  AND r.IsActive = 1  

END

CREATE PROCEDURE  [dbo].[sp_TAMS_GetParametersByLineandTracktype]
	@ParaCode AS nvarchar(50) = NULL,
	@Line AS nvarchar(350) = NULL,
	@TrackType AS nvarchar(350) = NULL
AS
BEGIN
	SELECT ID, ParaType, ParaCode, ParaDesc,ParaValue1, ParaValue2, ParaValue3, ParaDatetime, ParaDate, ParaTime, [Order]
	FROM TAMS_Parameters
	WHERE ParaCode = @ParaCode AND ParaValue1 = @Line and ParaValue3 = @TrackType
	AND EffectiveDate <= GETDATE() and ExpiryDate >= GETDATE()
	ORDER BY [Order] ASC

CREATE PROCEDURE  [dbo].[sp_TAMS_GetRosterRoleByLine]
	@Line AS nvarchar(10) = NULL,
	@TrackType AS nvarchar(50) = NULL,
	@OperationDate AS nvarchar(10) = NULL,
	@Shift AS nvarchar(1) = NULL
AS
BEGIN
	declare @count as int = 0;

	IF @Line = 'DTL'
	BEGIN
		SELECT @Count = COUNT(*) FROM TAMS_OCC_Duty_Roster WHERE Line = @Line AND TrackType = @TrackType and OperationDate = @OperationDate

		IF @Count = 0
		BEGIN
			SELECT ID, Line, RosterCode, RoleId, [Order]
			FROM TAMS_Roster_Role
			WHERE Line = @Line
			 and TrackType = @TrackType
			AND EffectiveDate <= GETDATE() and ExpiryDate >= GETDATE()
			ORDER BY [Order] ASC
		END;
		ELSE
		BEGIN
			SELECT ID, Line, RosterCode, [Shift] AS RoleId, ID AS [Order]
			FROM TAMS_OCC_Duty_Roster
			WHERE Line = @Line
			and TrackType = @TrackType
			AND OperationDate = @OperationDate
			AND [Shift] = @Shift
			ORDER BY [Order] ASC
		END;
	END;
	ELSE
	BEGIN
		SELECT @Count = COUNT(*) FROM TAMS_OCC_Duty_Roster WHERE Line = @Line AND TrackType = @TrackType and  OperationDate = @OperationDate

		IF @Count = 0
		BEGIN
			SELECT ID, Line, RosterCode, RoleId, [Order]
			FROM TAMS_Roster_Role
			WHERE Line = @Line
			and TrackType = @TrackType
			AND RosterCode <> 'SCO'
			AND EffectiveDate <= GETDATE() and ExpiryDate >= GETDATE()
			ORDER BY [Order] ASC
		END;
		ELSE
		BEGIN
			SELECT ID, Line, RosterCode, [Shift] AS RoleId, ID AS [Order]
			FROM TAMS_OCC_Duty_Roster
			WHERE Line = @Line
			and TrackType = @TrackType
			AND OperationDate = @OperationDate
			AND [Shift] = @Shift
			ORDER BY [Order] ASC
		END;
	END;
END

CREATE PROCEDURE [dbo].[sp_TAMS_GetSectorsByLineAndDirection]
	@Line AS nvarchar(10) = NULL,
	@Direction AS nvarchar(10) = NULL
AS
IF @Line = 'DTL'
BEGIN
	SELECT ID, Line, Direction, Sector, IsGap, IsEndSector, SameSector, IsspecialSector, [Order]
	FROM TAMS_Sector
	WHERE Line = @Line
	AND Direction = @Direction
	AND IsActive = 1 
	AND EffectiveDate <= GETDATE() and ExpiryDate >= GETDATE()

	ORDER BY [Order] ASC
END
ELSE IF @Line = 'NEL'
BEGIN
	SELECT ID, Line, Direction, Sector, IsGap, IsEndSector, SameSector, IsspecialSector, [Order]
	FROM TAMS_Sector
	WHERE Line = @Line
	AND Direction = @Direction
	AND IsActive = 1 
	AND EffectiveDate <= GETDATE() and ExpiryDate >= GETDATE()

	ORDER BY [Order] ASC
END

CREATE PROCEDURE [dbo].[sp_TAMS_GetTarAccessRequirementsByTarId]
	@TarId AS integer = 0
AS
BEGIN
  select tta.id, tarid, tar.operationrequirement, tta.ispower, tta.isselected, tta.[order]
  from tams_tar_accessreq tta, TAMS_Access_Requirement tar
  where tta.OperationRequirement = tar.id
  and tta.tarid = @TarId
  and IsSelected = 1
END

CREATE PROCEDURE [dbo].[sp_TAMS_GetTarApprovalsByTarId]
	@TarId as integer = 0

AS
BEGIN
	SELECT a.ID, Title, [Name], Remark, WFStatus FROM TAMS_TAR_Workflow a, TAMS_Endorser b, TAMS_User c
	WHERE a.TARId = @TarId
	AND a.WorkflowId = b.WorkflowId
	AND a.ActionBy = c.Userid
	AND a.EndorserId = b.ID
	ORDER BY a.ID ASC
END
