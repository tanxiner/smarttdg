CREATE PROCEDURE [dbo].[sp_TAMS_GetTarSectorsByTarId]
	@TarId as integer = 0

AS
BEGIN
	SELECT a.ID, a.Line, Direction, Sector, TARNo, AccessDate, AccessType, SectorId, IsBuffer, ColourCode, IsGap
	FROM TAMS_Sector a
	INNER JOIN TAMS_TAR_Sector c
	on a.id = c.SectorId 
	INNER JOIN TAMS_TAR b
	on b.Id = c.TARId
	AND c.TARId = @TarId
	AND c.IsBuffer <> 1
	ORDER BY [Order], Sector ASC
END

CREATE PROCEDURE [dbo].[sp_TAMS_GetTarStationsByTarId]
	@TarId as integer = 0

AS
BEGIN
	SELECT a.ID, a.Line, b.TarId, a.StationCode, a.StationName, a.StationLongName
	FROM TAMS_Station a
	INNER JOIN TAMS_TAR_Station b
	on a.id = b.StationId
	AND b.TARId = @TarId
	ORDER BY [Order] ASC
END

CREATE PROCEDURE [dbo].[sp_TAMS_GetWFStatusByLine]
	@Line AS nvarchar(10) = NULL
AS
BEGIN
	SELECT ID, Line, WFType, WFDescription, WFStatus, WFStatusId, [Order]
	FROM TAMS_WFStatus 
	WHERE Line = @Line
	AND IsActive = 1 
	ORDER BY [Order] ASC
END

CREATE PROCEDURE  [dbo].[sp_TAMS_GetWFStatusByLineAndType]
	@Line AS nvarchar(10) = NULL,
	@TrackType as nvarchar(50) = NULL,
	@Type AS nvarchar(50) = NULL
AS
BEGIN
	SELECT ID, Line, WFType, WFDescription, WFStatus, WFStatusId, [Order]
	FROM TAMS_WFStatus 
	WHERE Line = @Line
	and TrackType = @TrackType
	AND WFType = @Type
	AND IsActive = 1 
	ORDER BY [Order] ASC
END

CREATE PROCEDURE [dbo].[sp_TAMS_Get_All_Roles] 
	@IsExternal	BIT
AS
BEGIN

	SELECT * FROM TAMS_Role WHERE Line = 'DTL' AND Module = 'TAR' and TrackType = 'Mainline'
	IF @IsExternal = 0
	BEGIN
		SELECT * FROM TAMS_Role WHERE Line = 'DTL' AND Module = 'OCC' and TrackType = 'Mainline'
	END
	SELECT * FROM TAMS_Role WHERE Line = 'NEL' AND Module = 'TAR' and TrackType = 'Mainline'
	IF @IsExternal = 0
	BEGIN
	SELECT * FROM TAMS_Role WHERE Line = 'NEL' AND Module = 'OCC'  and TrackType = 'Mainline'
	END
	SELECT * FROM TAMS_Role WHERE Line = 'SPLRT' AND Module = 'TAR'  and TrackType = 'Mainline'
	IF @IsExternal = 0
	BEGIN
	SELECT * FROM TAMS_Role WHERE Line = 'SPLRT' AND Module = 'OCC'  and TrackType = 'Mainline'
	END
	SELECT * FROM TAMS_Role WHERE Line = 'DTL' AND Module = 'TAR' and TrackType = 'Depot'
	SELECT * FROM TAMS_Role WHERE Line = 'NEL' AND Module = 'TAR' and TrackType = 'Depot'
	SELECT * FROM TAMS_Role WHERE Line = 'SPLRT' AND Module = 'TAR' and TrackType = 'Depot'
	SELECT * FROM TAMS_Role WHERE Line = 'NEL' AND Module = 'DCC' and TrackType = 'Depot'
	 
	
END

			 SET @rsql = 'SELECT MenuID, MenuLevel, DispText,ToolTip, ProgURL, ParentMenuID '+

			'AND ParentMenuID = ''' + @MenuID +''' ' +

			'SELECT DISTINCT(MenuID) ' +

		end;
		else 
		begin 
		
			SET @rsql = 'SELECT MenuID, MenuLevel, DispText,ToolTip, ProgURL, ParentMenuID '+
			'FROM TAMS_Menu ' +
			'WHERE Active = 1  ' +
			--' AND IsInternet = ' + @IsInternet +
			'AND MenuLevel = 2  ' +
			'AND ParentMenuID = ''' + @MenuID +''' ' +
			'AND MenuID IN(	 ' +	
			'SELECT DISTINCT(MenuID) ' +
