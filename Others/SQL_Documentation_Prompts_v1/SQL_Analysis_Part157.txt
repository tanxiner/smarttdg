CREATE PROCEDURE [dbo].[sp_TAMS_TOA_Add_Parties]
	@PartiesFIN		AS NVARCHAR(50) = NULL,
	@PartiesName	AS NVARCHAR(200) = NULL,
	@IsTMC			AS NVARCHAR(5) = NULL,
	@NoOfParties	AS BIGINT = 0,
	@TOAID			AS BIGINT = 0,
	@Message		AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

DECLARE @PartCtr	BIGINT
DECLARE @DerIsTMC	BIGINT
SET @PartCtr = 0
SET @Message = ''

IF @IsTMC = 'Y'
	BEGIN
		SET @DerIsTMC = 1
	END
ELSE
	BEGIN
		SET @DerIsTMC = 0
	END
SELECT @PartCtr = COUNT(*) 
	FROM TAMS_TOA_Parties
		WHERE TOAId = @TOAID
			AND dbo.DecryptString(NRIC) = @PartiesFIN

	IF @PartCtr > 0
		BEGIN
			SET @Message = '1'
		END
	ELSE
		BEGIN
			UPDATE TAMS_TOA
				SET NoOfParties = @NoOfParties
			WHERE Id = @TOAID

			INSERT INTO [dbo].[TAMS_TOA_Parties]
				([TOAId], [Name], [NRIC], [IsInCharge], [IsWitness], [IsTMC], 
					[BookInTime], [BookOutTime], [BookInStatus])
			VALUES
				(@TOAID, @PartiesName, dbo.EncryptString(@PartiesFIN), 0, 0, @DerIsTMC,
					GETDATE(), NULL, 'In')

			SET @Message = ''
		END

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR INSERTING TAMS_TOA_Parties'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END

CREATE PROCEDURE [dbo].[sp_TAMS_TOA_Delete_Parties]
	@PartiesID	AS BIGINT = 0,
	@TOAID		AS BIGINT = 0,
	@Message	AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	


IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END


BEGIN TRY

SET @Message = ''
DECLARE @PartiesCtr BIGINT


SELECT @PartiesCtr = COUNT(*) 
	FROM TAMS_TOA_Parties
		WHERE TOAId = @TOAID

if @PartiesCtr<=2
	BEGIN
		RAISERROR('Please ensure there must be minimum 2 parties before deleting.',11,1)
	END
ELSE
	BEGIN

		DELETE FROM TAMS_TOA_Parties
		WHERE TOAId = @TOAID
			AND Id = @PartiesID

	-- DECLARE @PartiesCtr BIGINT

		UPDATE TAMS_TOA
			SET NoOfParties	= (
			SELECT COUNT(*) 
			FROM TAMS_TOA_Parties
				WHERE TOAId = @TOAID
			)
		WHERE Id = @TOAID
	END


IF @@ERROR <> 0 BEGIN
	PRINT 'ERROR INSERTING'
	SET @Message = 'ERROR INSERTING TAMS_TOA_Parties'
	GOTO TRAP_ERROR
END


END TRY
BEGIN CATCH
	DECLARE @ErrorMessage NVARCHAR(4000);
	DECLARE @ErrorSeverity INT;
	DECLARE @ErrorState INT;

	SELECT @ErrorMessage=ERROR_MESSAGE(),
	@ErrorSeverity=ERROR_SEVERITY(),
	@ErrorState=ERROR_STATE()

	RAISERROR(@ErrorMessage, @ErrorSeverity,@ErrorState)
END CATCH

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN

CREATE PROCEDURE [dbo].[sp_TAMS_TOA_Get_Parties]
	@TOAID	BIGINT
AS

BEGIN
	
	SELECT NoOfParties 
		FROM TAMS_TOA
			WHERE Id = @TOAID

	SELECT ROW_NUMBER() OVER(ORDER BY ID) as Sno,
		[Name] AS PartiesName,
		dbo.QTS_MaskText(dbo.DecryptString(NRIC), 4) AS PartiesNRIC,  
		CASE WHEN IsTMC = 1 THEN 'Yes' ELSE 'No' END AS PartiesIsTMC,
		CASE WHEN IsInCharge = 1 THEN 'Y' ELSE 'N' END AS InCharge,
		Id AS RecID,
		BookInStatus
		FROM TAMS_TOA_Parties
			WHERE TOAId = @TOAID
		ORDER BY ID ASC

	-- >> Witness List
	SELECT [Name] AS PartiesName,
			ID AS PartiesID
		FROM TAMS_TOA_Parties
			WHERE TOAId = @TOAID
				AND IsInCharge = 0
		ORDER BY ID ASC

	-- >> Selected Witness
	SELECT ID AS PartiesID
		FROM TAMS_TOA_Parties
			WHERE TOAId = @TOAID
				AND IsInCharge = 0
				AND IsWitness = 1
		
	SELECT COUNT(*) AS WorkPartiesBook
		FROM TAMS_TOA_Parties
			WHERE TOAId = @TOAID
				AND IsInCharge = 0
				AND BookInStatus = 'In'

	SELECT COUNT(*) AS WPCtr
		FROM TAMS_TOA_Parties
			WHERE TOAId = @TOAID
				AND BookInStatus = 'In'
END

CREATE PROCEDURE [dbo].[sp_TAMS_TOA_OnLoad]
	@TOAID	BIGINT
AS

BEGIN
	
	SELECT a.QRLocation, 
			b.TARNo, 
			dbo.QTS_MaskText(dbo.DecryptString(a.InChargeNRIC), 4) AS InchargeNRIC,
			ISNULL(a.InChargeName, '') AS InChargeName,
			ISNULL(a.MobileNo, '') AS MobileNo,
			ISNULL(a.TetraRadioNo, '') AS TetraRadioNo,
			ISNULL(b.TVFMode, 'N.A') AS TVFMode,
			ISNULL(a.NoOfParties, 1) AS NoOfParties,
			ISNULL(a.TOANo, '') AS TOANo,
			ISNULL(a.ProtectionType,'') As ProtectionType
		FROM TAMS_TOA a, TAMS_TAR b
			WHERE a.Id = @TOAID
				AND a.TARId = b.Id
		
END
