CREATE PROCEDURE [dbo].[sp_TAMS_SummaryReport_OnLoad_Trace]
	@Line		NVARCHAR(20) = NULL,
	@StrAccDate	NVARCHAR(20) = NULL
AS

BEGIN

DECLARE @CurDate		DATE
DECLARE @CurTime		TIME
DECLARE @OperationDate	DATE
DECLARE @AccessDate		DATE

SET @CurDate = GETDATE()
SET @CurTime = CONVERT(TIME, GETDATE(), 108)

DECLARE @CutOff			TIME

SELECT @CutOff = ParaTime 
	FROM TAMS_Parameters
		WHERE ParaCode = 'OpDateCutOffTime'
			AND ParaValue1 = @Line

IF @CurTime < @CutOff
	BEGIN
		SET @AccessDate		= DATEADD(DAY, -1, @CurDate)
	END
ELSE
	BEGIN
		SET @AccessDate		= @CurDate
	END

--SELECT @AccessDate, CONVERT(DATE, @StrAccDate, 103)

IF @AccessDate < CONVERT(DATE, @StrAccDate, 103)
	BEGIN
		SELECT 'Err' AS ErrStr -- >> ERROR -- >> Report is not ready for selected date << --
	END
ELSE
	BEGIN
		
		DECLARE @TARPossCtr	INT
		DECLARE @TARProtCtr	INT 
		DECLARE @TOAPossCtr	INT
		DECLARE @TOAProtCtr	INT 
		DECLARE @CancelPossCtr INT
		DECLARE @CancelProtCtr INT
		DECLARE @CancelPoss NVARCHAR(4000)
		DECLARE @CancelProt NVARCHAR(4000)
		DECLARE @ExtPossCtr INT
		DECLARE @ExtProtCtr INT

		SET @TARPossCtr = 0
		SET @TARProtCtr = 0
		SET @TOAPossCtr = 0
		SET @TOAProtCtr = 0
		SET @CancelPossCtr = 0
		SET @CancelProtCtr = 0
		SET @CancelPoss = ''
		SET @CancelProt = ''
		SET @ExtPossCtr = 0
		SET @ExtProtCtr = 0

		SELECT @TARPossCtr = COUNT(*)
			FROM TAMS_TAR a
				WHERE a.AccessDate = CONVERT(DATE, @StrAccDate, 103)
					AND a.AccessType = 'Possession'
					AND a.PowerOn = 0
					AND a.TARStatusId = CASE WHEN @Line = 'NEL' THEN 9 ELSE 8 END
					AND a.Line = @Line

		SELECT @TARProtCtr = COUNT(*)
			FROM TAMS_TAR a
				WHERE a.AccessDate = CONVERT(DATE, @StrAccDate, 103)
					AND a.AccessType = 'Protection'
					AND a.PowerOn = 0
					AND a.TARStatusId = CASE WHEN @Line = 'NEL' THEN 9 ELSE 8 END
					AND a.Line = @Line

		SELECT @TOAPossCtr = COUNT(*)
			FROM TAMS_TAR a, TAMS_TOA b
				WHERE a.AccessDate = CONVERT(DATE, @StrAccDate, 103)
					AND a.AccessType = 'Possession'
					AND a.PowerOn = 0
					AND a.TARStatusId = CASE WHEN @Line = 'NEL' THEN 9 ELSE 8 END
					AND a.Line = @Line
					AND a.Id = b.TARId

		SELECT @TOAProtCtr = COUNT(*)
			FROM TAMS_TAR a, TAMS_TOA b
				WHERE a.AccessDate = CONVERT(DATE, @StrAccDate, 103)
					AND a.AccessType = 'Protection'
					AND a.PowerOn = 0
					AND a.TARStatusId = CASE WHEN @Line = 'NEL' THEN 9 ELSE 8 END
					AND a.Line = @Line
					AND a.Id = b.TARId

	SET @CancelProtCtr = 0
	SET @CancelProt = ''

	-- >> Cursor TARNo ; Sep
	DECLARE @Cur01 CURSOR;

	DECLARE @C1TarNo	NVARCHAR(50),
		@C1ID			BIGINT

	SET @Cur01 = CURSOR FAST_FORWARD 
	FOR

		-- >> Cursor TARNo ; Sep
		SELECT DISTINCT a.Id, a.TARNo
			FROM TAMS_TAR a
				WHERE a.AccessDate = CONVERT(DATE, @StrAccDate, 103)
					AND a.AccessType = 'Protection'
					AND a.PowerOn = 0
					AND a.TARStatusId = CASE WHEN @Line = 'NEL' THEN 9 ELSE 8 END
					AND a.Line = @Line
					AND a.Id NOT IN (SELECT DISTINCT TARId FROM TAMS_TOA WHERE AccessDate = CONVERT(DATE, @StrAccDate, 103) AND TOAStatus <> 0)
				ORDER BY a.Id

	OPEN @Cur01
	
	FETCH NEXT FROM @Cur01 INTO @C1ID, @C1TarNo
	
	WHILE @@FETCH_STATUS = 0 
	
		BEGIN

			SET @CancelProtCtr = @CancelProtCtr + 1
			
			IF @CancelProt = ''
				BEGIN
					SET @CancelProt = @C1TarNo
				END 
			ELSE
				BEGIN
					SET @CancelProt = @CancelProt + '; ' + @C1TarNo
				END

			FETCH NEXT FROM @Cur01 INTO @C1ID, @C1TarNo
		END 	

	CLOSE @Cur01
	
	DEALLOCATE @Cur01


	SET @CancelPossCtr = 0
	SET @CancelPoss = ''
		
	-- >> Cursor TARNo ; Sep
	DECLARE @Cur02 CURSOR;

	DECLARE @C2TarNo	NVARCHAR(50),
		@C2ID			BIGINT

	SET @Cur02 = CURSOR FAST_FORWARD 
	FOR

		-- >> Cursor TARNo ; Sep
		SELECT DISTINCT a.Id, a.TARNo
			FROM TAMS_TAR a
				WHERE a.AccessDate = CONVERT(DATE, @StrAccDate, 103)
					AND a.AccessType = 'Possession'
					AND a.PowerOn = 0
					AND a.TARStatusId = CASE WHEN @Line = 'NEL' THEN 9 ELSE 8 END
					AND a.Line = @Line
					AND a.Id NOT IN (SELECT DISTINCT TARId FROM TAMS_TOA WHERE AccessDate = CONVERT(DATE, @StrAccDate, 103) AND TOAStatus <> 0)
				ORDER BY a.Id

	OPEN @Cur02
	
	FETCH NEXT FROM @Cur02 INTO @C2ID, @C2TarNo
	
	WHILE @@FETCH_STATUS = 0 
	
		BEGIN

			SET @CancelPossCtr = @CancelPossCtr + 1
			
			IF @CancelPoss = ''
				BEGIN
					SET @CancelPoss = @C2TarNo
				END 
			ELSE
				BEGIN
					SET @CancelPoss = @CancelPoss + '; ' + @C2TarNo
				END

			FETCH NEXT FROM @Cur02 INTO @C2ID, @C2TarNo
		END 	

	CLOSE @Cur02
	
	DEALLOCATE @Cur02


		SELECT @ExtPossCtr = COUNT(*)
			FROM TAMS_TAR a, TAMS_TOA b
				WHERE a.AccessDate = CONVERT(DATE, @StrAccDate, 103)
					AND a.AccessType = 'Possession'
					AND a.PowerOn = 0
					AND a.TARStatusId = CASE WHEN @Line = 'NEL' THEN 9 ELSE 8 END
					AND a.Line = @Line
					AND a.Id = b.TARId
					AND (b.TOAStatus = 3 OR ((b.TOAStatus = 4 OR b.TOAStatus = 4) AND CONVERT(TIME, b.SurrenderTime, 108) > CONVERT(TIME, '04:00:00', 108)))

		SELECT @ExtProtCtr = COUNT(*)
			FROM TAMS_TAR a, TAMS_TOA b
				WHERE a.AccessDate = CONVERT(DATE, @StrAccDate, 103)
					AND a.AccessType = 'Protection'
					AND a.PowerOn = 0
					AND a.TARStatusId = CASE WHEN @Line = 'NEL' THEN 9 ELSE 8 END
					AND a.Line = @Line
					AND a.Id = b.TARId
					AND (b.TOAStatus = 3 OR ((b.TOAStatus = 4 OR b.TOAStatus = 4) AND CONVERT(TIME, b.SurrenderTime, 108) > CONVERT(TIME, '04:00:00', 108)))

		SELECT @TARPossCtr AS TARPossCtr, @TARProtCtr AS TARProtCtr, 
			@TOAPossCtr AS TOAPossCtr, @TOAProtCtr AS TOAProtCtr, 
			@CancelPossCtr AS CancelPossCtr, @CancelProtCtr AS CancelProtCtr, @CancelPoss AS CancelPoss, @CancelProt AS CancelProt, 
			@ExtPossCtr AS ExtPossCtr, @ExtProtCtr AS ExtProtCtr

	END


END
