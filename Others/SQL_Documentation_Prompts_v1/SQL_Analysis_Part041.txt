CREATE PROCEDURE [dbo].[sp_TAMS_Email_CompanyRegistrationLinkByRegID] 
	@RegID NVARCHAR(200),
	@Cipher NVARCHAR(200)
AS
BEGIN
	IF EXISTS(SELECT * FROM TAMS_Registration WHERE ID = @RegID)
	BEGIN
		
		DECLARE @Sender NVARCHAR (100);
		DECLARE @SysID NVARCHAR(10);
		DECLARE @Subject NVARCHAR(500);    
		DECLARE @Sys NVARCHAR(100);
		DECLARE @Greetings  NVARCHAR(20);     
		DECLARE @ToList NVARCHAR(MAX);
		DECLARE @CCList NVARCHAR(MAX);
		DECLARE @BCCList NVARCHAR(MAX);
		DECLARE @Separator NVARCHAR(10);        
		DECLARE @AlertMsg NVARCHAR(MAX);
		DECLARE @AlertID AS INTEGER; 

		DECLARE @Body VARCHAR(max);
		DECLARE @Body1 VARCHAR(max);
		DECLARE @Body2 VARCHAR(max);
		DECLARE @Body3 VARCHAR(max);
		DECLARE @Link NVARCHAR(MAX);
		DECLARE @CutOff NVARCHAR(10);

		DECLARE @LoginPage	NVARCHAR(500)

		--SET @Link = 'http://localhost:54439/UserForms/AddCompany?ID=' + @Cipher
		SET @Link = 'https://sbstweb.sbstransit.com.sg/TAMS/UserForms/AddCompany?ID=' + @Cipher

		SELECT @CutOff = ParaValue1 FROM TAMS_Parameters WHERE ParaCode = 'CompanyRegCutOff'

		SELECT @ToList = Email FROM TAMS_Registration WHERE ID = @RegID;
			

		SET @Sender = 'TAMS Admin';
		SET @SysID = 'TAMS';
		SET @Subject = 'Link to register company '
		SET @Sys = 'Track Access Management System'   
		SET @Greetings = 'Dear Sir/Madam, '  
		--SET @ToList = ''
		SET @CCList = ''   
		SET @BCCList = ''
		SET @Separator = ','    
		SET @CCList = ''
		--SET @ToList = 'johnsontan@comfortdelgro.com';

		SET @Body1 = ''
		SET @Body2 = ''
		SET @Body3 = ''
			
		SET @Body1 = @Body1 + '<table width="100%">'
		SET @Body1 = @Body1 + '<tr align="left" valign="middle" style="font-size:10pt; font-face:Arial;">'
		SET @Body1 = @Body1 + '<td>Please click the Link to register the company details <a href=' + @Link + '>Open</a><br> Please note that this link is only valid for ' + @CutOff + ' Days.</td>'
		SET @Body1 = @Body1 + '</tr>'
		SET @Body1 = @Body1 + '</table>'
		SET @LoginPage = 'https://sbstweb.sbstransit.com.sg/TAMS/Login.aspx'

		SET @Body2 = @Body2 + '<br>'
		SET @Body2 = @Body2 + '<table width="100%">'
		SET @Body2 = @Body2 + '<tr align="left" valign="middle" style="font-size:10pt; font-face:Arial;">'
		SET @Body2 = @Body2 + '<td>Regards, <br> System Administrator <br></td>'
		SET @Body2 = @Body2 + '</tr>'
		SET @Body2 = @Body2 + '</table>'

		SET @Body3 = @Body3 + '<br>'
		SET @Body3 = @Body3 + '<table width="100%">'
		SET @Body3 = @Body3 + '<tr align="left" valign="middle" style="font-size:10pt; font-face:Arial;">'
		SET @Body3 = @Body3 + '<td>***********************************************<br>Do Not reply. This is a computer generated Email. <br>***********************************************</td>'
		SET @Body3 = @Body3 + '</tr>'
		SET @Body3 = @Body3 + '</table>'

		SET @Body = ''
		SET @Body = @Body1 + @Body2 + @Body3

		SET @BCCList = ''

		EXEC [dbo].EAlertQ_EnQueue 
			@Sender = @Sender, 
			@UserId = @SysID, 
			@Subject = @Subject,        
			@Sys = @Sys,  
			@Greetings = @Greetings, 
			@AlertMsg = @Body, 
			@SendTo = @ToList,        
			@CC = @CCList, 
			@BCC= @BCCList, 
			@Separator = @Separator, 
			@AlertID =  @AlertID output

	END;

END

CREATE PROCEDURE [dbo].[sp_TAMS_Email_Late_TAR]

	@Actor		AS NVARCHAR(100) = NULL,
	@ToSend		AS NVARCHAR(1000) = NULL,
	@Message	AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

-- >>
SET @Message = ''

DECLARE @Sender NVARCHAR (100);
DECLARE @SysID NVARCHAR(10);
DECLARE @Subject NVARCHAR(500);    
DECLARE @Sys NVARCHAR(100);
DECLARE @Greetings  NVARCHAR(20);

DECLARE @CCList NVARCHAR(MAX);

SET @Separator = ','    
SET @CCList = ''

SET @Subject = 'Late TAR ' + CAST(@TARNo AS NVARCHAR(50)) + ' has been ' + @TARStatus + ' by '

IF @Actor = 'Applicant HOD Endorsement' 
	BEGIN
		SET @Subject = @Subject + 'Applicant HOD' 
	END 
ELSE IF @Actor = 'TAP HOD Endorsement' 
	BEGIN
		SET @Subject = @Subject + 'TAP HOD' 
	END 
ELSE IF @Actor = 'Power Endorsement' 
	BEGIN
		SET @Subject = @Subject + 'Power' 
	END 
ELSE IF @Actor = 'TAP Authority Verification'

		SET @Subject = @Subject + 'TAP Verifier' 
	END 
ELSE IF @Actor = 'TAP Authority Approval'

		SET @Subject = @Subject + 'TAP Approver' 
	END 
ELSE IF @Actor = 'OCC Approval' 
	BEGIN
		SET @Subject = @Subject + 'OCC' 
	END 
ELSE
	BEGIN
		SET @Subject = @Subject
	END

	SET @Body1 = ''
	SET @Body2 = ''
	SET @Body3 = ''

	SET @Body1 = @Body1 + '<table width="100%">'
	SET @Body1 = @Body1 + '<tr align="left" valign="middle" style="font-size:10pt; font-face:Arial;">'
	IF @TARStatus = 'Approved'
		BEGIN
			SET @Body1 = @Body1 + '<td>Please note that Late TAR ' +  CAST(@TARNo AS NVARCHAR(50)) + ' has been APPROVED.</td>'
		END
	ELSE IF @TARStatus = 'Rejected'
		BEGIN
			SET @Body1 = @Body1 + '<td>Please note that Late TAR ' +  CAST(@TARNo AS NVARCHAR(50)) + ' has been REJECTED.</td>'
		END
	ELSE IF @TARStatus = 'Cancelled'
		BEGIN	
			SET @Body1 = @Body1 + '<td>Please note that Late TAR ' +  CAST(@TARNo AS NVARCHAR(50)) + ' has been Cancelled.</td>'
		END
	SET @Body1 = @Body1 + '</tr>'
	SET @Body1 = @Body1 + '<tr align="left" valign="middle" style="font-size:10pt; font-face:Arial;"><td>&nbsp;</td></tr>'
	SET @Body1 = @Body1 + '<tr align="left" valign="middle" style="font-size:10pt; font-face:Arial;">'
	SET @Body1 = @Body1 + '<td>Remark ' +  CAST(@Remarks AS NVARCHAR(1000)) + '</td>'
	SET @Body1 = @Body1 + '</tr>'
	SET @Body1 = @Body1 + '</table>'
	SET @LoginPage = 'https://sbstweb.sbstransit.com.sg/TAMS/Login.aspx'

	SET @Body2 = @Body2 + '<br>'
	SET @Body2 = @Body2 + '<table width="100%">'
	SET @Body2 = @Body2 + '<tr align="left" valign="middle" style="font-size:10pt; font-face:Arial;">'
	SET @Body2 = @Body2 + '<td>Click on the link <a href=' + @LoginPage + '>here</a> to access the TAR Form.</td>'
	SET @Body2 = @Body2 + '</tr>'
	SET @Body2 = @Body2 + '</table>'

	SET @Body2 = @Body2 + '<br>'
	SET @Body2 = @Body2 + '<table width="100%">'
	SET @Body2 = @Body2 + '<tr align="left" valign="middle" style="font-size:10pt; font-face:Arial;">'
	SET @Body2 = @Body2 + '<td>Regards, <br> System Administrator <br></td>'
	SET @Body2 = @Body2 + '</tr>'
	SET @Body2 = @Body2 + '</table>'

	SET @Body3 = @Body3 + '<br>'
	SET @Body3 = @Body3 + '<table width="100%">'
	SET @Body3 = @Body3 + '<tr align="left" valign="middle" style="font-size:10pt; font-face:Arial;">'
	SET @Body3 = @Body3 + '<td>********<br>Do Not reply. This is a computer generated Email. <br>********</td>'
	SET @Body3 = @Body3 + '</tr>'
	SET @Body3 = @Body3 + '</table>'

	SET @Body = ''
	SET @Body = @Body1 + @Body2 + @Body3

	EXEC [dbo].EAlertQ_EnQueue 
		@Sender = @Sender, 
		@UserId = @SysID, 
		@Subject = @Subject,        
		@Sys = @Sys,  
		@Greetings = @Greetings, 
		@AlertMsg = @Body, 
		@SendTo = @ToList,        
		@CC = @CCList, 
		@BCC= @BCCList, 
		@Separator = @Separator, 
		@AlertID =  @AlertID OUTPUT

-- >>

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR INSERTING INTO TAMS_TAR'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END
