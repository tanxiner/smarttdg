CREATE PROCEDURE [dbo].[sp_TAMS_Update_UserRegRole_SysOwnerApproval] 	
	@RegModID			INT,
	@RegRoleID			INT,
	@IsAssigned			BIT,
	@RejectRemarks		NVARCHAR(MAX),
	@UpdatedBy			INT
AS
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION;

		DECLARE @UserID INT;
		DECLARE @Line NVARCHAR(20);
		DECLARE @TrackType nvarchar(50);
		DECLARE @Module NVARCHAR(20);
		DECLARE @RegStatus INT;
		DECLARE @RegId INT;


		SELECT @UserID = UserID FROM TAMS_User WHERE LoginID = (SELECT LoginID FROM TAMS_Registration WHERE
		ID = (SELECT RegID FROM TAMS_Reg_Module WHERE ID = @RegModID));

		SELECT @Line = Line, @TrackType = TrackType FROM TAMS_Reg_Module WHERE ID = @RegModID;

		SELECT @RegId=RegID, @Line=Line, @Module=Module, @RegStatus=RegStatus FROM TAMS_Reg_Module WHERE ID = @RegModID

		SET @RegStatus = @RegStatus - 1;

		SELECT @RegModID=ID FROM TAMS_Reg_Module WHERE RegId = @RegId AND Line=@Line AND Module=@Module AND RegStatus=@RegStatus;

		
		IF EXISTS(SELECT * FROM TAMS_Reg_Role WHERE RegModID = @RegModID AND RegRoleID = @RegRoleID)
		BEGIN
			UPDATE TAMS_Reg_Role 
			SET IsAssigned = @IsAssigned,
			UpdatedOn = GETDATE(),
			RejectReason = @RejectRemarks, 
			UpdatedBy = @UpdatedBy
			WHERE RegModID = @RegModID
			AND RegRoleID = @RegRoleID
				
			IF @IsAssigned = 1
			BEGIN
				IF NOT EXISTS(SELECT * FROM TAMS_User_Role WHERE UserID = @UserID AND Line = @Line AND RoleID = @RegRoleID)
				BEGIN
						INSERT INTO TAMS_User_Role VALUES (@UserID, @RegRoleID, @Line,@TrackType, GETDATE(), 1, GETDATE(), @UpdatedBy);
				END;
			END;
		END;
		
		
		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		ROLLBACK TRANSACTION;
	END CATCH

END;

CREATE PROCEDURE [dbo].[sp_TAMS_User_CheckLastEmailRequest]
	@LoginID	NVARCHAR(200) = NULL,
	@Mode		NVARCHAR(200)

AS
BEGIN

	DECLARE @maxDate AS DATETIME;
	DECLARE @EmailID AS NVARCHAR(250);
	DECLARE @RateLimit AS INT;

	IF NOT @LoginID = ''
	BEGIN
		IF EXISTS (SELECT * FROM TAMS_Registration WHERE LoginId = @LoginID) 
		BEGIN

			IF @Mode = 'User Detail View' 
			BEGIN
				SELECT @maxDate = MAX(e.CreatedOn) 
				FROM EAlertQTo eto, eAlertQ e
				WHERE eto.alertID = e.alertID
				AND eto.recipient = (SELECT TOP 1 Email FROM TAMS_Registration where LoginId = @LoginID ORDER BY ID DESC)
				AND e.Subject LIKE 'Link to View Sign Up Status%'
			END
			ELSE IF @Mode = 'Forget Password'
			BEGIN
				SELECT @maxDate = MAX(e.CreatedOn) 
				FROM EAlertQTo eto, eAlertQ e
				WHERE eto.alertID = e.alertID
				AND eto.recipient = (SELECT TOP 1 Email FROM TAMS_Registration where LoginId = @LoginID ORDER BY ID DESC)
				AND e.Subject LIKE 'Link to reset password%'
			END

			
			SELECT @RateLimit = Paravalue1 FROM TAMS_Parameters WHERE ParaCode = 'Rate Limiting'

			IF DATEDIFF(SECOND, @maxDate, getdate()) < @RateLimit
			BEGIN
				SELECT -1
			END
			ELSE
			BEGIN
				SELECT 1
			END

		END
	END

END

CREATE PROCEDURE [dbo].[sp_TAMS_User_CheckLastUserRegistration]

BEGIN

				SELECT -1
			END
			ELSE
			BEGIN
				SELECT 1
			END

		END
		ELSE
		BEGIN 
			SELECT 1
		END

	END

END
