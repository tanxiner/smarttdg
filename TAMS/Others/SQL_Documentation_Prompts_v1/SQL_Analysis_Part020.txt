CREATE PROCEDURE [dbo].[sp_TAMS_Depot_Approval_OnLoad]
	@TARID		BIGINT,
	@LogInUser	NVARCHAR(20)
AS

BEGIN

	DECLARE @Line			NVARCHAR(10)
	DECLARE @TrackType		NVARCHAR(10)
	DECLARE @AccessDate		DATE
	DECLARE @AccessTimeSlot NVARCHAR(50)
	DECLARE @TARType		NVARCHAR(50)
	DECLARE @AccessType		NVARCHAR(50)
	DECLARE @IsExclusive	BIT

	SELECT @Line		= Line,
		@TrackType		= Tracktype,
		@AccessDate		= AccessDate,
		@AccessTimeSlot = AccessTimeSlot,
		@TARType		= TARType,
		@AccessType		= AccessType,
		@IsExclusive	= IsExclusive
	FROM TAMS_TAR
		WHERE ID = @TARID

	-- >> Table[0]
	SELECT [Id], [Line], [TARNo], [TARType],
		[Company], [Designation], [Name], [OfficeNo], [MobileNo], [Email],
		[SubmitDate], 
		CONVERT(NVARCHAR(20), [AccessDate], 103) AS [AccessDate], 
		[AccessTimeSlot],
		[AccessTimeFrom], 
		[AccessTimeTo], 
		[AccessLocation], [AccessType],
		[IsNeutralGap], [IsExclusive],
		[DescOfWork], [ARRemark], 
		[TARStatusId],
		[InvolvePower], [PowerOn], [Is13ASocket],
		[CrossOver], [TVFMode], ProtectionType
	FROM TAMS_TAR
		WHERE ID = @TARID

-- >> Sector

	DECLARE @SelSectorNotGap	NVARCHAR(1000)
	DECLARE @SelSectorGap		NVARCHAR(1000)
	DECLARE @PowerZone			NVARCHAR(1000)
	DECLARE @SPKSZone			NVARCHAR(1000)
	
	SET @SelSectorNotGap = ''
	SET @SelSectorGap = ''
	SET @PowerZone = ''
	SET @SPKSZone = ''

	-- EXEC sp_TAMS_Form_OnLoad 'NEL', '01/02/2021', 'Possession', '143;144;145;180;181;182'

	-- >> Selected Sector IsGap = 0
	SELECT @SelSectorNotGap = @SelSectorNotGap + ISNULL(b.Sector + ', ', '') 
		FROM TAMS_TAR_Sector a, TAMS_Sector b
			WHERE a.TARId = @TARID
				AND a.IsBuffer = 0
				AND a.SectorId = b.ID
				AND ISNULL(b.IsGap, 0) = 0
			GROUP BY b.Sector
			-- >> ORDER BY b.[Order]

	IF LEN(LTRIM(RTRIM(@SelSectorNotGap))) > 0
		BEGIN
			SET @SelSectorNotGap = LEFT(LTRIM(RTRIM(@SelSectorNotGap)), LEN(LTRIM(RTRIM(@SelSectorNotGap))) - 1)
		END

		SELECT @PowerZone = @PowerZone + ISNULL(PowerSector + ', ', '') 
			FROM TAMS_Power_Sector a, TAMS_TAR_Power_Sector b	
				WHERE a.ID = b.PowerSectorId
					AND a.IsActive = 1 
					and b.IsBuffer = 0 
					and b.TARId = @TARID
		GROUP BY PowerSector		

		IF LEN(LTRIM(RTRIM(@PowerZone))) > 0
		BEGIN
			SET @PowerZone = LEFT(LTRIM(RTRIM(@PowerZone)), LEN(LTRIM(RTRIM(@PowerZone))) - 1)
		END

		SELECT @SPKSZone = @SPKSZone + ISNULL(a.SPKSZone + ', ', '') 
			FROM TAMS_SPKSZone a, TAMS_TAR_SPKSZone b	
				WHERE a.ID = b.SPKSZoneId
					AND a.IsActive = 1	
					and b.TARId = @TARID
 		GROUP BY a.SPKSZone		

	IF LEN(LTRIM(RTRIM(@SPKSZone))) > 0
		BEGIN
			SET @SPKSZone = LEFT(LTRIM(RTRIM(@SPKSZone)), LEN(LTRIM(RTRIM(@SPKSZone))) - 1)
		END


	-- >> Table[1]
	SELECT @SelSectorNotGap AS SelSectorNotGap,
		@SelSectorGap AS SelSectorGap,
		@PowerZone AS PowerZone,@SPKSZone as SPKSZone


	-- >> Table[2]
	SELECT a.IsSelected, b.OperationRequirement, a.IsAttachment, a.ID AS ID, 
		(SELECT COUNT(*) FROM TAMS_TAR_Attachment c WHERE c.TARAccessReqId = a.ID) AS AttCtr 
	FROM [dbo].[TAMS_TAR_AccessReq] a, [dbo].[TAMS_Access_Requirement] b
		WHERE a.TARId = @TARID
			AND a.IsSelected = 1
			AND a.IsPower = 0
			AND a.OperationRequirement = b.ID
			AND b.IsActive = 1
		ORDER BY a.[Order]

	-- >> Table[3]
	SELECT a.IsSelected, b.OperationRequirement, b.ID 
	FROM [dbo].[TAMS_TAR_AccessReq] a, [dbo].[TAMS_Access_Requirement] b
		WHERE a.TARId = @TARID
			AND a.IsSelected = 1
			AND a.IsPower = 1
			AND a.OperationRequirement = b.ID
			AND b.IsActive = 1
		ORDER BY a.[Order]

	DECLARE @PossID		INT
	DECLARE @PowerOnOff	INT
	DECLARE @TypeOfWork	INT

	SELECT @PossID = [ID],
		@PowerOnOff = [PowerOnOff],
		@TypeOfWork = [TypeOfWorkId]
	  FROM [dbo].[TAMS_Possession]
		WHERE TARId = @TARID

	-- >> Table[4]
	SELECT [ID], [TARId],
		[Summary], [WorkDesc], [TypeOfWorkId], 
		[WorkWithinPossession], [TakePossession], [GiveUpPossession],
		[Remarks], 
		[PowerOnOff],
		[EngTrainFormation],
		[EngTrainArriveLoc], [EngTrainArriveTime],
		[EngTrainDepartLoc], [EngTrainDepartTime],
		[PCName]
	  FROM [dbo].[TAMS_Possession]
		WHERE TARId = @TARID

	-- >> Table[5]
	SELECT ID, TypeOfProtectionLimit, RedFlashingLampsLoc 
		FROM [dbo].[TAMS_Possession_Limit]
			WHERE PossessionId = @PossID
				ORDER BY ID

	-- >> Table[6]
	SELECT ID, RedFlashingLampsLoc 
		FROM [dbo].[TAMS_Possession_WorkingLimit]
			WHERE PossessionId = @PossID
				ORDER BY ID

	-- >> Table[7]
	SELECT ID, OtherProtection 
		FROM [dbo].[TAMS_Possession_OtherProtection]
			WHERE PossessionId = @PossID
				ORDER BY ID

	-- >> Table[8]
	SELECT ID, Sector, case when PowerOff = 1 then 'Y' else 'N' end AS poweroff, NoOFSCD, 
			CASE WHEN BreakerOut = 1 THEN 'Y' ELSE 'N' END AS BreakerOut
		FROM [dbo].[TAMS_Possession_DepotSector]
			WHERE PossessionId = @PossID
				ORDER BY ID
			
	-- >> Table[9]
	SELECT TypeOfWork 
		FROM TAMS_Type_Of_Work
			WHERE ID = @TypeOfWork
				AND IsActive = 1

	-- >> For Approval and Showing of Level

	DECLARE @MaxWFCtr		INT
	DECLARE @PendingWFCtr	INT
	DECLARE @ApprovedWFCtr	INT

	-- >> Get Max Workflow
	SELECT @MaxWFCtr = MAX(b.Level)
		FROM TAMS_TAR_Workflow a, TAMS_Endorser b
			WHERE a.WorkflowId = b.WorkflowId
				--AND a.EndorserId = b.ID
				AND a.TARId = @TARID

	SELECT @ApprovedWFCtr = COUNT(*)
		FROM TAMS_TAR_Workflow a, TAMS_Endorser b
			WHERE a.WorkflowId = b.WorkflowId
				AND a.EndorserId = b.ID
				AND a.TARId = @TARID
				AND a.WFStatus = 'Approved'

	SELECT @PendingWFCtr = COUNT(*)
		FROM TAMS_TAR_Workflow a, TAMS_Endorser b
			WHERE a.WorkflowId = b.WorkflowId
				AND a.EndorserId = b.ID
				AND a.TARId = @TARID
				AND a.WFStatus = 'Pending'

	-- >> Table[10]
	SELECT @MaxWFCtr AS MaxWFCtr, @ApprovedWFCtr AS AppWFCtr, @PendingWFCtr AS PendWFCtr

	-- >> Table[11]
	-- >> Workflow -- Approved
	SELECT a.TARId, a.UserId, a.WFStatus, a.Remark, a.ActionOn, a.ActionBy, c.[Name] AS ActionByName,
			b.RoleId, b.Title, b.RequireValidation, b.RequireTVF, b.WFStatusId, a.ID AS WFID, b.ID AS EID, b.Level
		FROM TAMS_TAR_Workflow a, TAMS_Endorser b, TAMS_User c
			WHERE a.WorkflowId = b.WorkflowId
				AND a.EndorserId = b.ID
				AND a.ActionBy = c.Userid
				AND a.TARId = @TARID
				AND a.WFStatus <> 'Pending'
			ORDER BY a.ID

	DECLARE @UserName	NVARCHAR(200)

	SELECT @UserName = [Name] FROM TAMS_User
		WHERE LoginId = @LogInUser

	-- >> Table[12]
	-- >> Workflow -- Pending
	SELECT a.TARId, a.UserId, a.WFStatus, a.Remark, a.ActionOn, a.ActionBy, c.[Name] AS ActionByName,
			b.RoleId, b.Title, b.RequireValidation, b.RequireTVF, b.RequireBuffer, b.WFStatusId, a.ID AS WFID, b.ID AS EID, b.Level, @UserName AS UserName
		FROM TAMS_TAR_Workflow a, TAMS_Endorser b, TAMS_User c
			WHERE a.WorkflowId = b.WorkflowId
				AND a.EndorserId = b.ID
				AND a.ActionBy = c.Userid
				AND a.TARId = @TARID
				AND a.WFStatus = 'Pending'
			ORDER BY a.ID

	--EXEC sp_TAMS_Approval_OnLoad 5, 'CSAPEX57'
	--EXEC sp_TAMS_Approval_OnLoad 4, 'CSAPEX57'
	
	-- >> Buffer Zone -- Additional
	-- >> Table[13]
	SELECT sect.Sector,
			sect.ID 
		FROM TAMS_Sector sect
			WHERE sect.ID NOT IN (SELECT SectorId 
					FROM TAMS_TAR_Sector ts LEFT JOIN TAMS_TAR t ON t.Id = ts.TARId
						WHERE (t.Id = @TarID OR
							(
							  ((t.TARStatusId = 8 and  t.Line = 'DTL') or (t.TARStatusId = 9 and  t.Line = 'NEL'))
							--t.TARStatusId = 8 
								AND t.AccessDate = @AccessDate
								AND t.AccessTimeSlot = @AccessTimeSlot
								AND t.Line = @Line)))
				AND sect.Line = @Line
				AND sect.TrackType = @TrackType
				AND sect.IsActive = 1
				AND @AccessDate BETWEEN sect.EffectiveDate AND sect.ExpiryDate
			ORDER BY sect.ID

	-- >> Buffer Zone -- Additional
	-- >> Table[14]
	SELECT a.Sector AS StrSector, a.ID AS SectorID
		FROM TAMS_Sector a, TAMS_TAR_Sector b
			WHERE a.ID = b.SectorId
				AND b.TARId = @TARID
				AND b.IsBuffer = 1
		ORDER BY a.ID

	 

	-- >> Get Exception TARs - Sector Conflict
	CREATE TABLE #TmpExc 
	(TARID			BIGINT,
	TARNo			NVARCHAR(20),
	TARType			NVARCHAR(10),
	AccessDate		NVARCHAR(20),
	AccessTimeSlot		NVARCHAR(50),
	AccessType		NVARCHAR(20),
	IsExclusive		INT)

	CREATE TABLE #TmpExcSector 
	(TARID			BIGINT,
	TARNo			NVARCHAR(20),
	TARType			NVARCHAR(10),
	AccessDate		NVARCHAR(20),
	AccessTimeSlot		NVARCHAR(50),
	AccessType		NVARCHAR(20),
	IsExclusive		INT,
	SectorID		INT,
	SectorStr		NVARCHAR(50),
	IsBuffer		INT)
	
	TRUNCATE TABLE #TmpExc
	TRUNCATE TABLE #TmpExcSector

	DECLARE @Cur01 CURSOR;

	DECLARE @CID			INTEGER
	DECLARE @CSectorID		INTEGER
	DECLARE @CIsBuffer		BIT
	DECLARE @CColourCode	NVARCHAR(20)
	DECLARE @CIsAddBuffer	BIT

	DECLARE @Cur02 CURSOR;

	DECLARE @C2TARID		INTEGER 
	DECLARE @C2TARNo		NVARCHAR(20) 
	DECLARE @C2TARType		NVARCHAR(20)
	DECLARE @C2AccessDate	DATE
	DECLARE @C2AccessTimeSlot nvarchar(50)
	DECLARE @C2AccessType	NVARCHAR(20) 
	DECLARE @C2IsExclusive	BIT
	DECLARE @C2SectorID		INTEGER 
	DECLARE @C2SectorStr	NVARCHAR(20) 
	DECLARE @C2IsBuffer		BIT 

	SET @Cur01 = CURSOR FAST_FORWARD 
	FOR
		SELECT ID, SectorId, IsBuffer, ColourCode, IsAddedBuffer FROM TAMS_TAR_Sector
			WHERE TARId = @TARID
				ORDER BY SectorId

	OPEN @Cur01
	
	FETCH NEXT FROM @Cur01 INTO @CID, @CSectorID, @CIsBuffer, @CColourCode, @CIsAddBuffer
	
	WHILE @@FETCH_STATUS = 0 
	
		BEGIN
			print @CSectorID;
			print '@CSectorID';
			IF @CIsBuffer = 1
				BEGIN
					-- >> Is Buffer
					 PRINT 'here 1';
					INSERT INTO #TmpExcSector
						(TARID, TARNo, TARType, AccessDate,AccessTimeSlot, AccessType, IsExclusive, SectorID, SectorStr, IsBuffer)
					SELECT t.Id, t.TARNo, t.TARType, t.AccessDate, t.AccessTimeSlot,t.AccessType, t.IsExclusive,
							ts.SectorId, s.Sector, ts.IsBuffer 
						FROM TAMS_TAR t, TAMS_TAR_Sector ts, TAMS_Sector s
							WHERE t.Id = ts.TARId
								AND s.ID = ts.SectorId
								AND t.AccessDate = @AccessDate
								AND t.AccessTimeSlot = @AccessTimeSlot
								AND ts.IsBuffer = 0
								AND ((t.TARStatusId = 8 and  t.Line = 'DTL') or (t.TARStatusId = 9 and  t.Line = 'NEL'))
								--AND t.TARStatusId = 8
								AND ts.SectorId = @CSectorID
								AND t.Id <> @TARID
								AND s.IsActive = 1
								AND @AccessDate BETWEEN s.EffectiveDate and S.ExpiryDate
				END
			ELSE
				BEGIN
					-- >> Not Buffer
					  PRINT 'here 2';
					INSERT INTO #TmpExcSector
						(TARID, TARNo, TARType, AccessDate, AccessType, IsExclusive, SectorID, SectorStr, IsBuffer)
					SELECT t.Id, t.TARNo, t.TARType, t.AccessDate, t.AccessType, t.IsExclusive,
							ts.SectorId, s.Sector, ts.IsBuffer 
						FROM TAMS_TAR t, TAMS_TAR_Sector ts, TAMS_Sector s
							WHERE t.Id = ts.TARId
								AND s.ID = ts.SectorId
								AND t.AccessDate = @AccessDate
								AND t.AccessTimeSlot = @AccessTimeSlot 
								-- >> AND ts.IsBuffer = 0
								AND ((t.TARStatusId = 8 and  t.Line = 'DTL') or (t.TARStatusId = 9 and  t.Line = 'NEL'))
								--AND t.TARStatusId = 8
								AND ts.SectorId = @CSectorID
								AND t.Id <> @TARID
								AND s.IsActive = 1
								AND @AccessDate BETWEEN s.EffectiveDate and S.ExpiryDate
 
				END
				
			-- >> #TmpExcSector
			SET @Cur02 = CURSOR FAST_FORWARD 
			FOR
				SELECT TARID, TARNo, TARType, AccessDate,AccessTimeSlot, AccessType, IsExclusive, SectorID, SectorStr, IsBuffer 
					FROM #TmpExcSector
							ORDER BY TARID, SectorID

			OPEN @Cur02
	
			FETCH NEXT FROM @Cur02 INTO @C2TARID, @C2TARNo, @C2TARType, @C2AccessDate,@C2AccessTimeSlot, @C2AccessType, @C2IsExclusive, @C2SectorID, @C2SectorStr,@C2IsBuffer
	
			WHILE @@FETCH_STATUS = 0 
	
				BEGIN
					IF @AccessType = 'Protection' AND @IsExclusive = 0
						BEGIN
							print 1
							IF @C2AccessType = 'Protection' AND @C2IsExclusive = 0
								BEGIN

									PRINT 'DO NOTHING'
								END
							ELSE
								BEGIN
									IF (SELECT COUNT(*) FROM #TmpExc WHERE TARID = @C2TARID) = 0
										BEGIN
											INSERT INTO #TmpExc
												(TARID, TARNo, TARType, AccessDate, AccessTimeSlot,AccessType, IsExclusive)
											VALUES 
												(@C2TARID, @C2TARNo, @C2TARType, @C2AccessDate,@C2AccessTimeSlot, @C2AccessType, @C2IsExclusive)
										END
								END
						END
					ELSE
						BEGIN
							IF (SELECT COUNT(*) FROM #TmpExc WHERE TARID = @C2TARID) = 0
								BEGIN
									INSERT INTO #TmpExc
										(TARID, TARNo, TARType, AccessDate,AccessTimeSlot, AccessType, IsExclusive)
									VALUES 
										(@C2TARID, @C2TARNo, @C2TARType, @C2AccessDate, @C2AccessTimeSlot,@C2AccessType, @C2IsExclusive)
								END
						END

					FETCH NEXT FROM @Cur02 INTO @C2TARID, @C2TARNo, @C2TARType, @C2AccessDate,@C2AccessTimeSlot, @C2AccessType, @C2IsExclusive, @C2SectorID, @C2SectorStr,@C2IsBuffer
				END 	

			CLOSE @Cur02
	
			DEALLOCATE @Cur02
			-- >> #TmpExcSector

			FETCH NEXT FROM @Cur01 INTO @CID, @CSectorID, @CIsBuffer, @CColourCode, @CIsAddBuffer
		END 	

	CLOSE @Cur01
	
	DEALLOCATE @Cur01

	-- >> Exception List. Sector Conflict, to show only if RequireValidation = True for current Level Endorser
	-- >> Table[15]
	SELECT TARID, TARNo, TARType, AccessDate, AccessType, IsExclusive 
		FROM #TmpExc

	-- >> Table[16]
	IF @AccessType = 'Protection'
		BEGIN
			SELECT ID, OperationRequirement 
				FROM TAMS_Access_Requirement
					WHERE Line = @Line
						AND IsPowerReq = 1
						and TrackType = @TrackType
						AND OperationRequirement = 'Traction Power ON'
						AND IsActive = 1
					ORDER BY [Order]
		END
	ELSE
		BEGIN
			SELECT ID, OperationRequirement 
				FROM TAMS_Access_Requirement
					WHERE Line = @Line
					and TrackType = @TrackType
						AND IsPowerReq = 1
						AND IsActive = 1
					ORDER BY [Order]
		END
END
