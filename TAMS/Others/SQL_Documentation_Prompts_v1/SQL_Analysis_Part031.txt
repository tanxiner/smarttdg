CREATE procedure  [dbo].[sp_TAMS_Depot_SectorBooking_OnLoad]
	@Line		AS NVARCHAR(10) = NULL,
	@TrackType	AS NVARCHAR(50) = NULL,
	@AccessDate	AS NVARCHAR(20) = NULL,
	@TARType	AS NVARCHAR(20) = NULL,
	@AccessType	AS NVARCHAR(20) = NULL	
AS

BEGIN

	CREATE TABLE #ListES
	(Line				NVARCHAR(10),
	SectorID			BIGINT,
	Sector				NVARCHAR(50),
	PowerZone			NVARCHAR(100),
	SPKSZone			NVARCHAR(100),
	FromTrack			NVARCHAR(200),
	ToTrack			NVARCHAR(200),
	IsSelected			BIT,
	IsEnabled			BIT,
	ColorCode			NVARCHAR(50),
	OrderID				INT 
	  )
	
	TRUNCATE TABLE #ListES

	DECLARE @CurID					BIGINT
	DECLARE @CurLine				NVARCHAR(10)
	 DECLARE @CurSector				NVARCHAR(50)
 	DECLARE @CurOrderID				INT
	DECLARE @PowerZone				NVARCHAR(100)
	DECLARE @SPKSZone				NVARCHAR(100)
	 
 

	DECLARE @ColorCode		NVARCHAR(50)
	SET @ColorCode = ''

	DECLARE @CCAccessType	NVARCHAR(50)
	SET @CCAccessType = ''

	-- >> Insert into temp table
	IF @Line = 'NEL'
		BEGIN
			INSERT INTO #ListES (Line,   SectorID, Sector, FromTrack,ToTrack, OrderID, IsEnabled)
			SELECT a.Line,  a.ID, a.Sector, a.FromTrack,a.ToTrack, a.[Order], 1
				FROM TAMS_Sector a
					WHERE a.Line = @Line
					and TrackType = @TrackType
						AND a.IsActive = 1 
							AND (a.EffectiveDate <= GETDATE() AND a.ExpiryDate >= GETDATE())
				ORDER BY a.[Order] ASC
		END
 

		DECLARE @Cur01 CURSOR;

		SET @Cur01 = CURSOR FAST_FORWARD 
		FOR

			SELECT a.ID, a.Line,  a.Sector,   a.[Order]
				FROM TAMS_Sector a
					WHERE a.Line = @Line
					and TrackType = @TrackType
						AND a.IsActive = 1 
						AND (a.EffectiveDate <= GETDATE() AND a.ExpiryDate >= GETDATE())
			ORDER BY a.[Order] ASC

		OPEN @Cur01
	
		FETCH NEXT FROM @Cur01 INTO @CurID, @CurLine,  @CurSector,   @CurOrderID
	
		WHILE @@FETCH_STATUS = 0 
	
			BEGIN

				SET @PowerZone = ''
				SET @ColorCode = ''
				SET @SPKSZone = ''
				 

				SELECT @SPKSZone = @SPKSZone + ISNULL(SPKSZone + ',', '') 
					FROM TAMS_SPKSZone a, TAMS_Track_SPKSZone b	
						WHERE a.ID = b.SPKSId
							AND a.IsActive = 1
							AND b.IsActive = 1
							 AND b.SectorId = @CurID
				SELECT @PowerZone = @PowerZone + ISNULL(PowerSector + ',', '') 
					FROM TAMS_Power_Sector a, TAMS_Track_Power_Sector b	
						WHERE a.ID = b.PowerSectorId
							AND a.IsActive = 1
							AND b.IsActive = 1
							AND b.IsBuffer = 0 
							 AND b.SectorId = @CurID
				 		

					IF LEN(LTRIM(RTRIM(@SPKSZone))) > 0
					BEGIN
						SET @SPKSZone = LEFT(LTRIM(RTRIM(@SPKSZone)), LEN(LTRIM(RTRIM(@SPKSZone))) - 1)
					END

					IF LEN(LTRIM(RTRIM(@PowerZone))) > 0
					BEGIN
						SET @PowerZone = LEFT(LTRIM(RTRIM(@PowerZone)), LEN(LTRIM(RTRIM(@PowerZone))) - 1)
					END
				
				SELECT @ColorCode = ISNULL(b.ColourCode, ''),
						@CCAccessType = ISNULL(a.AccessType, '') 
					FROM TAMS_TAR a, TAMS_TAR_Sector b
						WHERE a.Id = b.TARId
							AND a.AccessDate = CONVERT(DATETIME, @AccessDate, 103)
							AND ( (a.Line = 'NEL' and a.TARStatusId = 9 ) or (a.Line = 'DTL' and a.TARStatusId = 8)) -- >> Approved
							AND a.IsExclusive = 1 
							AND b.SectorId = @CurID 
							AND a.Line = @Line
							and TrackType = @TrackType
				
				IF @ColorCode = ''
					BEGIN
						UPDATE #ListES
							SET IsEnabled = 1,
								ColorCode = '',
								SPKSZone = @SPKSZone, 
								PowerZone = @PowerZone
						WHERE SectorID = @CurID
					END
				ELSE
					BEGIN
						UPDATE #ListES
							SET ColorCode = @ColorCode,
								SPKSZone = @SPKSZone, 
								PowerZone = @PowerZone
						WHERE SectorID = @CurID

						IF (@TARType = '2' OR @TARType = '3') 
								AND @AccessType = 'Protection' 
								AND @ColorCode = ''
								AND @CCAccessType = 'Possession'
							BEGIN
								UPDATE #ListES
									SET IsEnabled = 1,
										ColorCode = '',
										SPKSZone = @SPKSZone, 
									PowerZone = @PowerZone
								WHERE SectorID = @CurID
							END
						ELSE
							BEGIN
								UPDATE #ListES
									SET 
										--IsEnabled = 0,
										SPKSZone = @SPKSZone, 
										PowerZone = @PowerZone
								WHERE SectorID = @CurID
							END
					END

				FETCH NEXT FROM @Cur01 INTO @CurID, @CurLine,   @CurSector,  @CurOrderID
			END 	

		CLOSE @Cur01
	
		DEALLOCATE @Cur01
	
	SELECT Line,
		 SectorID,
		Sector,
 		IsSelected,
		IsEnabled,
		ColorCode,
 		OrderID,
		PowerZone,
		SPKSZone, 
		FromTrack,
		ToTrack
	FROM #ListES
	 	ORDER BY OrderID, SectorID

 

/*
	EXEC sp_TAMS_SectorBooking_OnLoad 'DTL', '01/02/2021', '2', 'Possession'
	EXEC sp_TAMS_SectorBooking_OnLoad 'DTL', '01/02/2021', '2', 'Protection'
	EXEC sp_TAMS_SectorBooking_OnLoad 'DTL', '01/02/2021', '1', 'Protection'
	EXEC sp_TAMS_SectorBooking_OnLoad 'NEL', '01/02/2021', '1', 'Protection'
*/

	IF @AccessType = 'Protection'
		BEGIN
			SELECT ID, OperationRequirement 
				FROM TAMS_Access_Requirement
					WHERE Line = @Line
					and TrackType = @TrackType
						AND IsPowerReq = 1
						AND OperationRequirement = 'Traction Power ON'
						AND IsActive = 1
					ORDER BY [Order]
		END
	ELSE
		BEGIN
			SELECT ID, OperationRequirement 
				FROM TAMS_Access_Requirement
					WHERE Line = @Line
					and TrackType = @TrackType
						AND IsPowerReq = 1
						AND IsActive = 1
					ORDER BY [Order]
		END

END

CREATE   procedure  [dbo].[sp_TAMS_Depot_SectorBooking_QTS_Chk] 
	@nric NVARCHAR(MAX) = null,
	@qualdate NVARCHAR(MAX) = null,
	@line NVARCHAR(MAX) = null,
	@TrackType NVARCHAR(50) = null
AS
BEGIN

/*
exec sp_TAMS_SectorBooking_QTS_Chk 'S1111111A',
	'10/05/2021',
	'DTL'
*/

DECLARE @Message NVARCHAR(500)

SET @Message = ''

DECLARE @EPCtr BIGINT
DECLARE @BLCtr BIGINT
DECLARE @RTCtr BIGINT

SET @EPCtr = 0
SET @BLCtr = 0
SET @RTCtr = 0

DECLARE @RetVal NVARCHAR(5)

SET @RetVal = '0'

CREATE TABLE #tmpqtsqc 
(id BIGINT IDENTITY(1,1) NOT NULL, 
access_id NVARCHAR(50),
pq_lastaccess_date	DATETIME NULL,
pq_validaccess_date	DATETIME NULL,
pq_validtill_date	DATETIME NULL,
suspened_till		DATETIME NULL)

CREATE TABLE #tmpnric 
(id BIGINT IDENTITY(1,1) NOT NULL, 
nric NVARCHAR(50),
namestr NVARCHAR(500),
qualdate NVARCHAR(50),
qualcode NVARCHAR(100),
line NVARCHAR(50),
qualstatus NVARCHAR(100))

CREATE TABLE #tmpqualcode 
( qual_code nvarchar(50))

TRUNCATE TABLE #tmpnric;
TRUNCATE TABLE #tmpqtsqc;
TRUNCATE TABLE #tmpqualcode;

 

insert into #tmpqualcode
SELECT   ParaValue2 
	FROM TAMS_Parameters
		WHERE ParaCode = 'QTSQualCode'
			AND ParaValue1 = @line
			 AND ParaValue3 = 'Depot Possession' 
			AND CONVERT(DATETIME, @qualdate, 103) BETWEEN EffectiveDate AND ExpiryDate

INSERT INTO #tmpnric (nric, namestr, qualdate, qualcode, line, qualstatus)
VALUES (@nric, NULL, @qualdate, '', @line, NULL)

DECLARE @CurID		BIGINT
DECLARE @CurNRIC	NVARCHAR(50)
DECLARE @CurQualDate NVARCHAR(50)
DECLARE @CurQualCode NVARCHAR(100)
DECLARE @CurLine	NVARCHAR(50)
DECLARE @CurNameStr		NVARCHAR(500)

DECLARE @QualCtr	BIGINT
SET @QualCtr = 0

DECLARE @Cur01 CURSOR;
SET @Cur01 = CURSOR FAST_FORWARD 
FOR

SELECT id, nric, qualdate, qualcode, line
FROM #tmpnric
	ORDER BY 1 ASC

OPEN @Cur01
	FETCH NEXT FROM @Cur01 INTO @CurID, @CurNRIC, @CurQualDate, @CurQualCode, @CurLine
	WHILE @@FETCH_STATUS = 0 
		BEGIN
			SET @QualCtr = 0

			SELECT @CurNameStr = p_name
				FROM [FLEXNETSKGSVR].[QTSDB].[dbo].QTS_Personnel
					WHERE p_active = 1
					AND dbo.DecryptStringQTS(p_access_id) = @CurNRIC

TRUNCATE TABLE #tmpqtsqc

print '@CurQualCode'; 
print @CurQualCode; 
INSERT INTO #tmpqtsqc
(access_id, pq_lastaccess_date, pq_validaccess_date, pq_validtill_date, suspened_till)
SELECT dbo.DecryptStringQTS(cc.p_access_id),
	aa.pq_lastaccess_date AS pq_lastaccess_date,
	aa.pq_validaccess_date AS pq_validaccess_date,
	aa.pq_validtill_date AS pq_validtill_date,
CASE WHEN aa.pq_suspend_to IS NOT NULL THEN
	CASE WHEN aa.pq_reinstate_eff IS NULL THEN
			CASE WHEN aa.pq_suspend_to > current_timestamp THEN aa.pq_suspend_to ELSE '' END
		WHEN aa.pq_suspend_to < aa.pq_reinstate_eff
			THEN CASE WHEN aa.pq_suspend_to > current_timestamp THEN aa.pq_suspend_to ELSE '' END
				ELSE CASE WHEN aa.pq_reinstate_eff > current_timestamp THEN aa.pq_reinstate_eff ELSE '' END
	END
END AS suspened_till
FROM [FLEXNETSKGSVR].[QTSDB].[dbo].QTS_Personnel_Qualification aa, 
	[FLEXNETSKGSVR].[QTSDB].[dbo].QTS_Qualification bb, 
	[FLEXNETSKGSVR].[QTSDB].[dbo].QTS_Personnel cc
WHERE cc.p_id = aa.pq_p_id
	AND aa.pq_qual_id = bb.qual_id
	AND cc.p_rail = @CurLine
	AND bb.qual_rail = @CurLine
	AND (bb.qual_access_period != 99
		AND bb.qual_code not like '%EPIC%'
		AND bb.qual_code not like '%OSM%'
		AND bb.qual_code  != 'TTP')
	AND bb.qual_code in (Select qual_code from #tmpqualcode)
	AND dbo.DecryptStringQTS(cc.p_access_id) = @CurNRIC

			SELECT @QualCtr = COUNT(*) 
				FROM #tmpqtsqc
					WHERE access_id = @CurNRIC

			IF @QualCtr = 0 -- >> IF NO RECORD, INVALID
				BEGIN
					-- >> SELECT '1'
					UPDATE #tmpnric
						SET qualstatus = 'InValid',
						namestr = @CurNameStr
					WHERE id = @CurID
				END
			ELSE
				BEGIN
					SELECT @QualCtr = COUNT(*) 
						FROM #tmpqtsqc
							WHERE access_id = @CurNRIC
								AND suspened_till IS NULL

					IF @QualCtr > 0 -- >> No Suspension Information
						BEGIN
							SELECT @QualCtr = COUNT(*) 
								FROM #tmpqtsqc
									WHERE access_id = @CurNRIC
										AND (CONVERT(DATETIME, @CurQualDate, 103) < pq_validaccess_date
										AND CONVERT(DATETIME, @CurQualDate, 103) < pq_validtill_date)

								IF @QualCtr > 0
									BEGIN
										-- >> SELECT '3'
										UPDATE #tmpnric
											SET qualstatus = 'Valid',
												namestr = @CurNameStr
										WHERE id = @CurID
									END
								ELSE
									BEGIN
										-- >> SELECT '4'
										UPDATE #tmpnric
											SET qualstatus = 'InValid',
												namestr = @CurNameStr
										WHERE id = @CurID
									END
						END
					ELSE
						BEGIN
							UPDATE #tmpnric
								SET qualstatus = 'InValid',
									namestr = @CurNameStr
							WHERE id = @CurID
						END
				END

			---SELECT @CurID, @CurNRIC, @CurQualDate, @CurQualCode, @CurLine

			FETCH NEXT FROM @Cur01 INTO @CurID, @CurNRIC, @CurQualDate, @CurQualCode, @CurLine
		END 	
	CLOSE @Cur01
	DEALLOCATE @Cur01

SELECT LTRIM(RTRIM(ISNULL(nric, ''))) AS nric, 
		LTRIM(RTRIM(ISNULL(namestr, ''))) AS namestr,
		LTRIM(RTRIM(ISNULL(line, ''))) AS line,
		LTRIM(RTRIM(ISNULL(qualdate, ''))) AS qualdate, 
		LTRIM(RTRIM(ISNULL(qualcode, ''))) AS qualcode, 
		LTRIM(RTRIM(ISNULL(qualstatus, ''))) AS qualstatus
	FROM #tmpnric
ORDER BY ID
