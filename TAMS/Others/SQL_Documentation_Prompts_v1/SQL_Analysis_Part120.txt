CREATE PROCEDURE [dbo].[sp_TAMS_RGS_Cancel_20230308]
	@TARID		AS BIGINT = 0,
	@CancelRemarks	AS NVARCHAR(1000) = NULL,
	@UserID		AS NVARCHAR(500),
	@Message	AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

SET @Message = ''

DECLARE @NewLine AS CHAR(2) = CHAR(30) + CHAR(10)

UPDATE TAMS_TOA
	SET	TOAStatus = 6,
		CancelRemark = @CancelRemarks,
		UpdatedOn = GETDATE(),
		UpdatedBy = @UserID
WHERE TARId = @TARID
		
INSERT INTO TAMS_TOA_Audit
SELECT @UserID, GETDATE(), 'U', * FROM TAMS_TOA
	WHERE TARId = @TARID

DECLARE @IDUserID BIGINT

SELECT @IDUserID = Userid
	FROM TAMS_User
		WHERE LoginId = @UserID
			
DECLARE @SMSMsg	NVARCHAR(4000)
SET @SMSMsg = ''

DECLARE @lv_IsAllAckSurrender BIT
DECLARE @TARNo NVARCHAR(50)
DECLARE @TOANo NVARCHAR(50)
DECLARE @Line NVARCHAR(5)
DECLARE @CurrDate NVARCHAR(20)
DECLARE @CurrTime NVARCHAR(20)
DECLARE @AccessDate DATE
DECLARE @OperationDate DATE
DECLARE @HPNo NVARCHAR(20)

SET @lv_IsAllAckSurrender = 1

SET @TARNo = ''
SET @TOANo = ''
SET @Line = ''
SET @CurrDate = ''
SET @HPNo = ''

SET @CurrDate = CONVERT(NVARCHAR(20), GETDATE(), 103)
SET @CurrTime = CONVERT(NVARCHAR(20), GETDATE(), 108)

SELECT @TARNo = ISNULL(a.TARNo, ''),
		@TOANo = ISNULL(b.TOANo, ''), 
		@Line = a.Line, 
		@AccessDate = b.AccessDate,
		@OperationDate = b.OperationDate,
		@HPNo = b.MobileNo
	FROM TAMS_TAR a, TAMS_TOA b	
		WHERE a.Id = b.TARId
			AND a.Id = @TARID

DECLARE @LogMsg NVARCHAR(100)
SET @LogMsg = ''
SET @LogMsg = CAST(@TARNo AS NVARCHAR(20)) + ' ' + CAST(@TOANo AS NVARCHAR(20)) + ' RGS Cancel Action Log for checking'
-- >> SELECT @LogMsg

INSERT INTO TAMS_Action_Log
VALUES(@Line, 'RGS', 'RGS-Cancel', @TarId, @LogMsg, GETDATE(), 88);

DECLARE @Endorser1 INT
DECLARE @Endorser2 INT

IF @Line = 'DTL'
	BEGIN
		SELECT @Endorser1 = a.ID 
			FROM TAMS_Endorser a, TAMS_Workflow b
				WHERE b.WorkflowType = 'OCCAuth'
					AND a.WorkflowId = b.ID
					AND b.Line = 'DTL'
					AND a.Level = 10

		SELECT @Endorser2 = a.ID 
			FROM TAMS_Endorser a, TAMS_Workflow b
				WHERE b.WorkflowType = 'OCCAuth'
					AND a.WorkflowId = b.ID
					AND b.Line = 'DTL'
					AND a.Level = 12
	END
ELSE
	BEGIN
		SELECT @Endorser1 = a.ID 
			FROM TAMS_Endorser a, TAMS_Workflow b
				WHERE b.WorkflowType = 'OCCAuth'
					AND a.WorkflowId = b.ID
					AND b.Line = 'NEL'
					AND a.Level = 7
	END

DECLARE @Cur01 CURSOR;
DECLARE @TOAStatus	INT

SET @Cur01 = CURSOR FAST_FORWARD 
FOR
SELECT b.TOAStatus
	FROM TAMS_TAR a, TAMS_TOA b
		WHERE a.Id = b.TARId
			AND b.AccessDate = @AccessDate
			AND b.TOAStatus NOT IN (0, 6)
			AND a.Line = @Line
	ORDER BY b.Id
OPEN @Cur01
FETCH NEXT FROM @Cur01 INTO @TOAStatus
WHILE @@FETCH_STATUS = 0 
	BEGIN
		IF @TOAStatus <> 5
			BEGIN
				SET @lv_IsAllAckSurrender = 0
			END
		FETCH NEXT FROM @Cur01 INTO @TOAStatus
	END 	
CLOSE @Cur01
DEALLOCATE @Cur01

IF @lv_IsAllAckSurrender = 1
	BEGIN
		IF @Line = 'DTL'
			BEGIN
				DECLARE @Cur02 CURSOR;
				DECLARE @Cur03 CURSOR;
				DECLARE @DTLOCCAuthID	INT

				SET @Cur02 = CURSOR FAST_FORWARD 
				FOR
				SELECT a.ID
					FROM TAMS_OCC_Auth a
						WHERE a.Line = @Line
							AND a.OperationDate = @OperationDate
							AND a.AccessDate = @AccessDate
							AND a.OCCAuthStatusId = 10
							AND a.PowerOn = 0
				OPEN @Cur02
				FETCH NEXT FROM @Cur02 INTO @DTLOCCAuthID
				WHILE @@FETCH_STATUS = 0 
					BEGIN
						UPDATE TAMS_OCC_Auth
							SET OCCAuthStatusId = 11,
								UpdatedOn = GETDATE(),
								UpdatedBy = @IDUserID
							WHERE ID = @DTLOCCAuthID

						INSERT INTO [dbo].[TAMS_OCC_Auth_Workflow]
							([OCCAuthId], [OCCAuthEndorserId], [WFStatus], 
								[StationId], [FISTestResult], [ActionOn], [ActionBy])
						VALUES (@DTLOCCAuthID, @Endorser1, 'Pending',
							0, NULL, GETDATE(), @IDUserID)

						FETCH NEXT FROM @Cur02 INTO @DTLOCCAuthID
					END 	
				CLOSE @Cur02
				DEALLOCATE @Cur02


				-- >> Buffer Zone
				-- >> 
				SET @Cur03 = CURSOR FAST_FORWARD 
				FOR
				SELECT a.ID
					FROM TAMS_OCC_Auth a
						WHERE a.Line = @Line
							AND a.OperationDate = @OperationDate
							AND a.AccessDate = @AccessDate
							AND a.OCCAuthStatusId = 7 -- >> Check with Reno how to know this one end already
				OPEN @Cur03
				FETCH NEXT FROM @Cur03 INTO @DTLOCCAuthID
				WHILE @@FETCH_STATUS = 0 
					BEGIN
						UPDATE TAMS_OCC_Auth
							SET OCCAuthStatusId = 13,
								UpdatedOn = GETDATE(),
								UpdatedBy = @IDUserID
							WHERE ID = @DTLOCCAuthID

						INSERT INTO [dbo].[TAMS_OCC_Auth_Workflow]
							([OCCAuthId], [OCCAuthEndorserId], [WFStatus], 
								[StationId], [FISTestResult], [ActionOn], [ActionBy])
						VALUES (@DTLOCCAuthID, @Endorser2, 'Pending', -- >> Check with Reno what is the EndorserID
							0, NULL, GETDATE(), @IDUserID)

						FETCH NEXT FROM @Cur03 INTO @DTLOCCAuthID
					END 	
				CLOSE @Cur03
				DEALLOCATE @Cur03
				-- >> 
				-- >> Buffer Zone
			END
		ELSE
			BEGIN
				DECLARE @Cur04 CURSOR;
				DECLARE @NELOCCAuthID	INT

				SET @Cur02 = CURSOR FAST_FORWARD 
				FOR
				SELECT a.ID
					FROM TAMS_OCC_Auth a
						WHERE a.Line = @Line
							AND a.OperationDate = @OperationDate
							AND a.AccessDate = @AccessDate
							AND a.OCCAuthStatusId = 8
							AND a.PowerOn = 0
				OPEN @Cur02
				FETCH NEXT FROM @Cur02 INTO @NELOCCAuthID
				WHILE @@FETCH_STATUS = 0 
					BEGIN
						UPDATE TAMS_OCC_Auth
							SET OCCAuthStatusId = 9,
								UpdatedOn = GETDATE(),
								UpdatedBy = @IDUserID
							WHERE ID = @NELOCCAuthID

						INSERT INTO [dbo].[TAMS_OCC_Auth_Workflow]
							([OCCAuthId], [OCCAuthEndorserId], [WFStatus], 
								[StationId], [FISTestResult], [ActionOn], [ActionBy])
						VALUES (@NELOCCAuthID, @Endorser1, 'Pending',
							0, NULL, GETDATE(), @IDUserID)

						FETCH NEXT FROM @Cur02 INTO @NELOCCAuthID
					END 	
				CLOSE @Cur02
				DEALLOCATE @Cur02
			END
	END


	-- >>

--	END

--			+ ' hrs on ' + CAST(@CurrDate AS NVARCHAR(20)) + '. Thank you!'
--	END

IF @Line = 'DTL'
	BEGIN
		-- >> SET @SMSMsg = CAST(@NoStr AS NVARCHAR(50)) + ' has been cancelled on TAMS due to TPO/PC inactivity. Please call back OCC at 67187110 if you have surrendered TOA on TAMS.'
		IF CAST(@TOANo AS NVARCHAR(50)) = ''
			BEGIN
				SET @SMSMsg = CAST(@TARNo AS NVARCHAR(50)) + ' has been cancelled on TAMS & contact OCC 67187110, if required.'
			END
		ELSE
			BEGIN
				SET @SMSMsg = CAST(@TOANo AS NVARCHAR(50)) + ' has been cancelled on TAMS & contact OCC 67187110, IMMEDIATELY.'
			END	
	END
ELSE
	BEGIN
		-- >> NEL
		IF CAST(@TOANo AS NVARCHAR(50)) = ''
			BEGIN
				SET @SMSMsg = CAST(@TARNo AS NVARCHAR(50)) + ' has been cancelled on TAMS & contact OCC 64132001, if required.'
			END
		ELSE
			BEGIN
				SET @SMSMsg = CAST(@TOANo AS NVARCHAR(50)) + ' has been cancelled on TAMS & contact OCC 64132001, IMMEDIATELY.'
			END
	END

-- >> SEND SMS
DECLARE @RetVal NVARCHAR(5)
SET @RetVal = ''

	IF @HPNo <> ''
		BEGIN
			EXEC sp_api_send_sms @HPNo, 'TAMS RGS', @SMSMsg, @RetVal OUTPUT
		END

-- >> 20220818 Quickly trigger SMS Sending
	DECLARE @SMSOutputMsg NVARCHAR(500) 

	SET @SMSOutputMsg = ''

	EXEC SP_Call_SMTP_Send_SMSAlert @SMSOutputMsg OUTPUT

	IF @SMSOutputMsg <> ''
		BEGIN
			SET @Message = 'Error SMS Sending'
			GOTO TRAP_ERROR
		END
-- >> 20220818 Quickly trigger SMS Sending

IF @@ERROR <> 0 BEGIN
	SET @Message = 'Error RGS Cancel'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END
