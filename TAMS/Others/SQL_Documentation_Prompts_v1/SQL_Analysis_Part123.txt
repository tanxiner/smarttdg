CREATE PROCEDURE [dbo].[sp_TAMS_RGS_GrantTOA_001]
	@TARID		AS BIGINT = 0,
	@EncTARID	AS NVARCHAR(250),
	@UserID		AS NVARCHAR(500),
	@Message	AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

SET @Message = ''

DECLARE @NewLine AS CHAR(2) = CHAR(30) + CHAR(10)

DECLARE @SMSMsg	NVARCHAR(4000)
SET @SMSMsg = ''

DECLARE @OPDate NVARCHAR(20)
DECLARE @TARNo NVARCHAR(50)
DECLARE @Line NVARCHAR(5)
DECLARE @CurrDate NVARCHAR(20)
DECLARE @AccessType NVARCHAR(20)
DECLARE @Site NVARCHAR(200)
DECLARE @HPNo NVARCHAR(20)

SET @TARNo = ''
SET @Line = ''
SET @CurrDate = ''
SET @AccessType = ''
SET @Site = ''
SET @HPNo = ''

SELECT @TARNo = a.TARNo, 
		@Line = a.Line, 
		@OPDate = CONVERT(NVARCHAR(20), b.OperationDate, 103),
		@AccessType = a.AccessType,
		@HPNo = b.MobileNo
	FROM TAMS_TAR a, TAMS_TOA b	
		WHERE a.Id = b.TARId
			AND a.Id = @TARID

DECLARE @RefNum		NVARCHAR(20)
DECLARE @RefNumMsg	NVARCHAR(20)
SET @RefNum = ''
SET @RefNumMsg = ''
EXEC sp_Generate_Ref_Num_TOA 'TOA', @Line, @OPDate, @RefNum OUTPUT, @RefNumMsg OUTPUT

UPDATE TAMS_TOA
	SET	TOAStatus = 3,
		TOANo = @RefNum,
		GrantTOATime = GETDATE(),
		UpdatedOn = GETDATE(),
		UpdatedBy = @UserID
WHERE TARId = @TARID

INSERT INTO TAMS_TOA_Audit
SELECT @UserID, GETDATE(), 'U', * FROM TAMS_TOA
	WHERE TARId = @TARID

SET @CurrDate = CONVERT(NVARCHAR(20), GETDATE(), 103)

--			+ @Site + 'https://pulsewebuat.sbstransit.com.sg/TAMS/RGSAckSMS.aspx?TARID=' + CAST(@EncTARID AS NVARCHAR(500)) + '&SMSType=1'
--	END

IF @AccessType = 'Possession'
	BEGIN
		SET @SMSMsg = CAST(@RefNum AS NVARCHAR(50)) + ' has been granted for TAR ID ' + CAST(@TARNo AS NVARCHAR(20))
			+ '. Click on the link to acknowledge TOA :~'
			+ @Site + 'https://sbstweb.sbstransit.com.sg/TAMS/RGSAckSMS.aspx?TARID=' + CAST(@EncTARID AS NVARCHAR(500)) + '&SMSType=2'
	END
