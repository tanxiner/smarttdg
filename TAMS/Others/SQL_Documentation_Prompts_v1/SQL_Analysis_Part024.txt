CREATE   procedure [dbo].[sp_TAMS_Depot_Form_Update_Access_Details]
	@Company		AS NVARCHAR(50) = NULL,
	@Designation	AS NVARCHAR(50) = NULL,
	@Name			AS NVARCHAR(50) = NULL,
	@OfficeNo		AS NVARCHAR(50) = NULL,
	@MobileNo		AS NVARCHAR(50) = NULL,
	@Email			AS NVARCHAR(50) = NULL,
	@AccessTimeFrom AS NVARCHAR(50) = NULL,
	@AccessTimeTo	AS NVARCHAR(50) = NULL,
	@IsExclusive	AS INT= 0,
	@DescOfWork		AS NVARCHAR(100) = NULL,
	@ARRemark		AS NVARCHAR(1000) = NULL,
	@InvolvePower	AS INT = 0,
	@PowerOn		AS INT = 0,
	@Is13ASocket	AS INT = 0,
	@CrossOver		AS INT = 0, 
	@UserID			AS NVARCHAR(20) = NULL,
	@ProtectionType	AS NVARCHAR(50) = NULL,
	@TARID			AS BIGINT = 0,
	@Message		AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

DECLARE @UserIDID	INT = 1;

SELECT @UserIDID = Userid
	FROM TAMS_User
		WHERE LoginId = @UserID
-- >>

UPDATE [dbo].[TAMS_TAR]
SET [Company] = @Company, 
	[Designation] = @Designation, 
	[Name] = @Name, 
	[OfficeNo] = @OfficeNo, 
	[MobileNo] = @MobileNo, 
	[Email] = @Email,
	[AccessTimeFrom] = @AccessTimeFrom, 
	[AccessTimeTo] = @AccessTimeTo,
	[IsExclusive] = @IsExclusive,
	[DescOfWork] = @DescOfWork, 
	[ARRemark] = @ARRemark,
	[InvolvePower] = @InvolvePower, 
	[PowerOn] = @PowerOn, 
	[Is13ASocket] = @Is13ASocket, 
	[CrossOver] = @CrossOver,
	[ProtectionType] = @ProtectionType,
	[UpdatedOn] = GETDATE(), 
	[UpdatedBy] = @UserIDID
WHERE ID = @TARID

-- >>

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR INSERTING INTO TAMS_TAR'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END

CREATE   procedure [dbo].[sp_TAMS_Depot_GetBlockedTarDates]
	@Line AS nvarchar(10) = NULL,
	@AccessDate AS date = NULL
AS
BEGIN
	SELECT ID, Line, BlockDate, BlockReason
	FROM TAMS_Block_TARDate
	WHERE Line = @Line
	AND BlockDate = @AccessDate
	AND IsActive = 1 
	ORDER BY BlockDate ASC
END

CREATE PROCEDURE [dbo].[sp_TAMS_Depot_GetTarByTarId]
	@TarId AS integer = 0
AS
BEGIN

	DECLARE @PowerZone			NVARCHAR(1000)
	DECLARE @SPKSZone			NVARCHAR(1000)
	SET @PowerZone = ''
	SET @SPKSZone = ''

	SELECT @PowerZone = @PowerZone + ISNULL(PowerSector + ', ', '') 
			FROM TAMS_Power_Sector a, TAMS_TAR_Power_Sector b	
				WHERE a.ID = b.PowerSectorId
					AND a.IsActive = 1 
					and b.IsBuffer = 0 
					and b.TARId = @TARID
		GROUP BY PowerSector		

		IF LEN(LTRIM(RTRIM(@PowerZone))) > 0
		BEGIN
			SET @PowerZone = LEFT(LTRIM(RTRIM(@PowerZone)), LEN(LTRIM(RTRIM(@PowerZone))) - 1)
		END

		SELECT @SPKSZone = @SPKSZone + ISNULL(a.SPKSZone + ', ', '') 
			FROM TAMS_SPKSZone a, TAMS_TAR_SPKSZone b	
				WHERE a.ID = b.SPKSZoneId
					AND a.IsActive = 1	
					and b.TARId = @TARID
 		GROUP BY a.SPKSZone		

	IF LEN(LTRIM(RTRIM(@SPKSZone))) > 0
		BEGIN
			SET @SPKSZone = LEFT(LTRIM(RTRIM(@SPKSZone)), LEN(LTRIM(RTRIM(@SPKSZone))) - 1)
		END

	SELECT [Id],
	[Line],
	[TARNo],
	[TARType],
	[Company],
	[Designation],
	[Name],
	[OfficeNo],
	[MobileNo],
	[Email],
	[SubmitDate],
	[AccessDate],
	isnull(AccessTimeSlot,'') as AccessTimeSlot,
	[AccessTimeFrom],
	[AccessTimeTo],
	[AccessLocation],
	[AccessType],
	[IsNeutralGap],
	[IsExclusive],
	[DescOfWork],
	[ARRemark],
	[Is13ASocket],
	[CrossOver],
	[ProtectionType], 
	@SPKSZone as SPKSZone, 
	@PowerZone as PowerZone,
	[ProtectionType], 
	[WithdrawRemark], 
	Convert(varchar,WithdrawDate,103) as WithdrawDate,
	(Select top 1 Name from TAMS_User where Userid = WithdrawBy) as WithdrawBy, 
	TARStatusId, 
	(Select top 1 WFStatus from TAMS_WFStatus where WFStatusId = TARStatusId and TAMS_WFStatus.Line = TAMS_TAR.Line and TAMS_WFStatus.TrackType =TAMS_TAR.TrackType  and WFType='TARWFStatus')  as TARStatus
	
	FROM TAMS_TAR
	WHERE [Id] = @TarId;
END

CREATE procedure  [dbo].[sp_TAMS_Depot_GetTarEnquiryResult_Department]
	@uid AS integer = 0,
	@Line AS nvarchar(50) = NULL,
	@TrackType AS nvarchar(50) = NULL,
	@TarType AS nvarchar(50) = NULL,
	@AccessType AS nvarchar(50) = NULL,
	@TarStatusId AS integer = 0,
	@AccessDateFrom as nvarchar(50) = NULL,
	@AccessDateTo as nvarchar(50) = NULL 
 
AS
BEGIN
	DECLARE  @Dept as nvarchar(50),
			@StatusId as int
	declare @IsAll as int
	declare @IsPower as int
	declare @IsDep as int

	if ( ( Select count(*) 
	From TAMS_User u,TAMS_User_Role ur,TAMS_Role r
	Where u.Userid = ur.UserID and ur.RoleID= r.ID and r.TrackType = @TrackType
	and r.Role in ('NEL_ChiefController','NEL_DCC','NEL_ApplicantHOD' ,'NEL_SDS','NEL_TAPApprover','NEL_TAPHOD','NEL_TAPVerifier' )
	and u.Userid = @uid
	) ) > 0 
	begin
		set @IsAll = 1 ;
	end;

	if ( ( Select count(*) 
	From TAMS_User u,TAMS_User_Role ur,TAMS_Role r
	Where u.Userid = ur.UserID and ur.RoleID= r.ID and r.TrackType = @TrackType
	and r.Role in ('NEL_PowerEndorser' ,'NEL_PowerHOD','NEL_PFR' )
	and u.Userid = @uid
	) ) > 0 
	begin 
		set @IsPower = 1 
	end;

	if ( ( Select count(*) 
	From TAMS_User u,TAMS_User_Role ur,TAMS_Role r
	Where u.Userid = ur.UserID and ur.RoleID= r.ID and r.TrackType = @TrackType
	and r.Role in ('NEL_ApplicantHOD'   )
	and u.Userid = @uid
	) ) > 0 
	begin 
		set @IsDep = 1 
	end;

	DECLARE @cond as varchar(max);

	SET @cond = '';

	 
	IF @Line IS NOT NULL AND @Line != ''  AND @Line != '-1' 
	BEGIN 
		SET @cond = ' AND t.Line='''+@Line+'''';
	END;
	IF @TarType IS NOT NULL AND @TarType != ''  AND @TarType != '-1' 
	BEGIN 
		SET @cond = ' AND t.TarType='''+@TarType+'''';
	END;

	IF @AccessType IS NOT NULL AND @AccessType != ''  AND @AccessType != '-1' 
	BEGIN 
		SET @cond = @cond + ' AND AccessType='''+@AccessType+'''';
	END;

	IF @TarStatusId IS NOT NULL AND @TarStatusId != ''  AND @TarStatusId != '-1' 
	BEGIN 
		SET @cond = @cond + ' AND TarStatusId='''+CAST(@TarStatusId as nvarchar(10))+'''';
	END;

	IF NOT ISNULL(@AccessDateFrom, '') = '' AND NOT @AccessDateFrom = ''
	BEGIN			
		SET @cond = @cond + ' AND AccessDate >= convert(datetime,'''+@AccessDateFrom+''',103)';
	END;

	IF NOT ISNULL(@AccessDateTo, '') = '' AND NOT @AccessDateTo = ''
	BEGIN			
		SET @cond = @cond + ' AND AccessDate <= convert(datetime,'''+@AccessDateTo+''',103)';
	END;	

	 

	DECLARE @sql nvarchar(max);
	 
	SET @sql = 'SELECT ROW_NUMBER() OVER(ORDER BY t.company  ) AS sno,   t.company FROM ( '


	if @IsAll = 1 
	begin 
		SET @sql = @sql + 'SELECT distinct    t.company
				from TAMS_TAR t, TAMS_WFStatus s
				where t.TARStatusId = s.WFStatusId
				and t.Line = s.Line
				and t.TrackType = ''' +  @TrackType +'''
				and s.WFType = ''TARWFStatus''
				 ' + @cond
	end;
	else if @IsPower = 1 
	begin 
		SET @sql = @sql + 'SELECT distinct   t.company
				from TAMS_TAR t, TAMS_WFStatus s
				where t.TARStatusId = s.WFStatusId
				and t.Line = s.Line
				and s.WFType = ''TARWFStatus''
				and (t.InvolvePower = 1 OR t.createdby = ''' + CAST(@uid as nvarchar(10)) +'''
				and t.TrackType = ''' +  @TrackType +'''' + @cond 
	end; 
	else if @IsDep = 1 
	begin 
		SET @sql = @sql + 'SELECT distinct   t.company
				from TAMS_TAR t, TAMS_WFStatus s
				where t.TARStatusId = s.WFStatusId
				and t.Line = s.Line
				and s.WFType = ''TARWFStatus''
				and t.TrackType = ''' +  @TrackType +'''
				and ( t.company in (select TARQueryDept FROM TAMS_User_QueryDept WHERE SUBSTRING(TARQueryDept,1,4) = ''NEL'' AND UserID = ''' + CAST(@uid AS nvarchar(10)) + ''')
				  OR  t.createdby = ''' + CAST(@uid as nvarchar(10)) +''') ' + @cond
	end;
	else 
	begin 
		SET @sql = @sql + 'SELECT distinct   t.company
				from TAMS_TAR t, TAMS_WFStatus s
				where t.TARStatusId = s.WFStatusId
				and t.Line = s.Line
				and t.TrackType = ''' +  @TrackType +'''
				and s.WFType = ''TARWFStatus''
				and t.createdby = ''' + CAST(@uid as nvarchar(10)) +''''  + @cond 
				 
	end;
	---- applicant can view own created tar

	--- PFR, PowerHOD, power_endorser InvolvePower = 1 
	--- applicant hod role can view TAR under own department 
	--	TAP verifier/approver/HOD/scheduler/SDS/CC/TC can see all TAR under own Line 


	  

	SET @sql = @sql + ') as t';
	PRINT (@sql);

	EXEC (@sql);	

END

CREATE PROCEDURE [dbo].[sp_TAMS_Depot_GetTarSectorsByAccessDateAndLine]
	@AccessDate as date,
	@Line AS nvarchar(10) = NULL 
AS

CREATE TABLE #TMP
(
	ID int not null,	
	Line nvarchar(10),	
	SameSector int,
	IsSpecialSector bit,
	Direction nvarchar(10),
	Sector nvarchar(50),
	TarNo nvarchar(50),
	AccessDate date,
	AccessType nvarchar(50),
	SectorId int,
	IsBuffer bit,
	ColourCode nvarchar(50),
	IsGap bit,
	[Order] int

);
IF @Line = 'NEL'
BEGIN
	insert into #TMP(ID, Line, SameSector, IsSpecialSector, Direction, Sector, TarNo, AccessDate, AccessType, SectorId, IsBuffer, ColourCode, IsGap, [Order])
	SELECT a.ID, a.Line, a.SameSector, a.IsSpecialSector, Direction, Sector, TarNo, AccessDate, AccessType, SectorId, IsBuffer, ColourCode, IsGap, [Order]
	FROM TAMS_Sector a
	LEFT JOIN TAMS_TAR_Sector c
	LEFT JOIN TAMS_TAR b
	on b.Id = c.TarId
	on a.id = c.SectorId 
	AND AccessDate = @AccessDate
	AND TARStatusId = 9
	AND (AccessType = 'Possession' OR (AccessType = 'Protection' AND  isExclusive = 1))
	WHERE a.Line = @Line
	AND a.TrackType = 'Depot'
	--AND Direction = @Direction
	AND IsActive = 1 
	AND EffectiveDate <= GETDATE() AND ExpiryDate >= GETDATE()
	ORDER BY [Order] ASC
END

update #TMP
set ColourCode = (select ColourCode from #TMP where SameSector is not null and TarNo is not null and ColourCode is not null)
where SameSector = (select SameSector from #TMP where SameSector is not null and TarNo is not null and ColourCode is not null)


SELECT * FROM #TMP ORDER BY [Order] ASC;

--select ColourCode from #TMP where SameSector is not null and TarNo is not null and ColourCode is not null
DROP TABLE #TMP;

CREATE   procedure [dbo].[sp_TAMS_Depot_GetTarSectorsByTarId]
	@TarId as integer = 0

AS
BEGIN
	SELECT a.ID, a.Line, Direction, Sector, TARNo, AccessDate, AccessType, SectorId, IsBuffer, ColourCode, IsGap
	FROM TAMS_Sector a
	INNER JOIN TAMS_TAR_Sector c
	on a.id = c.SectorId 
	INNER JOIN TAMS_TAR b
	on b.Id = c.TARId
	AND c.TARId = @TarId
	AND c.IsBuffer <> 1
	ORDER BY [Order], Sector ASC
END
