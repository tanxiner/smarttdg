CREATE PROCEDURE  [dbo].[SMSEAlertQ_EnQueue]
(  
  @Sender as nvarchar(100),
  @Subject as nvarchar(500),
  @Sys as nvarchar(100),
  @Greetings as ntext,
  @AlertMsg as ntext,
  @UserId as nvarchar(50),
  @SendTo as ntext,
  @CC as ntext,
  @BCC as ntext,
  @Separator as nvarchar(1),
  @AlertID as decimal(18, 0) output,
  @From as nvarchar(250) = null
)
as
  SET NOCOUNT ON 
  
  declare @ptrval as binary(16)
  declare @startreadpos as int
  declare @endreadpos as int
  declare @fndloc as int
  declare @datalen as int
  declare @emailid as nvarchar(100)
   
  
begin

  BEGIN TRAN

if @SendTo is null
begin
return
end

  -- Status default is Q
  -- Active default is 1
  
  insert into SMSEAlertQ ([From], Sender, Subject, Sys, Greetings, AlertMsg, CreatedOn, CreatedBy, LastUpdatedOn, LastUpdatedBy)
  values (@From, @Sender, @Subject, @Sys, @Greetings, @AlertMsg, getdate(), @UserId, getdate(), @UserId)
  
  set @AlertID = @@Identity   
  
  /*----------------------------------SendTo Start---------------------------------------------*/
  -- insert into EAletQTo
  create table #tsendto (n ntext)
  insert #tsendto values(@SendTo)
  
  -- get the ntext pointer
  SELECT @ptrval = TEXTPTR(n) FROM #tsendto
  SELECT @fndloc  = PATINDEX('%' + @Separator + '%', n) FROM #tsendto
  
  while @fndloc <> 0
  Begin
    --print @fndloc
    set @startreadpos = 0
    set @endreadpos = @fndloc - 1
 
    --print @endreadpos

    --READTEXT #tsendto.n @ptrval @startreadpos @endreadpos
  
    select @emailid = ltrim(rtrim(substring(n, 1, @endreadpos))) from #tsendto
    --print @emailid
    
    insert into SMSEAlertQTo (AlertID, Recipient, CreatedOn, CreatedBy, LastUpdatedOn, LastUpdatedBy)
    values(@AlertID, @emailid, getdate(), @UserId, getdate(), @UserId)
    
    UPDATETEXT #tsendto.n @ptrval @startreadpos @fndloc ''
  
    SELECT @fndloc  = PATINDEX('%' + @Separator + '%', n) FROM #tsendto
  End

  set @datalen = (select DATALENGTH(n) from #tsendto) / 2 -- due ntext 2 bytes/character
  --print @datalen

  if @datalen > 0
  Begin
    set @startreadpos = 0
    --READTEXT #tsendto.n @ptrval @startreadpos @datalen
    select @emailid = ltrim(rtrim(substring(n, 1, @datalen))) from #tsendto
    --print @emailid
    insert into SMSEAlertQTo (AlertID, Recipient, CreatedOn, CreatedBy, LastUpdatedOn, LastUpdatedBy)
    values(@AlertID, @emailid, getdate(), @UserId, getdate(), @UserId)
  End 
  
  drop table #tsendto   
  /*-----------------------------------SendTo End------------------------------------------*/
  
  
  
  /*-----------------------------------CC Start--------------------------------------------*/
  -- insert into EAletQCC
  create table #tcc (n ntext)
  insert #tcc values(@CC)
  
  -- get the ntext pointer
  SELECT @ptrval = TEXTPTR(n) FROM #tcc
  SELECT @fndloc  = PATINDEX('%' + @Separator + '%', n) FROM #tcc
  
  while @fndloc <> 0
  Begin
    --print @fndloc
    set @startreadpos = 0
    set @endreadpos = @fndloc - 1
 
    --print @endreadpos

    --READTEXT #tcc.n @ptrval @startreadpos @endreadpos
  
    select @emailid = ltrim(rtrim(substring(n, 1, @endreadpos))) from #tcc
    --print @emailid
    
    insert into SMSEAlertQCC (AlertID, CC, CreatedOn, CreatedBy, LastUpdatedOn, LastUpdatedBy)
    values(@AlertID, @emailid, getdate(), @UserId, getdate(), @UserId)
    
    UPDATETEXT #tcc.n @ptrval @startreadpos @fndloc ''
  
    SELECT @fndloc  = PATINDEX('%' + @Separator + '%', n) FROM #tcc
  End

  set @datalen = (select DATALENGTH(n) from #tcc) / 2 -- due ntext 2 bytes/character
  --print @datalen

  if @datalen > 0
  Begin
    set @startreadpos = 0
    --READTEXT #tcc.n @ptrval @startreadpos @datalen
    select @emailid = ltrim(rtrim(substring(n, 1, @datalen))) from #tcc
    --print @emailid
    insert into SMSEAlertQCC (AlertID, CC, CreatedOn, CreatedBy, LastUpdatedOn, LastUpdatedBy)
    values(@AlertID, @emailid, getdate(), @UserId, getdate(), @UserId)
  End 
  
  drop table #tcc   
   /*-----------------------------------CC End--------------------------------------------*/
  
  
  
  /*------------------------------------BCC Start-----------------------------------------*/
  -- insert into EAletQBCC
  create table #tbcc (n ntext)
  insert #tbcc values(@BCC)
  
  -- get the ntext pointer
  SELECT @ptrval = TEXTPTR(n) FROM #tbcc
  SELECT @fndloc  = PATINDEX('%' + @Separator + '%', n) FROM #tbcc
  
  while @fndloc <> 0
  Begin
    --print @fndloc
    set @startreadpos = 0
    set @endreadpos = @fndloc - 1
 
    --print @endreadpos

    --READTEXT #tbcc.n @ptrval @startreadpos @endreadpos
  
    select @emailid = ltrim(rtrim(substring(n, 1, @endreadpos))) from #tbcc
    --print @emailid
    
    insert into SMSEAlertQBCC (AlertID, BCC, CreatedOn, CreatedBy, LastUpdatedOn, LastUpdatedBy)
    values(@AlertID, @emailid, getdate(), @UserId, getdate(), @UserId)
    
    UPDATETEXT #tbcc.n @ptrval @startreadpos @fndloc ''
  
    SELECT @fndloc  = PATINDEX('%' + @Separator + '%', n) FROM #tbcc
  End

  set @datalen = (select DATALENGTH(n) from #tbcc) / 2 -- due ntext 2 bytes/character
  --print @datalen

  if @datalen > 0
  Begin
    set @startreadpos = 0
    --READTEXT #tbcc.n @ptrval @startreadpos @datalen
    select @emailid = ltrim(rtrim(substring(n, 1, @datalen))) from #tbcc
    --print @emailid
    insert into SMSEAlertQBCC (AlertID, BCC, CreatedOn, CreatedBy, LastUpdatedOn, LastUpdatedBy)
    values(@AlertID, @emailid, getdate(), @UserId, getdate(), @UserId)
  End 
  
  drop table #tbcc

end

RETURN

CREATE  PROCEDURE [dbo].[SMTP_GET_Email_Lists]
 
AS
--------------------------------------------------------------------------------------------------------------
-- Procedure Name 	: SMTP_GET_Email_Lists 
-- Date Created		: 16/03/2018
-- Purpose			: To Send Email Alert using SMTP
---------------------------------------------------------------------------------------------------------------



SET TRANSACTION ISOLATION LEVEL SERIALIZABLE

DELETE FROM EAlertQBCC
WHERE AlertID IN (SELECT AlertID FROM EAlertQ WHERE AlertMsg IS NULL)

DELETE FROM EAlertQCC
WHERE AlertID IN (SELECT AlertID FROM EAlertQ WHERE AlertMsg IS NULL)

DELETE FROM EAlertQTo
WHERE AlertID IN (SELECT AlertID FROM EAlertQ WHERE AlertMsg IS NULL)

DELETE FROM EAlertQ
WHERE AlertMsg IS NULL

 select A.GREETINGS,A.SUBJECT,
	    A.AlertMsg ,A.SENDER,A.ALERTID,a.SYS  ,
		[dbo].[SMTP_GET_EMAIL_CC_LISTS](A.ALERTID) as CCRecipient,
		[dbo].[SMTP_GET_EMAIL_BCC_LISTS](A.ALERTID) as BCCRecipient,				 
        X.RECIPIENT             
 from EALERTQ A , ( SELECT ALERTID, 
					STUFF((SELECT CAST(',' AS varchar(max)) + LTRIM(RTRIM(RECIPIENT))
					FROM EALERTQTO  as b 
					WHERE b.ALERTID = a.ALERTID 
					AND b.Active=1
					AND LTRIM(rtrim(RECIPIENT)) <> ''
					FOR XML PATH(''), TYPE 
					 ).value('.', 'varchar(max)' 
					  ),1, 1,'') AS RECIPIENT 
					FROM 
					(SELECT DISTINCT ALERTID FROM EALERTQTO ) AS a  
					  
				 ) X
 where A.status='Q'
 and A.Active = 1 
 and A.ALERTID=X.ALERTID
 ORDER BY A.ALERTID;

CREATE PROCEDURE [dbo].[SP_Call_SMTP_Send_SMSAlert]
	 @Message		AS NVARCHAR(500) = NULL OUTPUT
AS
------------------------------------------------------------------------------------------
-- Procedure Name 		: [SP_SMTP_Send_SMSAlert]
-- Date Created			: 31/05/2017
-- Purpose				: Send SMS ALert using SMTP
-- Version   	Date         		 		Changes
-- ----------------------------------------------------------------------------------------
-- 1.0    		 31/05/2017			Initial version
--									 
 -------------------------------------------------------------------------------------------

BEGIN
	 
DECLARE	@IntrnlTrans	TINYINT		
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

SET @Message = ''

DECLARE @p_Alertid INT,
	@p_from VARCHAR(100),
	@p_To VARCHAR(100),
	@p_Alertmsg VARCHAR(MAX),
	@p_sysname VARCHAR(50)
	
DECLARE @CurMain CURSOR
SET @CurMain = CURSOR FAST_FORWARD FOR
	
	SELECT a.AlertID, a.[From], a.AlertMsg, 'SMTP Alert'
	FROM dbo.SMSEAlertQ a 	  
		WHERE a.status = 'Q' 		
	
OPEN @CurMain 
FETCH NEXT FROM @CurMain INTO @p_Alertid,@p_from,@p_Alertmsg,@p_sysname
WHILE (@@FETCH_STATUS = 0 )
	BEGIN 
		
		DECLARE @CurDetl CURSOR
		SET @CurDetl = CURSOR FAST_FORWARD FOR
				 
		SELECT b.Recipient
			FROM dbo.SMSEAlertQTo b
		WHERE b.AlertID=@p_Alertid

		OPEN @CurDetl 
		FETCH NEXT FROM @CurDetl INTO @p_To
		WHILE (@@FETCH_STATUS = 0 )
			BEGIN 
		 
				EXEC [dbo].[SP_SMTP_SMS_NetPage] @p_from, @p_To, @p_Alertmsg, @p_Alertid, @p_sysname
		
				FETCH NEXT FROM @CurDetl INTO @p_To
			END
		CLOSE @CurDetl
		DEALLOCATE @CurDetl
	
		----------------------
		UPDATE SMSEAlertQ 
			SET status='S', 
				LastUpdatedOn = getdate(),
				LastUpdatedBy = @p_sysname
			WHERE AlertID = @p_Alertid
		----------------------
	FETCH NEXT FROM @CurMain INTO  @p_Alertid, @p_from, @p_Alertmsg, @p_sysname
	END

CLOSE @CurMain
DEALLOCATE @CurMain

IF @@ERROR <> 0 BEGIN
	SET @Message = 'Error Sending SMS'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END
