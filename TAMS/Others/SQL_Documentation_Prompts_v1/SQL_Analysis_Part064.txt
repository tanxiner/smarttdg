create PROCEDURE [dbo].[sp_TAMS_Get_ChildMenuByUserRole_20231009] 
	@UserID AS NVARCHAR(100) = null,
	@MenuID AS NVARCHAR(100) = null
AS
BEGIN

	DECLARE @urole AS NVARCHAR(2000);
	DECLARE @rc2 AS INT;
	DECLARE @loop2 AS INT;
	SET @urole = '';

	--To Keep Role ID List	
	CREATE TABLE #RoleTbl
	(
		ROWID [int] IDENTITY(1,1),
		RoleCode VARCHAR(50)
	);	

	INSERT INTO #RoleTbl
	SELECT Distinct(Role) 
	FROM [TAMS_User_Role] ur , [TAMS_User] u, [TAMS_Role] r
	WHERE  ur.UserID = u.UserID
	AND r.ID = ur.RoleID
	AND u.LoginID = @UserID
	--AND u.userid IN (SELECT UserID FROM [TAMS_User] WHERE LoginID = @UserID AND IsActive = 1);


	SELECT  @urole = @urole + '''' +  RoleCode  + '''' + ','
	FROM #RoleTbl;

    SET @urole = LEFT(@urole, NULLIF(LEN(@urole) - 1, -1));
    PRINT('role: ' + @urole); 
		
	IF @urole IS NOT NULL AND @urole != '' 
	  BEGIN
		DECLARE @rsql AS NVARCHAR(2000);
		SET @rsql = 'SELECT MenuID, MenuLevel, DispText,ToolTip, ProgURL, ParentMenuID '+
		'FROM TAMS_Menu ' +
		'WHERE Active = 1  ' +
		'AND MenuLevel = 2  ' +
		'AND ParentMenuID = ''' + @MenuID +''' ' +
		'AND MenuID IN(	 ' +	
		'SELECT DISTINCT(MenuID) ' +
		'FROM TAMS_Menu_Role ' +
		'WHERE Role IN (' + @urole + '))' +
		'AND ModuleID IN(	 ' +	
		'SELECT DISTINCT(ModuleID) ' +
		'FROM TAMS_Menu_Role ' +
		'WHERE Role IN (' + @urole + ')) ORDER BY ID';
			
		PRINT @rsql;
		EXEC (@rsql);
		
		DROP TABLE #RoleTbl;
	  END;
	  ELSE 
	  BEGIN 
			SELECT MenuID, MenuLevel, DispText, ToolTip, ProgURL, ParentMenuID 
				FROM Menu WHERE Active =1 AND MenuLevel = 2
				AND ParentMenuID = @MenuID
				AND MenuID IN ('Application','Apply TRF','Logout')
	  END; 
    
END

CREATE procedure  [dbo].[sp_TAMS_Get_Depot_TarEnquiryResult_Header]
	@uid AS integer = 0,
	@Line AS nvarchar(50) = NULL,
	@TrackType AS nvarchar(50) = NULL,
	@TarType AS nvarchar(50) = NULL,
	@AccessType AS nvarchar(50) = NULL,
	@TarStatusId AS integer = 0,
	@AccessDateFrom as nvarchar(50) = NULL,
	@AccessDateTo as nvarchar(50) = NULL , 
	@Department as nvarchar(50) = NULL
 
AS
BEGIN
	DECLARE  @Dept as nvarchar(50),
			@StatusId as int
	declare @IsAll as int
	declare @IsPower as int
	declare @IsDep as int

	if ( ( Select count(*) 
	From TAMS_User u,TAMS_User_Role ur,TAMS_Role r
	Where u.Userid = ur.UserID and ur.RoleID= r.ID and r.TrackType = @TrackType
	and r.Role in ('NEL_ApplicantHOD' ,'NEL_SDS','NEL_TAPApprover','NEL_TAPHOD','NEL_TAPVerifier' )
	and u.Userid = @uid
	) ) > 0 
	begin
		set @IsAll = 1 ;
	end;

	if ( ( Select count(*) 
	From TAMS_User u,TAMS_User_Role ur,TAMS_Role r
	Where u.Userid = ur.UserID and ur.RoleID= r.ID and r.TrackType = @TrackType
	and r.Role in ('NEL_PowerEndorser' ,'NEL_PowerHOD','NEL_PFR' )
	and u.Userid = @uid
	) ) > 0 
	begin 
		set @IsPower = 1 
	end;

	if ( ( Select count(*) 
	From TAMS_User u,TAMS_User_Role ur,TAMS_Role r
	Where u.Userid = ur.UserID and ur.RoleID= r.ID and r.TrackType = @TrackType
	and r.Role in ('NEL_ApplicantHOD'   )
	and u.Userid = @uid
	) ) > 0 
	begin 
		set @IsDep = 1 
	end;

	DECLARE @cond as varchar(max);

	SET @cond = '';

	 
	IF @Line IS NOT NULL AND @Line != ''  AND @Line != '-1' 
	BEGIN 
		SET @cond = ' AND t.Line='''+@Line+'''';
	END;
	IF @TarType IS NOT NULL AND @TarType != ''  AND @TarType != '-1' 
	BEGIN 
		SET @cond = ' AND t.TarType='''+@TarType+'''';
	END;

	IF @AccessType IS NOT NULL AND @AccessType != ''  AND @AccessType != '-1' 
	BEGIN 
		SET @cond = @cond + ' AND AccessType='''+@AccessType+'''';
	END;

	IF @TarStatusId IS NOT NULL AND @TarStatusId != ''  AND @TarStatusId != '-1' 
	BEGIN 
		SET @cond = @cond + ' AND TarStatusId='''+CAST(@TarStatusId as nvarchar(10))+'''';
	END;

	IF NOT ISNULL(@AccessDateFrom, '') = '' AND NOT @AccessDateFrom = ''
	BEGIN			
		SET @cond = @cond + ' AND AccessDate >= convert(datetime,'''+@AccessDateFrom+''',103)';
	END;

	IF NOT ISNULL(@AccessDateTo, '') = '' AND NOT @AccessDateTo = ''
	BEGIN			
		SET @cond = @cond + ' AND AccessDate <= convert(datetime,'''+@AccessDateTo+''',103)';
	END;	

	IF NOT ISNULL(@Department, '') = '' AND NOT @Department = ''
	BEGIN			
		SET @cond = @cond + ' AND Company = ''' + @Department + '''';
	END;

	DECLARE @sql nvarchar(max);
	 
	SET @sql = 'SELECT ROW_NUMBER() OVER(ORDER BY t.tarno desc) AS sno, t.id, t.line, t.tarno, t.tartype, t.accesstype, t.accessdate,t.AccessTimeSlot, tarstatus, t.company FROM ( '


	if @IsAll = 1 
	begin 
		SET @sql = @sql + 'SELECT distinct t.id, t.line, t.tarno, t.tartype, t.accesstype, t.accessdate,t.AccessTimeSlot, s.wfstatus as tarstatus, t.company
				from TAMS_TAR t, TAMS_WFStatus s
				where t.TARStatusId = s.WFStatusId
				and t.Line = s.Line
				and t.TrackType = ''' +  @TrackType +'''
				and s.WFType = ''TARWFStatus''
				 ' + @cond
	end;
	else if @IsPower = 1 
	begin 
		SET @sql = @sql + 'SELECT distinct t.id, t.line, t.tarno, t.tartype, t.accesstype, t.accessdate,t.AccessTimeSlot, s.wfstatus as tarstatus, t.company
				from TAMS_TAR t, TAMS_WFStatus s
				where t.TARStatusId = s.WFStatusId
				and t.Line = s.Line
				and s.WFType = ''TARWFStatus''
				and (t.InvolvePower = 1 OR t.createdby = ''' + CAST(@uid as nvarchar(10)) +'''
				and t.TrackType = ''' +  @TrackType +'''' + @cond 
	end; 
	else if @IsDep = 1 
	begin 
		SET @sql = @sql + 'SELECT distinct t.id, t.line, t.tarno, t.tartype, t.accesstype, t.accessdate,t.AccessTimeSlot, s.wfstatus as tarstatus, t.company
				from TAMS_TAR t, TAMS_WFStatus s
				where t.TARStatusId = s.WFStatusId
				and t.Line = s.Line
				and s.WFType = ''TARWFStatus''
				and t.TrackType = ''' +  @TrackType +'''
				and ( t.company in (select TARQueryDept FROM TAMS_User_QueryDept WHERE SUBSTRING(TARQueryDept,1,4) = ''NEL'' AND UserID = ''' + CAST(@uid AS nvarchar(10)) + ''')
				  OR  t.createdby = ''' + CAST(@uid as nvarchar(10)) +''') ' + @cond
	end;
	else 
	begin 
		SET @sql = @sql + 'SELECT distinct t.id, t.line, t.tarno, t.tartype, t.accesstype, t.accessdate,t.AccessTimeSlot, s.wfstatus as tarstatus, t.company
				from TAMS_TAR t, TAMS_WFStatus s
				where t.TARStatusId = s.WFStatusId
				and t.Line = s.Line
				and t.TrackType = ''' +  @TrackType +'''
				and s.WFType = ''TARWFStatus''
				and t.createdby = ''' + CAST(@uid as nvarchar(10)) +''''  + @cond 
				 
	end;
	---- applicant can view own created tar

	--- PFR, PowerHOD, power_endorser InvolvePower = 1 
	--- applicant hod role can view TAR under own department 
	--	TAP verifier/approver/HOD/scheduler/SDS/CC/TC can see all TAR under own Line 


	  

	SET @sql = @sql + ') as t';
	PRINT (@sql);

	EXEC (@sql);	

END

CREATE PROCEDURE [dbo].[sp_TAMS_Get_ParaValByParaCode]
	@paraCode AS NVARCHAR(200) = NULL,
	@paraValue1 AS NVARCHAR(200) = NULL
AS
BEGIN

	SELECT * FROM TAMS_Parameters
	WHERE ParaCode = @paraCode
	AND EffectiveDate <= GETDATE()
	AND ExpiryDate >= GETDATE()
	AND ParaValue1 LIKE @paraValue1
	ORDER BY ParaValue1, ParaValue2, [Order]
END

CREATE PROCEDURE [dbo].[sp_TAMS_Get_ParentMenuByUserRole] 
	@UserID AS NVARCHAR(100) = null,
	@IsInternet AS NVARCHAR(1) = null
AS
BEGIN
	DECLARE @rsql AS NVARCHAR(2000);

	DECLARE @usql AS NVARCHAR(2000);
	SET @usql ='SELECT DISTINCT Role ' +
	'FROM [TAMS_User_Role] ur, [TAMS_User] u, [TAMS_Role] r ' +
	'WHERE  ur.UserID = u.UserID ' + 
	'AND ur.RoleID = r.ID ' +
	'AND u.LoginID = '+ '''' +  @UserID   + '''' + 
	'AND u.isactive = 1';

	print (@usql);


	IF EXISTS(SELECT u.UserID FROM TAMS_User_Role ur, TAMS_User u WHERE  ur.UserID = u.UserID AND u.LoginID = @UserID AND u.IsActive = 1 )
	BEGIN 

		IF @IsInternet = 0
		BEGIN
			SET @rsql ='SELECT MenuID, MenuLevel, DispText, ToolTip, ProgURL, ParentMenuID ' +
			'FROM TAMS_Menu ' +
			'WHERE Active = 1 ' + 
			'AND MenuLevel = 1 ' + 
			'AND (IsInternet = 0 OR IsInternet = 1)' +
			'AND DispText IN (	 ' +	
			'SELECT DISTINCT(menuID) ' +
			'FROM TAMS_Menu_Role ' +
			'WHERE Role in (' + @usql + ')) ORDER BY ID';
			PRINT @rsql;
			EXEC (@rsql);
		END;
		ELSE IF @IsInternet = 1
		BEGIN
			SET @rsql ='SELECT MenuID, MenuLevel, DispText, ToolTip, ProgURL, ParentMenuID ' +
			'FROM TAMS_Menu ' +
			'WHERE Active = 1 ' + 
			'AND MenuLevel = 1 ' + 
			'AND IsInternet = 1' +
			'AND DispText IN (	 ' +	
			'SELECT DISTINCT(menuID) ' +
			'FROM TAMS_Menu_Role ' +
			'WHERE Role in (' + @usql + ')) ORDER BY ID';
			PRINT @rsql;
			EXEC (@rsql);
		END;
    END;
	ELSE 
	BEGIN 
		SELECT MenuID, MenuLevel, DispText, ToolTip, ProgURL, ParentMenuID 
		FROM TAMS_Menu WHERE Active = 1 AND MenuLevel = 1 
		AND MenuID IN ('Application','Apply TRF','Logout')
	END; 
END;
