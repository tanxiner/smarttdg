CREATE PROCEDURE [dbo].[sp_TAMS_Update_UserRegModule_ApplicantRegisterCompany] 	
	@RegID			INT,
	@Company		NVARCHAR(200),
	@UENNo			NVARCHAR(200),
	@BizOwner		NVARCHAR(200),
	@OfficeTel		NVARCHAR(20),
	@Mobile			NVARCHAR(20),
	@Email			NVARCHAR(200)
AS
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION;

			CREATE TABLE #TMP_RegModule
			(
				ID INT,
				RegId INT,
				Line NVARCHAR(10),
				TrackType NVARCHAR(50),
				Module VARCHAR(50),
				RegStatus INT
			);

			IF EXISTS(SELECT * FROM TAMS_Reg_Module WHERE RegID = @RegID And RegStatus = 1) 
			BEGIN
				-- Update Company Details in TAMS_Registration
				UPDATE TAMS_Registration
				SET UENNo = @UENNo,
				Company = @Company, 
				BizOwner = @BizOwner, 
				CompanyOfficeNo = @OfficeTel,
				CompanyMobileNo = @Mobile,
				CompanyEmail = @Email
				WHERE ID = @RegID;

				DECLARE @ID INT;
				DECLARE @Line NVARCHAR(20);
				DECLARE @TrackType NVARCHAR(50);
				DECLARE @Module NVARCHAR(20);
				DECLARE @RegStatus INT;
				DECLARE @RegModID INT;

				INSERT INTO #TMP_RegModule(ID, RegId, Line,TrackType, Module, RegStatus)   
				SELECT ID, RegId, Line,TrackType, Module, RegStatus FROM TAMS_Reg_Module WHERE WFStatus = 'Pending' AND RegID = @RegID


				DECLARE cur CURSOR FOR
				SELECT ID, RegId, Line,TrackType, Module, RegStatus FROM #TMP_RegModule WHERE RegID = @RegID

				OPEN cur;
				FETCH NEXT FROM cur INTO @ID, @RegModID, @Line, @TrackType, @Module, @RegStatus;

				WHILE @@FETCH_STATUS = 0
				BEGIN
					
					DECLARE @WorkflowID INT;
					DECLARE @NewWFStatusID INT;
					DECLARE @NewWFRoleID INT;
					DECLARE @EndorserID		INT;
					DECLARE @WFStatus       NVARCHAR(50);


					-- Get next stageID for this TAMS_Reg_Module
					SELECT @WorkflowID = ID 
					FROM TAMS_Workflow
					WHERE Line = @Line 
					AND TrackType = @TrackType
					AND WorkFlowType = 'ExtUser'
					AND GETDATE() >= EffectiveDate 
					AND GETDATE() <= ExpiryDate 
					AND IsActive = 1
		
					SELECT @NewWFStatusID = WFStatusID, @NewWFRoleID = RoleID, @EndorserID = ID
					FROM TAMS_Endorser 
					WHERE WorkflowID = @WorkflowID
					AND Line = @Line
					AND GETDATE() >= EffectiveDate
					AND GETDATE() <= ExpiryDate
					AND IsActive = 1
					AND [Level] = 
					(
						SELECT [Level] + 1 
						FROM TAMS_Endorser
						WHERE WorkflowID = @WorkflowID
						AND Line = @Line
						AND WFStatusID = @RegStatus
						AND GETDATE() >= EffectiveDate
						AND GETDATE() <= ExpiryDate
						AND IsActive = 1
					)

					SELECT @WFStatus = WFStatus FROM TAMS_WFStatus
					WHERE Line = @Line
					AND TrackType = @TrackType
					AND WFType = 'UserRegStatus'
					AND WFStatusId = @NewWFStatusID
					AND IsActive = 1

					SELECT  @RegModID, @RegID, @Line,@TrackType, @Module, @NewWFStatusID, @WorkflowID, @EndorserID, 'Pending', @WFStatus, GETDATE(), 1, GETDATE(), 1


					INSERT INTO TAMS_Reg_Module 
					VALUES (@RegID, @Line,@TrackType, @Module, @NewWFStatusID, @WorkflowID, @EndorserID, 'Pending', @WFStatus, GETDATE(), 1, GETDATE(), 1);

					UPDATE TAMS_Reg_Module
					SET WFStatus = 'Approved',
					UpdatedOn = GETDATE()
					WHERE ID = @ID;

					-- Sends email
					DECLARE @Sender NVARCHAR (100);
					DECLARE @SysID NVARCHAR(10);
					DECLARE @Subject NVARCHAR(500);    
					DECLARE @Sys NVARCHAR(100);
					DECLARE @Greetings  NVARCHAR(20);     
					DECLARE @ToList NVARCHAR(MAX);
					DECLARE @CCList NVARCHAR(MAX);
					DECLARE @BCCList NVARCHAR(MAX);
					DECLARE @Separator NVARCHAR(10);        
					DECLARE @AlertMsg NVARCHAR(MAX);
					DECLARE @AlertID AS INTEGER; 
					SET @ToList = '';

					DECLARE @Body VARCHAR(max);
					DECLARE @Body1 VARCHAR(max);
					DECLARE @Body2 VARCHAR(max);
					DECLARE @Body3 VARCHAR(max);
					DECLARE @Link NVARCHAR(MAX);

					DECLARE @LoginPage	NVARCHAR(500)
					
					DECLARE @SendTo	NVARCHAR(200);	

					-- Get Send to list
					DECLARE curSendTo CURSOR FOR
					(
						SELECT Email FROM TAMS_User u, TAMS_User_Role ur
						WHERE u.UserID = ur.UserID
						AND ur.RoleID IN (@NewWFRoleID
						--SELECT ID FROM TAMS_Role WHERE Line = @Line AND Module = @Module AND Role LIKE '%SysAdmin%'
						)
					)

					OPEN curSendTo;
					FETCH NEXT FROM curSendTo INTO @SendTo;

					WHILE @@FETCH_STATUS = 0
					BEGIN
						IF LEN(@ToList) =0 
						BEGIN
							SET @ToList = @SendTo
						END

						SET @ToList = @ToList + ',' + @SendTo;
						
						FETCH NEXT FROM curSendTo INTO @SendTo;
					END;
				

					CLOSE curSendTo;
					DEALLOCATE curSendTo;


					SET @Sender = 'TAMS Admin';
					SET @SysID = 'TAMS';
					SET @Subject = 'For your approval: Company registration approval '
					SET @Sys = 'Track Access Management System'   
					SET @Greetings = 'Dear Sir/Madam, '  
					--SET @ToList = ''
					SET @CCList = ''   
					SET @BCCList = ''
					SET @Separator = ','    
					SET @CCList = ''
					--SET @ToList = '';

					SET @Body1 = ''
					SET @Body2 = ''
					SET @Body3 = ''
			
					SET @Body1 = @Body1 + '<table width="100%">'
					SET @Body1 = @Body1 + '<tr align="left" valign="middle" style="font-size:10pt; font-face:Arial;">'
					SET @Body1 = @Body1 + '<td>' + @Company + ' (' + @UENNo + ') has been registered. Please access TAMS to approve/reject the registration.</td>'
					SET @Body1 = @Body1 + '</tr>'
					SET @Body1 = @Body1 + '</table>'
					--SET @LoginPage = 'https://pulsewebuat.sbstransit.com.sg/TAMS/Login.aspx'
					SET @LoginPage = 'https://sbstweb.sbstransit.com.sg/TAMS/Login.aspx'
					 
					SET @Body2 = @Body2 + '<br>'
					SET @Body2 = @Body2 + '<table width="100%">'
					SET @Body2 = @Body2 + '<tr align="left" valign="middle" style="font-size:10pt; font-face:Arial;">'
					SET @Body2 = @Body2 + '<td>Click on the link <a href=' + @LoginPage + '>here</a> to access tams.</td>'
					SET @Body2 = @Body2 + '</tr>'
					SET @Body2 = @Body2 + '</table>'

					SET @Body3 = @Body3 + '<br>'
					SET @Body3 = @Body3 + '<table width="100%">'
					SET @Body3 = @Body3 + '<tr align="left" valign="middle" style="font-size:10pt; font-face:Arial;">'
					SET @Body3 = @Body3 + '<td>***********************************************<br>Do Not reply. This is a computer generated Email. <br>***********************************************</td>'
					SET @Body3 = @Body3 + '</tr>'
					SET @Body3 = @Body3 + '</table>'

					SET @Body = ''
					SET @Body = @Body1 + @Body2 + @Body3

					SET @BCCList = ''

					EXEC [dbo].EAlertQ_EnQueue 
						@Sender = @Sender, 
						@UserId = @SysID, 
						@Subject = @Subject,        
						@Sys = @Sys,  
						@Greetings = @Greetings, 
						@AlertMsg = @Body, 
						@SendTo = @ToList,        
						@CC = @CCList, 
						@BCC= @BCCList, 
						@Separator = @Separator, 
						@AlertID =  @AlertID

						SELECT @ToList

					FETCH NEXT FROM cur INTO @ID, @RegModID, @Line,@TrackType, @Module, @RegStatus;
					
				END


				CLOSE cur;
				DEALLOCATE cur;

			END;

		-- Insert Audit Log
		INSERT INTO TAMS_Action_Log VALUES 
		(
			'All', 
			'All', 
			'UserReg-Registration', 
			@RegID, 
			'External User has registered company details for this User Registration', 
			GETDATE(), 
			1
		)

		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		ROLLBACK TRANSACTION;
	END CATCH

END;
