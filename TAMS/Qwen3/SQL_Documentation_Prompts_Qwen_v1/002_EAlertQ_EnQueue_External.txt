CREATE   procedure [dbo].[EAlertQ_EnQueue_External]
(
	@From		nvarchar(250),
	@Sender		nvarchar(100),
	@Subject	nvarchar(500),
	@Sys		nvarchar(100),
	@Greetings	ntext,
	@AlertMsg	ntext,
	@UserId		nvarchar(50),
	@SendTo		ntext,
	@CC		ntext,
	@BCC		ntext,
	@Attachment	nvarchar(500),
	@Separator	nvarchar(1),
	@AlertID	decimal(18, 0)	output
)
as
	SET NOCOUNT ON
  
	declare @ptrval		binary(16)
	declare @startreadpos	int
	declare @endreadpos	int
	declare @fndloc		int
	declare @datalen	int
	declare @emailid	nvarchar(100)
	declare @CurrDt		datetime

BEGIN
	BEGIN TRAN

	-- Status default is Q
	-- Active default is 1

	SET @CurrDt = getdate()

	insert into EAlertQ
	( [From], Sender, Subject, Sys, Greetings, AlertMsg, CreatedOn, CreatedBy, LastUpdatedOn, LastUpdatedBy )
	values
	( @From, @Sender, @Subject, @Sys, @Greetings, @AlertMsg, @CurrDt, @UserId, @CurrDt, @UserId )

	set @AlertID = @@Identity

	IF @AlertID <> 0
	BEGIN
		INSERT EAlertQAtt
		( ALERTID, FPATH, ACTIVE, CREATEDON, CREATEDBY, LASTUPDATEDON, LASTUPDATEDBY )
		VALUES
		( @AlertID, @Attachment, '1', @CurrDt, @UserId, @CurrDt, @UserId )
	END

	/*----------------------------------SendTo Start---------------------------------------------*/
	-- insert into EAletQTo
	create table #tsendto (n ntext)
	insert #tsendto values(@SendTo)

	-- get the ntext pointer
	SELECT @ptrval = TEXTPTR(n) FROM #tsendto
	SELECT @fndloc  = PATINDEX('%' + @Separator + '%', n) FROM #tsendto

	while @fndloc <> 0
	Begin
		set @startreadpos = 0
		set @endreadpos = @fndloc - 1
		select @emailid = ltrim(rtrim(substring(n, 1, @endreadpos))) from #tsendto

		insert into EAlertQTo 
		( AlertID, Recipient, CreatedOn, CreatedBy, LastUpdatedOn, LastUpdatedBy )
		values
		( @AlertID, @emailid, @CurrDt, @UserId, @CurrDt, @UserId )

		UPDATETEXT #tsendto.n @ptrval @startreadpos @fndloc ''

		SELECT @fndloc  = PATINDEX('%' + @Separator + '%', n) FROM #tsendto
	End

	set @datalen = ( select DATALENGTH(n) from #tsendto ) / 2 -- due ntext 2 bytes/character

	if @datalen > 0
	Begin
		set @startreadpos = 0
		--READTEXT #tsendto.n @ptrval @startreadpos @datalen
		select @emailid = ltrim(rtrim(substring(n, 1, @datalen))) from #tsendto

		insert into EAlertQTo
		( AlertID, Recipient, CreatedOn, CreatedBy, LastUpdatedOn, LastUpdatedBy )
		values
		( @AlertID, @emailid, @CurrDt, @UserId, @CurrDt, @UserId )
	End

	drop table #tsendto
	/*-----------------------------------SendTo End------------------------------------------*/




	/*-----------------------------------CC Start--------------------------------------------*/
	-- insert into EAletQCC
	create table #tcc (n ntext)
	insert #tcc values(@CC)

	-- get the ntext pointer
	SELECT @ptrval = TEXTPTR(n) FROM #tcc
	SELECT @fndloc  = PATINDEX('%' + @Separator + '%', n) FROM #tcc

	while @fndloc <> 0
	Begin
		set @startreadpos = 0
		set @endreadpos = @fndloc - 1

		select @emailid = ltrim(rtrim(substring(n, 1, @endreadpos))) from #tcc

		insert into EAlertQCC
		( AlertID, CC, CreatedOn, CreatedBy, LastUpdatedOn, LastUpdatedBy )
		values
		( @AlertID, @emailid, @CurrDt, @UserId, @CurrDt, @UserId )

		UPDATETEXT #tcc.n @ptrval @startreadpos @fndloc ''

		SELECT @fndloc  = PATINDEX('%' + @Separator + '%', n) FROM #tcc
	End

	set @datalen = (select DATALENGTH(n) from #tcc) / 2 -- due ntext 2 bytes/character

	if @datalen > 0
	Begin
		set @startreadpos = 0
		--READTEXT #tcc.n @ptrval @startreadpos @datalen
		select @emailid = ltrim(rtrim(substring(n, 1, @datalen))) from #tcc

		insert into EAlertQCC
		( AlertID, CC, CreatedOn, CreatedBy, LastUpdatedOn, LastUpdatedBy )
		values
		( @AlertID, @emailid, @CurrDt, @UserId, @CurrDt, @UserId )
	End

	drop table #tcc
	/*-----------------------------------CC End--------------------------------------------*/



	/*------------------------------------BCC Start-----------------------------------------*/
	-- insert into EAletQBCC
	create table #tbcc (n ntext)
	insert #tbcc values(@BCC)

	-- get the ntext pointer
	SELECT @ptrval = TEXTPTR(n) FROM #tbcc
	SELECT @fndloc  = PATINDEX('%' + @Separator + '%', n) FROM #tbcc

	while @fndloc <> 0
	Begin
		set @startreadpos = 0
		set @endreadpos = @fndloc - 1

		select @emailid = ltrim(rtrim(substring(n, 1, @endreadpos))) from #tbcc

		insert into EAlertQBCC
		( AlertID, BCC, CreatedOn, CreatedBy, LastUpdatedOn, LastUpdatedBy )
		values
		( @AlertID, @emailid, @CurrDt, @UserId, @CurrDt, @UserId )
    
		UPDATETEXT #tbcc.n @ptrval @startreadpos @fndloc ''

		SELECT @fndloc  = PATINDEX('%' + @Separator + '%', n) FROM #tbcc
	End

	set @datalen = (select DATALENGTH(n) from #tbcc) / 2 -- due ntext 2 bytes/character

	if @datalen > 0
	Begin
		set @startreadpos = 0
		--READTEXT #tbcc.n @ptrval @startreadpos @datalen
		select @emailid = ltrim(rtrim(substring(n, 1, @datalen))) from #tbcc

		insert into EAlertQBCC
		( AlertID, BCC, CreatedOn, CreatedBy, LastUpdatedOn, LastUpdatedBy )
		values
		( @AlertID, @emailid, getdate(), @UserId, getdate(), @UserId )
	End

	drop table #tbcc   

	/*------------------------------------BCC End-----------------------------------------*/

	COMMIT TRAN
END

RETURN
