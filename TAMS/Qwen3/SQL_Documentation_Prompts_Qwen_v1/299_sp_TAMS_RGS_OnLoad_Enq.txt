### ⬇️ RAW SQL INPUT ⬇️
CREATE PROCEDURE  [dbo].[sp_TAMS_RGS_OnLoad_Enq]
	@Line	NVARCHAR(20) = NULL,
	@TrackType nvarchar(50) = null,
	@OPDate	NVARCHAR(20) = NULL
AS

BEGIN
	
/*
Notes :
OCCAuthStatusId 
DTL = 10
NEL = 7
*/

CREATE TABLE #TmpRGS 
(Sno				INT,
TARNo				NVARCHAR(30),
ElectricalSections	NVARCHAR(4000),
PowerOffTime		NVARCHAR(20),
CircuitBreakOutTime	NVARCHAR(20),
PartiesName			NVARCHAR(4000),
NoOfPersons			NVARCHAR(20),
WorkDescription		NVARCHAR(4000),
ContactNo			NVARCHAR(50),
TOANo				NVARCHAR(30),
CallBackTime		NVARCHAR(20),
RadioMsgTime		NVARCHAR(20),
LineClearMsgTime	NVARCHAR(20),
Remarks				NVARCHAR(4000),
TOAStatus			NVARCHAR(20),
IsTOAAuth			BIT,
ColourCode			NVARCHAR(20),
IsGrantTOAEnable	BIT,
UpdQTSTime			NVARCHAR(20),
AccessType			NVARCHAR(20),
AckGrantTOATime		NVARCHAR(20),
AckProtLimitTime	NVARCHAR(20),
TARID				INT,
TOAID				INT,
InchargeNRIC		NVARCHAR(100))

CREATE TABLE #TmpRGSSectors 
(OCCAuthID			INT,
OperationDate		DATE,
AccessDate			DATE,
OCCAuthStatus		NVARCHAR(50),
TractionPowerID		INT,
TractionPowerType	NVARCHAR(50),
SectorID			INT,
Sector				NVARCHAR(100),
WorkFlowTime		DATETIME)

TRUNCATE TABLE #TmpRGS
TRUNCATE TABLE #TmpRGSSectors

DECLARE @CurDate		DATE
DECLARE @CurTime		TIME
DECLARE @OperationDate	DATE
DECLARE @AccessDate		DATE

SET @CurDate = GETDATE()
SET @CurTime = CONVERT(TIME, GETDATE(), 108)

DECLARE @CutOff				TIME
DECLARE @TOACallBackTime	NVARCHAR(10)
DECLARE @RGSPossBG			NVARCHAR(20)
DECLARE @RGSProtBG			NVARCHAR(20)
DECLARE @RGSCancBG			NVARCHAR(20)

--SET @TOACallBackTime = '04:00'
SET @CutOff = '06:00:00'

SELECT @TOACallBackTime = CAST(ParaTime AS NVARCHAR(5))
	FROM TAMS_Parameters
		WHERE ParaCode = 'TOACallbackTime'
			AND ParaValue1 = @Line
			AND ParaValue2 = @TrackType

SELECT @RGSPossBG = ParaValue2 
	FROM TAMS_Parameters
		WHERE ParaCode = 'RGSPossessionBG'
			AND ParaValue1 = @Line

SELECT @RGSProtBG = ParaValue2 
	FROM TAMS_Parameters
		WHERE ParaCode = 'RGSProtectionBG'
			AND ParaValue1 = @Line

SELECT @RGSCancBG = ParaValue2 
	FROM TAMS_Parameters
		WHERE ParaCode = 'RGSCancelledBG'
			AND ParaValue1 = @Line

IF @CurTime > @CutOff
	BEGIN
		SET @OperationDate	= @CurDate
		SET @AccessDate		= DATEADD(DAY, 1, @CurDate)
	END
ELSE
	BEGIN
		SET @OperationDate	= DATEADD(DAY, -1, @CurDate)
		SET @AccessDate		= @CurDate
	END

SET @OperationDate = CONVERT(DATE, @OPDate, 103)
SET @AccessDate = DATEADD(DAY, 1, CONVERT(DATE, @OPDate, 103))

if UPPER(@TrackType)='DEPOT' BEGIN
	SET @AccessDate = @OperationDate
END

--SELECT @OperationDate, @AccessDate

DECLARE @Cur01 CURSOR;
DECLARE @Cur02 CURSOR;

DECLARE @TARID BIGINT,
@TOAID	BIGINT,
@ARRemark NVARCHAR(1000),
@TVFMode NVARCHAR(20),
@AccessType NVARCHAR(30),
@TOAStatus INT,
@ProtTimeLimit NVARCHAR(10),
@TARNo NVARCHAR(30),
@NoOfParties INT,
@DescOfWork NVARCHAR(1000),
@MobileNo NVARCHAR(20),
@TetraRadioNo NVARCHAR(50),
@TOANo NVARCHAR(50),
@GrantTOATime NVARCHAR(10),
@AckSurrenderTime NVARCHAR(10),
@AckGrantTOATime NVARCHAR(10),
@UpdateQTSTime NVARCHAR(10),
@InchargeNRIC NVARCHAR(100), 
@CancelRemark NVARCHAR(1000)

DECLARE @lv_Sno INT,
	@lv_IsTOAAuth BIT,
	@lv_PartiesName	NVARCHAR(4000),
	@lv_Remarks	NVARCHAR(4000),
	@lv_IsPowerOff BIT,
	@lv_PowerOffTime NVARCHAR(20),
	@lv_IsCircuitBreak BIT,
	@lv_CircuitBreakTime NVARCHAR(20),
	@lv_ColourCode NVARCHAR(20),
	@lv_ES NVARCHAR(100),
	@lv_SectorID INT,
	@lv_TVFStations NVARCHAR(1000),
	@lv_PossessionCtr INT,
	@lv_IsGrantTOAEnable BIT,
	@lv_ContactNo NVARCHAR(200), 
	@lv_Callbacktime datetime

DECLARE @SqlCtr1 INT,
	@SqlCtr2 INT,
	@SqlCtr3 INT,
	@WorkFlowTime NVARCHAR(20)

SET @lv_Sno = 0
SET @SqlCtr1 = 0
SET @SqlCtr2 = 0
SET @SqlCtr3 = 0
SET @lv_PossessionCtr = 0

SELECT @lv_PossessionCtr = COUNT(*)
	FROM TAMS_TAR a, TAMS_TOA b
		WHERE a.Id = b.TARId
			AND a.AccessDate = @AccessDate
			AND b.TOAStatus NOT IN (0, 6)
			AND a.Line = @Line
			and a.TrackType = @TrackType
			AND a.AccessType = 'Possession'

DECLARE @NewLine AS NVARCHAR(6)
SET @NewLine = '<br />'

SET @Cur01 = CURSOR FAST_FORWARD 
FOR

SELECT a.Id, b.Id, a.TARNo, a.ARRemark, a.TVFMode, a.AccessType, b.TOAStatus, 
		-- >> ISNULL(CONVERT(NVARCHAR(20), b.AckProtectionLimitTime, 108), '00:00:00'), 
		ISNULL(CONVERT(NVARCHAR(5), b.AckProtectionLimitTime, 108), ''), 
		b.NoOfParties,
		a.DescOfWork, b.MobileNo, b.TetraRadioNo, ISNULL(b.TOANo, ''),
		ISNULL(CONVERT(NVARCHAR(5), b.GrantTOATime, 108), ''),
		ISNULL(CONVERT(NVARCHAR(5), b.AckSurrenderTime, 108), ''),
		ISNULL(CONVERT(NVARCHAR(8), b.AckGrantTOATime, 108), ''),
		ISNULL(CONVERT(NVARCHAR(5), b.UpdateQTSTime, 108), ''),
		dbo.DecryptString(b.InChargeNRIC),
		b.CancelRemark,b.TOACallbackTime
	FROM TAMS_TAR a, TAMS_TOA b
		WHERE a.Id = b.TARId
			AND a.AccessDate = @AccessDate
			AND b.TOAStatus <> 0
			AND a.Line = @Line
			and a.TrackType = @TrackType
	-- >> ORDER BY a.AccessType, b.TOAType, a.Id
	ORDER BY a.AccessType, b.TOAStatus DESC, a.Id

OPEN @Cur01
	
FETCH NEXT FROM @Cur01 INTO @TARID, @TOAID, @TARNo, @ARRemark, @TVFMode, @AccessType, @TOAStatus, 
	@ProtTimeLimit, 
	@NoOfParties,
	@DescOfWork, @MobileNo, @TetraRadioNo, @TOANo,
	@GrantTOATime, @AckSurrenderTime, @AckGrantTOATime, @UpdateQTSTime,
	@InchargeNRIC, @CancelRemark,@lv_Callbacktime
	
WHILE @@FETCH_STATUS = 0 
	
	BEGIN

		SET @lv_Sno = @lv_Sno + 1
		SET	@lv_IsTOAAuth = 0
		SET @lv_PartiesName	= ''
		SET @lv_Remarks	= ''
		SET @lv_IsPowerOff = 0
		SET @lv_PowerOffTime = ''
		SET @lv_IsCircuitBreak = 0
		SET @lv_CircuitBreakTime = ''
		SET @lv_ColourCode = ''
		SET @lv_ES = ''
		SET @lv_ContactNo = ''

		SELECT @lv_ES = dbo.TAMS_Get_ES_NoBufferZone(@TARID)
		SELECT @lv_PartiesName = dbo.TAMS_Get_TOA_Parties(@TOAID)
		SET @lv_ContactNo = @MobileNo + @NewLine + @TetraRadioNo

		-- >> Electrical Sector Cursor
		-- >> 	EXEC sp_TAMS_RGS_OnLoad_20210819_M 'NEL'

		IF @Line = 'DTL'
			BEGIN

			set @lv_CircuitBreakTime = '';
				set @lv_IsTOAAuth = 0;
				set @lv_PowerOffTime = '';
				set @lv_IsPowerOff = 0 ;

			if exists(SELECT a.ID
							FROM TAMS_OCC_Auth a 
								WHERE  a.Line = @Line
									and a.TrackType = @TrackType
									AND a.OperationDate = @OperationDate
									AND a.AccessDate = @AccessDate)
			begin
				if(
					SELECT Count(a.ID)
									FROM TAMS_OCC_Auth a, TAMS_Traction_Power_Detail b 
										WHERE a.TractionPowerId = b.TractionPowerId
											AND b.SectorId in (
												SELECT a.SectorId
											FROM TAMS_TAR_Sector a, TAMS_Sector b
												WHERE a.SectorId = b.ID
													AND a.TARId = @TARID
													AND ISNULL(b.IsGap, 0) = 0
													AND a.IsBuffer = 0
											)
											AND b.TractionPowerType = 'Sector'
											AND a.OperationDate = @OperationDate
											AND a.AccessDate = @AccessDate
											 
											AND ((a.OCCAuthStatusId < 10 and a.IsBuffer = 0) OR ((a.OCCAuthStatusId < 7 and a.IsBuffer = 1)))
											 
								) = 0 
								begin 
									set @lv_IsTOAAuth = 1;
								end;
				end;


				-- To check whether PFR has done action for all the sector ; Power OFF type is NULL means they has not completed the action
				if not exists(
					select * 
					FROM TAMS_OCC_Auth a, TAMS_Traction_Power_Detail b 
						WHERE a.TractionPowerId = b.TractionPowerId
							 AND b.SectorId in (
												SELECT a.SectorId
											FROM TAMS_TAR_Sector a, TAMS_Sector b
												WHERE a.SectorId = b.ID
													AND a.TARId = @TARID
													AND ISNULL(b.IsGap, 0) = 0
													AND a.IsBuffer = 0
											)
							AND b.TractionPowerType = 'Sector'
							AND a.OperationDate = @OperationDate
							AND a.AccessDate = @AccessDate
							 AND a.PowerOffType IS NULL  
						)
					begin
							SELECT  @lv_PowerOffTime = ISNULL(CONVERT(NVARCHAR(5), max(a.PowerOffTime), 108), '')
					FROM TAMS_OCC_Auth a, TAMS_Traction_Power_Detail b 
						WHERE a.TractionPowerId = b.TractionPowerId
							 AND b.SectorId in (
												SELECT a.SectorId
											FROM TAMS_TAR_Sector a, TAMS_Sector b
												WHERE a.SectorId = b.ID
													AND a.TARId = @TARID
													AND ISNULL(b.IsGap, 0) = 0
													AND a.IsBuffer = 0
											)
							AND b.TractionPowerType = 'Sector'
							AND a.OperationDate = @OperationDate
							AND a.AccessDate = @AccessDate
							 AND a.PowerOffTime IS NOT NULL and a.PowerOffType='Completed'
				 	
					end;		 
				if @lv_PowerOffTime <> ''
				begin 
					set @lv_IsPowerOff = 1 ;
				end; 
				
				-- To check whether PFR has done action for all the sector ; Rack Out type is NULL means they has not completed the action
				if not exists(
					select *
					FROM TAMS_OCC_Auth a, TAMS_Traction_Power_Detail b 
								WHERE a.TractionPowerId = b.TractionPowerId
									 AND b.SectorId in (
														SELECT a.SectorId
													FROM TAMS_TAR_Sector a, TAMS_Sector b
														WHERE a.SectorId = b.ID
															AND a.TARId = @TARID
															AND ISNULL(b.IsGap, 0) = 0
															AND a.IsBuffer = 0
													)
									AND b.TractionPowerType = 'Sector'
									AND a.OperationDate = @OperationDate
									AND a.AccessDate = @AccessDate
									 AND a.RackedOutType IS  NULL 

				)
				begin 
						SELECT  @lv_CircuitBreakTime = ISNULL(CONVERT(NVARCHAR(5), max(a.RackedOutTime), 108), '')
							FROM TAMS_OCC_Auth a, TAMS_Traction_Power_Detail b 
								WHERE a.TractionPowerId = b.TractionPowerId
									 AND b.SectorId in (
														SELECT a.SectorId
													FROM TAMS_TAR_Sector a, TAMS_Sector b
														WHERE a.SectorId = b.ID
															AND a.TARId = @TARID
															AND ISNULL(b.IsGap, 0) = 0
															AND a.IsBuffer = 0
													)
									AND b.TractionPowerType = 'Sector'
									AND a.OperationDate = @OperationDate
									AND a.AccessDate = @AccessDate
									 AND a.RackedOutTime IS NOT NULL and a.RackedOutType='Completed'
 
				end;
 			
			END
		ELSE
			BEGIN
			 	-----For NEL Checking .. 

				set @lv_CircuitBreakTime = '';
				set @lv_IsTOAAuth = 0;
				set @lv_PowerOffTime = '';
				set @lv_IsPowerOff = 0 ;


				if exists(select a.ID FROM TAMS_OCC_Auth a, TAMS_Traction_Power_Detail b 
					WHERE a.TractionPowerId = b.TractionPowerId
						AND b.PowerSectionId in (
							select PowerSectorId from TAMS_TAR_Power_Sector where TARId = @TARID
						)
						AND b.TractionPowerType = 'Sector'
						AND a.OperationDate = @OperationDate
						AND a.AccessDate = @AccessDate)
				begin
				 	if(SELECT COUNT(a.ID)
					FROM TAMS_OCC_Auth a, TAMS_Traction_Power_Detail b 
						WHERE a.TractionPowerId = b.TractionPowerId
							AND b.PowerSectionId in (
								select PowerSectorId from TAMS_TAR_Power_Sector where TARId = @TARID
							)
							AND b.TractionPowerType = 'Sector'
							AND a.OperationDate = @OperationDate
							AND a.AccessDate = @AccessDate
							 AND a.OCCAuthStatusId < 8)= 0 
					begin 
						set @lv_IsTOAAuth = 1 ; 
					end;
				end;


			-- To check whether PFR has done action for all the sector ; Power OFF type is NULL means they has not completed the action
			if not exists(
				select * 
				FROM TAMS_OCC_Auth a, TAMS_Traction_Power_Detail b 
					WHERE a.TractionPowerId = b.TractionPowerId
						AND b.PowerSectionId in (
							select PowerSectorId from TAMS_TAR_Power_Sector where TARId = @TARID
						)
						AND b.TractionPowerType = 'Sector'
						AND a.OperationDate = @OperationDate
						AND a.AccessDate = @AccessDate
						 AND a.PowerOffType is   null   
			)
			begin
				SELECT  @lv_PowerOffTime = ISNULL(CONVERT(NVARCHAR(5), max(a.PowerOffTime), 108), '')
				FROM TAMS_OCC_Auth a, TAMS_Traction_Power_Detail b 
					WHERE a.TractionPowerId = b.TractionPowerId
						AND b.PowerSectionId in (
							select PowerSectorId from TAMS_TAR_Power_Sector where TARId = @TARID
						)
						AND b.TractionPowerType = 'Sector'
						AND a.OperationDate = @OperationDate
						AND a.AccessDate = @AccessDate
						 AND a.PowerOffTime is not null  and a.PowerOffType = 'Completed' 
				end;		
				
				if @lv_PowerOffTime <> ''
				begin 
					set @lv_IsPowerOff = 1 ;
				end; 

				-- To check whether PFR has done action for all the sector ; Power OFF type is NULL means they has not completed the action
			if not exists(
				select *
				FROM TAMS_OCC_Auth a, TAMS_Traction_Power_Detail b 
					WHERE a.TractionPowerId = b.TractionPowerId
						AND b.PowerSectionId in (
							select PowerSectorId from TAMS_TAR_Power_Sector where TARId = @TARID
						)
						AND b.TractionPowerType = 'Sector'
						AND a.OperationDate = @OperationDate
						AND a.AccessDate = @AccessDate
						 AND a.RackedOuttype is   null 
			)
			begin
				SELECT  @lv_CircuitBreakTime = ISNULL(CONVERT(NVARCHAR(5), max(a.RackedOutTime), 108), '')
				FROM TAMS_OCC_Auth a, TAMS_Traction_Power_Detail b 
					WHERE a.TractionPowerId = b.TractionPowerId
						AND b.PowerSectionId in (
							select PowerSectorId from TAMS_TAR_Power_Sector where TARId = @TARID
						)
						AND b.TractionPowerType = 'Sector'
						AND a.OperationDate = @OperationDate
						AND a.AccessDate = @AccessDate
						 AND a.RackedOutTime is not null and a.RackedOutType = 'Completed' 
				print @TARID; 	 
				print @lv_CircuitBreakTime;
			end;
			END
		-- >>
		-- >> Electrical Sector Cursor

		IF @Line = 'DTL'
			BEGIN
				SET @lv_Remarks = LTRIM(RTRIM(@ARRemark)) + @NewLine + LTRIM(RTRIM(ISNULL(@TVFMode, '')))
			END
		ELSE
			BEGIN

					-- DTL also will follow the same rule as NEL (meaning if TAR is without Rackout, RGS screen show blank in Rackout column)
		
				DECLARE @RackOutCtr	INT

				SELECT @RackOutCtr = COUNT(*) 
					FROM TAMS_TAR_AccessReq a, TAMS_Access_Requirement b
						WHERE a.IsPower = 1
							AND a.OperationRequirement = b.ID
							AND b.ID = 27	-- >> 27 FOR NEL
							AND a.IsSelected = 1
							AND a.TARId = @TARID

				IF @RackOutCtr > 0
					BEGIN
						SET @lv_Remarks = 'Rack Out' + @NewLine + @NewLine + LTRIM(RTRIM(@ARRemark)) + @NewLine + LTRIM(RTRIM(ISNULL(@TVFMode, '')))
					END
				ELSE
					BEGIN
						-- >> Don't show RackedOutTime for NEL if Power Requirement is with out Racked Out
						SET @lv_CircuitBreakTime = ''
						SET @lv_Remarks = LTRIM(RTRIM(@ARRemark)) + @NewLine + LTRIM(RTRIM(ISNULL(@TVFMode, '')))
					END
			 END

		DECLARE @SCDCtr	INT

		SELECT @SCDCtr = COUNT(*) 
			FROM TAMS_TAR_AccessReq a, TAMS_Access_Requirement b
				WHERE -- >> a.IsPower = 1 AND  -- >> 07082023 To remove this condition Jeffrey
					a.OperationRequirement = b.ID
					AND b.OperationRequirement = 'SCD Application'
					AND a.IsSelected = 1
					AND a.TARId = @TARID

		IF @SCDCtr > 0
			BEGIN
				SET @lv_Remarks = 'SCD' + @NewLine + @NewLine + LTRIM(RTRIM(@lv_Remarks))
			END

		SELECT @lv_TVFStations = dbo.TAMS_Get_TOA_TVF_Stations(@TOAID)

		IF @lv_Remarks = ''
			BEGIN
				SET @lv_Remarks = @lv_TVFStations
			END
		ELSE
			BEGIN
				SET @lv_Remarks = @lv_Remarks + @NewLine + @lv_TVFStations
			END

		 
		IF @Line = 'DTL'	-- >> DTL to continue with Possession Priority Rule. 
			BEGIN
				IF @AccessType = 'Possession'
					BEGIN
						SET @lv_ColourCode = @RGSPossBG

						IF @TOAStatus = 6
							BEGIN
								IF @lv_PossessionCtr = 0
									BEGIN
										SET @lv_PossessionCtr = @lv_PossessionCtr
									END
								ELSE
									BEGIN
										SET @lv_PossessionCtr = @lv_PossessionCtr - 1
									END
							END
						ELSE
							BEGIN
								IF @ProtTimeLimit = '' OR @ProtTimeLimit = '00:00:00'
									BEGIN
										-- >> SELECT '1'
										SET @lv_PossessionCtr = @lv_PossessionCtr + 1
										-- >> SET @lv_PossessionCtr = @lv_PossessionCtr
									END
								ELSE
									BEGIN
										-- >> SELECT '2'
										IF @lv_PossessionCtr = 0
											BEGIN
												-- >> SELECT '3'
												SET @lv_PossessionCtr = @lv_PossessionCtr
											END
										ELSE
											BEGIN
												-- >> SELECT '4'
												SET @lv_PossessionCtr = @lv_PossessionCtr - 1
											END
									END
							END

						SET @lv_IsGrantTOAEnable = 1

					END
				ELSE
					BEGIN
						-- >> SELECT @lv_PossessionCtr AS PossCtr
						-- 	EXEC sp_TAMS_RGS_OnLoad 'DTL'
						SET @lv_ColourCode = @RGSProtBG

						IF @TOAStatus = 6
							BEGIN
								SET @lv_IsGrantTOAEnable = 1
							END
						ELSE
							BEGIN
								IF @lv_PossessionCtr > 0
									BEGIN
										SET @lv_IsGrantTOAEnable = 0
									END
								ELSE
									BEGIN
										SET @lv_IsGrantTOAEnable = 1
									END
							END
					END
			END
		ELSE	-- >> NEL to change with no Priority
			BEGIN
				IF @AccessType = 'Possession'
					BEGIN
						SET @lv_ColourCode = @RGSPossBG
					END
				ELSE
					BEGIN
						SET @lv_ColourCode = @RGSProtBG
					END

				SET @lv_IsGrantTOAEnable = 1
			END

		-- >> 2023 change highlight for cancelled records
		-- >> Include Cancel remarks after new line
		IF @TOAStatus = 6
			BEGIN
				SET @lv_ColourCode = @RGSCancBG
				SET @lv_Remarks = @lv_Remarks + @NewLine + @NewLine + 'Cancel Remark: ' + LTRIM(RTRIM(@CancelRemark))
			END

		INSERT INTO #TmpRGS
			(Sno, TARNo, ElectricalSections,
				PowerOffTime, CircuitBreakOutTime,
				PartiesName, NoOfPersons, 
				WorkDescription, ContactNo, TOANo,
				CallBackTime, 
				RadioMsgTime, LineClearMsgTime,
				Remarks, TOAStatus, IsTOAAuth, ColourCode, IsGrantTOAEnable, 
				UpdQTSTime, AccessType,
				AckGrantTOATime, AckProtLimitTime, 
				TARID, TOAID, InchargeNRIC)
		VALUES (@lv_Sno, @TARNo, @lv_ES, 
			@lv_PowerOffTime, @lv_CircuitBreakTime, 
			@lv_PartiesName, @NoOfParties, 
			@DescOfWork, @lv_ContactNo, @TOANo, 
			CASE WHEN @TrackType = 'Depot' THEN format(@lv_Callbacktime,'HH:mm') when @TOANo = '' THEN '' ELSE @TOACallBackTime END, 
			@GrantTOATime, @AckSurrenderTime, 
			@lv_Remarks, @TOAStatus, @lv_IsTOAAuth, @lv_ColourCode, @lv_IsGrantTOAEnable, 
			@UpdateQTSTime, @AccessType, 
			@AckGrantTOATime, @ProtTimeLimit, 
			@TARID, @TOAID, @InchargeNRIC)

		FETCH NEXT FROM @Cur01 INTO @TARID, @TOAID, @TARNo, @ARRemark, @TVFMode, @AccessType, @TOAStatus, 
			@ProtTimeLimit, 
			@NoOfParties,
			@DescOfWork, @MobileNo, @TetraRadioNo, @TOANo,
			@GrantTOATime, @AckSurrenderTime, @AckGrantTOATime, @UpdateQTSTime,
			@InchargeNRIC, @CancelRemark,@lv_Callbacktime
	END 	

CLOSE @Cur01
	
DEALLOCATE @Cur01


-- >> RGS List
SELECT Sno, TARNo, ElectricalSections,
	PowerOffTime, CircuitBreakOutTime,
	PartiesName, NoOfPersons, 
	WorkDescription, ContactNo, TOANo,
	CallBackTime, RadioMsgTime, LineClearMsgTime,
	Remarks, TOAStatus, IsTOAAuth, ColourCode, IsGrantTOAEnable, 
	UpdQTSTime, AccessType,
	AckGrantTOATime, AckProtLimitTime, 
	TARID, TOAID,
