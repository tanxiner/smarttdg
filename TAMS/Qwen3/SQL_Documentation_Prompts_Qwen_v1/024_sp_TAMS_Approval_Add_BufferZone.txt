CREATE PROCEDURE [dbo].[sp_TAMS_Approval_Add_BufferZone]
	@TARID		AS BIGINT = 0,
	@SectorID	AS BIGINT = 0,
	@Message	AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT

SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

IF (SELECT COUNT(*) 
		FROM [dbo].[TAMS_TAR_Sector]
			WHERE TARId = @TARID
				AND SectorId = @SectorID) = 0
	BEGIN

		DECLARE @Line	NVARCHAR(20)

		SELECT @Line = Line
			FROM TAMS_TAR
				WHERE Id = @TARID

		-- >> TAMS_TAR_Sector
		DECLARE @SecColourCode	NVARCHAR(20)

		SET @SecColourCode = ''

		SELECT @SecColourCode = ColourCode 
			FROM TAMS_Type_Of_Work
				WHERE Line = @Line
					AND TypeOfWork = 'Buffer Zone'
					AND IsActive = 1

		INSERT INTO [dbo].[TAMS_TAR_Sector]
			([TARId], [SectorId], [IsBuffer], [ColourCode], [IsAddedBuffer])
		SELECT @TARID, ID, 1, @SecColourCode, 1
			FROM TAMS_Sector
				WHERE ID = @SectorID;
		-- >> TAMS_TAR_Sector

	END

-- >>

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR INSERTING INTO TAMS_TAR'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message


END
