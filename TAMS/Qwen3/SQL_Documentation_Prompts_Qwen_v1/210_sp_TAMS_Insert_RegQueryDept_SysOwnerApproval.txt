CREATE PROCEDURE [dbo].[sp_TAMS_Insert_RegQueryDept_SysOwnerApproval] 	
	@RegModID			INT,
	@RegRoleID			INT,
	@Dept				NVARCHAR(200),
	@UpdatedBy			INT
AS
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION;

		DECLARE @UserID INT;
		SELECT @UserID = UserID FROM TAMS_User WHERE LoginID = (SELECT LoginID FROM TAMS_Registration WHERE ID = 
		(SELECT RegID FROM TAMS_Reg_Module WHERE ID = @RegModID))

		INSERT INTO TAMS_Reg_QueryDept 
		VALUES (@RegModID, @RegRoleID, @Dept, GETDATE(), @UpdatedBy, GETDATE(), @UpdatedBy);

		IF NOT EXISTS(SELECT * FROM TAMS_User_QueryDept WHERE UserID = @UserID AND RoleID = @RegRoleID AND 
		TARQueryDept = @Dept)
		BEGIN
			INSERT INTO TAMS_User_QueryDept 
			VALUES (@UserID, @RegRoleID, @Dept, GETDATE(), @UpdatedBy, GETDATE(), @UpdatedBy);
		END;

		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		ROLLBACK TRANSACTION;
	END CATCH

END;
