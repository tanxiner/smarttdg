CREATE PROCEDURE [dbo].[sp_TAMS_Get_CompanyInfo_by_ID] 
	@CompanyID NVARCHAR(100) = NULL
AS
BEGIN

	IF EXISTS (SELECT * FROM TAMS_Company WHERE ID = @CompanyID)
	BEGIN
		SELECT * 
		FROM TAMS_Company c
		WHERE c.ID = @CompanyID

	END
	ELSE
	BEGIN
		SELECT * from TAMS_Company where 1=2
	END
END
