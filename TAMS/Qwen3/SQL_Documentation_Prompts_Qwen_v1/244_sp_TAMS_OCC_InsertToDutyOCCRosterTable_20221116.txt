CREATE PROCEDURE [dbo].[sp_TAMS_OCC_InsertToDutyOCCRosterTable_20221116]
	@TAMS_OCC_DutyRoster [dbo].[TAMS_OCC_DutyRoster] READONLY
AS
BEGIN

	declare @count as int = 0;
	declare @loop as int = 0;
	declare @operationdate as date;
	declare @shift as nvarchar(1);
	declare @line as nvarchar(10);

	select @operationdate = operationdate, @shift = [shift], @line = [line] from @TAMS_OCC_DutyRoster where 1=1;

	select @count = count(*) from TAMS_OCC_Duty_Roster where operationdate = @operationdate and [shift] = @shift and [line] = @line;

	if @count = 0
	begin
		INSERT INTO TAMS_OCC_Duty_Roster(Line,
		OperationDate,
		[Shift],
		RosterCode,
		DutyStaffId,
		IsActive,
		CreatedOn,
		CreatedBy,
		UpdatedOn,
		UpdatedBy
		)
		SELECT Line,
		OperationDate,
		[Shift],
		RosterCode,
		DutyStaffId,
		IsActive,
		CreatedOn,
		CreatedBy,
		UpdatedOn,
		UpdatedBy
		FROM @TAMS_OCC_DutyRoster

		INSERT INTO TAMS_OCC_Duty_Roster_Audit(
		ActionBy,
		ActionOn,
		AuditAction,
		DutyRosterId,
		Line,
		OperationDate,
		[Shift],
		RosterCode,
		DutyStaffId,
		IsActive,
		CreatedOn,
		CreatedBy,
		UpdatedOn,
		UpdatedBy
		)
		SELECT 
		CreatedBy,
		UpdatedOn,
		'I',
		DutyStaffId,
		Line,
		OperationDate,
		[Shift],
		RosterCode,
		DutyStaffId,
		IsActive,
		CreatedOn,
		CreatedBy,
		UpdatedOn,
		UpdatedBy
		FROM @TAMS_OCC_DutyRoster
	end;
	else
	begin
		update TAMS_OCC_Duty_Roster
		set DutyStaffId = dr.DutyStaffId,
		UpdatedOn = dr.UpdatedOn,
		UpdatedBy = dr.UpdatedBy
		from  @TAMS_OCC_DutyRoster dr
		where TAMS_OCC_Duty_Roster.ID = dr.ID
		and TAMS_OCC_Duty_Roster.Line = dr.Line
		and TAMS_OCC_Duty_Roster.OperationDate = dr.operationdate
		and TAMS_OCC_Duty_Roster.[shift] = dr.[shift]
		and TAMS_OCC_Duty_Roster.RosterCode = dr.RosterCode

		INSERT INTO TAMS_OCC_Duty_Roster_Audit(
		ActionBy,
		ActionOn,
		AuditAction,
		DutyRosterId,
		Line,
		OperationDate,
		[Shift],
		RosterCode,
		DutyStaffId,
		IsActive,
		CreatedOn,
		CreatedBy,
		UpdatedOn,
		UpdatedBy
		)
		SELECT 
		CreatedBy,
		UpdatedOn,
		'U',
		DutyStaffId,
		Line,
		OperationDate,
		[Shift],
		RosterCode,
		DutyStaffId,
		IsActive,
		CreatedOn,
		CreatedBy,
		UpdatedOn,
		UpdatedBy
		FROM @TAMS_OCC_DutyRoster
	end;
END
