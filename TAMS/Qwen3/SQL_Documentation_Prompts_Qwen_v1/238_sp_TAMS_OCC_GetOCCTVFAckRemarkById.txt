CREATE PROCEDURE [dbo].[sp_TAMS_OCC_GetOCCTVFAckRemarkById]
	@ID AS INT = 0
AS
DECLARE @TVFMode as nvarchar(50),
		@TVFDirection as nvarchar(10)
BEGIN

	SELECT ID, 
	TVFAckId, 
	Remark, 
	tvf.CreatedOn, 
	[Name] AS CreatedBy,
	tvf.UpdatedOn,
	[Name] AS UpdatedBy
	FROM TAMS_TVF_Ack_Remark tvf, TAMS_User u
	WHERE TVFAckId = @ID
	AND tvf.CreatedBy = u.Userid

END
