CREATE PROCEDURE  [dbo].[sp_TAMS_SectorBooking_OnLoad]
	@Line		AS NVARCHAR(10) = NULL,
	@TrackType	AS NVARCHAR(50) = NULL,
	@AccessDate	AS NVARCHAR(20) = NULL,
	@TARType	AS NVARCHAR(20) = NULL,
	@AccessType	AS NVARCHAR(20) = NULL	
AS

BEGIN

	CREATE TABLE #ListES
	(Line				NVARCHAR(10),
	Direction			NVARCHAR(10),
	SectorID			BIGINT,
	Sector				NVARCHAR(50),
	EntryStation		NVARCHAR(50),
	CrossSector			NVARCHAR(50),
	IsSelected			BIT,
	IsEnabled			BIT,
	ColorCode			NVARCHAR(50),
	IsGap				BIT,
	IsEndSector			BIT,
	OrderID				INT,
	DirID				INT,
	IsChkPair			BIT)
	
	TRUNCATE TABLE #ListES

	DECLARE @CurID					BIGINT
	DECLARE @CurLine				NVARCHAR(10)
	DECLARE @CurDirection			NVARCHAR(10)
	DECLARE @CurSector				NVARCHAR(50)
	DECLARE @CurIsGap				BIT
	DECLARE @CurIsEndSector			BIT
	DECLARE @CurOrderID				INT

	DECLARE @EntryStation	NVARCHAR(200)
	SET @EntryStation = ''

	DECLARE @ColorCode		NVARCHAR(50)
	SET @ColorCode = ''

	DECLARE @CCAccessType	NVARCHAR(50)
	SET @CCAccessType = ''

	-- >> Insert into temp table
	IF @Line = 'DTL'
		BEGIN
			INSERT INTO #ListES (Line, Direction, SectorID, Sector, IsGap, IsEndSector, OrderID, IsEnabled, DirID, IsChkPair)
			SELECT a.Line, a.Direction, a.ID, a.Sector, ISNULL(a.IsGap, 0), a.IsEndSector, a.[Order], 1, CASE WHEN Direction = 'BB' THEN 1 ELSE 2 END, 1
				FROM TAMS_Sector a
					WHERE a.Line = @Line
					and TrackType = @TrackType
						AND a.IsActive = 1 
							AND (a.EffectiveDate <= GETDATE() AND a.ExpiryDate >= GETDATE())
				ORDER BY a.[Order] ASC
		END
	ELSE IF @Line = 'NEL'
		BEGIN
			INSERT INTO #ListES (Line, Direction, SectorID, Sector, IsGap, IsEndSector, OrderID, IsEnabled, DirID, IsChkPair)
			SELECT a.Line, a.Direction, a.ID, a.Sector, ISNULL(a.IsGap, 0), a.IsEndSector, a.[Order], 1, CASE WHEN Direction = 'NB' THEN 1 ELSE 2 END, CASE WHEN Sector IN ('801_RT1', '803_RT3', '802_RT2') THEN 0 ELSE 1 END
				FROM TAMS_Sector a
					WHERE a.Line = @Line
					and TrackType = @TrackType
						AND a.IsActive = 1 
							AND (a.EffectiveDate <= GETDATE() AND a.ExpiryDate >= GETDATE())
				ORDER BY a.[Order] ASC
		END

		DECLARE @Cur01 CURSOR;

		SET @Cur01 = CURSOR FAST_FORWARD 
		FOR

			SELECT a.ID, a.Line, a.Direction, a.Sector, a.IsGap, a.IsEndSector, a.[Order]
				FROM TAMS_Sector a
					WHERE a.Line = @Line
					and TrackType = @TrackType
						AND a.IsActive = 1 
						AND (a.EffectiveDate <= GETDATE() AND a.ExpiryDate >= GETDATE())
			ORDER BY a.[Order] ASC

		OPEN @Cur01
	
		FETCH NEXT FROM @Cur01 INTO @CurID, @CurLine, @CurDirection, @CurSector, @CurIsGap, @CurIsEndSector, @CurOrderID
	
		WHILE @@FETCH_STATUS = 0 
	
			BEGIN

				SET @EntryStation = ''
				SET @ColorCode = ''

				SELECT @EntryStation = @EntryStation + ISNULL(StationCode + ',', '') 
					FROM TAMS_Station a, TAMS_Entry_Station b	
						WHERE a.ID = b.StationId
							AND a.IsActive = 1
							AND b.IsActive = 1
							AND a.IsStation = 1
							AND b.SectorId = @CurID
				GROUP BY StationCode		

				IF LEN(LTRIM(RTRIM(@EntryStation))) > 0
					BEGIN
						SET @EntryStation = LEFT(LTRIM(RTRIM(@EntryStation)), LEN(LTRIM(RTRIM(@EntryStation))) - 1)
					END
				
				SELECT @ColorCode = ISNULL(b.ColourCode, ''),
						@CCAccessType = ISNULL(a.AccessType, '') 
					FROM TAMS_TAR a, TAMS_TAR_Sector b
						WHERE a.Id = b.TARId
							AND a.AccessDate = CONVERT(DATETIME, @AccessDate, 103)
							AND ( (a.Line = 'NEL' and a.TARStatusId = 9 ) or (a.Line = 'DTL' and a.TARStatusId = 8)) -- >> Approved
							AND a.IsExclusive = 1 
							AND b.SectorId = @CurID 
							AND a.Line = @Line
							and TrackType = @TrackType
				
				IF @ColorCode = ''
					BEGIN
						UPDATE #ListES
							SET IsEnabled = 1,
								ColorCode = '',
								EntryStation = @EntryStation
						WHERE SectorID = @CurID
					END
				ELSE
					BEGIN
						UPDATE #ListES
							SET ColorCode = @ColorCode,
								EntryStation = @EntryStation
						WHERE SectorID = @CurID

						IF (@TARType = '2' OR @TARType = '3') 
								AND @AccessType = 'Protection' 
								AND @ColorCode = ''
								AND @CCAccessType = 'Possession'
							BEGIN
								UPDATE #ListES
									SET IsEnabled = 1,
										ColorCode = '',
										EntryStation = @EntryStation
								WHERE SectorID = @CurID
							END
						ELSE
							BEGIN
								UPDATE #ListES
									SET IsEnabled = 0,
										EntryStation = @EntryStation
								WHERE SectorID = @CurID
							END
					END

				FETCH NEXT FROM @Cur01 INTO @CurID, @CurLine, @CurDirection, @CurSector, @CurIsGap, @CurIsEndSector, @CurOrderID
			END 	

		CLOSE @Cur01
	
		DEALLOCATE @Cur01
	
	SELECT Line,
		Direction,
		SectorID,
		Sector,
		EntryStation,
		CrossSector,
		IsSelected,
		IsEnabled,
		ColorCode,
		IsGap,
		IsEndSector,
		OrderID,
		DirID,
		IsChkPair
	FROM #ListES
		WHERE DirID = 1
		ORDER BY OrderID, SectorID

	SELECT Line,
		Direction,
		SectorID,
		Sector,
		EntryStation,
		CrossSector,
		IsSelected,
		IsEnabled,
		ColorCode,
		IsGap,
		IsEndSector,
		OrderID,
		DirID,
		IsChkPair
	FROM #ListES
		WHERE DirID = 2
		ORDER BY OrderID, SectorID

/*
	EXEC sp_TAMS_SectorBooking_OnLoad 'DTL', '01/02/2021', '2', 'Possession'
	EXEC sp_TAMS_SectorBooking_OnLoad 'DTL', '01/02/2021', '2', 'Protection'
	EXEC sp_TAMS_SectorBooking_OnLoad 'DTL', '01/02/2021', '1', 'Protection'
	EXEC sp_TAMS_SectorBooking_OnLoad 'NEL', '01/02/2021', '1', 'Protection'
*/

	IF @AccessType = 'Protection'
		BEGIN
			SELECT ID, OperationRequirement 
				FROM TAMS_Access_Requirement
					WHERE Line = @Line
					and TrackType = @TrackType
						AND IsPowerReq = 1
						AND OperationRequirement <> 'Traction Power ON'
						AND IsActive = 1
					ORDER BY [Order]
		END
	ELSE
		BEGIN
			SELECT ID, OperationRequirement 
				FROM TAMS_Access_Requirement
					WHERE Line = @Line
					and TrackType = @TrackType
						AND IsPowerReq = 1
						AND IsActive = 1
					ORDER BY [Order]
		END

END
