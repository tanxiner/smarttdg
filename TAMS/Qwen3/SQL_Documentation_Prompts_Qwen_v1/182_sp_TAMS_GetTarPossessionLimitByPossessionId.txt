CREATE PROCEDURE [dbo].[sp_TAMS_GetTarPossessionLimitByPossessionId]
	@PossessionId AS integer = 0
AS
BEGIN
  select id, possessionid, typeofprotectionlimit, redflashinglampsloc
  from TAMS_Possession_Limit 
  where possessionid = @PossessionId
  order by id asc
END
