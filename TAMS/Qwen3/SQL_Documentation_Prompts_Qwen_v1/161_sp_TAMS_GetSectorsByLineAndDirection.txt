### ⬇️ RAW SQL INPUT ⬇️
CREATE PROCEDURE [dbo].[sp_TAMS_GetSectorsByLineAndDirection]
	@Line AS nvarchar(10) = NULL,
	@Direction AS nvarchar(10) = NULL
AS
IF @Line = 'DTL'
BEGIN
	SELECT ID, Line, Direction, Sector, IsGap, IsEndSector, SameSector, IsspecialSector, [Order]
	FROM TAMS_Sector
	WHERE Line = @Line
	AND Direction = @Direction
	AND IsActive = 1 
	AND EffectiveDate <= GETDATE() and ExpiryDate >= GETDATE()

	ORDER BY [Order] ASC
END
ELSE IF @Line = 'NEL'
BEGIN
	SELECT ID, Line, Direction, Sector, IsGap, IsEndSector, SameSector, IsspecialSector, [Order]
	FROM TAMS_Sector
	WHERE Line = @Line
	AND Direction = @Direction
	AND IsActive = 1 
	AND EffectiveDate <= GETDATE() and ExpiryDate >= GETDATE()

	ORDER BY [Order] ASC
END
