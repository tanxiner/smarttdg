CREATE PROCEDURE [dbo].[sp_TAMS_Get_SignUpStatusByLoginID] 
	@LoginID NVARCHAR(100) = NULL
AS
BEGIN
	
	IF EXISTS(SELECT * FROM TAMS_Registration WHERE LoginID = @LoginID)
	BEGIN
		DECLARE @ID INT;

		SELECT TOP 1 * FROM TAMS_Registration WHERE LoginID = @LoginID ORDER BY CreatedOn Desc
		SELECT @ID = ID FROM TAMS_Registration WHERE LoginID = @LoginID ORDER BY CreatedOn Desc
	

		CREATE TABLE #AccessStatus
		(
			Line NVARCHAR(10),
			Module NVARCHAR(10),
			Status NVARCHAR(100),
			Pending NVARCHAR(100),
			NotifiedAt DATETIME
		)
	
		DECLARE @Line NVARCHAR(10);
		DECLARE @Module NVARCHAR(10);

		DECLARE cur CURSOR FOR
			SELECT DISTINCT rm.Line, rm.Module 
			FROM TAMS_Registration r, TAMS_Reg_Module rm
			WHERE r.ID = rm.RegID
			AND r.LoginID = @LoginID

		OPEN cur;

		FETCH NEXT FROM cur INTO @Line, @Module;

		WHILE @@FETCH_STATUS = 0
		BEGIN
			DECLARE @RegStatusID INT;
			DECLARE @UserType NVARCHAR(20);
			DECLARE @WFStatusID INT;
			DECLARE @WFStatus NVARCHAR(100);
			DECLARE @WFPending NVARCHAR(100);
			DECLARE @Notified DATETIME;
			DECLARE @IsExternal INT;

			SELECT TOP 1 @RegStatusID = rm.RegStatus, @Notified = rm.UpdatedOn, @UserType = IIF(r.IsExternal = 1, 'ExtUser', 'IntUser'), @IsExternal = r.IsExternal
			FROM TAMS_Registration r, TAMS_Reg_Module rm
			WHERE r.ID = rm.RegID
			AND r.LoginID = @LoginID
			AND rm.Line = @Line
			AND rm.Module = @Module 
			ORDER BY r.CreatedOn DESC

			--SELECT @WFStatus = WFStatus, @WFStatusID = WFStatusID 
			--FROM TAMS_WFStatus
			--WHERE Line = @Line
			--AND WFType = 'UserRegStatus'
			--AND IsActive = 1
			--AND ID = @RegStatusID

			SELECT @WFStatus = WFStatus, @WFStatusID = WFStatusID 
			FROM TAMS_WFStatus
			WHERE Line = @Line
			AND WFType = 'UserRegStatus'
			AND IsActive = 1
			AND ID IN (SELECT RegStatus from  TAMS_Reg_Module WHERE RegId = @ID)

			IF @IsExternal = 0
			BEGIN
				SET @UserType = @Module + @UserType;
			END

			IF @WFStatus IN ('Rejected', 'Company Rejected', 'Approved')
			BEGIN
				SET @WFPending = '-';
			END
			ELSE
			BEGIN
				SELECT @WFPending = r.Role
				FROM TAMS_Endorser e, TAMS_Workflow w, TAMS_Role r
				WHERE e.WFStatusID = @WFStatusID
				AND w.ID = e.WorkflowID
				AND w.WorkflowType = @UserType
				AND w.IsActive = 1
				AND w.EffectiveDate <= GETDATE()
				AND w.ExpiryDate >= GETDATE()
				AND e.RoleID = r.ID
				AND e.Line = @Line
			END

			INSERT INTO #AccessStatus VALUES (@Line, @Module, @WFStatus, @WFPending, @Notified);


			FETCH NEXT FROM cur INTO @Line, @Module;

		END;

		CLOSE cur;
		DEALLOCATE cur;


		SELECT * FROM #AccessStatus ORDER BY Line, Module DESC;
		DROP TABLE #AccessStatus;

	END;
END
