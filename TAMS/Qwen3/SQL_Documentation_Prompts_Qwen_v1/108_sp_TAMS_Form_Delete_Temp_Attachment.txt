CREATE PROCEDURE  [dbo].[sp_TAMS_Form_Delete_Temp_Attachment]
(@TARId			INTEGER			= NULL, 
@TARAccessReqId	INTEGER			= NULL)

AS
BEGIN

DECLARE @ret AS NVARCHAR(50); 
SET @ret = ''; 

BEGIN TRY
	BEGIN TRANSACTION;

	DELETE 
		FROM TAMS_TAR_Attachment_Temp
			WHERE [TARId] = @TARId
				AND [TARAccessReqId] = @TARAccessReqId

	COMMIT TRANSACTION;
END TRY

BEGIN CATCH
	PRINT 'Inside exception';
	SELECT ERROR_NUMBER() AS ErrorNumber, 
		ERROR_MESSAGE() AS ErrorMessage;
	ROLLBACK TRANSACTION;
	SET @ret = 'Errror' 
END CATCH

SELECT @ret AS ReturnValue 

End;
