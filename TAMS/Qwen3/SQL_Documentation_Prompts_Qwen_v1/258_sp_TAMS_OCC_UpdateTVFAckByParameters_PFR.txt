CREATE PROCEDURE [dbo].[sp_TAMS_OCC_UpdateTVFAckByParameters_PFR]
	@OperationDate datetime = null,
	@AccessDate datetime = null,
	@UserID as int = 0,
	@StationId as int = 0,	
	@TVFMode as varchar(10) = null,	
	@TVFDirection1 as bit = null,	
	@TVFDirection2 as bit = null



AS
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION;

			UPDATE TAMS_TVF_Acknowledge
			SET TVFMode = @TVFMode, TVFDirection1 = @TVFDirection1, TVFDirection2 = @TVFDirection2,
			OperatedBy = @UserID, TVFOnTime = GETDATE(), UpdatedOn = GETDATE(), UpdatedBy = @UserID
			WHERE StationId = @StationId
			AND OperationDate = @OperationDate
			AND AccessDate = @AccessDate

			INSERT INTO TAMS_TVF_Acknowledge_Audit(
			   [ActionBy]
			  ,[ActionOn]
			  ,[AuditAction]
			  ,[TVFAckId]
			  ,[AccessDate]
			  ,[OperationDate]
			  ,[StationId]
			  ,[TVFMode]
			  ,[TVFDirection1]
			  ,[TVFDirection2]
			  ,[AcknowledgedBy]
			  ,[AcknowledgedOn]
			  ,[VerifiedBy]
			  ,[VerifiedOn]
			  ,[OperatedBy]
			  ,[TVFOnTime]
			  ,[CreatedOn]
			  ,[CreatedBy]
			  ,[UpdatedOn]
			  ,[UpdatedBy])
			  SELECT 
			   @UserID
			  ,GETDATE()
			  ,'O'
			  ,[ID]
			  ,[AccessDate]
			  ,[OperationDate]
			  ,[StationId]
			  ,[TVFMode]
			  ,[TVFDirection1]
			  ,[TVFDirection2]
			  ,[AcknowledgedBy]
			  ,[AcknowledgedOn]
			  ,[VerifiedBy]
			  ,[VerifiedOn]
			  ,[OperatedBy]
			  ,[TVFOnTime]
			  ,[CreatedOn]
			  ,[CreatedBy]
			  ,[UpdatedOn]
			  ,[UpdatedBy]
			  FROM TAMS_TVF_Acknowledge
			  WHERE StationId = @StationId
			  AND OperationDate = @OperationDate
			  AND AccessDate = @AccessDate

		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		PRINT ERROR_MESSAGE();
		ROLLBACK TRANSACTION;
	END CATCH
END
