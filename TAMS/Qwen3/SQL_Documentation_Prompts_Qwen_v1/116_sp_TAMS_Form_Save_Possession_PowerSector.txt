CREATE PROCEDURE [dbo].[sp_TAMS_Form_Save_Possession_PowerSector]
	@PowerSector	AS NVARCHAR(4000) = NULL,
	@NoOFSCD		AS INT = 0,
	@BreakerOut		AS NVARCHAR(5) = NULL,
	@PossID			AS BIGINT = 0,
	@Message		AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT

SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

IF (SELECT COUNT(*) 
		FROM [dbo].[TAMS_Possession_PowerSector]
			WHERE PossessionId = @PossID
				AND PowerSector = LTRIM(RTRIM(@PowerSector))) = 0
	BEGIN
		INSERT INTO [dbo].[TAMS_Possession_PowerSector]
		([PossessionId], [PowerSector], [NoOFSCD], [BreakerOut])
		VALUES (@PossID, LTRIM(RTRIM(@PowerSector)), @NoOFSCD, CASE WHEN @BreakerOut = 'Y' THEN 1 ELSE 0 END)
	END

-- >>

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR INSERTING INTO TAMS_TAR'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message


END
