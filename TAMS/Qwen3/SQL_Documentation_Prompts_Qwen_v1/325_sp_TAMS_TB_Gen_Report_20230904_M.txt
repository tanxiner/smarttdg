CREATE PROCEDURE  [dbo].[sp_TAMS_TB_Gen_Report_20230904_M]
	@Line			AS NVARCHAR(10) = NULL,
	@TrackType		AS NVARCHAR(50) = NULL,
	@AccessDateFrom	AS NVARCHAR(20) = NULL,
	@AccessDateTo	AS NVARCHAR(20) = NULL,
	@AccessType		AS NVARCHAR(20) = NULL

AS

BEGIN

	SELECT a.TARNo AS [TAR ID],
			a.Company AS [Company/Dept],
			a.AccessType AS [Access Type],
			a.Name AS [Name],
			dbo.TAMS_Get_Station(a.Id) AS [Access Stations], 
			CONVERT(NVARCHAR(20), CONVERT(DATETIME, a.AccessDate, 101), 105) AS [TAR Date], 
			dbo.TAMS_Get_ES_NoBufferZone(a.Id) AS [Electrical Section],
			a.DescOfWork AS [Nature of Work],
			a.ARRemark AS [Remarks]
		FROM TAMS_TAR a
			WHERE CONVERT(DATETIME, a.AccessDate, 103) BETWEEN CONVERT(DATETIME, @AccessDateFrom, 103) AND CONVERT(DATETIME, @AccessDateTo, 103) 
				AND a.TARStatusId = 8
				AND (a.AccessType = @AccessType OR ISNULL(@AccessType, '') = '')
				AND a.Line = @Line
				and a.TrackType = @TrackType
	ORDER BY CONVERT(DATETIME, a.AccessDate, 101), a.TARNo

END
