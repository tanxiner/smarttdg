CREATE PROCEDURE [dbo].[sp_TAMS_TOA_Get_PointNo]
	@TOAID	BIGINT
AS

BEGIN
	SELECT ProtectionType 
	FROM TAMS_TOA
	WHERE Id = @TOAID

	SELECT ID as Sno,
		PointNo
		FROM TAMS_TOA_PointNo
			WHERE TOAID = @TOAID
		ORDER BY ID ASC

END
