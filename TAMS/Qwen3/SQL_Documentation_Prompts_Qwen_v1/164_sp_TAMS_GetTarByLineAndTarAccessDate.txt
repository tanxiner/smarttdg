CREATE PROCEDURE [dbo].[sp_TAMS_GetTarByLineAndTarAccessDate]
	@Line AS nvarchar(10) = NULL,
	@AccessDate as nvarchar(50) = NULL
AS
BEGIN
	SELECT [Id],
	[Line],
	[TARNo],
	[TARType],
	[Company],
	[Designation],
	[Name],
	[OfficeNo],
	[MobileNo],
	[Email],
	[SubmitDate],
	[AccessDate],
	[AccessTimeFrom],
	[AccessTimeTo],
	[AccessLocation],
	[AccessType],
	[IsNeutralGap],
	[IsExclusive],
	[DescOfWork],
	[ARRemark],
	[Is13ASocket],
	[CrossOver]
	FROM TAMS_TAR
	WHERE [Line] = @Line
	AND [AccessDate] = convert(datetime,@AccessDate,103)
END
