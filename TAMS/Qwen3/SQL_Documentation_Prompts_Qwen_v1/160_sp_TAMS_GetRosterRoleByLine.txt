CREATE PROCEDURE  [dbo].[sp_TAMS_GetRosterRoleByLine]
	@Line AS nvarchar(10) = NULL,
	@TrackType AS nvarchar(50) = NULL,
	@OperationDate AS nvarchar(10) = NULL,
	@Shift AS nvarchar(1) = NULL
AS
BEGIN
	declare @count as int = 0;

	IF @Line = 'DTL'
	BEGIN
		SELECT @Count = COUNT(*) FROM TAMS_OCC_Duty_Roster WHERE Line = @Line AND TrackType = @TrackType and OperationDate = @OperationDate

		IF @Count = 0
		BEGIN
			SELECT ID, Line, RosterCode, RoleId, [Order]
			FROM TAMS_Roster_Role
			WHERE Line = @Line
			 and TrackType = @TrackType
			AND EffectiveDate <= GETDATE() and ExpiryDate >= GETDATE()
			ORDER BY [Order] ASC
		END;
		ELSE
		BEGIN
			SELECT ID, Line, RosterCode, [Shift] AS RoleId, ID AS [Order]
			FROM TAMS_OCC_Duty_Roster
			WHERE Line = @Line
			and TrackType = @TrackType
			AND OperationDate = @OperationDate
			AND [Shift] = @Shift
			ORDER BY [Order] ASC
		END;
	END;
	ELSE
	BEGIN
		SELECT @Count = COUNT(*) FROM TAMS_OCC_Duty_Roster WHERE Line = @Line AND TrackType = @TrackType and  OperationDate = @OperationDate

		IF @Count = 0
		BEGIN
			SELECT ID, Line, RosterCode, RoleId, [Order]
			FROM TAMS_Roster_Role
			WHERE Line = @Line
			and TrackType = @TrackType
			AND RosterCode <> 'SCO'
			AND EffectiveDate <= GETDATE() and ExpiryDate >= GETDATE()
			ORDER BY [Order] ASC
		END;
		ELSE
		BEGIN
			SELECT ID, Line, RosterCode, [Shift] AS RoleId, ID AS [Order]
			FROM TAMS_OCC_Duty_Roster
			WHERE Line = @Line
			and TrackType = @TrackType
			AND OperationDate = @OperationDate
			AND [Shift] = @Shift
			ORDER BY [Order] ASC
		END;
	END;
END
