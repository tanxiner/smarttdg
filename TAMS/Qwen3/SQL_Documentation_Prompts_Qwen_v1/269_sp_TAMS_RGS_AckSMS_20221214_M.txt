CREATE PROCEDURE [dbo].[sp_TAMS_RGS_AckSMS_20221214_M]
	@TARID		AS BIGINT = 0,
	@EncTARID	AS NVARCHAR(250),
	@SMSType	AS NVARCHAR(5),
	@Message	AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

SET @Message = ''

DECLARE @NewLine AS CHAR(2) = CHAR(30) + CHAR(10)

DECLARE @SMSMsg	NVARCHAR(4000)
SET @SMSMsg = ''

DECLARE @TARNo	NVARCHAR(50)
DECLARE @Line NVARCHAR(5)
DECLARE @AccessType NVARCHAR(50)
DECLARE @CurrDate NVARCHAR(20)
DECLARE @TOANo NVARCHAR(50)
DECLARE @Site NVARCHAR(200)
DECLARE @HPNo NVARCHAR(20)

SET @TARNo = ''
SET @Line = ''
SET @CurrDate = ''
SET @TOANo = ''
SET @Site = ''
SET @HPNo = ''

SELECT @TARNo = a.TARNo, 
		@Line = a.Line, 
		@AccessType = a.AccessType,
		@TOANo = b.TOANo,
		@HPNo = '96734774' -- b.MobileNo
	FROM TAMS_TAR a, TAMS_TOA b	
		WHERE a.Id = b.TARId
			AND a.Id = @TARID

SET @CurrDate = CONVERT(NVARCHAR(20), GETDATE(), 103)

IF @AccessType = 'Possession'
	BEGIN
		IF @SMSType = '2'
			BEGIN
				-- Logging

				UPDATE TAMS_TOA
					SET AckGrantTOATime = GETDATE(),
						ReqProtectionLimitTime = GETDATE()
				WHERE TARId = @TARID

				SET @SMSMsg = CAST(@TOANo AS NVARCHAR(50)) + ' acknowledgement has been received. '
					+ ' Click on the link to report once protection limit has been set up. :~'
					+ @Site + 'https://sbstweb.sbstransit.com.sg/TAMS/RGSAckSMS.aspx?TARID=' + CAST(@EncTARID AS NVARCHAR(250)) + '&SMSType=3'
			
				-- Logging

			END
		ELSE
			BEGIN
				UPDATE TAMS_TOA
					SET AckProtectionLimitTime = GETDATE()
				WHERE TARId = @TARID
			END
	END
ELSE
	BEGIN
		UPDATE TAMS_TOA
			SET AckGrantTOATime = GETDATE()
		WHERE TARId = @TARID
	END

SELECT @SMSMsg

INSERT INTO TAMS_TOA_Audit
SELECT '', GETDATE(), 'U', * FROM TAMS_TOA
	WHERE TARId = @TARID

IF @SMSMsg <> ''
	BEGIN
		-- >> SEND SMS
		DECLARE @RetVal NVARCHAR(5)
		SET @RetVal = ''

			IF LTRIM(RTRIM(ISNULL(@HPNo, ''))) <> '' AND LTRIM(RTRIM(ISNULL(@SMSMsg, ''))) <> ''
				BEGIN
					EXEC sp_api_send_sms @HPNo, 'TAMS RGS', @SMSMsg, @RetVal OUTPUT
				END

		-- >> 20220818 Quickly trigger SMS Sending
			DECLARE @SMSOutputMsg NVARCHAR(500) 

			SET @SMSOutputMsg = ''

			EXEC SP_Call_SMTP_Send_SMSAlert @SMSOutputMsg OUTPUT

			IF @SMSOutputMsg <> ''
				BEGIN
					SET @Message = 'Error SMS Sending'
					-- >> GOTO TRAP_ERROR
				END
		-- >> 20220818 Quickly trigger SMS Sending
	END

IF @@ERROR <> 0 BEGIN
	SET @Message = 'Error RGS Ack SMS'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END
