CREATE PROCEDURE [dbo].[sp_TAMS_Depot_RGS_AckSurrender]
	@TARID		AS BIGINT = 0,
	@UserID		AS NVARCHAR(500),
	@Message	AS NVARCHAR(500) = NULL OUTPUT

AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

SET @Message = ''

DECLARE @NewLine AS CHAR(2) = CHAR(30) + CHAR(10)

DECLARE @UserIDID	INT = 1;

SELECT @UserIDID = Userid
	FROM TAMS_User
		WHERE LoginId = @UserID

DECLARE @TOAStatusChk INT

SELECT @TOAStatusChk = b.TOAStatus
	FROM TAMS_TAR a, TAMS_TOA b	
		WHERE a.Id = b.TARId
			AND a.Id = @TARID

IF @TOAStatusChk = 4
	BEGIN
		UPDATE TAMS_TOA
			SET	TOAStatus = 5,
				AckSurrenderTime = GETDATE(),
				UpdatedOn = GETDATE(),
				UpdatedBy = @UserID
		WHERE TARId = @TARID

		INSERT INTO TAMS_TOA_Audit
		SELECT @UserID, GETDATE(), 'U', * FROM TAMS_TOA
			WHERE TARId = @TARID

		DECLARE @SMSMsg	NVARCHAR(4000)
		SET @SMSMsg = ''

		DECLARE @lv_IsAllAckSurrender BIT
		DECLARE @AckSurrenderTime NVARCHAR(20)
		DECLARE @TARNo NVARCHAR(50)
		DECLARE @TOANo NVARCHAR(50)
		DECLARE @Line NVARCHAR(5)
		DECLARE @CurrDate NVARCHAR(20)
		DECLARE @CurrTime NVARCHAR(20)
		DECLARE @AccessDate DATE
		DECLARE @OperationDate DATE
		DECLARE @HPNo NVARCHAR(20)

		SET @lv_IsAllAckSurrender = 1

		SET @AckSurrenderTime = ''
		SET @TARNo = ''
		SET @Line = ''
		SET @CurrDate = ''
		SET @HPNo = ''

		SET @CurrDate = CONVERT(NVARCHAR(20), GETDATE(), 103)
		SET @CurrTime = CONVERT(NVARCHAR(20), GETDATE(), 108)

		SELECT @TARNo = a.TARNo,
				@TOANo = b.TOANo, 
				@Line = a.Line, 
				@AckSurrenderTime = CONVERT(NVARCHAR(20), b.AckSurrenderTime, 108),
				@AccessDate = b.AccessDate,
				@OperationDate = b.OperationDate,
				@HPNo = b.MobileNo
			FROM TAMS_TAR a, TAMS_TOA b	
				WHERE a.Id = b.TARId
					AND a.Id = @TARID

		
		DECLARE @Cur01 CURSOR;
		DECLARE @TOAStatus	INT

		SET @Cur01 = CURSOR FAST_FORWARD 
		FOR
		SELECT b.TOAStatus
			FROM TAMS_TAR a, TAMS_TOA b
				WHERE a.Id = b.TARId
					AND b.AccessDate = @AccessDate
					AND b.TOAStatus NOT IN (0, 6)
					AND a.Line = @Line
			ORDER BY b.Id
		OPEN @Cur01
		FETCH NEXT FROM @Cur01 INTO @TOAStatus
		WHILE @@FETCH_STATUS = 0 
			BEGIN
				IF @TOAStatus <> 5
					BEGIN
						SET @lv_IsAllAckSurrender = 0
					END
				FETCH NEXT FROM @Cur01 INTO @TOAStatus
			END 	
		CLOSE @Cur01
		DEALLOCATE @Cur01

		

		/***********************Changes for NEL , to start second phase of OCC once the TAR sectors has cleared********************************/
		IF @Line = 'NEL'
		BEGIN
			iF EXISTS(Select * from TAMS_TOA INNER JOIN TAMS_Depot_Auth ON
			TAMS_Depot_Auth.TarID=TAMS_TOA.TARId
			where TAMS_TOA.TARId=@TARID and TAMS_TOA.TOAStatus>=5)
			BEGIN
				DECLARE @authid as int;
				DECLARE @statusid as int;
				DECLARE @newstatusid as int;
				DECLARE @newworkflowid as int;
				DECLARE @workflowid as int;


				SELECT @authid=TAMS_Depot_Auth.ID,@statusid=DepotAuthStatusId FROM TAMS_Depot_Auth where TarID=@TARID

				if @statusid<>8 
				BEGIN
					SET @Message = 'Depot Authentication status is incorrect.'
					-- Auth status is not 8
					GOTO TRAP_ERROR
				END

				-- get next status
				select @newstatusid=WFStatusId+1 from TAMS_WFStatus where WFType='dtcauth' and WFStatusId=@statusid
				and isActive=1
				order by [Order] ASC;

				SELECT @workflowid=ID from TAMS_WFStatus where WFType='dtcauth' and WFStatusId=@statusid
				Select @newworkflowid=ID from TAMS_WFStatus where WFType='dtcauth' and WFStatusId=@newstatusid


				UPDATE TAMS_Depot_Auth_Workflow
				SET WFStatus='Completed', ActionBy=@UserID, ActionOn=getDate(),CheckBoxValue=1
				where WorkflowID=@workflowid

				INSERT INTO TAMS_Depot_Auth_Workflow
				SELECT @authid, null, null, null, @newworkflowid,null,0

				UPDATE TAMS_Depot_Auth
				SET DepotAuthStatusId=@newstatusid,UpdatedOn=getDate(),UpdatedBy=@UserID
				where ID=@authid
			END
		 			 

		END;
		/***********************END of Changes for NEL , to start second phase of OCC once the TAR sectors has cleared********************************/
		
		IF @Line = 'NEL'
			BEGIN
				SET @SMSMsg = CAST(@TOANo AS NVARCHAR(50)) + ' surrender has been acknowledged by NEL DCC @ ' + CAST(@CurrTime AS NVARCHAR(20))
					+ ' hrs on ' + CAST(@CurrDate AS NVARCHAR(20)) + '. Thank you!'
			END

		-- >> SEND SMS
		DECLARE @RetVal NVARCHAR(5)
		SET @RetVal = ''

			IF LTRIM(RTRIM(ISNULL(@HPNo, ''))) <> '' AND LTRIM(RTRIM(ISNULL(@SMSMsg, ''))) <> ''
				BEGIN
					EXEC sp_api_send_sms @HPNo, 'TAMS RGS', @SMSMsg, @RetVal OUTPUT
				END

	END
ELSE
	BEGIN
		IF @TOAStatus = 5
			BEGIN
				SET @Message = '1'
				-- TOA already granted. Please refresh RGS.
				GOTO TRAP_ERROR
			END
		ELSE
			BEGIN
				SET @Message = 'Invalid TAR status. Please refresg RGS.'
				GOTO TRAP_ERROR
			END
	END


IF @@ERROR <> 0 BEGIN
	SET @Message = 'Error RGS Ack Surrender'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END
