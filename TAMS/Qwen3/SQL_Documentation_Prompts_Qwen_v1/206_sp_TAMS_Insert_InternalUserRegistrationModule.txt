CREATE PROCEDURE  [dbo].[sp_TAMS_Insert_InternalUserRegistrationModule] 	
	@RegID		INT,
	@Line		NVARCHAR(20),
	@TrackType		NVARCHAR(50),
	@Module		NVARCHAR(20)
	
AS
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION;

		DECLARE @NextStageTitle NVARCHAR(100);
		DECLARE @NextStageID	INT;
		DECLARE @WorkflowID		INT;
		DECLARE @NewWFRoleID INT;
		DECLARE @EndorserID		INT;
		DECLARE @WFStatus       NVARCHAR(50);

		-- Get the next stage in flow
		--SELECT @NextStageID = e.ID, @NextStageTitle = e.Title
		--FROM TAMS_Workflow w, TAMS_Endorser e, TAMS_WFStatus wf
		--WHERE w.ID = e.WorkflowId
		--AND w.Line = @Line
		--AND w.WorkflowType = 'IntUser'
		--AND w.EffectiveDate <= GETDATE()
		--AND w.ExpiryDate >= GETDATE()
		--AND w.IsActive = 1
		--AND e.[Level] = 1
		--ORDER BY e.[Level]
		IF @Module = 'TAR'
		BEGIN 
			SELECT @WorkflowID = ID 
			FROM TAMS_Workflow 
			WHERE Line = @Line 
			and TrackType = @TrackType
			AND WorkflowType = 'TARIntUser' 
			AND GETDATE() >= EffectiveDate 
			AND GETDATE() <= ExpiryDate 
			AND IsActive = 1
		END;
		Else IF @Module = 'DCC'
		BEGIN 
		print 'here dcc module'
			SELECT @WorkflowID = ID 
			FROM TAMS_Workflow 
			WHERE Line = @Line 
			and TrackType = @TrackType
			AND WorkflowType = 'DCCIntUser' 
			AND GETDATE() >= EffectiveDate 
			AND GETDATE() <= ExpiryDate 
			AND IsActive = 1

		END;
		ELSE 
		BEGIN
			SELECT @WorkflowID = ID 
			FROM TAMS_Workflow 
			WHERE Line = @Line 
			and TrackType = @TrackType
			AND WorkflowType = 'OCCIntUser' 
			AND GETDATE() >= EffectiveDate 
			AND GETDATE() <= ExpiryDate 
			AND IsActive = 1
		END;

		SELECT @NextStageID = WFStatusId, @NewWFRoleID = RoleID, @EndorserID = ID 
		FROM TAMS_Endorser 
		WHERE Line = @Line
		AND WorkflowId = @WorkflowID
		AND [Level] = 1 
		AND GETDATE() >= EffectiveDate 
		AND GETDATE() <= ExpiryDate 
		AND IsActive = 1

		SELECT @NextStageID = WFStatusId, @WFStatus = WFStatus FROM TAMS_WFStatus
		WHERE Line = @Line
		and TrackType = @TrackType
		AND WFType = 'UserRegStatus'
		AND WFStatusId = @NextStageID
		AND IsActive = 1
		
		--SELECT @NextStageID = e.ID, @NextStageTitle = e.Title
		--FROM TAMS_Workflow w, TAMS_Endorser e, TAMS_WFStatus wf
		--WHERE w.ID = e.WorkflowId
		--AND w.Line = @Line
		--AND w.WorkflowType = 'IntUser'
		--AND w.EffectiveDate <= GETDATE()
		--AND w.ExpiryDate >= GETDATE()
		--AND w.IsActive = 1
		--AND e.[Level] = 1
		--AND e.WorkFlowID = w.ID
		--AND wf.Line = @Line
		--AND wf.WFType = 'UserRegStatus'
		--ORDER BY e.[Level]  

		INSERT INTO TAMS_Reg_Module 
		VALUES (@RegID, @Line,@TrackType, @Module, @NextStageID, @WorkflowID, @EndorserID, 'Pending', @WFStatus, GETDATE(), 1, GETDATE(), 1);

		-- Insert Audit Log
		INSERT INTO TAMS_Action_Log VALUES 
		(
			@Line, 
			 @Module, 
			'UserReg-Registration', 
			@RegID, 
			'Internal User submitted User Registration with this module', 
			GETDATE(), 
			1
		)

		-- Sends email
		DECLARE @Sender NVARCHAR (100);
		DECLARE @SysID NVARCHAR(10);
		DECLARE @Subject NVARCHAR(500);    
		DECLARE @Sys NVARCHAR(100);
		DECLARE @Greetings  NVARCHAR(20);     
		DECLARE @ToList NVARCHAR(MAX);
		DECLARE @CCList NVARCHAR(MAX);
		DECLARE @BCCList NVARCHAR(MAX);
		DECLARE @Separator NVARCHAR(10);        
		DECLARE @AlertMsg NVARCHAR(MAX);
		DECLARE @AlertID AS INTEGER; 

		DECLARE @Body VARCHAR(max);
		DECLARE @Body1 VARCHAR(max);
		DECLARE @Body2 VARCHAR(max);
		DECLARE @Body3 VARCHAR(max);
		DECLARE @Link NVARCHAR(MAX);

		DECLARE @LoginPage	NVARCHAR(500)
					
		DECLARE @SendTo	NVARCHAR(200);	
		SET @ToList = '';

		-- Get Send to list
		DECLARE curSendTo CURSOR FOR
		SELECT Email FROM TAMS_User u, TAMS_User_Role ur
		WHERE u.UserID = ur.UserID
		AND ur.RoleID IN (
		SELECT ID FROM TAMS_Role WHERE Line = @Line AND TrackType = @TrackType AND Module = @Module AND Role LIKE '%SysApprover%')
					
		OPEN curSendTo;
		FETCH NEXT FROM curSendTo INTO @SendTo;

		WHILE @@FETCH_STATUS = 0
		BEGIN
			IF LEN(@ToList) =0 
			BEGIN
				SET @ToList = @SendTo
			END

			SET @ToList = @ToList + ',' + @SendTo;
						
			FETCH NEXT FROM curSendTo INTO @SendTo;
		END;
				

		CLOSE curSendTo;
		DEALLOCATE curSendTo;


		SET @Sender = 'TAMS Admin';
		SET @SysID = 'TAMS';
		SET @Subject = 'For your approval: User Registration Request '
		SET @Sys = 'Track Access Management System'   
		SET @Greetings = 'Dear Sir/Madam, '  
		--SET @ToList = ''
		SET @CCList = ''   
		SET @BCCList = ''
		SET @Separator = ','    
		SET @CCList = ''
		--SET @ToList = '';

		SET @Body1 = ''
		SET @Body2 = ''
		SET @Body3 = ''
			
		SET @Body1 = @Body1 + '<table width="100%">'
		SET @Body1 = @Body1 + '<tr align="left" valign="middle" style="font-size:10pt; font-face:Arial;">'
		SET @Body1 = @Body1 + '<td>Please access TAMS to approve/reject the User registration.</td>'
		SET @Body1 = @Body1 + '</tr>'
		SET @Body1 = @Body1 + '</table>'
		SET @LoginPage = 'https://pulsewebuat.sbstransit.com.sg/TAMS/Login.aspx'
	 
		SET @Body2 = @Body2 + '<br>'
		SET @Body2 = @Body2 + '<table width="100%">'
		SET @Body2 = @Body2 + '<tr align="left" valign="middle" style="font-size:10pt; font-face:Arial;">'
		SET @Body2 = @Body2 + '<td>Click on the link <a href=' + @LoginPage + '>here</a> to access tams.</td>'
		SET @Body2 = @Body2 + '</tr>'
		SET @Body2 = @Body2 + '</table>'

		SET @Body3 = @Body3 + '<br>'
		SET @Body3 = @Body3 + '<table width="100%">'
		SET @Body3 = @Body3 + '<tr align="left" valign="middle" style="font-size:10pt; font-face:Arial;">'
		SET @Body3 = @Body3 + '<td>***********************************************<br>Do Not reply. This is a computer generated Email. <br>***********************************************</td>'
		SET @Body3 = @Body3 + '</tr>'
		SET @Body3 = @Body3 + '</table>'

		SET @Body = ''
		SET @Body = @Body1 + @Body2 + @Body3

		SET @BCCList = ''

		EXEC [dbo].EAlertQ_EnQueue 
			@Sender = @Sender, 
			@UserId = @SysID, 
			@Subject = @Subject,        
			@Sys = @Sys,  
			@Greetings = @Greetings, 
			@AlertMsg = @Body, 
			@SendTo = @ToList,        
			@CC = @CCList, 
			@BCC= @BCCList, 
			@Separator = @Separator, 
			@AlertID =  @AlertID output
	

		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		ROLLBACK TRANSACTION;
	END CATCH

END;
