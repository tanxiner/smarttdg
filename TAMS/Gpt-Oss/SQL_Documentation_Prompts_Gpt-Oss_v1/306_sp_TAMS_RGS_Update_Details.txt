CREATE PROCEDURE  [dbo].[sp_TAMS_RGS_Update_Details]
	@TARID			AS BIGINT,
	@InchargeNRIC	AS NVARCHAR(50),
	@MobileNo		AS NVARCHAR(20),
	@TetraRadioNo	AS NVARCHAR(50),
	@UserID			AS NVARCHAR(50),
	@Message		AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT

SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

CREATE TABLE #tmpnric 
(id BIGINT IDENTITY(1,1) NOT NULL, 
nric NVARCHAR(50),
namestr NVARCHAR(500),
line NVARCHAR(50),
qualdate NVARCHAR(50),
qualcode NVARCHAR(100),
qualstatus NVARCHAR(100))

TRUNCATE TABLE #tmpnric

DECLARE @ExecString	NVARCHAR(MAX)
SET @ExecString = ''
DECLARE @QualDate	NVARCHAR(20)
SET @QualDate = ''

DECLARE @InChargeName	NVARCHAR(500)
DECLARE @InChargeStatus	NVARCHAR(20)

SET @InChargeName = ''
SET @InChargeStatus = ''

DECLARE @Line NVARCHAR(20)
DECLARE @AccessDate DATE
DECLARE @AccessType NVARCHAR(50)
DECLARE @TOAID BIGINT

SELECT @Line = a.Line,
	@AccessDate = a.AccessDate,
	@TOAID = a.Id,
	@AccessType = b.AccessType
FROM TAMS_TOA a, TAMS_TAR b 
	WHERE a.TARId = @TARID
		AND a.TARId = b.Id

DECLARE @QTSQualCode		NVARCHAR(20)
DECLARE @QTSQualCodeProt	NVARCHAR(20)

SELECT @QTSQualCode = ParaValue2
	FROM TAMS_Parameters
		WHERE ParaCode = 'QTSQualCode'
			AND ParaValue1 = @Line
			AND ParaValue3 = 'Mainline Possession'
			AND GETDATE() BETWEEN EffectiveDate AND ExpiryDate

SELECT @QTSQualCodeProt = ParaValue2
	FROM TAMS_Parameters
		WHERE ParaCode = 'QTSQualCode'
			AND ParaValue1 = @Line
			AND ParaValue3 = 'Mainline'
			AND GETDATE() BETWEEN EffectiveDate AND ExpiryDate

SET @QualDate = CONVERT(NVARCHAR(20), @AccessDate, 103)
SET @ExecString = 'sp_TAMS_TOA_QTS_Chk '+ '''' + @InchargeNRIC + ''', ''' + @QualDate + ''', ''' + @Line + ''', ''' + @QTSQualCode + ''''

INSERT INTO #tmpnric
EXEC(@ExecString)

SELECT @InChargeName = namestr,
	@InChargeStatus = qualstatus
FROM #tmpnric

DECLARE @QTSFinStatus NVARCHAR(20)
SET @QTSFinStatus = ''

IF @InChargeStatus	= 'InValid'
	BEGIN
		IF @AccessType = 'Protection'
			BEGIN
				TRUNCATE TABLE #tmpnric
				SET @InChargeName = ''
				SET @InChargeStatus = ''

				SET @ExecString = 'sp_TAMS_TOA_QTS_Chk '+ '''' + @InchargeNRIC + ''', ''' + @QualDate + ''', ''' + @Line + ''', ''' + @QTSQualCodeProt + ''''

				INSERT INTO #tmpnric
				EXEC(@ExecString)

				SELECT @InChargeName = namestr,
					@InChargeStatus = qualstatus
				FROM #tmpnric

				IF @InChargeStatus	= 'InValid'
					BEGIN
						SET @QTSFinStatus = 'InValid'
					END
				ELSE
					BEGIN
						SET @QTSFinStatus = 'Valid'
					END
			END
		ELSE
			BEGIN
				SET @QTSFinStatus = 'InValid'
			END
	END
ELSE
	BEGIN
		SET @QTSFinStatus = 'Valid'
	END

DROP TABLE #tmpnric

IF @QTSFinStatus = 'InValid'
	BEGIN
		SET @Message = '1' -- >> Invalid Qualification
	END
ELSE
	BEGIN
		DECLARE @InchargeCtr INT
		SET @InchargeCtr = 0
		SELECT @InchargeCtr = COUNT(*)
			FROM TAMS_TOA
				WHERE TARId = @TARID
					AND dbo.DecryptString(InChargeNRIC) = @InchargeNRIC

		IF @InchargeCtr = 0 -- >> New Incharge need to delete TOA_Parties
			BEGIN
				UPDATE TAMS_TOA
					SET InChargeNRIC = dbo.EncryptString(@InchargeNRIC),
						InChargeName = @InChargeName,
						MobileNo = @MobileNo,
						TetraRadioNo = @TetraRadioNo,
						UpdatedOn = GETDATE(),
						UpdatedBy = @UserID
					WHERE TARId = @TARID

				INSERT INTO TAMS_TOA_Audit
				SELECT @UserID, GETDATE(), 'U', * FROM TAMS_TOA
					WHERE TARId = @TARID

				INSERT INTO TAMS_TOA_Parties_Audit
				SELECT @UserID, GETDATE(), 'D', * FROM TAMS_TOA_Parties
					WHERE TOAId = @TOAID
						AND IsInCharge = 1

				DELETE FROM TAMS_TOA_Parties
					WHERE TOAId = @TOAID
						AND IsInCharge = 1

				INSERT INTO TAMS_TOA_Parties
					(TOAId, [Name], NRIC, IsInCharge, IsWitness, IsTMC, BookInTime, BookOutTime, BookInStatus)
				VALUES (@TOAID, @InChargeName, dbo.EncryptString(@InchargeNRIC), 1, 0, 0, GETDATE(), NULL, 'In')

				INSERT INTO TAMS_TOA_Parties_Audit
				SELECT @UserID, GETDATE(), 'I', * FROM TAMS_TOA_Parties
					WHERE TOAId = @TOAID
						AND IsInCharge = 1

			END
		ELSE
			BEGIN
				UPDATE TAMS_TOA
					SET MobileNo = @MobileNo,
						TetraRadioNo = @TetraRadioNo,
						UpdatedOn = GETDATE(),
						UpdatedBy = @UserID
					WHERE TARId = @TARID

				INSERT INTO TAMS_TOA_Audit
				SELECT @UserID, GETDATE(), 'U', * FROM TAMS_TOA
					WHERE TARId = @TARID
			END

		SET @Message = '0' -- >> Updated
	END

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR INSERTING INTO TAMS_TOA'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END
