CREATE PROCEDURE [dbo].[sp_TAMS_TOA_Get_Parties]
	@TOAID	BIGINT
AS

BEGIN
	
	SELECT NoOfParties 
		FROM TAMS_TOA
			WHERE Id = @TOAID

	SELECT ROW_NUMBER() OVER(ORDER BY ID) as Sno,
		[Name] AS PartiesName,
		dbo.QTS_MaskText(dbo.DecryptString(NRIC), 4) AS PartiesNRIC,  
		CASE WHEN IsTMC = 1 THEN 'Yes' ELSE 'No' END AS PartiesIsTMC,
		CASE WHEN IsInCharge = 1 THEN 'Y' ELSE 'N' END AS InCharge,
		Id AS RecID,
		BookInStatus
		FROM TAMS_TOA_Parties
			WHERE TOAId = @TOAID
		ORDER BY ID ASC

	-- >> Witness List
	SELECT [Name] AS PartiesName,
			ID AS PartiesID
		FROM TAMS_TOA_Parties
			WHERE TOAId = @TOAID
				AND IsInCharge = 0
		ORDER BY ID ASC

	-- >> Selected Witness
	SELECT ID AS PartiesID
		FROM TAMS_TOA_Parties
			WHERE TOAId = @TOAID
				AND IsInCharge = 0
				AND IsWitness = 1
		
	SELECT COUNT(*) AS WorkPartiesBook
		FROM TAMS_TOA_Parties
			WHERE TOAId = @TOAID
				AND IsInCharge = 0
				AND BookInStatus = 'In'

	SELECT COUNT(*) AS WPCtr
		FROM TAMS_TOA_Parties
			WHERE TOAId = @TOAID
				AND BookInStatus = 'In'
END
