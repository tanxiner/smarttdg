CREATE PROCEDURE  [dbo].[sp_TAMS_GetBlockedTarDates]
	@Line AS nvarchar(10) = NULL,
	@TrackType as nvarchar(50) = NULL,
	@AccessDate AS date = NULL
AS
BEGIN
	SELECT ID, Line,TrackType, BlockDate, BlockReason
	FROM TAMS_Block_TARDate
	WHERE Line = @Line
	AND TrackType = @TrackType
	AND BlockDate = @AccessDate
	AND IsActive = 1 
	ORDER BY BlockDate ASC
END
