CREATE procedure  [dbo].[sp_TAMS_GetTarEnquiryResult_Header_20240905]
	@uid AS integer = 0,
	@Line AS nvarchar(50) = NULL,
	@TrackType AS nvarchar(50) = NULL,
	@TarType AS nvarchar(50) = NULL,
	@AccessType AS nvarchar(50) = NULL,
	@TarStatusId AS integer = 0,
	@AccessDateFrom as nvarchar(50) = NULL,
	@AccessDateTo as nvarchar(50) = NULL , 
	@Department as nvarchar(50) = NULL,
	@Userid as nvarchar(50)=NULL
 
AS
BEGIN
	DECLARE  @Dept as nvarchar(50),
			@StatusId as int
	declare @IsAll as int
	declare @IsPower as int
	declare @IsDep as int


	if ( ( Select count(*) 
	From TAMS_User u,TAMS_User_Role ur,TAMS_Role r
	Where u.Userid = ur.UserID and ur.RoleID= r.ID and r.TrackType = @TrackType
	and r.Role in ('NEL_OCCScheduler','DTL_OCCScheduler','NEL_TrafficController','DTL_TrafficController','NEL_DCC','NEL_ChiefController','NEL_SDS','NEL_TAPApprover','NEL_TAPHOD','NEL_TAPVerifier', 'DTL_TAPApprover','DTL_TAPHOD','DTL_TAPVerifier','DTL_ChiefController')
	and u.Userid = @uid
	) ) > 0 
	begin
		set @IsAll = 1 ;
	end;

	if ( ( Select count(*) 
	From TAMS_User u,TAMS_User_Role ur,TAMS_Role r
	Where u.Userid = ur.UserID and ur.RoleID= r.ID and r.TrackType = @TrackType
	and r.Role in ('NEL_PowerEndorser' ,'NEL_PowerHOD','NEL_PFR', 'DTL_PowerEndorser','DTL_PFR')
	and u.Userid = @uid
	) ) > 0 
	begin 
		set @IsPower = 1 
	end;

	if ( ( Select count(*) 
	From TAMS_User u,TAMS_User_Role ur,TAMS_Role r
	Where u.Userid = ur.UserID and ur.RoleID= r.ID and r.TrackType = @TrackType
	and r.Role in ('NEL_ApplicantHOD','NEL_Applicant','DTL_Applicant','DTL_ApplicantHOD')
	and u.Userid = @uid
	) ) > 0 
	begin 
		set @IsDep = 1 
	end;


	-- For external User
	if ( ( Select count(*) 
	From TAMS_User u,TAMS_User_Role ur,TAMS_Role r
	Where u.Userid = ur.UserID and ur.RoleID= r.ID and r.TrackType = @TrackType
	and u.IsExternal=1
	and u.Userid = @uid
	) ) > 0 
	begin 
		set @IsDep = 0
		set @IsPower = 0
		set @IsAll = 0
	end;

	PRINT @IsDep
	PRINT @IsPower
	PRINT @IsAll

	DECLARE @cond as varchar(max);

	SET @cond = '';

	 PRINT @Line

	IF @Line IS NOT NULL AND @Line != ''  AND @Line != '-1' 
	BEGIN 
		SET @cond = @cond + ' AND t.Line='''+@Line+'''';

	END;
	IF @TarType IS NOT NULL AND @TarType != ''  AND @TarType != '-1' 
	BEGIN 
		SET @cond  = @cond + ' AND t.TarType='''+@TarType+'''';
	END;

	IF @AccessType IS NOT NULL AND @AccessType != ''  AND @AccessType != '-1' 
	BEGIN 
		SET @cond = @cond + ' AND AccessType='''+@AccessType+'''';
	END;

	IF @TarStatusId IS NOT NULL AND @TarStatusId != ''  AND @TarStatusId != '-1' 
	BEGIN 
		SET @cond = @cond + ' AND TarStatusId='''+CAST(@TarStatusId as nvarchar(10))+'''';
	END;

	IF NOT ISNULL(@AccessDateFrom, '') = '' AND NOT @AccessDateFrom = ''
	BEGIN			
		SET @cond = @cond + ' AND AccessDate >= convert(datetime,'''+@AccessDateFrom+''',103)';
	END;

	IF NOT ISNULL(@AccessDateTo, '') = '' AND NOT @AccessDateTo = ''
	BEGIN			
		SET @cond = @cond + ' AND AccessDate <= convert(datetime,'''+@AccessDateTo+''',103)';
	END;	

	IF NOT ISNULL(@Department, '') = '' AND NOT @Department = ''
	BEGIN			
		SET @cond = @cond + ' AND Company = ''' + @Department + '''';
	END;

	IF NOT ISNULL(@Userid, '') = '' AND NOT @Userid = ''
	BEGIN			
		SET @cond = @cond + ' AND Userid = ''' + @Userid + '''';
	END;


	DECLARE @sql nvarchar(max);
	 
	SET @sql = 'SELECT ROW_NUMBER() OVER(ORDER BY t.tarno desc) AS sno, t.id, t.line, t.tarno, t.tartype, t.accesstype, t.accessdate,t.AccessTimeSlot, tarstatus, t.company,t.createdby,t.Name  FROM ( '


	if @IsAll = 1 
	begin 
		SET @sql = @sql + 'SELECT distinct t.id, t.line, t.tarno, t.tartype, t.accesstype, t.accessdate,t.AccessTimeSlot, s.wfstatus as tarstatus, t.company,t.createdby,u.Name
				from TAMS_TAR t, TAMS_WFStatus s, TAMS_User u
				where t.TARStatusId = s.WFStatusId
				and t.Line = s.Line
				and t.TrackType = ''' +  @TrackType +''' and s.TrackType=t.TrackType
				and t.createdby=u.Userid
				and s.WFType = ''TARWFStatus''
				 ' + @cond
	end;
	else if @IsPower = 1 
	begin 
		SET @sql = @sql + 'SELECT distinct t.id, t.line, t.tarno, t.tartype, t.accesstype, t.accessdate,t.AccessTimeSlot, s.wfstatus as tarstatus, t.company,t.createdby,u.Name
				from TAMS_TAR t, TAMS_WFStatus s, TAMS_User u
				where t.TARStatusId = s.WFStatusId
				and t.Line = s.Line
				and s.WFType = ''TARWFStatus''
				and t.InvolvePower = 1 OR t.createdby = ''' + CAST(@uid as nvarchar(10)) +'''
				and t.createdby=u.Userid
				and s.TrackType=t.TrackType and t.TrackType = ''' +  @TrackType +'''' + @cond 
	end; 
	else if @IsDep = 1 
	begin 


		SET @sql = @sql + 'SELECT distinct t.id, t.line, t.tarno, t.tartype, t.accesstype, t.accessdate,t.AccessTimeSlot, s.wfstatus as tarstatus, t.company,t.createdby,u.Name
				from TAMS_TAR t, TAMS_WFStatus s, TAMS_User u
				where t.TARStatusId = s.WFStatusId
				and t.Line = s.Line
				and s.WFType = ''TARWFStatus''
				and t.TrackType = ''' +  @TrackType +'''
				and s.TrackType=t.TrackType
				and t.createdby=u.Userid
				and ( t.company in (select Department FROM TAMS_User where UserID = ''' + CAST(@uid AS nvarchar(10)) + ''')
				  OR  t.createdby = ''' + CAST(@uid as nvarchar(10)) +''') ' + @cond
	end;
	else 
	begin 
		SET @sql = @sql + 'SELECT distinct t.id, t.line, t.tarno, t.tartype, t.accesstype, t.accessdate,t.AccessTimeSlot, s.wfstatus as tarstatus, t.company,t.createdby,u.Name
				from TAMS_TAR t, TAMS_WFStatus s, TAMS_User u
				where t.TARStatusId = s.WFStatusId
				and t.Line = s.Line
				and t.TrackType = ''' +  @TrackType +'''
				and s.TrackType=t.TrackType
				and s.WFType = ''TARWFStatus''
				and t.createdby=u.Userid
				and t.createdby = ''' + CAST(@uid as nvarchar(10)) +''''  + @cond 
				 
	end;
	---- applicant can view own created tar

	--- PFR, PowerHOD, power_endorser InvolvePower = 1 
	--- applicant hod role can view TAR under own department 
	--	TAP verifier/approver/HOD/scheduler/SDS/CC/TC can see all TAR under own Line 


	  

	SET @sql = @sql + ') as t';
	PRINT (@sql);

	EXEC (@sql);	

END
