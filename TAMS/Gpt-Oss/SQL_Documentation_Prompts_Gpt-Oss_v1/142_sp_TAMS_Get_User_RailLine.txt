CREATE PROCEDURE [dbo].[sp_TAMS_Get_User_RailLine] 
	@UserId NVARCHAR(100) = NULL
AS
BEGIN

	IF EXISTS (SELECT * FROM TAMS_User_Role ur, TAMS_User u WHERE ur.UserID = u.Userid and ur.Line = 'All'
				AND u.LoginID = @UserId)
	BEGIN
		SELECT 'DTL' AS 'RailLine' UNION SELECT 'NEL' AS 'RailLine' UNION SELECT 'SPLRT' AS 'RailLine' 
	END
	ELSE
	BEGIN
		SELECT DISTINCT(Line) AS 'RailLine'
		FROM TAMS_User_Role ur, TAMS_User u
		WHERE ur.UserID = u.Userid
		AND u.LoginID = @UserID

	END;
END
