CREATE PROCEDURE [dbo].[sp_TAMS_OCC_GetTarSectorByLineAndTarAccessDate]
	@Line AS nvarchar(10) = NULL,
	@AccessDate as nvarchar(50) = NULL

AS
BEGIN
	IF @Line = 'DTL'
	BEGIN
		SELECT s.id, tarid, d.tractionpowerid, s.sectorid, isbuffer, colourcode, isaddedbuffer, t.poweron
		FROM tams_tar t, TAMS_TAR_Sector s, TAMS_Traction_Power_Detail d
		WHERE t.id = s.TARId
		AND s.sectorid = d.sectorid
		AND t.TARStatusId = 8
		AND t.Line = @Line
		AND t.AccessDate = convert(datetime,@AccessDate,103)
		ORDER by d.tractionpowerid ASC
	END;
	ELSE IF @Line = 'NEL'
	BEGIN
		SELECT s.id, tarid, d.PowerSection as tractionpowerid, s.powersectorid as sectorid, isbuffer, '' as colourcode, '' as isaddedbuffer, t.poweron
		FROM tams_tar t, TAMS_TAR_Power_Sector s, TAMS_Power_Sector d
		WHERE t.id = s.TARId
		AND s.PowerSectorId = d.ID
		AND t.TARStatusId = 9
		AND t.Line = @Line
		AND t.AccessDate = convert(datetime,@AccessDate,103)
		ORDER by s.powersectorid ASC
	END;

	--ORDER by s.id ASC, d.tractionpowerid ASC, s.SectorId ASC
