CREATE PROCEDURE [dbo].[sp_TAMS_GetTarAccessRequirementsByTarId]
	@TarId AS integer = 0
AS
BEGIN
  select tta.id, tarid, tar.operationrequirement, tta.ispower, tta.isselected, tta.[order]
  from tams_tar_accessreq tta, TAMS_Access_Requirement tar
  where tta.OperationRequirement = tar.id
  and tta.tarid = @TarId
  and IsSelected = 1
END
