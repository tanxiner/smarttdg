CREATE PROCEDURE  [dbo].[sp_TAMS_Approval_Endorse20250120]
	@TARID		AS INTEGER,						-- >> TAR ID
	@TARWFID	AS INTEGER,						-- >> Current WF ID
	@EID		AS INTEGER,						-- >> Current Endorser ID
	@ELevel		AS INTEGER,						-- >> Current Endorser Level
	@Remarks	AS NVARCHAR(1000) = NULL,		-- >> Remarks -- Mandatory for Reject / Optional for Approved / Endorse
	@TVFRunMode			AS NVARCHAR(50) = NULL,	-- >> New Column to be confirmed with Adeline
	@TVFRunModeUpdInd	AS NVARCHAR(5) = NULL,	-- >> Indicator to Update TVF Run Mode or Not.
	@UserLI		AS NVARCHAR(100) = NULL,			-- >> User Login ID	
	@Message	AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

-- >>
SET @Message = ''
DECLARE @UserID		INTEGER
DECLARE @UserName	NVARCHAR(500)

DECLARE @EMTARID		INTEGER
DECLARE @EMTARStatus	NVARCHAR(20)
DECLARE @EMTARNo		NVARCHAR(50)
DECLARE @EMRemarks		NVARCHAR(1000)
DECLARE @EMActor		NVARCHAR(100)
DECLARE @EMToSend		NVARCHAR(1000)
DECLARE @EMCCSend		NVARCHAR(1000)
DECLARE @EMCompany		NVARCHAR(100)
DECLARE @EMMsg			NVARCHAR(500)
DECLARE @TrackType		NVARCHAR(50)

DECLARE @TARDate		DATE

SELECT @UserID = Userid, @UserName = [Name]
FROM TAMS_User
	WHERE LoginId = @UserLI

-- >>>
DECLARE @TARWFCtr	INTEGER
SET @TARWFCtr = 0

SELECT @TARWFCtr = COUNT(*) 
	FROM TAMS_TAR_Workflow
		WHERE TARId = @TARID
			AND ActionBy = @UserID
			AND WFStatus = 'Approved'

IF @TARWFCtr > 0
	BEGIN
		--SET @Message = 'Error! User ID have already approved this TAR.'
		SET @Message = '1001'
		GOTO FORCE_EXIT_PROC
	END
-- >>>

UPDATE TAMS_TAR_Workflow
	SET WFStatus = 'Approved',
		ActionBy = @UserID,
		ActionOn = GETDATE(),
		Remark = @Remarks
	WHERE ID =  @TARWFID

IF @TVFRunModeUpdInd = 1
	BEGIN
		UPDATE TAMS_TAR
			SET TVFMode = @TVFRunMode
		WHERE ID = @TARID
	END

DECLARE @NextEndID		INTEGER
DECLARE @NextEndTitle	NVARCHAR(100)
DECLARE @NextWFID		INTEGER
DECLARE @NextWFStatID	INTEGER
DECLARE @NextRoleID		INTEGER
DECLARE @TARType	NVARCHAR(20)
DECLARE @WFID		INTEGER
DECLARE @Line		NVARCHAR(10)

SELECT @TARType = TARType,
		@Line = Line,
		@EMToSend = Email,		-- >> ToSendEmail
		@EMCCSend = Email,
		@EMTARID = Id,
		@EMTARNo = TARNo,
		@EMRemarks = ARRemark,
		@EMCompany = Company,
		@TARDate = AccessDate, 
		@TrackType = TrackType
	FROM TAMS_TAR
		WHERE ID = @TARID

print '@Line'; 
print @Line;

SELECT @WFID = WorkflowId 
	FROM TAMS_TAR_Workflow
		WHERE ID = @TARWFID

-- >> Current Level Endorser
SELECT @EMActor = Title
	FROM TAMS_Endorser
		WHERE WorkflowId = @WFID
			AND [Level] = @ELevel

-- >> Next Level Endorser
SELECT @NextEndID = ID, 
		@NextEndTitle = Title,
		@NextWFID = WorkflowId,
		@NextWFStatID = WFStatusId,
		@NextRoleID = RoleId
	FROM TAMS_Endorser
		WHERE WorkflowId = @WFID
			AND [Level] = @ELevel + 1

IF ISNULL(@NextEndTitle, '') = ''
	BEGIN
		-- >> Last Level Approved TAR
		
		-- >> If NEL Approved = 9 Else DTL or LRT = 8
		UPDATE TAMS_TAR
			SET TARStatusId = CASE WHEN @Line = 'NEL' THEN 9 ELSE 8 END,
				UpdatedOn = GETDATE(),
				UpdatedBy = @UserID
			WHERE ID = @TARID

		IF @TARType = 'Urgent'
			BEGIN
				-- >> APPROVED EMAIL
				-- >> PRINT 'Urgent Email'
				SET @EMTARStatus = 'Approved'
				SET @EMMsg = ''

					print 'before sending email sp_TAMS_Email_Urgent_TAR';
					print @EMTARStatus; 
				EXEC sp_TAMS_Email_Urgent_TAR @EMTARID, @EMTARStatus, @EMTARNo, @EMRemarks, @EMActor, @EMToSend, @EMMsg OUTPUT
			END

	END
ELSE
	BEGIN
		IF (SELECT COUNT(*) FROM TAMS_TAR_Workflow
			WHERE TARId = @TARID
				AND WorkflowId = @NextWFID
				AND EndorserId = @NextEndID) = 0
			BEGIN	
				INSERT INTO [dbo].[TAMS_TAR_Workflow]
					([TARId], [WorkflowId], [EndorserId], [UserId], [WFStatus], [Remark], [ActionOn], [ActionBy])
				VALUES
					(@TARID, @NextWFID, @NextEndID, NULL, 'Pending', '', GETDATE(), @UserID)
			END

		UPDATE TAMS_TAR
			SET TARStatusId = @NextWFStatID,
				UpdatedOn = GETDATE(),
				UpdatedBy = @UserID
			WHERE Id = @TARID

		IF @TARType = 'Urgent'
			BEGIN
				-- >> NEXT ROLE ID EMAIL ALERT
				-- >> PRINT 'Urgent Email'
				DECLARE @RoleEmail	NVARCHAR(1000)
				SET @RoleEmail = ''
				SET @EMMsg = ''

				SELECT @RoleEmail = @RoleEmail + ISNULL(a.Email + ', ', '') 
					FROM TAMS_User a, TAMS_User_Role b
						WHERE a.Userid = b.UserID
							AND a.IsActive = 1
							AND @TARDate BETWEEN a.ValidFrom AND a.ValidTo
							-- >> AND b.RoleID = @NextEndID -- >> Orig, Wrongone
							AND b.RoleID = @NextRoleID

				IF LEN(LTRIM(RTRIM(@RoleEmail))) > 0
					BEGIN
						SET @RoleEmail = LEFT(LTRIM(RTRIM(@RoleEmail)), LEN(LTRIM(RTRIM(@RoleEmail))) - 1)
					END

				-- >> Select all email where role id = next role id
				SET @EMToSend = @RoleEmail

				-- >> 2024-01-31 Changes to add nelocc email to the recipient list for Urgent After and if next level is OCC Approval
				DECLARE @WFTypeChk	NVARCHAR(50)

				SELECT @WFTypeChk = WorkflowType
					FROM TAMS_Workflow
						WHERE ID = @WFID

				IF @WFTypeChk = 'UrgentAfter' AND @NextEndTitle = 'OCC Approval'
					BEGIN 
						declare @OCCEmailForUrgent as nvarchar(100);
						set @OCCEmailForUrgent = '';
						
						SELECT @OCCEmailForUrgent = ParaValue3 
							FROM TAMS_Parameters
								WHERE ParaCode = 'OCCContact'
									AND ParaValue1 = @Line
									AND LTRIM(RTRIM(ISNULL(ParaValue3, ''))) <> ''

						IF LEN(LTRIM(RTRIM(@EMToSend))) > 0
							BEGIN
								-- >> 2024-01-16 Changes to add nelocc email to the recipient list 
						 		SET @EMToSend = @EMToSend + ', ' + @OCCEmailForUrgent
							END
					END;
				ELSE IF @WFTypeChk = 'UrgentAfter' AND @NextEndTitle = 'SDS Approval'
					BEGIN 
						declare @SDSEmailForUrgent as nvarchar(100);
						set @SDSEmailForUrgent = '';
						
						SELECT @SDSEmailForUrgent = ParaValue3 
							FROM TAMS_Parameters
								WHERE ParaCode = 'SDSContact'
									AND ParaValue1 = @Line
									AND LTRIM(RTRIM(ISNULL(ParaValue3, ''))) <> ''

						IF LEN(LTRIM(RTRIM(@EMToSend))) > 0
							BEGIN
								-- >> 2024-01-16 Changes to add nelocc email to the recipient list 
						 		SET @EMToSend = @EMToSend + ', ' + @SDSEmailForUrgent
							END
					END;
				-- >> 2024-01-31 Changes to add nelocc email to the recipient list for Urgent After and if next level is OCC Approval

				print 'before sending email sp_TAMS_Email_Urgent_TAR 2';
					print @NextEndTitle; 
				EXEC sp_TAMS_Email_Apply_Urgent_TAR 2, @EMCompany, @EMTARNo, @NextEndTitle, @EMToSend, @EMCCSend, @EMMsg OUTPUT
			END
	END

-- >> Action Log
DECLARE @LogMsg	NVARCHAR(100)
-- >> SET @LogMsg = 'TAR Form has approved by : ' + '' + ' On ' + GETDATE()	
	SET @LogMsg = 'TAR Form Approved by : ' + CAST(@UserName AS NVARCHAR(500)) + ' On ' + CAST(CONVERT(NVARCHAR(20), GETDATE(), 103) AS NVARCHAR(20)) + ' ' + CAST(CONVERT(NVARCHAR(20), GETDATE(), 108) AS NVARCHAR(20))
		
INSERT INTO [dbo].[TAMS_Action_Log]
	([Line], [Module], [Function], [TransactionID], [LogMessage], [CreatedOn], [CreatedBy])
VALUES 
	(@Line, 'TAR', 'Approved TAR', @TARID, @LogMsg, GETDATE(), @UserID)

DECLARE @WFType	NVARCHAR(50)
DECLARE @InvPow	BIT

SELECT @WFType = WorkflowType,
		@InvPow = InvolvePower 
	FROM TAMS_Workflow
		WHERE ID = @WFID

IF @WFType = 'UrgentAfter' AND @InvPow = 1 AND @ELevel = 2
	BEGIN
		-- >> TAMS_DTL_PFR
		-- >> PRINT 'EMAIL'

		DECLARE @OCCEmail	NVARCHAR(1000)
		DECLARE @OCCRole	NVARCHAR(50)
		DECLARE @OCCEmailAdd	NVARCHAR(100)

		SET @OCCEmail = ''
		SET @OCCEmailAdd = ''
		SET @OCCRole = @Line + '_PFR'
		SET @EMTARStatus = 'Approved'
		SET @EMMsg = ''

		SELECT @OCCEmail = @OCCEmail + ISNULL(a.Email + ', ', '') 
			FROM TAMS_User a, TAMS_User_Role b, TAMS_Role c
				WHERE a.Userid = b.UserID
					AND b.RoleID = c.ID
					AND a.IsActive = 1
					AND @TARDate BETWEEN a.ValidFrom AND a.ValidTo
					AND c.Role = @OCCRole

		-- >> 2024-01-16 Changes to add nelocc email to the recipient list 
		if @TrackType = 'Mainline'
		Begin
			SELECT @OCCEmailAdd = ParaValue3 
			FROM TAMS_Parameters
				WHERE ParaCode = 'OCCContact'
					AND ParaValue1 = @Line
					AND LTRIM(RTRIM(ISNULL(ParaValue3, ''))) <> ''
		end;
		else  -- Depot 
		begin 
			SELECT @OCCEmailAdd = ParaValue3 
			FROM TAMS_Parameters
				WHERE ParaCode = 'SDSContact'
					AND ParaValue1 = @Line
					AND LTRIM(RTRIM(ISNULL(ParaValue3, ''))) <> ''
		end;
		

		IF LEN(LTRIM(RTRIM(@OCCEmail))) > 0
			BEGIN
				-- >> 2024-01-16 Changes to add nelocc email to the recipient list 
				SET @OCCEmail = LEFT(LTRIM(RTRIM(@OCCEmail)), LEN(LTRIM(RTRIM(@OCCEmail))) - 1)
				SET @OCCEmail = @OCCEmail + ', ' + @OCCEmailAdd
			END
		print 'before sending email to OCC';
		print @EMTARStatus; 
		EXEC sp_TAMS_Email_Urgent_TAR_OCC @TARID, @EMTARStatus, @EMTARNo, @Remarks, @OCCEmail, @EMMsg OUTPUT
	END

-- >>

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR INSERTING INTO TAMS_TAR'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END
