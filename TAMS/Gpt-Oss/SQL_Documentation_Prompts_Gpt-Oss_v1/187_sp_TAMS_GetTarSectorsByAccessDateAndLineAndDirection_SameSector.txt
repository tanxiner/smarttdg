CREATE PROCEDURE [dbo].[sp_TAMS_GetTarSectorsByAccessDateAndLineAndDirection_SameSector]
	@AccessDate as date,
	@Line AS nvarchar(10) = NULL,
	@Direction AS nvarchar(10) = NULL
AS

CREATE TABLE #TMP
(
	ID int not null,	
	Line nvarchar(10),	
	SameSector int,
	IsSpecialSector bit,
	Direction nvarchar(10),
	Sector nvarchar(50),
	TarNo nvarchar(50),
	AccessDate date,
	AccessType nvarchar(50),
	SectorId int,
	IsBuffer bit,
	ColourCode nvarchar(50),
	IsGap bit

);
IF @Line = 'DTL'
BEGIN
	insert into #TMP(ID, Line, SameSector, IsSpecialSector, Direction, Sector, TarNo, AccessDate, AccessType, SectorId, IsBuffer, ColourCode, IsGap)
	SELECT a.ID, a.Line, a.SameSector, a.IsSpecialSector, Direction, Sector, TarNo, AccessDate, AccessType, SectorId, IsBuffer, ColourCode, IsGap
	FROM TAMS_Sector a
	LEFT JOIN TAMS_TAR_Sector c
	LEFT JOIN TAMS_TAR b
	on b.Id = c.TarId
	on a.id = c.SectorId 
	AND AccessDate = @AccessDate
	AND TARStatusId = 8
	AND (AccessType = 'Possession' OR (AccessType = 'Possession' OR isExclusive = 1))
	WHERE a.Line = @Line
	AND Direction = @Direction
	AND IsActive = 1 
	AND EffectiveDate <= GETDATE() AND ExpiryDate >= GETDATE()
	ORDER BY [Order] ASC
END
ELSE IF @Line = 'NEL'
BEGIN
	insert into #TMP(ID, Line, SameSector, IsSpecialSector, Direction, Sector, TarNo, AccessDate, AccessType, SectorId, IsBuffer, ColourCode, IsGap)
	SELECT a.ID, a.Line, a.SameSector, a.IsSpecialSector, Direction, Sector, TarNo, AccessDate, AccessType, SectorId, IsBuffer, ColourCode, IsGap
	FROM TAMS_Sector a
	LEFT JOIN TAMS_TAR_Sector c
	LEFT JOIN TAMS_TAR b
	on b.Id = c.TarId
	on a.id = c.SectorId 
	AND AccessDate = @AccessDate
	AND TARStatusId = 9
	AND (AccessType = 'Possession' OR (AccessType = 'Possession' OR isExclusive = 1))
	WHERE a.Line = @Line
	AND Direction = @Direction
	AND IsActive = 1 
	AND EffectiveDate <= GETDATE() AND ExpiryDate >= GETDATE()
	ORDER BY [Order] ASC
