CREATE PROCEDURE [dbo].[sp_TAMS_TOA_Delete_Parties]
	@PartiesID	AS BIGINT = 0,
	@TOAID		AS BIGINT = 0,
	@Message	AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	


IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END


BEGIN TRY

SET @Message = ''
DECLARE @PartiesCtr BIGINT


SELECT @PartiesCtr = COUNT(*) 
	FROM TAMS_TOA_Parties
		WHERE TOAId = @TOAID

if @PartiesCtr<=2
	BEGIN
		RAISERROR('Please ensure there must be minimum 2 parties before deleting.',11,1)
	END
ELSE
	BEGIN

		DELETE FROM TAMS_TOA_Parties
		WHERE TOAId = @TOAID
			AND Id = @PartiesID

	-- DECLARE @PartiesCtr BIGINT

		UPDATE TAMS_TOA
			SET NoOfParties	= (
			SELECT COUNT(*) 
			FROM TAMS_TOA_Parties
				WHERE TOAId = @TOAID
			)
		WHERE Id = @TOAID
	END


IF @@ERROR <> 0 BEGIN
	PRINT 'ERROR INSERTING'
	SET @Message = 'ERROR INSERTING TAMS_TOA_Parties'
	GOTO TRAP_ERROR
END


END TRY
BEGIN CATCH
	DECLARE @ErrorMessage NVARCHAR(4000);
	DECLARE @ErrorSeverity INT;
	DECLARE @ErrorState INT;

	SELECT @ErrorMessage=ERROR_MESSAGE(),
	@ErrorSeverity=ERROR_SEVERITY(),
	@ErrorState=ERROR_STATE()

	RAISERROR(@ErrorMessage, @ErrorSeverity,@ErrorState)
END CATCH

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message



END
