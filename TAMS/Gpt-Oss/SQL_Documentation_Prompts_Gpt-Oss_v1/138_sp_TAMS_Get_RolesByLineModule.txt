CREATE PROCEDURE  [dbo].[sp_TAMS_Get_RolesByLineModule] 
	@Line	NVARCHAR(100) = NULL,
	@TrackType	NVARCHAR(50) = NULL,
	@Module NVARCHAR(100) = NULL
AS
BEGIN
	
	SELECT ID, Line,TrackType, Module, Role, RoleDesc
	FROM TAMS_Role
	WHERE Line LIKE @Line
	AND TrackType LIKE @TrackType
	AND Module LIKE @Module;
	
END
