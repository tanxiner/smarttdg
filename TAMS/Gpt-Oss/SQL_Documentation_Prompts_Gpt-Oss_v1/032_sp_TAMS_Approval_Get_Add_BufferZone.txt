CREATE PROCEDURE [dbo].[sp_TAMS_Approval_Get_Add_BufferZone]
	@TARID		BIGINT
AS

BEGIN

-- >> Buffer Zone -- Additional
-- >> Table[0]
SELECT a.Sector AS StrSector, a.ID AS SectorID
	FROM TAMS_Sector a, TAMS_TAR_Sector b
		WHERE a.ID = b.SectorId
			AND b.TARId = @TARID
			AND b.IsBuffer = 1
	ORDER BY a.ID

END
