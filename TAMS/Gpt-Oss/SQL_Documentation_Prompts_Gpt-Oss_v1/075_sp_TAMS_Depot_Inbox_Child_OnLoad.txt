CREATE procedure  [dbo].[sp_TAMS_Depot_Inbox_Child_OnLoad]
	@Line		AS NVARCHAR(10) = NULL,
	@TrackType		AS NVARCHAR(50) = NULL,
	@AccessDate	AS NVARCHAR(20) = NULL,
	@TARType	AS NVARCHAR(20) = NULL,
	@LoginUser	AS NVARCHAR(50) = NULL,
	@SectorID	AS INT
AS

BEGIN

DECLARE @UserID INT

SELECT @UserID = Userid 
	FROM TAMS_USER
		WHERE LoginId = @LoginUser

DECLARE @CurrDate	DATETIME
SET @CurrDate = CONVERT(DATETIME, CONVERT(NVARCHAR(20), GETDATE(), 103), 103)

CREATE TABLE #TmpSector 
(Line		NVARCHAR(10),
SectorID	INT,
SectorStr	NVARCHAR(50),
SectorOrder	INT )

CREATE TABLE #TmpInbox 
(TARID			BIGINT,
TARNo			NVARCHAR(20),
TARType			NVARCHAR(10),
AccessDate		NVARCHAR(20),
AccessTimeSlot		NVARCHAR(50),
AccessType		NVARCHAR(20),
SectorID		INT,
SectorStr		NVARCHAR(50) )

CREATE TABLE #TmpInboxList 
(TARID			BIGINT,
TARNo			NVARCHAR(20),
TARType			NVARCHAR(10),
AccessDate		NVARCHAR(20),
AccessTimeSlot		NVARCHAR(50),
AccessType		NVARCHAR(20),
SectorID		INT,
SectorStr		NVARCHAR(50) )

TRUNCATE TABLE #TmpSector
TRUNCATE TABLE #TmpInbox
TRUNCATE TABLE #TmpInboxList

INSERT INTO #TmpSector 
	(Line, SectorID, SectorStr, SectorOrder)
SELECT Line, ID, Sector, [Order] 
	FROM TAMS_Sector
		WHERE Line = @Line
		AND TrackType = @TrackType
			AND IsActive = 1
			AND @CurrDate BETWEEN EffectiveDate AND ExpiryDate
		ORDER BY [Order]

INSERT INTO #TmpInbox
	(TARID, TARNo, TARType, AccessDate,AccessTimeSlot, AccessType, SectorID, SectorStr)
SELECT a.[Id], a.[TARNo], a.[TARType], CONVERT(NVARCHAR(20), a.[AccessDate], 103),a.AccessTimeSlot, a.[AccessType], b.[Id], b.[Sector] 
	FROM TAMS_TAR a, TAMS_Sector b, TAMS_TAR_Sector c, TAMS_TAR_Workflow d, TAMS_Endorser e
		WHERE b.[Id] = c.[SectorId]
			AND c.[TarId] = a.[Id]
			AND a.[Id] = d.[TARid]
			AND d.[EndorserId] = e.[Id]
			AND d.[WFStatus] like '%Pending%'
			-- >> AND ISNULL(b.[IsGap], 0) = 0
			--AND  d.[UserId]  is null
			AND ISNULL(d.[UserId], 0) = 0
			AND e.[RoleId] IN (SELECT f.RoleID FROM TAMS_User_Role f WHERE f.UserID = @UserID) 
			AND (a.AccessDate = CONVERT(DATE, @AccessDate, 103) OR (ISNULL(@AccessDate, '') = '' AND a.AccessDate>GETDATE()))
			AND (a.TARType = @TARType OR ISNULL(@TARType, '') = '')
			AND c.[IsBuffer] = 0
			AND a.TrackType = @TrackType
UNION
SELECT a.[Id], a.[TARNo], a.[TARType], CONVERT(NVARCHAR(20), a.[AccessDate], 103),a.AccessTimeSlot, a.[AccessType], b.[Id], b.[Sector] 
	FROM TAMS_TAR a, TAMS_Sector b, TAMS_TAR_Sector c, TAMS_TAR_Workflow d, TAMS_Endorser e
		WHERE b.[Id] = c.[SectorId]
			AND c.[TarId] = a.[Id]
			AND a.[Id] = d.[TARid]
			AND d.[EndorserId] = e.[Id]
			AND d.[WFStatus] like '%Pending%'
			AND d.[UserId] = @UserID
			AND (a.AccessDate = CONVERT(DATE, @AccessDate, 103) OR (ISNULL(@AccessDate, '') = '' AND a.AccessDate>GETDATE()))
			AND (a.TARType = @TARType OR ISNULL(@TARType, '') = '')
			AND c.[IsBuffer] = 0
			AND a.TrackType = @TrackType

DECLARE @Cur01 CURSOR;
DECLARE @CTARID			BIGINT
DECLARE @CTARNo			NVARCHAR(20)
DECLARE @CTARType		NVARCHAR(10)
DECLARE @CAccessDate	NVARCHAR(20)
DECLARE @CAccessTimeSlot	NVARCHAR(50)
DECLARE @CAccessType	NVARCHAR(20)
DECLARE @CSectorID		INT
DECLARE @CSectorStr		NVARCHAR(50)
DECLARE @CDirection		INT

DECLARE @Cur02 CURSOR;
DECLARE @C2ActionBy		INT
DECLARE @ActionByChk	INT

SET @Cur01 = CURSOR FAST_FORWARD 
FOR
	SELECT TARID, TARNo, TARType, AccessDate,AccessTimeSlot, AccessType, SectorID, SectorStr
		FROM #TmpInbox
			ORDER BY TARID, SectorID;		

OPEN @Cur01
	
FETCH NEXT FROM @Cur01 INTO @CTARID, @CTARNo, @CTARType, @CAccessDate,@CAccessTimeSlot, @CAccessType, @CSectorID, @CSectorStr
	
WHILE @@FETCH_STATUS = 0 
	
	BEGIN

		SET @ActionByChk = 0

		IF (SELECT COUNT(*) 
				FROM TAMS_TAR_Workflow
					WHERE TARId = @CTARID
						AND WFStatus <> 'Pending'
				) = 0
			BEGIN
				INSERT INTO #TmpInboxList
					(TARID, TARNo, TARType, AccessDate,AccessTimeSlot, AccessType, SectorID, SectorStr)
				VALUES
					(@CTARID, @CTARNo, @CTARType, @CAccessDate,@CAccessTimeSlot, @CAccessType, @CSectorID, @CSectorStr)
			END
		ELSE
			BEGIN
				-- >>
				SET @Cur02 = CURSOR FAST_FORWARD 
				FOR
					SELECT ActionBy
						FROM TAMS_TAR_Workflow
							WHERE TARId = @CTARID
								AND WFStatus <> 'Pending'

				OPEN @Cur02
	
				FETCH NEXT FROM @Cur02 INTO @C2ActionBy
	
				WHILE @@FETCH_STATUS = 0 
	
					BEGIN

						 -- >> Comment for 1 Man Testing
						IF (@UserID = @C2ActionBy)
							BEGIN
								SET @ActionByChk = @ActionByChk + 1
							END

						FETCH NEXT FROM @Cur02 INTO @C2ActionBy
					END 	

				CLOSE @Cur02
	
				DEALLOCATE @Cur02
				-- >>
			END

		IF (@ActionByChk = 0)
			BEGIN
				INSERT INTO #TmpInboxList
					(TARID, TARNo, TARType, AccessDate, AccessTimeSlot,AccessType, SectorID, SectorStr)
				VALUES
					(@CTARID, @CTARNo, @CTARType, @CAccessDate, @CAccessTimeSlot,@CAccessType, @CSectorID, @CSectorStr)
			END

		FETCH NEXT FROM @Cur01 INTO @CTARID, @CTARNo, @CTARType, @CAccessDate,@CAccessTimeSlot, @CAccessType, @CSectorID, @CSectorStr
	END 	

CLOSE @Cur01
	
DEALLOCATE @Cur01

SELECT b.TARID, b.TARNo, b.TARType, b.AccessDate,b.AccessTimeSlot, b.AccessType
	FROM #TmpInboxList b
		WHERE b.SectorID = @SectorID
	GROUP BY b.TARID, b.TARNo, b.TARType, b.AccessDate,b.AccessTimeSlot, b.AccessType
	ORDER BY b.TARID, b.TARNo, b.TARType, b.AccessDate,b.AccessTimeSlot, b.AccessType
 
DROP TABLE #TmpSector
DROP TABLE #TmpInbox
DROP TABLE #TmpInboxList

END
