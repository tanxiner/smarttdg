### ⬇️ RAW SQL INPUT ⬇️
CREATE PROCEDURE  [dbo].[sp_TAMS_GetDutyOCCRosterByParameters]
	@Line AS nvarchar(10) = NULL,
	@TrackType AS nvarchar(50) = NULL,
	@OperationDate as date,
	@Shift as nvarchar(1) = NULL,
	@RosterCode as nvarchar(50) = NULL,
	@ID as int = 0

AS
BEGIN

  select ID, Line, OperationDate, [Shift], RosterCode, DutyStaffId, [Name] as DutyStaffName
  from TAMS_OCC_Duty_Roster r , TAMS_User u
  where r.Line = @Line 
  and r.TrackType = @TrackType
  and r.DutyStaffId = u.Userid
  and r.OperationDate = @OperationDate
  and r.[shift] = @Shift
  and r.RosterCode = @RosterCode
  and r.id = @ID
  AND r.IsActive = 1  

END
