CREATE PROCEDURE [dbo].[sp_TAMS_WithdrawTarByTarID] 	
	@TarId AS integer = 0,
	@UID As integer = 0, 
	@Remark as nvarchar(1000) = null
AS
BEGIN
	DECLARE @Line as varchar(3),
			@StatusId as int,
			@Name as nvarchar(500)
	BEGIN TRY
		BEGIN TRANSACTION;
			BEGIN
				select @Line = line from TAMS_TAR where Id = @TarId;
				select @StatusId = WFStatusId from TAMS_WFStatus where Line = @Line and WFType = 'TARWFStatus' and WFStatus = 'Withdraw';
				select @Name = [Name] from TAMS_User where userid = @UID;

				UPDATE TAMS_TAR 
				SET TARStatusId = @StatusId, 
				WithdrawRemark= @Remark,
				WithdrawBy = @UID, 
				WithdrawDate = GETDATE(),
				UpdatedOn = GETDATE() ,
				UpdatedBy =@UID
				WHERE Id = @TarId;



				insert into TAMS_Action_Log
				values(@Line, 'TAR', 'Enquiry-Withdraw', @TarId, 'TAR Form has been withdrawn by ' + @Name + ' on ' + convert(varchar, getdate(), 103), GETDATE(), @UID);
			END;

		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		ROLLBACK TRANSACTION;
	END CATCH

END;
