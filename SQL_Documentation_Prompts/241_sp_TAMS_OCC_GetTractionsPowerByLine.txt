CREATE PROCEDURE [dbo].[sp_TAMS_OCC_GetTractionsPowerByLine]
	@Line as nvarchar(10) = null

AS
BEGIN
	SELECT [ID],
	[Line],
	[TractionPowerSection],
	[HSCBName],
	[TC],
	[Train],
	[Order]
	FROM [dbo].[TAMS_Traction_Power]
	WHERE [Line] = @Line
	AND EffectiveDate <= GETDATE() and ExpiryDate >= GETDATE()
	AND IsActive = 1
	ORDER BY [Order] ASC
END
