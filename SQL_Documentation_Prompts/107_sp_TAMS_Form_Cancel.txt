CREATE PROCEDURE [dbo].[sp_TAMS_Form_Cancel]
	@TARID			AS BIGINT = 0,
	@Message		AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT,
	@ErrorMsg		NVARCHAR(500)			

SET @TARID = 0
SET @ErrorMsg = ''

SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

-- >>
DELETE FROM TAMS_TAR
	WHERE Id = @TARID

DELETE FROM TAMS_TAR_Attachment_Temp
	WHERE TARId = @TARID

-- >>

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR DELETING TAMS_TAR'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message


END
