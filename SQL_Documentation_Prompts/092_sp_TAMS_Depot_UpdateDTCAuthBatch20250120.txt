CREATE PROCEDURE [dbo].[sp_TAMS_Depot_UpdateDTCAuthBatch20250120]
	-- Add the parameters for the stored procedure here
	--@username as nvarchar(50),
	--@authid as int,
	--@workflowid as int,
	--@statusid as int,
	--@val as bit=NULL,
	--@valstr as nvarchar(50)=NULL,
	--@powerzoneid as int=NULL,
	--@type as int, -- type is 1 if checkbox, 2 if dropdown
	--@spksid as int=NULL,
	 @success as bit=NULL OUTPUT,
	 @Message AS NVARCHAR(500) = NULL OUTPUT,
	 @str AS TAMS_DTC_AUTH READONLY
AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	SET NOCOUNT ON;

	DECLARE	@IntrnlTrans	TINYINT
	SELECT 	@IntrnlTrans 	= 0	

	IF (@@TRANCOUNT = 0) 
	BEGIN
		SET @IntrnlTrans = 1
		BEGIN TRAN
	END DECLARE @valstr as nvarchar(50)=NULL;
	 DECLARE @powerzoneid as int=NULL;
	 DECLARE @type as int; -- type is 1 if checkbox, 2 if dropdown
	 DECLARE @spksid as int=NULL;


	 DECLARE @username as nvarchar(50);
	 DECLARE @authid as int;
	 DECLARE @workflowid as int;
     DECLARE @statusid as int;
	 DECLARE @val as bit=NULL;
	
	DECLARE C CURSOR FOR
	SELECT username, authid, workflowid, statusid, val, valstr, powerzoneid, [type], spksid from @str 
	

	OPEN C;

	FETCH NEXT FROM C INTO @username, @authid, @workflowid, @statusid, @val, @valstr, @powerzoneid, @type, @spksid 
	WHILE @@FETCH_STATUS=0
	BEGIN 
		DECLARE @access as bit=0;
		DECLARE @WFStatusID as int;
		SELECT @WFStatusID=TAMS_WFStatus.WFStatusId from TAMS_WFStatus where ID=@workflowid

		DECLARE @toastatus as int;
		SELECT @toastatus=TAMS_TOA.TOAStatus from TAMS_Depot_Auth INNER JOIN TAMS_TOA ON TAMS_TOA.TARId=TAMS_Depot_Auth.TarID 
		where TAMS_Depot_Auth.ID=@authid

		if @statusid > @WFStatusID
		BEGIN
			SET @WFStatusID=@statusid
		END
		

		DECLARE @accessDate as DateTime
		Select @accessDate=AccessDate from TAMS_Depot_Auth where id=@authid

		DECLARE @mworkflowid as INT
		Select @mworkflowid=ID from TAMS_Workflow where WorkflowType='OCCAuth' and TrackType='DEPOT' and Line='NEL'

		IF EXISTS(Select TAMS_Roster_Role.RoleId,TAMS_OCC_Duty_Roster.DutyStaffId,TAMS_User.LoginId from TAMS_OCC_Duty_Roster INNER JOIN TAMS_Roster_Role ON TAMS_Roster_Role.RosterCode=TAMS_OCC_Duty_Roster.RosterCode
		and TAMS_OCC_Duty_Roster.Line=TAMS_Roster_Role.Line and TAMS_OCC_Duty_Roster.TrackType=TAMS_Roster_Role.TrackType
		INNER JOIN TAMS_User ON TAMS_User.Userid=TAMS_OCC_Duty_Roster.DutyStaffId 
		INNER JOIN TAMS_Endorser ON TAMS_Endorser.RoleId=TAMS_Roster_Role.RoleId
		where TAMS_OCC_Duty_Roster.tracktype='Depot' and OperationDate=@accessDate
		and LoginId=@username and TAMS_Endorser.WorkflowId=@mworkflowid and TAMS_Endorser.WFStatusId=@WFStatusID)
		BEGIN
			SET @access=1
		END

		--IF not EXISTS(select TAMS_User.LoginId from TAMS_Endorser INNER JOIN TAMS_User_Role 
		--ON TAMS_User_Role.ID=TAMS_Endorser.RoleId INNER JOIN TAMS_User
		--ON TAMS_User.LoginId=@username 
		--where TAMS_Endorser.ID=@workflowid 
		--and TAMS_User.LoginId IN (SELECT DutyStaffId from TAMS_OCC_Duty_Roster where TrackType='DEPOT' and OperationDate=@accessDate))
		--BEGIN
		--	SET @access=1
		--END

		if @access=0
		BEGIN
			--throw exception
			set @Message='User ' + @username + ' do not have access to update the information';
		
			GOTO TRAP_ERROR
		END
		
		
		if  @WFStatusID<>6 and @WFStatusID<>13
		BEGIN
			if not Exists(Select * from TAMS_Depot_Auth_Workflow where WorkflowID=@workflowid)
			BEGIN
			--	if @WFStatusID=5 
			--	BEGIN
			--		if not Exists(Select * from TAMS_Depot_Auth_Powerzone where StatusID=@WFStatusID)
			--		BEGIN
			--			set @Message='Workflow not found or is invalid';
					
			--			GOTO TRAP_ERROR
			--		END
			--	END
			--	ELSE
			--	BEGIN
					set @Message='Workflow not found or is invalid';
					GOTO TRAP_ERROR
			--	END
			END
		END


		-- check if spks zone is still in used by another tar 
		if @WFStatusID=14
		BEGIN
			if Exists(
			Select * from (
			Select B.* from (Select AccessDate,ID from TAMS_Depot_Auth) A  
					join TAMS_Depot_Auth B ON B.AccessDate=A.AccessDate
				where A.ID=@authid) C JOIN TAMS_Depot_DTCAuth_SPKS ON C.ID=TAMS_Depot_DTCAuth_SPKS.AuthID 
				where C.ID<>@authid and TAMS_Depot_DTCAuth_SPKS.SPKSID=@spksid and DepotAuthStatusId=8)
				BEGIN
					set @Message='The SPKS Zone ' + CONVERT(nvarchar(50),@spksid) + ' is being access by another TAR';
					GOTO TRAP_ERROR
				END
		END
		
		-- check if power zone is still in used by another tar 
		--if @WFStatusID=11
		--BEGIN 
		--	if Exists(
		--	Select * from (
		--	Select B.* from (Select AccessDate,ID from TAMS_Depot_Auth) A  
		--			join TAMS_Depot_Auth B ON B.AccessDate=A.AccessDate
		--		where A.ID=@authid) C JOIN TAMS_Depot_Auth_Powerzone ON C.ID=TAMS_Depot_Auth_Powerzone.AuthID 
		--		where C.ID<>@authid and TAMS_Depot_Auth_Powerzone.PowerzoneID=@powerzoneid and DepotAuthStatusId=8)
		--		BEGIN
		--			DECLARE @powersector as nvarchar(50)
		--			SELECT DISTINCT @powersector=PowerSector from TAMS_Depot_Auth_Powerzone INNER JOIN 
		--			TAMS_Power_Sector ON TAMS_Depot_Auth_Powerzone.PowerzoneID=TAMS_Power_Sector.ID where 
		--			TAMS_Depot_Auth_Powerzone.PowerzoneID=@powerzoneid and AuthID=@authid


		--			set @Message='The power Zone ' + @powersector + ' is being access by another TAR';
		--			GOTO TRAP_ERROR
		--		END
		--END


		--check if workflow is already updated
		IF Exists(Select * from TAMS_Depot_Auth_Workflow where DTCAuthId=@authid and WorkflowID=@workflowid
		and WFStatus is not NULL and CheckBoxValue is not null)
		BEGIN
			-- already updated
			-- throw error
			set @Message='Workflow ID ' + CONVERT(nvarchar(10),@workflowid) + ' is already updated';
			GOTO TRAP_ERROR

		END

		DECLARE @newstatusid as int;
		-- get next status
		select @newstatusid=WFStatusId+1 from TAMS_WFStatus where WFType='dtcauth' and WFStatusId=@statusid
		and isActive=1
		order by [Order] ASC;


		--update current workflow
		UPDATE TAMS_Depot_Auth_Workflow
		SET WFStatus=CASE when @type=1 THEN 'Completed' ELSE @valstr END, ActionBy=@username, ActionOn=getDate(),CheckBoxValue=CASE WHEN @type=1 THEN @val ELSE NULL END
		where WorkflowID=@workflowid and DTCAuthId=@authid and
		@WFStatusID<>4 and @WFStatusID<>5 and @WFStatusID<>6 and @WFStatusID<>11 and @WFStatusID<>12 and @WFStatusID<>14 and @WFStatusID<>7



		-- get next workflow
		DECLARE @currentStatus as int;
		DECLARE @newworkflowid as int;

		select @currentStatus=WFStatusId from TAMS_WFStatus where WFType='dtcauth' and Id=@workflowid
		Select @newworkflowid=ID from TAMS_WFStatus where WFType='dtcauth' and WFStatusId=(@currentStatus+1)

		-- insert next workflow 
		INSERT INTO TAMS_Depot_Auth_Workflow
		SELECT @authid, null, null, null, @newworkflowid,null,0
		where @WFStatusID<>7 and @WFStatusID<>4 and @WFStatusID<>5 and @WFStatusID<>6 and @WFStatusID<>11 and @WFStatusID<>12 and @WFStatusID<>14

		-- if surrender, enable line clear stage
		if @WFStatusID=7
		BEGIN

				UPDATE TAMS_Depot_Auth_Workflow
				SET WFStatus=CASE when @type=1 THEN 'Completed' ELSE @valstr END, ActionBy=@username, ActionOn=getDate(),CheckBoxValue=CASE WHEN @type=1 THEN @val ELSE NULL END
				where WorkflowID=@workflowid and DTCAuthId=@authid 

				INSERT INTO TAMS_Depot_Auth_Workflow
				SELECT @authid, null, null, null, @newworkflowid,null,0

				UPDATE TAMS_Depot_Auth
				SET DepotAuthStatusId=@newstatusid,UpdatedOn=getDate(),UpdatedBy=@username
				where ID=@authid
			
		END


		if @WFStatusID=10
		BEGIN
			-- set powerzone id to next id
			UPDATE TAMS_Depot_Auth_Powerzone
			SET TAMS_Depot_Auth_Powerzone.StatusID=TAMS_Depot_Auth_Powerzone.StatusID+1 
			where AuthID=@authid
		END

		if @WFStatusID=3
		BEGIN
			UPDATE TAMS_Depot_DTCAuth_SPKS
			SET ProtectOffTiming=getDate(), ProtectOffType=@valstr,ProtectOffActionBy=@username
			where AuthID=@authid
		END
	


		if @WFStatusID=4
		BEGIN

			UPDATE TAMS_Depot_Auth_Powerzone
			SET 
			--RackedOutTiming=getDate(),RackedOutType=@valstr,RackedOffActionBy=@username,
			PowerOffTiming=getDate(),PowerOffType=@valstr,PowerOffActionBy=@username,
			statusID=@newstatusid
			where AuthID=@authid and PowerzoneID=@powerzoneid



		END

		if @WFStatusID=11
		BEGIN

			UPDATE TAMS_Depot_Auth_Powerzone
			SET  PowerOnTiming=getDate(),PowerOnType=@valstr,PowerOnActionBy=@username,statusID=@newstatusid
			where AuthID=@authid and PowerzoneID=@powerzoneid

		END


		if @WFStatusID=14
		BEGIN
			UPDATE TAMS_Depot_DTCAuth_SPKS
			SET ProtectOnTiming=getDate(), ProtectOnType=@valstr,ProtectOnActionBy=@username,StatusID=@newstatusid
			where AuthID=@authid and SPKSID=@spksid
		END

		if @WFStatusID=5 or @WFStatusID=6
		BEGIN
			UPDATE TAMS_Depot_Auth_Powerzone
			SET RackedOutTiming=getDate(),RackedOutType=@valstr,RackedOffActionBy=@username,statusID=@newstatusid
			where AuthID=@authid and PowerzoneID=@powerzoneid

		END


		if @WFStatusID=12
		BEGIN
			UPDATE TAMS_Depot_Auth_Powerzone
			SET RackedInTiming=getDate(),RackedInType=@valstr,RackedInActionBy=@username,statusID=@newstatusid
			where AuthID=@authid and PowerzoneID=@powerzoneid

			UPDATE TAMS_Depot_DTCAuth_SPKS
			SET StatusID=@newstatusid+1
			where AuthID=@authid and SPKSID NOT IN 
			(
				Select DISTINCT SPKSId from (
				(Select PowerzoneID,TAMS_Depot_Auth_Powerzone.StatusID from TAMS_Depot_Auth_Powerzone
				where TAMS_Depot_Auth_Powerzone.AuthID=@authid) B
				LEFT JOIN
				(Select DISTINCT PowerSectorId, PowerSector,SPKSId  from 
				TAMS_Track_Power_Sector INNER JOIN TAMS_Track_SPKSZone 
				ON TAMS_Track_Power_Sector.SectorId=TAMS_Track_SPKSZone.SectorId 
				INNER JOIN TAMS_Power_Sector 
				ON TAMS_Power_Sector.ID=TAMS_Track_Power_Sector.PowerSectorId 
				) A ON A.PowerSectorId=B.PowerzoneID)
				where StatusID<>13
				group by SPKSId,PowerSectorId,StatusID
			) 
			--and SPKSID not in
			--(
			--	Select TAMS_Depot_DTCAuth_SPKS.SPKSID from (
			--	Select B.* from (Select AccessDate,ID from TAMS_Depot_Auth) A  
			--		join TAMS_Depot_Auth B ON B.AccessDate=A.AccessDate
			--	where A.ID=@authid ) C JOIN TAMS_Depot_DTCAuth_SPKS ON C.ID=TAMS_Depot_DTCAuth_SPKS.AuthID 
			--	where C.ID<>@authid and TAMS_Depot_DTCAuth_SPKS.StatusID<14 and TAMS_Depot_DTCAuth_SPKS.StatusID is null
			--) 
			and SPKSID not in
			(
				Select SPKSID from TAMS_Depot_DTCAuth_SPKS 
				where authID=@authid and StatusID=15
			)
		
		END
	
	

		if @WFStatusID=4 or @WFStatusID=5 or @WFStatusID=6 or @WFStatusID=11 or @WFStatusID=12 or @WFStatusID=14
		BEGIN
			DECLARE @checkstatus as int;
		
			if @WFStatusID=4 or @WFStatusID=11
			BEGIN
				Select @checkstatus=WFStatusId from TAMS_WFStatus where WFStatusId=@WFStatusID
			END
			ELSE
			BEGIN
				Select @checkstatus=WFStatusId+1 from TAMS_WFStatus where WFStatusId=@WFStatusID
			END

		
			DECLARE @newstatus as int

			if @WFStatusID=14
			BEGIN
			if not exists(Select * from TAMS_Depot_DTCAuth_SPKS where AuthID=@authid and (StatusID<@checkstatus or StatusID is null))
				BEGIN
					SET @newstatus=@checkstatus+1;

					UPDATE TAMS_Depot_Auth
					SET DepotAuthStatusId=@newstatus,UpdatedOn=getDate(),UpdatedBy=@username
					where ID=@authid

					UPDATE TAMS_Depot_Auth_Workflow
					SET WFStatus=CASE when @type=1 THEN 'Completed' ELSE @valstr END, ActionBy=@username, ActionOn=getDate(),CheckBoxValue=CASE WHEN @type=1 THEN @val ELSE NULL END
					where WorkflowID=@workflowid and DTCAuthId=@authid 

					Select @newworkflowid=ID from TAMS_WFStatus where WFType='dtcauth' and WFStatusId=@newstatus


					INSERT INTO TAMS_Depot_Auth_Workflow
					VALUES(@authid, 'Completed', getDate(), @username, @newworkflowid,null,0)
				END
			END
			ELSE
			BEGIN
			if not exists(Select * from TAMS_Depot_Auth_Powerzone where AuthID=@authid and (StatusID<@checkstatus or StatusID is null))
				BEGIN
					SET @newstatus=@checkstatus+1;

					UPDATE TAMS_Depot_Auth
					SET DepotAuthStatusId=@newstatus,UpdatedOn=getDate(),UpdatedBy=@username
					where ID=@authid

					UPDATE TAMS_Depot_Auth_Workflow
					SET WFStatus=CASE when @type=1 THEN 'Completed' ELSE @valstr END, ActionBy=@username, ActionOn=getDate(),CheckBoxValue=CASE WHEN @type=1 THEN @val ELSE NULL END
					where WorkflowID=@workflowid and DTCAuthId=@authid 

					Select @newworkflowid=ID from TAMS_WFStatus where WFType='dtcauth' and WFStatusId=@newstatus

					INSERT INTO TAMS_Depot_Auth_Workflow
					VALUES(@authid, null, null, null, @newworkflowid,null,0)
				END
			END
		END
		ELSE
		BEGIN

			--check for cancel tar
			if @toastatus=6 
			BEGIN
				DECLARE @notrainsds INT;
				SELECT @notrainsds=WFStatusID from TAMS_WFStatus where WFStatus='Pending No Train Movement (SDS)'  and TrackType='Depot' and WFType='DTCAuth' and Line='NEL'

				DECLARE @notraindtc INT;
				SELECT @notraindtc=WFStatusID from TAMS_WFStatus where WFStatus='Pending No Train Movement (DTC)' and TrackType='Depot' and WFType='DTCAuth' and Line='NEL'
			

				DECLARE @lineclearcerdtctstatus INT;
				SELECT @lineclearcerdtctstatus=WFStatusID from TAMS_WFStatus where WFStatus='Line Clear Certification (TOA/SCD) (DTC)' and TrackType='Depot' and WFType='DTCAuth' and Line='NEL'

				DECLARE @lineclearcertccstatus INT;
				SELECT @lineclearcertccstatus=WFStatusID from TAMS_WFStatus where WFStatus='Line Clear Certification (TOA/SCD) (CC)' and TrackType='Depot' and WFType='DTCAuth' and Line='NEL'
			
				DECLARE @completestatus INT;
				SELECT @completestatus=WFStatusID from TAMS_WFStatus where WFStatus='Completed' 

		
				-- check if cancel toa
				DECLARE @cancelstatusid as int
				SELECT @cancelstatusid=TAMS_WFStatus.WFStatusId-1 from TAMS_Depot_Auth_Workflow INNER JOIN TAMS_WFStatus 
				ON TAMS_Depot_Auth_Workflow.WorkflowID=TAMS_WFStatus.ID
				where DTCAuthId=@authid and TAMS_Depot_Auth_Workflow.isCancelled=1 

				if @cancelstatusid=@notrainsds
				BEGIN
					SET @newstatusid=CASE WHEN @WFStatusID=@lineclearcerdtctstatus THEN  @completestatus+1 ELSE @newstatusid END
				END
				ELSE if @cancelstatusid=@notraindtc
				BEGIN 
					SET @newstatusid=CASE WHEN @WFStatusID=@lineclearcertccstatus THEN  @completestatus+1 ELSE @newstatusid END
				END
			END
		
			UPDATE TAMS_Depot_Auth
			SET DepotAuthStatusId=@newstatusid,UpdatedOn=getDate(),UpdatedBy=@username
			where ID=@authid and @WFStatusID<>7
			
		END


		FETCH NEXT FROM C INTO @username, @authid, @workflowid, @statusid, @val, @valstr, @powerzoneid, @type, @spksid 
	END
	CLOSE C;
	DEALLOCATE C;

	IF @@ERROR <> 0 BEGIN
		SET @Message = 'ERROR INSERTING INTO Depot Authorization module.'
		
		PRINT 'trap error'
		GOTO TRAP_ERROR
	END

	FORCE_EXIT_PROC:	
	
		PRINT 'trap error 1'
		IF @IntrnlTrans = 1 COMMIT TRAN
		set @success=1;
		RETURN;
		--RETURN @Message 

	TRAP_ERROR:
	
		PRINT 'trap error 2'
		IF @IntrnlTrans = 1 ROLLBACK TRAN
		set @success=0;
		RETURN;
		--RETURN @Message

END
