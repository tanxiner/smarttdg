CREATE PROCEDURE [dbo].[sp_TAMS_GetTarStationsByTarId]
	@TarId as integer = 0

AS
BEGIN
	SELECT a.ID, a.Line, b.TarId, a.StationCode, a.StationName, a.StationLongName
	FROM TAMS_Station a
	INNER JOIN TAMS_TAR_Station b
	on a.id = b.StationId
	AND b.TARId = @TarId
	ORDER BY [Order] ASC
END
