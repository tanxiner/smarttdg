CREATE PROCEDURE  [dbo].[sp_TAMS_Insert_UserRoleByUserIDRailModule] 	
	@UserID			INT,
	@Rail			NVARCHAR(10),
	@TrackType		NVARCHAR(50),
	@Module			NVARCHAR(10),
	@RoleID			INT,
	@UpdatedBy		INT
AS
BEGIN
	BEGIN TRY
		BEGIN TRANSACTION;
			-- TAMS_User_Role
			IF NOT EXISTS(SELECT * FROM TAMS_User_Role WHERE UserID = @UserID AND RoleID = @RoleID)
			BEGIN
				INSERT INTO TAMS_User_Role 
				VALUES (@UserID, @RoleID, @Rail,@TrackType, GETDATE(), @UpdatedBy, GETDATE(), @UpdatedBy);

			END;
		COMMIT TRANSACTION;
	END TRY
	BEGIN CATCH
		ROLLBACK TRANSACTION;
	END CATCH

END;
