CREATE PROCEDURE  [dbo].[sp_TAMS_Get_RegistrationInboxByUserID] 
	@UserID INT
AS
BEGIN
	--SELECT IIF(r.IsExternal = 1, 'External', 'Internal') AS 'UserType', r.Name, r.Email, r.UENNo, r.Company,
	--r.SBSTContactPersonName, r.CreatedOn, rm.ID, rm.RegID, rm.Line, rm.Module, e.Title, wf.WFStatus
	--FROM TAMS_Registration r, TAMS_Reg_Module rm, TAMS_Endorser e, TAMS_WFStatus wf
	--WHERE r.id = rm.RegId 
	--AND e.ID = rm.RegStatus
	--AND e.RoleId IN 
	--(
	--	SELECT ur.RoleID 
	--	FROM TAMS_User_Role ur 
	--	WHERE ur.UserID = @UserID
	--)
	--AND wf.WFType = 'UserRegStatus'
	--AND wf.Line = rm.Line
	--AND wf.WFStatusID = e.WFStatusId


	-- Option 2 - fail as tams_endorser is generic for OCC/TAR. There is no indicator
	--SELECT IIF(r.IsExternal = 1, 'External', 'Internal') AS 'UserType', r.Name, r.Email, r.UENNo, r.Company,
	--r.SBSTContactPersonName, r.CreatedOn, rm.ID, rm.RegID, rm.Line, rm.Module, e.Title, wf.WFStatus
	--FROM TAMS_Registration r, TAMS_Reg_Module rm, TAMS_Endorser e, TAMS_WFStatus wf
	--WHERE r.id = rm.RegId 
	----AND e.ID = rm.RegStatus
	--AND e.RoleId IN 
	--(
	--	SELECT ur.RoleID 
	--	FROM TAMS_User_Role ur 
	--	WHERE ur.UserID = @UserID
	--)
	--AND wf.WFType = 'UserRegStatus'
	--AND wf.Line = rm.Line
	--AND wf.WFStatusID = rm.regStatus
	--AND e.WFStatusID = rm.regStatus
	--AND e.Line = rm.Line
	--AND e.WorkflowID = (SELECT ID FROM TAMS_Workflow 
	--					WHERE Line = rm.Line 
	--					AND WorkFlowType = (SELECT IIF(r.IsExternal = 1, 'ExtUser', 'IntUser')) 
	--					AND GETDATE() >= EffectiveDate AND GETDATE() <= ExpiryDate AND IsActive = 1)

	--To Keep Role ID List	
	CREATE TABLE #RegistrationTable
	(
		--ROWID [int] IDENTITY(1,1),
		RegID INT,
		ID INT,
		Line NVARCHAR(10),
		TrackType NVARCHAR(50),
		Module NVARCHAR(10),
		WFStatus NVARCHAR(100),
		UserType NVARCHAR(20),
		Name NVARCHAR(500),
		UENNo	NVARCHAR(100),
		Company	NVARCHAR(200),
		SBSTContactPersonName NVARCHAR(200),
		CreatedOn DATETIME
	);	

	DECLARE @Line NVARCHAR(10);
	DECLARE @TrackType NVARCHAR(50);
	DECLARE @Module NVARCHAR(10);
	DECLARE @Role NVARCHAR(100);
	
	DECLARE @RegId INT;
	DECLARE @RegStatus INT;
	DECLARE @WorkflowId INT;
	DECLARE @EndorseId INT;

	DECLARE @UpdatedBy INT;

	DECLARE cur CURSOR FOR

	SELECT r.Line,r.TrackType,  r.Module, r.Role FROM TAMS_User_Role ur, TAMS_Role r
	WHERE ur.RoleID = r.ID
	AND ur.Userid = @UserID
	AND (r.Role LIKE '%_SysAdmin' OR r.Role LIKE '%_SysApprover')

	OPEN cur;

	FETCH NEXT FROM cur INTO @Line,@TrackType, @Module, @Role;

	WHILE @@FETCH_STATUS = 0
	BEGIN
		
		If @Role LIKE '%_SysAdmin'
		BEGIN
			INSERT INTO #RegistrationTable 
			
				SELECT rm.RegID, rm.ID, rm.Line,rm.TrackType, rm.Module, wf.WFStatus, IIF(r.IsExternal = 1, 'External', 'Internal') AS 'UserType', r.name, r.UENNo, r.Company, r.SBSTContactPersonName, r.CreatedOn FROM TAMS_Registration r, TAMS_Reg_Module rm, TAMS_WFStatus wf
				WHERE r.ID = rm.RegID 
				AND wf.WFType = 'UserRegStatus'
				AND wf.Line = rm.Line
				AND rm.TrackType = @TrackType
				AND wf.WFStatusId = rm.RegStatus
				AND rm.RegStatus in (SELECT DISTINCT(WFStatusId) FROM TAMS_WFStatus WHERE WFStatus = 'Pending Company Registration' AND LINE = @Line and TrackType = @TrackType)
				AND rm.WFStatus = 'Pending'
			

			INSERT INTO #RegistrationTable 
				SELECT rm.RegID, rm.ID, rm.Line,rm.TrackType, rm.Module, wf.WFStatus, IIF(r.IsExternal = 1, 'External', 'Internal') AS 'UserType', r.name, r.UENNo, r.Company, r.SBSTContactPersonName, r.CreatedOn FROM TAMS_Registration r, TAMS_Reg_Module rm, TAMS_WFStatus wf
				WHERE r.ID = rm.RegID 
				AND wf.WFType = 'UserRegStatus'
				AND wf.Line = rm.Line
				AND rm.TrackType = @TrackType
				AND wf.WFStatusId = rm.RegStatus
				AND rm.RegStatus in (SELECT DISTINCT(WFStatusId) FROM TAMS_WFStatus WHERE WFStatus = 'Pending Company Approval' AND LINE = @Line and TrackType = @TrackType)
			    AND rm.WFStatus = 'Pending'

			INSERT INTO #RegistrationTable
				SELECT rm.RegID, rm.ID, rm.Line,rm.TrackType, rm.Module, wf.WFStatus, IIF(r.IsExternal = 1, 'External', 'Internal') AS 'UserType', r.name, r.UENNo, r.Company, r.SBSTContactPersonName, r.CreatedOn FROM TAMS_Registration r, TAMS_Reg_Module rm, TAMS_WFStatus wf
				WHERE r.ID = rm.RegID 
				AND wf.WFType = 'UserRegStatus'
				AND wf.Line = rm.Line
				AND rm.TrackType = @TrackType
				AND wf.WFStatusId = rm.RegStatus
				AND rm.RegStatus in (SELECT DISTINCT(WFStatusId) FROM TAMS_WFStatus WHERE WFStatus = 'Pending System Admin Approval' AND LINE = @Line and TrackType = @TrackType)
				AND rm.Line = @Line
				AND rm.Module = @Module
				AND rm.WFStatus = 'Pending'
		END
		
		IF @Role LIKE '%_SysApprover'
		BEGIN

			DECLARE cur1 CURSOR FOR

			SELECT rm.RegID,  rm.Line,rm.TrackType, rm.Module, rm.RegStatus, rm.WorkflowId, rm.EndorserId FROM TAMS_Registration r, TAMS_Reg_Module rm, TAMS_WFStatus wf
			WHERE r.ID = rm.RegID 
			AND wf.WFType = 'UserRegStatus'
			AND wf.Line = rm.Line
			AND rm.TrackType = @TrackType
			AND wf.WFStatusId = rm.RegStatus
			AND rm.RegStatus in (SELECT DISTINCT(WFStatusId) FROM TAMS_WFStatus WHERE WFStatus = 'Pending System Approver Approval' AND LINE = @Line and TrackType = @TrackType)
			AND rm.Line = @Line
			AND rm.Module = @Module
			AND rm.WFStatus = 'Pending'

			OPEN cur1;

			FETCH NEXT FROM cur1 INTO @RegId, @Line,@TrackType, @Module, @RegStatus, @WorkflowId, @EndorseId;

			WHILE @@FETCH_STATUS = 0
			BEGIN
				SET @RegStatus = @RegStatus - 1;
				SELECT @UpdatedBy=UpdatedBy FROM TAMS_Reg_Module WHERE RegId=@RegId AND Line=@Line and TrackType = @TrackType AND Module=@Module AND RegStatus=@RegStatus

				IF @UpdatedBy != @UserID
				BEGIN

					INSERT INTO #RegistrationTable
						SELECT rm.RegID, rm.ID, rm.Line,rm.TrackType, rm.Module, wf.WFStatus, IIF(r.IsExternal = 1, 'External', 'Internal') AS 'UserType', r.name, r.UENNo, r.Company, r.SBSTContactPersonName, r.CreatedOn FROM TAMS_Registration r, TAMS_Reg_Module rm, TAMS_WFStatus wf
						WHERE r.ID = rm.RegID 
						AND wf.WFType = 'UserRegStatus'
						AND wf.Line = rm.Line
						AND rm.TrackType = @TrackType
						AND wf.WFStatusId = rm.RegStatus
						AND rm.RegStatus in (SELECT DISTINCT(WFStatusId) FROM TAMS_WFStatus WHERE WFStatus = 'Pending System Approver Approval' AND LINE = @Line and TrackType = @TrackType)
						AND rm.Line = @Line
						AND rm.Module = @Module
						AND rm.WFStatus = 'Pending'
						AND rm.RegId = @RegId
				END

				FETCH NEXT FROM cur1 INTO @RegId, @Line,@TrackType, @Module, @RegStatus, @WorkflowId, @EndorseId;
			END
			CLOSE cur1;
			DEALLOCATE cur1;
		END 

		FETCH NEXT FROM cur INTO @Line,@TrackType, @Module, @Role;
	END;

	CLOSE cur;
	DEALLOCATE cur;

	SELECT DISTINCT(RegID), ID, Line,TrackType, Module, WFStatus, UserType, Name, UENNo, Company, SBSTContactPersonName, CreatedOn 
	FROM #RegistrationTable

	DROP TABLE #RegistrationTable;

END
