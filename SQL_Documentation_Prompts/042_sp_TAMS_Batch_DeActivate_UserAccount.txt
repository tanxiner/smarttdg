CREATE PROCEDURE [dbo].[sp_TAMS_Batch_DeActivate_UserAccount]
-- -----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
-- Procedure Name 	: MMS_DeActivate_UserAccount
-- Author		: jeffrey
-- Date Created		: 26/08/2021
-- Purpose		: 
AS

BEGIN
		
	DECLARE @DeAct	BIGINT
	SET @DeAct = 0

	SELECT @DeAct = CAST(ParaValue1 AS BIGINT) FROM TAMS_Parameters
	WHERE ParaCode = 'DeActivateAcct'
	AND GETDATE() BETWEEN EffectiveDate AND ExpiryDate

	UPDATE TAMS_User 
	SET IsActive = 0,
		UpdatedBy = 1,
		UpdatedOn = GETDATE()
			WHERE DATEDIFF(MONTH, CONVERT(DATETIME, CONVERT(NVARCHAR(10), LastLogin, 103), 103), GETDATE()) >= @DeAct 

END
