CREATE procedure  [dbo].[sp_TAMS_Depot_Form_OnLoad]
	@Line		AS NVARCHAR(10) = NULL,
	@TrackType	AS NVARCHAR(50) = NULL,
	@AccessDate	AS NVARCHAR(20) = NULL,
	@AccessType	AS NVARCHAR(20) = NULL,
	@Sectors	AS NVARCHAR(2000) = NULL,
	@PowerSelTxt	AS NVARCHAR(100) = NULL
AS

BEGIN

	DECLARE @isDWONLY as int; 
	set @isDWONLY = 0; 
	DECLARE @DWSectors as nvarchar(500);
	SELECT @DWSectors = ParaValue2 
	FROM TAMS_Parameters
	WHERE ParaCode = 'DWSectors'
	AND ParaValue1 = @Line
	AND CONVERT(DATETIME, @AccessDate, 103) BETWEEN EffectiveDate AND ExpiryDate


	-- check the selection sector is DW only ?
	SELECT @isDWONLY = CASE WHEN (
	   SELECT ISNULL( COUNT(*),0)
	   FROM STRING_SPLIT(@Sectors, ';')
	   WHERE VALUE not IN (
	   SELECT VALUE
	   FROM STRING_SPLIT(@DWSectors, ';'))
	 ) = 0
	 AND (
	   SELECT ISNULL( COUNT(*),0)
	   FROM STRING_SPLIT(@DWSectors, ';')
	   WHERE VALUE not IN (
	   SELECT VALUE
	   FROM STRING_SPLIT(@Sectors, ';'))
	 ) = 0
	THEN 1
	ELSE 0 END  


	SELECT ParaValue1, ParaValue2 
		FROM TAMS_Parameters
			WHERE ParaCode = 'Company'
				AND ParaValue1 = @Line
				AND CONVERT(DATETIME, @AccessDate, 103) BETWEEN EffectiveDate AND ExpiryDate
			ORDER BY [Order]

	SELECT ROW_NUMBER() OVER(ORDER BY ID) as Sno, ID, OperationRequirement, IsAttachment 
		FROM TAMS_Access_Requirement
			WHERE Line = @Line
			AND TrackType= @TrackType
				AND IsPowerReq = 0
				AND IsActive = 1
			ORDER BY [Order]

	IF @AccessType = 'Protection'
		BEGIN
			SELECT ID, OperationRequirement 
				FROM TAMS_Access_Requirement
					WHERE Line = @Line
					and TrackType = @TrackType
						AND IsPowerReq = 1
						AND OperationRequirement = 'Traction Power ON'
						AND IsActive = 1
					ORDER BY [Order]
		END
	ELSE
		BEGIN
			SELECT ID, OperationRequirement 
				FROM TAMS_Access_Requirement
					WHERE Line = @Line
					AND TrackType = @TrackType
						AND IsPowerReq = 1
						AND IsActive = 1
					ORDER BY [Order]
		END
	
	SELECT ID, TypeOfWork, ColourCode 
		FROM [dbo].[TAMS_Type_Of_Work]
			WHERE Line = @Line
			and TrackType = @TrackType
				AND ForSelection = 1
				AND IsActive = 1
			ORDER BY [Order]

	DECLARE @HODRole	 NVARCHAR(50)
	SET @HODRole = ''
	SET @HODRole = @Line + '_ApplicantHOD'

	SELECT a.Userid, a.LoginId, a.[Name] AS EmpName 
		FROM TAMS_User a, TAMS_User_Role b, TAMS_Role c
			WHERE a.Userid = b.UserID
				AND b.RoleID = c.ID
				AND b.Line = c.Line
				AND c.Line = @Line
				and c.TrackType = @TrackType
				AND c.Role = @HODRole
				AND a.IsActive = 1
				AND c.IsActive = 1
	ORDER BY 3

	DECLARE @SelSectorNotGap	NVARCHAR(1000)
	DECLARE @SelSectorGap		NVARCHAR(1000)
	DECLARE @PowerZone			NVARCHAR(1000)
	DECLARE @SPKSZone			NVARCHAR(1000)
	
	SET @SelSectorNotGap = ''
	SET @SelSectorGap = ''
	SET @PowerZone = ''
	SET @SPKSZone = ''

	-- EXEC sp_TAMS_Form_OnLoad 'NEL', '01/02/2021', 'Possession', '143;144;145;180;181;182'

	-- >> Selected Sector IsGap = 0
	SELECT @SelSectorNotGap = @SelSectorNotGap + ISNULL(Sector + ', ', '') 
		FROM TAMS_Sector
			WHERE ID IN (SELECT [VALUE] FROM [dbo].[SPLIT](';', @Sectors))
			AND ISNULL(IsGap, 0) = 0
		GROUP BY Sector

	IF LEN(LTRIM(RTRIM(@SelSectorNotGap))) > 0
		BEGIN
			SET @SelSectorNotGap = LEFT(LTRIM(RTRIM(@SelSectorNotGap)), LEN(LTRIM(RTRIM(@SelSectorNotGap))) - 1)
		END

	  

	SELECT @PowerZone = @PowerZone + ISNULL(PowerSector + ', ', '') 
			FROM TAMS_Power_Sector a, TAMS_Track_Power_Sector b	
				WHERE a.ID = b.PowerSectorId
					AND a.IsActive = 1
					AND b.IsActive = 1
					and b.IsBuffer = 0 
					 AND b.SectorId IN (SELECT [VALUE] FROM [dbo].[SPLIT](';', @Sectors))
		GROUP BY PowerSector		

		IF LEN(LTRIM(RTRIM(@PowerZone))) > 0
		BEGIN
			SET @PowerZone = LEFT(LTRIM(RTRIM(@PowerZone)), LEN(LTRIM(RTRIM(@PowerZone))) - 1)
		END

		SELECT @SPKSZone = @SPKSZone + ISNULL(SPKSZone + ', ', '') 
			FROM TAMS_SPKSZone a, TAMS_Track_SPKSZone b	
				WHERE a.ID = b.SPKSId
					AND a.IsActive = 1
					AND b.IsActive = 1
					 AND b.SectorId IN (SELECT [VALUE] FROM [dbo].[SPLIT](';', @Sectors))
		GROUP BY SPKSZone		

	IF LEN(LTRIM(RTRIM(@SPKSZone))) > 0
		BEGIN
			SET @SPKSZone = LEFT(LTRIM(RTRIM(@SPKSZone)), LEN(LTRIM(RTRIM(@SPKSZone))) - 1)
		END


		SELECT @SelSectorNotGap AS SelSectorNotGap,
		@SelSectorGap AS SelSectorGap,
		@PowerZone AS PowerZone,@SPKSZone as SPKSZone

 

	CREATE TABLE #TmpPossPowSector 
		(PowerSector	NVARCHAR(4000),
		PowerOnOff		NVARCHAR(5),
		NoOFSCD			INT,
		BreakerOut		NVARCHAR(5))
	
	TRUNCATE TABLE #TmpPossPowSector

	DECLARE @StrPowerOnOff NVARCHAR(5)
	DECLARE @StrBreakerOut NVARCHAR(5)
	DECLARE @PowerOnInd	INT

	SET @StrPowerOnOff = ''
	SET @StrBreakerOut = ''

	SET @StrPowerOnOff = CASE WHEN @PowerSelTxt = 'Traction Power ON' THEN 'N' ELSE 'Y' END
	SET @PowerOnInd = CASE WHEN @PowerSelTxt = 'Traction Power ON' THEN 1 ELSE 0 END
	SET @StrBreakerOut = CASE WHEN @PowerSelTxt LIKE '%With Rack Out%' THEN 'Y' ELSE 'N' END

 
	 
			 
			 SELECT @StrPowerOnOff as poweroff,(select top 1 Sector from TAMS_Sector where ID=[VALUE]) as sector,@StrBreakerOut as breaker,0 as NoOFSCD FROM [dbo].[SPLIT](';', @Sectors) 
	
	

	--GET Timeslot based on the weekend/weekday/PH

	 
	DECLARE @DayName	NVARCHAR(20)
	SET @DayName = UPPER(SUBSTRING(DATENAME(dw, CONVERT(DATE, @AccessDate, 103)), 1, 3))

	DECLARE @PHInd	NVARCHAR(5)
	IF (SELECT COUNT(*) FROM TAMS_Calendar
		WHERE HolidayCode = 1
			AND CalendarDate = CONVERT(DATE, GETDATE(), 103)) > 0
		BEGIN
			SET @PHInd = '1'
		END
	ELSE
		BEGIN
			SET @PHInd = '0'
		END
	-- >> If Saturday, Sunday and PH -- Flow to OCC
	-- if Protection, can work on buffer zone, so only need to check for the book sectors has any approved TAR or not. 
	-- if Possession, current sectors are book sectors , so we have to check for all the sectors in TAMS_TAR_Sectors has any approved TAR or not . 
	if @AccessType = 'Protection'
	begin 
			IF (@DayName = 'SAT' OR @DayName = 'SUN' OR @PHInd = '1')
			begin 
				-- get weekend;
				  SELECT   ParaValue2 , CASE WHEN((Select count(*) FROM TAMS_TAR a, TAMS_TAR_Sector b where a.Id = b.TARId AND  Line = @Line and TrackType='Depot' and AccessDate = convert(date,@AccessDate,103)  and AccessTimeSlot=Paravalue2 and TARStatusId = 9 
				 and b.SectorId in (select VALUE  FROM STRING_SPLIT(@Sectors, ';') )
				 and b.IsBuffer = 0
				 and a.AccessType = 'Possession'
				 ) > 0  ) THEN 'false'  ELSE 'true' end as DDLEnable
				 FROM TAMS_Parameters
					WHERE ParaCode = 'AccessTimeSlot_Weekend'
						AND ParaValue1 = @Line
						AND CONVERT(DATETIME, @AccessDate, 103) BETWEEN EffectiveDate AND ExpiryDate
					ORDER BY [Order]
			end;
			else if(@isDWONLY = 1 )  
			begin 
				-- get weekday DW;
				 SELECT   ParaValue2 , CASE WHEN((Select count(*) FROM TAMS_TAR a, TAMS_TAR_Sector b where a.Id = b.TARId AND  Line = @Line and TrackType='Depot' and AccessDate = convert(date,@AccessDate,103)  and AccessTimeSlot=Paravalue2 and TARStatusId = 9 
				 and b.SectorId in (select VALUE  FROM STRING_SPLIT(@Sectors, ';') )
				 and b.IsBuffer = 0
				  and a.AccessType = 'Possession'
				 ) > 0  ) THEN 'false'  ELSE 'true' end as DDLEnable
				FROM TAMS_Parameters
					WHERE ParaCode = 'AccessTimeSlot_Weekday_DW'
						AND ParaValue1 = @Line
						AND CONVERT(DATETIME, @AccessDate, 103) BETWEEN EffectiveDate AND ExpiryDate
					ORDER BY [Order]
			end; 
			else 
			begin 
				-- get weekday 
				 SELECT   ParaValue2 , CASE WHEN((Select count(*) FROM TAMS_TAR a, TAMS_TAR_Sector b where a.Id = b.TARId AND  Line = @Line and TrackType='Depot' and AccessDate = convert(date,@AccessDate,103)  and AccessTimeSlot=Paravalue2 and TARStatusId = 9 
				 and b.SectorId in (select VALUE  FROM STRING_SPLIT(@Sectors, ';') )
				 and b.IsBuffer = 0
				  and a.AccessType = 'Possession'
				 ) > 0  ) THEN 'false'  ELSE 'true' end as DDLEnable
				FROM TAMS_Parameters
					WHERE ParaCode = 'AccessTimeSlot_Weekday'
						AND ParaValue1 = @Line
						AND CONVERT(DATETIME, @AccessDate, 103) BETWEEN EffectiveDate AND ExpiryDate
					ORDER BY [Order]
			end;
	end; 
	else 
	begin 

		 IF (@DayName = 'SAT' OR @DayName = 'SUN' OR @PHInd = '1')
			begin 
				-- get weekend;
				  SELECT   ParaValue2 , CASE WHEN((Select count(*) FROM TAMS_TAR a, TAMS_TAR_Sector b where a.Id = b.TARId AND  Line = @Line and TrackType='Depot' and AccessDate = convert(date,@AccessDate,103)  and AccessTimeSlot=Paravalue2 and TARStatusId = 9 
				 and b.SectorId in (select VALUE  FROM STRING_SPLIT(@Sectors, ';') )
				  ) > 0  ) THEN 'false'  ELSE 'true' end as DDLEnable
				 FROM TAMS_Parameters
					WHERE ParaCode = 'AccessTimeSlot_Weekend'
						AND ParaValue1 = @Line
						AND CONVERT(DATETIME, @AccessDate, 103) BETWEEN EffectiveDate AND ExpiryDate
					ORDER BY [Order]
			end;
			else if(@isDWONLY = 1 )  
			begin 
				-- get weekday DW;
				 SELECT   ParaValue2 , CASE WHEN((Select count(*) FROM TAMS_TAR a, TAMS_TAR_Sector b where a.Id = b.TARId AND  Line = @Line and TrackType='Depot' and AccessDate = convert(date,@AccessDate,103)  and AccessTimeSlot=Paravalue2 and TARStatusId = 9 
				 and b.SectorId in (select VALUE  FROM STRING_SPLIT(@Sectors, ';') )
				  ) > 0  ) THEN 'false'  ELSE 'true' end as DDLEnable
				FROM TAMS_Parameters
					WHERE ParaCode = 'AccessTimeSlot_Weekday_DW'
						AND ParaValue1 = @Line
						AND CONVERT(DATETIME, @AccessDate, 103) BETWEEN EffectiveDate AND ExpiryDate
					ORDER BY [Order]
			end; 
			else 
			begin 
				-- get weekday 
				 SELECT   ParaValue2 , CASE WHEN((Select count(*) FROM TAMS_TAR a, TAMS_TAR_Sector b where a.Id = b.TARId AND  Line = @Line and TrackType='Depot' and AccessDate = convert(date,@AccessDate,103)  and AccessTimeSlot=Paravalue2 and TARStatusId = 9 
				 and b.SectorId in (select VALUE  FROM STRING_SPLIT(@Sectors, ';') )
				  ) > 0  ) THEN 'false'  ELSE 'true' end as DDLEnable
				FROM TAMS_Parameters
					WHERE ParaCode = 'AccessTimeSlot_Weekday'
						AND ParaValue1 = @Line
						AND CONVERT(DATETIME, @AccessDate, 103) BETWEEN EffectiveDate AND ExpiryDate
					ORDER BY [Order]
			end;

	end;

*/
END
