CREATE PROCEDURE [dbo].[sp_TAMS_TOA_Save_ProtectionType]
	@toaid as int,
	@protectiontype as nvarchar(50),
	@Message		AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

SET @Message = ''

		BEGIN

			if @protectiontype<>'B'
			BEGIN
				delete from TAMS_TOA_PointNo
				where TAMS_TOA_PointNo.TOAID=@TOAID
			END
		
			UPDATE TAMS_TOA
			SET ProtectionType = @protectiontype
			WHERE Id = @TOAID

			SET @Message = ''
		END

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR SELECTING PROTECTION TYPE'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END
