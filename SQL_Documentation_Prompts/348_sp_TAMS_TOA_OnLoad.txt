### ⬇️ RAW SQL INPUT ⬇️
CREATE PROCEDURE [dbo].[sp_TAMS_TOA_OnLoad]
	@TOAID	BIGINT
AS

BEGIN
	
	SELECT a.QRLocation, 
			b.TARNo, 
			dbo.QTS_MaskText(dbo.DecryptString(a.InChargeNRIC), 4) AS InchargeNRIC,
			ISNULL(a.InChargeName, '') AS InChargeName,
			ISNULL(a.MobileNo, '') AS MobileNo,
			ISNULL(a.TetraRadioNo, '') AS TetraRadioNo,
			ISNULL(b.TVFMode, 'N.A') AS TVFMode,
			ISNULL(a.NoOfParties, 1) AS NoOfParties,
			ISNULL(a.TOANo, '') AS TOANo,
			ISNULL(a.ProtectionType,'') As ProtectionType
		FROM TAMS_TOA a, TAMS_TAR b
			WHERE a.Id = @TOAID
				AND a.TARId = b.Id
		
END
