CREATE PROCEDURE [dbo].[sp_TAMS_TOA_Get_Station_Name]
	@Line			NVARCHAR(10),
	@StationName	NVARCHAR(20)
AS

BEGIN
	
	SELECT StationCode 
		FROM TAMS_Station
			WHERE Line = @Line 
				AND StationName = @StationName

		
END
