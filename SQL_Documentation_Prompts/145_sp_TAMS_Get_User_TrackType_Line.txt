create PROCEDURE [dbo].[sp_TAMS_Get_User_TrackType_Line] 
	@Line as nvarchar(100) = null,
	@UserId NVARCHAR(100) = NULL
AS
BEGIN

 
		SELECT DISTINCT(TrackType) AS 'TrackType'
		FROM TAMS_User_Role ur, TAMS_User u
		WHERE ur.UserID = u.Userid
		AND u.LoginID = @UserID
		AND ur.Line = @Line;
		 
		  
	 
	 end;
