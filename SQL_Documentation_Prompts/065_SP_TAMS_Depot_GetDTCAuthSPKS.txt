CREATE PROCEDURE [dbo].[SP_TAMS_Depot_GetDTCAuthSPKS]
	-- Add the parameters for the stored procedure here
	@accessDate as Date
AS
BEGIN
	-- SET NOCOUNT ON added to prevent extra result sets from
	-- interfering with SELECT statements.
	SET NOCOUNT ON;

	Select AuthID,TAMS_Depot_DTCAuth_SPKS.SPKSID,TAMS_Depot_DTCAuth_SPKS.ProtectOffType,TAMS_Depot_DTCAuth_SPKS.ProtectOffTiming,TAMS_Depot_DTCAuth_SPKS.ProtectOnType,TAMS_Depot_DTCAuth_SPKS.ProtectOnTiming,TAMS_Depot_DTCAuth_SPKS.StatusID,ProtectOffActionBy.Name as ProtectOffActionBy,ProtectOnActionBy.Name as ProtectOnActionBy,Z.ID as workflowid from TAMS_Depot_Auth 
	LEFT JOIN TAMS_Depot_DTCAuth_SPKS ON TAMS_Depot_Auth.ID=TAMS_Depot_DTCAuth_SPKS.AuthID
	LEFT JOIN (Select * from TAMS_WFStatus where WFType='DTCAuth') Z ON Z.WFStatusId=StatusID
	LEFT JOIN TAMS_User ProtectOffActionBy ON ProtectOffActionBy.LoginId=ProtectOffActionBy
	LEFT JOIN TAMS_User ProtectOnActionBy ON ProtectOnActionBy.LoginId=ProtectOnActionBy
	where TAMS_Depot_Auth.AccessDate=@accessDate

END
