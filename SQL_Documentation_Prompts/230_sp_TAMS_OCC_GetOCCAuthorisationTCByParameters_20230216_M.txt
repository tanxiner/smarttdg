CREATE PROCEDURE [dbo].[sp_TAMS_OCC_GetOCCAuthorisationTCByParameters_20230216_M]
	@UserID		as int,
	@Line AS nvarchar(10) = NULL,
	@OperationDate as date,
	@AccessDate as date

AS

DECLARE @WorkflowId as int,
		@OCCAuthID as int,
		@EndorserID as int,
		@EndorserLevel as int,
		@EndorserTitle as nvarchar(250),
		@WFStatus as nvarchar(50),
		@ActionOn as datetime

CREATE TABLE #TMP
(
	TractionPowerId int not null,	
	StationName nvarchar(50)
);

CREATE TABLE #TMP_Endorser
(
	ID int not null,	
	Line nvarchar(10),
	[Level] int,
	RoleId int,
	Title nvarchar(250)
);

CREATE TABLE #TMP_OCCAuthTC
(
	SNO int not null,
	ID int not null,	
	OCCAuthID int not null,
	Remark nvarchar(1000),
	PFRRemark nvarchar(1000),
	IsBuffer bit,
	PowerOn bit,
	Line nvarchar(10),
	TractionPowerSection nvarchar(100),
	HSCBName nvarchar(100),
	StationName nvarchar(50),
	TC nvarchar(10),
	Train nvarchar(50),
	TrainClearCert nvarchar(8),
	AuthForTrackAccess nvarchar(8),
	LineClearCertTOA nvarchar(8),
	LineClearCertSCD nvarchar(8),
	AuthForTrainInsert nvarchar(8)
);

IF @Line = 'DTL'
BEGIN

	SELECT @WorkflowId = ID FROM [TAMS_Workflow]
						WHERE Line = @Line
						AND  WorkflowType = 'OCCAuth'
						AND IsActive = 1
						AND EffectiveDate <= GETDATE()
						AND ExpiryDate >= GETDATE()

	INSERT INTO #TMP_Endorser(ID, Line, [Level], RoleId, Title)
	SELECT ID, Line, [Level], RoleId, Title FROM [TAMS_Endorser]
	WHERE Line = 'DTL'
	AND WorkflowId = @WorkflowId
	AND RoleId = 13
	--AND Level in (1,9,10,11,20)

	INSERT INTO #TMP(TractionPowerId, StationName)
	SELECT 
	   Id,
	   STUFF((SELECT ';' + S.StationName
			  FROM [TAMS_Traction_Power_Detail] TPD, [TAMS_Station] S
			  WHERE TPD.TractionPowerId = TP.ID
			  AND TPD.StationId = S.ID
			  AND  TPD.TractionPowerType = 'Station'
			  FOR XML PATH('')), 1, 1, '') [StationName]
	FROM [TAMS_Traction_Power] TP
	WHERE TP.Line = 'DTL'
	GROUP BY TP.ID

	INSERT INTO #TMP_OCCAuthTC(
	SNO, 
	ID, 
	OCCAuthID,
	Remark,
	PFRRemark,
	IsBuffer,
	PowerOn,
	Line, 
	TractionPowerSection,  
	HSCBName,
	StationName,
	TC,
	Train,
	TrainClearCert,
	AuthForTrackAccess,
	LineClearCertTOA,
	LineClearCertSCD,
	AuthForTrainInsert)
	SELECT ROW_NUMBER() OVER(ORDER BY tp.ID ASC) AS SNO
		  ,tp.[ID]
		  ,oa.[ID]
		  ,oa.[Remark]
		  ,oa.[PFRRemark]
		  ,oa.[IsBuffer]
		  ,oa.[PowerOn]
		  ,tp.[Line]
		  ,[TractionPowerSection]
		  ,[HSCBName]
		  ,[StationName]
		  ,[TC]
		  ,[Train]
		  ,'' AS TrainClearCert
		  ,'' AS AuthForTrackAccess
		  ,'' AS LineClearCertTOA
		  ,'' AS LineClearCertSCD
		  ,'' AS AuthForTrainInsert
	  FROM [TAMS_Traction_Power] tp, #TMP tmp, [TAMS_OCC_Auth] oa
	  WHERE tp.ID = tmp.TractionPowerId
	  AND tmp.TractionPowerId = oa.TractionPowerId
	  AND EffectiveDate <= GETDATE() and ExpiryDate >= GETDATE()
	  AND IsActive = 1
	  AND oa.OperationDate = @OperationDate
	  AND oa.AccessDate = @AccessDate
	  AND TC in 
	  (SELECT RosterCode FROM TAMS_OCC_Duty_Roster 
	   WHERE OperationDate = CAST(@OperationDate as DATE)
	   and [shift] = 3
	   and [IsActive] = 1
	   and [DutyStaffId] = @UserID
	   )


	DECLARE cur CURSOR FOR
	SELECT OCCAuthID FROM #TMP_OCCAuthTC


	OPEN cur;

	FETCH NEXT FROM cur INTO @OCCAuthID
	WHILE @@FETCH_STATUS = 0
	BEGIN

		DECLARE cur1 CURSOR FOR
		SELECT ID, [Level], Title FROM #TMP_Endorser

		OPEN Cur1;

		FETCH NEXT FROM Cur1 INTO @EndorserID, @EndorserLevel, @EndorserTitle;
		WHILE @@FETCH_STATUS = 0
		BEGIN
			IF @EndorserID = 97
			BEGIN
				SELECT @WFStatus = WFStatus, @ActionOn = ActionOn 
				FROM [TAMS_OCC_Auth_Workflow] 
				WHERE OCCAuthId = @OCCAuthID AND OCCAuthEndorserId = 97;
				IF @WFStatus = 'Pending'
				BEGIN
					UPDATE #TMP_OCCAuthTC
					SET TrainClearCert = @WFStatus
					WHERE OCCAuthID = @OCCAuthID
				END;
				IF @WFStatus = 'Completed'
				BEGIN
					UPDATE #TMP_OCCAuthTC
					SET TrainClearCert = CONVERT(TIME, @ActionOn)
					WHERE OCCAuthID = @OCCAuthID
				END;

				SET @WFStatus = '';
				SET @ActionOn = '';
			END;

			IF @EndorserID = 105
			BEGIN
				SELECT @WFStatus = WFStatus, @ActionOn = ActionOn 
				FROM [TAMS_OCC_Auth_Workflow] 
				WHERE OCCAuthId = @OCCAuthID AND OCCAuthEndorserId = 105;
				IF @WFStatus = 'Pending'
				BEGIN
					UPDATE #TMP_OCCAuthTC
					SET AuthForTrackAccess = @WFStatus
					WHERE OCCAuthID = @OCCAuthID
				END;
				IF @WFStatus = 'Completed'
				BEGIN
					UPDATE #TMP_OCCAuthTC
					SET AuthForTrackAccess = CONVERT(TIME, @ActionOn)
					WHERE OCCAuthID = @OCCAuthID
				END;
				SET @WFStatus = '';
				SET @ActionOn = '';
			END;
			IF @EndorserID = 106
			BEGIN
				SELECT @WFStatus = WFStatus, @ActionOn = ActionOn 
				FROM [TAMS_OCC_Auth_Workflow] 
				WHERE OCCAuthId = @OCCAuthID AND OCCAuthEndorserId = 106;
				IF @WFStatus = 'Pending'
				BEGIN
					UPDATE #TMP_OCCAuthTC
					SET LineClearCertTOA = @WFStatus
					WHERE OCCAuthID = @OCCAuthID
				END;
				IF @WFStatus = 'Completed'
				BEGIN
					UPDATE #TMP_OCCAuthTC
					SET LineClearCertTOA = CONVERT(TIME, @ActionOn)
					WHERE OCCAuthID = @OCCAuthID
				END;
				SET @WFStatus = '';
				SET @ActionOn = '';
			END;

			IF @EndorserID = 107
			BEGIN
				SELECT @WFStatus = WFStatus, @ActionOn = ActionOn 
				FROM [TAMS_OCC_Auth_Workflow] 
				WHERE OCCAuthId = @OCCAuthID AND OCCAuthEndorserId = 107;
				IF @WFStatus = 'Pending'
				BEGIN
					UPDATE #TMP_OCCAuthTC
					SET LineClearCertSCD = @WFStatus
					WHERE OCCAuthID = @OCCAuthID
				END;
				IF @WFStatus = 'Remove' OR @WFStatus = 'N.A.'
				BEGIN
					UPDATE #TMP_OCCAuthTC
					SET LineClearCertSCD = CONVERT(TIME, @ActionOn)
					WHERE OCCAuthID = @OCCAuthID
				END;
				SET @WFStatus = '';
				SET @ActionOn = '';
			END;

			IF @EndorserID = 116
			BEGIN
				SELECT @WFStatus = WFStatus, @ActionOn = ActionOn 
				FROM [TAMS_OCC_Auth_Workflow] 
				WHERE OCCAuthId = @OCCAuthID AND OCCAuthEndorserId = 116;
				IF @WFStatus = 'Pending'
				BEGIN
					UPDATE #TMP_OCCAuthTC
					SET AuthForTrainInsert = @WFStatus
					WHERE OCCAuthID = @OCCAuthID
				END;
				IF @WFStatus = 'Completed'
				BEGIN
					UPDATE #TMP_OCCAuthTC
					SET AuthForTrainInsert = CONVERT(TIME, @ActionOn)
					WHERE OCCAuthID = @OCCAuthID
				END;
				SET @WFStatus = '';
				SET @ActionOn = '';
			END;

			FETCH NEXT FROM Cur1 INTO @EndorserID, @EndorserLevel, @EndorserTitle;
		END

		CLOSE Cur1;
		DEALLOCATE Cur1;

		FETCH NEXT FROM cur INTO @OCCAuthID;

	END;

	CLOSE cur;
	DEALLOCATE cur;


	SELECT * FROM #TMP_OCCAuthTC;
	--SELECT * FROM #TMP;

END

DROP TABLE #TMP;

DROP TABLE #TMP_Endorser;

DROP TABLE #TMP_OCCAuthTC;
