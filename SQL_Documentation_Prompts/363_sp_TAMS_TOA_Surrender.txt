CREATE PROCEDURE [dbo].[sp_TAMS_TOA_Surrender]
	@TOAID			AS BIGINT = 0,
	@Message		AS NVARCHAR(500) = NULL OUTPUT
AS

BEGIN

DECLARE	@IntrnlTrans	TINYINT
SELECT 	@IntrnlTrans 	= 0	

IF (@@TRANCOUNT = 0) 
BEGIN
	SET @IntrnlTrans = 1
	BEGIN TRAN
END

-- >>
UPDATE [dbo].[TAMS_TOA]
SET TOAStatus	= 4,
	SurrenderTime	=	GETDATE(),
	UpdatedOn		=	GETDATE()
WHERE ID = @TOAID and TOAStatus<>6

INSERT INTO TAMS_TOA_Audit
SELECT '', GETDATE(), 'U', * FROM TAMS_TOA
	WHERE Id = @TOAID
-- >>

IF @@ERROR <> 0 BEGIN
	SET @Message = 'ERROR UPDATING TAMS_TOA'
	GOTO TRAP_ERROR
END

FORCE_EXIT_PROC:	
	IF @IntrnlTrans = 1 COMMIT TRAN
	RETURN @Message 

TRAP_ERROR:
	IF @IntrnlTrans = 1 ROLLBACK TRAN
	RETURN @Message

END
