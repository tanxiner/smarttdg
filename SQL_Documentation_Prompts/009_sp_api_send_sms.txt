CREATE PROCEDURE [dbo].[sp_api_send_sms] 
	@contactno nvarchar(MAX) = null,
	@subject nvarchar(500) = null,
	@msg nvarchar(MAX) = null,
	@ret nvarchar(5) output
AS

BEGIN

DECLARE @AlertID AS INTEGER;

DECLARE @RowID INTEGER
DECLARE @Value NVARCHAR(100)
DECLARE @DerContactNo NVARCHAR(100)

--SET @contactno = @contactno + '@sbstsms2.com,'
SET @contactno = @contactno + ',83681010'
--SET @contactno = '96734774'

/*
	DECLARE @ret nvarchar(5)
	SET @ret = ''
	exec sp_api_send_sms '96734774,91832674', 'TAMS SMS', 'TOA...', @ret OUTPUT
	SELECT @ret
*/

SET @DerContactNo = ''

DECLARE @Cur02 CURSOR;	

	SET @Cur02 = CURSOR FAST_FORWARD 
	FOR

		SELECT * FROM [dbo].[SPLIT] (',', @contactno)
				
	OPEN @Cur02
	
	FETCH NEXT FROM @Cur02 INTO @RowID, @Value
	
	WHILE @@FETCH_STATUS = 0 
	
		BEGIN
			SET @DerContactNo = @DerContactNo + @Value + ','
			
			FETCH NEXT FROM @Cur02 INTO @RowID, @Value
		END 	

	CLOSE @Cur02
	
	DEALLOCATE @Cur02

EXEC  [dbo].SMSEAlertQ_EnQueue 'TAMSadmin', --@Sender,
	@Subject,		-- @Subject,
	'',				--@Sys,
	'',				--@Greetings,   
	@msg,			--@AlertMsg,   -- do not send null for ntext
	'TAMSadmin',		--@UserId,
	@DerContactNo,		--@SendTo,
	'',				--@CC, -- do not send null for ntext
	'',				--@BCC, -- do not send null for ntext
	',',			--@Separator,
	@AlertID OUTPUT

SET @ret = '0'

RETURN @ret 

END
