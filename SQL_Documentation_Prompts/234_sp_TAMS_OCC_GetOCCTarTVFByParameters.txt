CREATE PROCEDURE [dbo].[sp_TAMS_OCC_GetOCCTarTVFByParameters]
	@StationId	as int,
	@AccessDate as date

AS

DECLARE @TVF_Ack_CNT int;
DECLARE @Id as int,
		@TarNo as nvarchar(50),
		@PersonInCharge nvarchar(500),
		@TVFMode as nvarchar(50),
		@TVFDirection as nvarchar(10)


CREATE TABLE #TMP_TVF
(
	Id int not null,	
	TarNo nvarchar(50),
	PersonInCharge nvarchar(500),
	TVFMode nvarchar(20),
	TVFDirection nvarchar(10)
);

CREATE TABLE #TMP_TAR_TVF
(
	Id int not null,	
	TarNo nvarchar(50),
	PersonInCharge nvarchar(500),
	TVFDirection1 bit,
	TVFDirection2 bit,
	TVFMode nvarchar(20)
);


BEGIN

	TRUNCATE TABLE #TMP_TVF;

	TRUNCATE TABLE #TMP_TAR_TVF;

	INSERT INTO #TMP_TVF(Id, TarNo, PersonInCharge, TVFDirection, TVFMode)
	SELECT tar.id, tar.tarno, inchargename, tvf.TVFDirection, tar.TVFMode
	FROM TAMS_TAR tar
	INNER JOIN TAMS_TAR_TVF tvf
	ON tar.Id = tvf.TARID
	LEFT JOIN TAMS_TOA toa
	ON  tar.id = toa.TARId
	WHERE tvf.TVFStationId = @StationId
	AND tar.AccessDate = @AccessDate

	DECLARE cur CURSOR FOR
	SELECT Id, TarNo, PersonInCharge, TVFDirection, TVFMode FROM #TMP_TVF ORDER BY ID ASC

	OPEN cur;

	FETCH NEXT FROM cur INTO @Id, @TarNo, @PersonInCharge, @TVFDirection, @TVFMode

	WHILE @@FETCH_STATUS = 0
	BEGIN
		IF NOT EXISTS
		(
			SELECT ID
			FROM #TMP_TAR_TVF
			WHERE ID = @ID
		)
			BEGIN
				IF @TVFDirection = 'XB'
				BEGIN
					INSERT INTO #TMP_TAR_TVF(Id, TarNo, PersonInCharge, TVFDirection1, TVFDirection2, TVFMode)
					VALUES(@Id, @TarNo, @PersonInCharge, 1, 0, @TVFMode)
				END;
				IF @TVFDirection = 'BB'
				BEGIN
					INSERT INTO #TMP_TAR_TVF(Id, TarNo, PersonInCharge, TVFDirection1, TVFDirection2, TVFMode)
					VALUES(@Id, @TarNo, @PersonInCharge, 0, 1, @TVFMode)
				END;
			END
		ELSE
			BEGIN
				IF @TVFDirection = 'XB'
				BEGIN
					UPDATE #TMP_TAR_TVF
					SET TVFDirection1 = 1;
				END;
				IF @TVFDirection = 'BB'
				BEGIN
					UPDATE #TMP_TAR_TVF
					SET TVFDirection2 = 1;
				END;
			END;
		FETCH NEXT FROM Cur INTO @Id, @TarNo, @PersonInCharge, @TVFDirection, @TVFMode;
	END;
	CLOSE Cur;
	DEALLOCATE Cur;

	SELECT * FROM #TMP_TAR_TVF;
	--SELECT * FROM #TMP;
END;

DROP TABLE #TMP_TVF;

DROP TABLE #TMP_TAR_TVF;
