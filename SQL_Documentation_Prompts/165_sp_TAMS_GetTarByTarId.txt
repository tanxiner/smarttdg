CREATE PROCEDURE [dbo].[sp_TAMS_GetTarByTarId]
	@TarId AS integer = 0
AS
BEGIN
	SELECT [Id],
	[Line],
	[TARNo],
	[TARType],
	[Company],
	[Designation],
	[Name],
	[OfficeNo],
	[MobileNo],
	[Email],
	[SubmitDate],
	[AccessDate],
	[AccessTimeFrom],
	[AccessTimeTo],
	[AccessLocation],
	[AccessType],
	[IsNeutralGap],
	[IsExclusive],
	[DescOfWork],
	[ARRemark],
	[Is13ASocket],
	[CrossOver], 
	[ProtectionType], 
	[WithdrawRemark], 
	Convert(varchar,WithdrawDate,103) as WithdrawDate,
	(Select top 1 Name from TAMS_User where Userid = WithdrawBy) as WithdrawBy, 
	TARStatusId, 
	(Select top 1 WFStatus from TAMS_WFStatus where WFStatusId = TARStatusId and TAMS_WFStatus.Line = TAMS_TAR.Line and TAMS_WFStatus.TrackType =TAMS_TAR.TrackType  and WFType='TARWFStatus')  as TARStatus
	FROM TAMS_TAR
	WHERE [Id] = @TarId;
END
